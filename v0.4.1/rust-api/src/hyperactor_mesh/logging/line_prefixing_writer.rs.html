10use std::task::Context as TaskContext;
11use std::task::Poll;

13use tokio::io;

15/// A line-prefixing AsyncWrite wrapper that adds "[<local_rank>] " prefix to each line.
16pub struct LinePrefixingWriter<W>
17where
18    W: io::AsyncWrite + Unpin,
19{
20    inner: W,
21    prefix: Vec<u8>,   // Pre-formatted prefix like "[42] "
22    need_prefix: bool, // true when we need to write a prefix before the next data
23}
24
25impl<W> LinePrefixingWriter<W>
26where
27    W: io::AsyncWrite + Unpin,
28{
29    /// Creates a new LinePrefixingWriter that will prefix each line with "[<local_rank>] ".
30    #[allow(dead_code)]
31    pub fn new(local_rank: usize, inner: W) -> Self {
32        let prefix = format!("[{}] ", local_rank).into_bytes();
33        Self {
34            inner,
35            prefix,
36            need_prefix: true, // Start by needing a prefix
37        }
38    }
39}
40
41impl<W> io::AsyncWrite for LinePrefixingWriter<W>
42where
43    W: io::AsyncWrite + Unpin,
44{
45    fn poll_write(
46        self: Pin<&mut Self>,
47        cx: &mut TaskContext<'_>,
48        buf: &[u8],
49    ) -> Poll<Result<usize, io::Error>> {
50        let this = self.get_mut();
51        let mut output = Vec::with_capacity(buf.len());
52
53        let mut pos = 0;
54        while pos < buf.len() {
55            if this.need_prefix {
56                output.extend_from_slice(&this.prefix);
57                this.need_prefix = false;
58            }
59
60            let remaining = &buf[pos..];
61            if let Some(newline_offset) = remaining.iter().position(|&b| b == b'\n') {
62                // Write up to and including the newline
63                let end_pos = newline_offset + 1;
64                output.extend_from_slice(&remaining[..end_pos]);
65                pos += end_pos;
66                this.need_prefix = true;
67            } else {
68                output.extend_from_slice(remaining);
69                break;
70            }
71        }
72
73        match Pin::new(&mut this.inner).poll_write(cx, &output) {
74            Poll::Ready(Ok(n)) => {
75                if n == output.len() {
76                    // Need to return buf.len() bytes written to the outer call,
77                    // which is unaware that we are making the actual write
78                    // slightly larger
79                    Poll::Ready(Ok(buf.len()))
80                } else {
81                    // This case is annoying to handle, so just return an error as
82                    // partial writes are extremely rare
83                    Poll::Ready(Err(io::Error::other("Partial write to inner writer")))
84                }
85            }
86            Poll::Ready(err) => Poll::Ready(err),
87            Poll::Pending => Poll::Pending,
88        }
89    }
90
91    fn poll_flush(self: Pin<&mut Self>, cx: &mut TaskContext<'_>) -> Poll<Result<(), io::Error>> {
92        let this = self.get_mut();
93        Pin::new(&mut this.inner).poll_flush(cx)
94    }
95
96    fn poll_shutdown(
97        self: Pin<&mut Self>,
98        cx: &mut TaskContext<'_>,
99    ) -> Poll<Result<(), io::Error>> {
100        let this = self.get_mut();
101        Pin::new(&mut this.inner).poll_shutdown(cx)
102    }
103}

152    // Mock writer that can simulate partial writes and pending states
153    struct PartialMockWriter {
154        data: Arc<Mutex<Vec<u8>>>,
155        write_behavior: Arc<Mutex<Vec<WriteResult>>>,
156        call_count: Arc<Mutex<usize>>,
157    }
158
159    #[derive(Clone, Debug)]
160    enum WriteResult {
161        Full,           // Write all bytes successfully
162        Partial(usize), // Write only N bytes
163        Pending,        // Return Poll::Pending
164        Error,          // Return an error
165    }
166
167    impl PartialMockWriter {
168        fn new(write_behavior: Vec<WriteResult>) -> (Self, Arc<Mutex<Vec<u8>>>) {
169            let data = Arc::new(Mutex::new(Vec::new()));
170            (
171                Self {
172                    data: data.clone(),
173                    write_behavior: Arc::new(Mutex::new(write_behavior)),
174                    call_count: Arc::new(Mutex::new(0)),
175                },
176                data,
177            )
178        }
179    }
180
181    impl io::AsyncWrite for PartialMockWriter {
182        fn poll_write(
183            self: Pin<&mut Self>,
184            _cx: &mut TaskContext<'_>,
185            buf: &[u8],
186        ) -> Poll<Result<usize, io::Error>> {
187            let mut call_count = self.call_count.lock().unwrap();
188            let behaviors = self.write_behavior.lock().unwrap();
189
190            if *call_count >= behaviors.len() {
191                // Default to writing all bytes if no more behaviors specified
192                let mut data = self.data.lock().unwrap();
193                data.extend_from_slice(buf);
194                return Poll::Ready(Ok(buf.len()));
195            }
196
197            let behavior = behaviors[*call_count].clone();
198            *call_count += 1;
199
200            match behavior {
201                WriteResult::Full => {
202                    let mut data = self.data.lock().unwrap();
203                    data.extend_from_slice(buf);
204                    Poll::Ready(Ok(buf.len()))
205                }
206                WriteResult::Partial(n) => {
207                    let n = n.min(buf.len());
208                    let mut data = self.data.lock().unwrap();
209                    data.extend_from_slice(&buf[..n]);
210                    Poll::Ready(Ok(n))
211                }
212                WriteResult::Pending => Poll::Pending,
213                WriteResult::Error => Poll::Ready(Err(io::Error::other("Mock error"))),
214            }
215        }
216
217        fn poll_flush(
218            self: Pin<&mut Self>,
219            _cx: &mut TaskContext<'_>,
220        ) -> Poll<Result<(), io::Error>> {
221            Poll::Ready(Ok(()))
222        }
223
224        fn poll_shutdown(
225            self: Pin<&mut Self>,
226            _cx: &mut TaskContext<'_>,
227        ) -> Poll<Result<(), io::Error>> {
228            Poll::Ready(Ok(()))
229        }
230    }

373    async fn test_line_prefixing_writer_partial_prefix_write() {
374        // Test case: prefix itself is written partially - should return error
375        let (mock_writer, data) = PartialMockWriter::new(vec![
376            WriteResult::Partial(2), // Only "[4" of "[42] " is written
377        ]);
378        let mut writer = LinePrefixingWriter::new(42, mock_writer);
379
380        // This should return an error for partial prefix write
381        let result = std::future::poll_fn(|cx| {
382            use tokio::io::AsyncWrite;

384        })
385        .await;
