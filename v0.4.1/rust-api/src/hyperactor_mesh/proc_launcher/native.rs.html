9//! Native OS process launcher.
10//!
11//! This module provides [`NativeProcLauncher`], a [`ProcLauncher`]
12//! backend that runs bootstrap procs as ordinary local OS processes
13//! using [`tokio::process::Command`].
14//!
15//! ## Role in the proc-launching architecture
16//!
17//! [`BootstrapProcManager`](crate::bootstrap::BootstrapProcManager)
18//! owns the *lifecycle* model (readiness tracking, supervision,
19//! bookkeeping) and delegates only the *mechanics* of
20//! starting/stopping a proc to a launcher. `NativeProcLauncher` is
21//! the simplest such backend and serves as the reference
22//! implementation for the [`ProcLauncher`] contract.
23//!
24//! In particular:
25//! - **Readiness is callback-driven.** A proc is considered ready
26//!   only when it invokes the bootstrap callback address carried in
27//!   [`Bootstrap`]. This module does not determine readiness.
28//! - **Terminal status is channel-driven.** `launch` returns a
29//!   [`LaunchResult`] whose `exit_rx` resolves exactly once with a
30//!   [`ProcExitResult`]. The native backend implements this by
31//!   awaiting `child.wait()` in a background task and mapping the OS
32//!   [`ExitStatus`] into a [`ProcExitKind`].
33//! - **Termination is initiation-only.** [`ProcLauncher::terminate`]
34//!   and [`ProcLauncher::kill`] send signals and return without
35//!   waiting for exit; completion is observed by awaiting `exit_rx`
36//!   (or a higher-level handle built on it).
37//!
38//! ## Bootstrap + diagnostics plumbing
39//!
40//! The bootstrap payload and a human-readable process name are passed
41//! to the child via environment variables:
42//! - [`BOOTSTRAP_MODE_ENV`] carries an env-safe encoding of [`Bootstrap`].
43//! - [`PROCESS_NAME_ENV`] carries a diagnostic name derived from the
44//!   [`ProcId`] and hostname (useful for logs/`ps`/systemd).
45//! - [`BOOTSTRAP_LOG_CHANNEL`] is optionally set when the manager
46//!   provisions a mesh log-forwarding channel in [`LaunchOptions`].
47//!
48//! ## Stdio behavior
49//!
50//! Stdio configuration is driven by [`LaunchOptions::want_stdio`]:
51//! - When true, stdout/stderr are piped and returned as
52//!   [`StdioHandling::Captured`] so the manager can attach forwarding
53//!   / tailing infrastructure.
54//! - When false, stdio is inherited from the parent process.
55//!
56//! The native launcher itself does not compute stderr tails; the
57//! manager can do that when it attaches log forwarding for captured
58//! pipes.
59//!
60//! ## Termination semantics
61//!
62//! Termination is implemented with POSIX signals:
63//! - `terminate` sends `SIGTERM` and schedules escalation to
64//!   `SIGKILL` after a timeout if the proc still appears present.
65//! - `kill` sends `SIGKILL` immediately.
66//!
67//! A small best-effort PID registry is maintained to support
68//! signaling and to avoid leaking long-lived children if the launcher
69//! is dropped during teardown.

105/// Native OS process launcher.
106///
107/// This launcher runs bootstrap procs as ordinary local OS processes
108/// via `tokio::process::Command`.
109///
110/// It is the "baseline" backend for [`ProcLauncher`]:
111/// - it spawns a child process directly,
112/// - optionally captures stdout/stderr pipes for the manager,
113/// - and reports terminal status by awaiting `child.wait()` and
114///   sending a [`ProcExitResult`] through the `exit_rx` channel.
115///
116/// Termination is implemented by sending POSIX signals (`SIGTERM`
117/// with optional escalation to `SIGKILL`, or immediate `SIGKILL`).
118pub(crate) struct NativeProcLauncher {
119    /// Best-effort PID registry for processes launched by this
120    /// launcher.
121    ///
122    /// The map is used to:
123    /// - look up the OS pid when `terminate`/`kill` are requested,
124    ///   and
125    /// - perform best-effort cleanup on `Drop`.
126    ///
127    /// Entries are inserted immediately after a successful spawn and
128    /// removed by the exit-monitor task after `wait()` completes.
129    /// Missing entries are treated as idempotent success (already
130    /// exited / unknown).
131    pid_table: Arc<Mutex<HashMap<hyperactor_reference::ProcId, u32>>>,
132}
133
134impl NativeProcLauncher {
135    /// Create a new native launcher.
136    /// `command` as the base "bootstrap command".
137    ///
138    /// A fresh `tokio::process::Command` is constructed per launch
139    /// configured with:
140    /// - the bootstrap payload / diagnostics env vars, and
141    /// - stdio behavior as directed by [`LaunchOptions`].
142    ///
143    /// The internal PID table starts empty and is populated as procs
144    /// are spawned.
145    pub fn new() -> Self {
146        Self {
147            pid_table: Arc::new(Mutex::new(HashMap::new())),
148        }
149    }
150
151    /// Send a POSIX signal to `pid`.
152    ///
153    /// Semantics:
154    /// - `Ok(())` if delivered successfully.
155    /// - `Ok(())` if the process is already gone (`ESRCH`).
156    /// - Otherwise returns a launcher error in the provided `kind`.
157    ///
158    /// Note: We signal the process *group* (negative PID) because the
159    /// child is started in its own process group via `setpgid(0, 0)`.
160    /// This ensures we kill the entire process tree, including any
161    /// sub-processes spawned by wrappers (e.g., shell scripts, python
162    /// launchers).
163    fn send_signal(
164        pid: u32,
165        sig: i32,
166        kind: fn(String) -> ProcLauncherError,
167    ) -> Result<(), ProcLauncherError> {
168        // We use negative PID to signal the process group (PGID = pid)
169        // rather than just the leader.
170        let pgid = -(pid as i32);
171        // SAFETY: `libc::kill` is a simple syscall wrapper with no
172        // memory safety concerns. Passing a negative PID signals the
173        // process group, which is valid POSIX semantics.
174        let rc = unsafe { libc::kill(pgid, sig) };
175        if rc == 0 {
176            return Ok(());
177        }
178
179        let e = std::io::Error::last_os_error();
180        if e.raw_os_error() == Some(libc::ESRCH) {
181            Ok(())
182        } else {
183            Err(kind(format!("signal(pgid={pgid}, {sig}) failed: {e}")))
184        }
185    }
186}

189/// Wrap a [`BootstrapCommand`] with `numactl` or `taskset` based on the
190/// binding config. On NUMA-capable systems the numactl keys
191/// (`cpunodebind`, `membind`, `physcpubind`) take precedence. On Linux,
192/// `cpus` falls back to `taskset -c`.
193fn wrap_command_for_binding(mut cmd: BootstrapCommand, bind: &ProcBind) -> BootstrapCommand {
194    let numa = std::path::Path::new("/sys/devices/system/node/node0").exists();
195
196    if numa {
197        let mut numactl_args = Vec::new();
198        if let Some(v) = bind.cpunodebind.as_deref() {
199            numactl_args.push(format!("--cpunodebind={v}"));
200        }
201        if let Some(v) = bind.membind.as_deref() {
202            numactl_args.push(format!("--membind={v}"));
203        }
204        if let Some(v) = bind.physcpubind.as_deref() {
205            numactl_args.push(format!("--physcpubind={v}"));
206        }
207        if !numactl_args.is_empty() {
208            numactl_args.push(cmd.program.to_string_lossy().into_owned());
209            numactl_args.append(&mut cmd.args);
210            cmd.program = "numactl".into();
211            cmd.arg0 = None;
212            cmd.args = numactl_args;
213            return cmd;
214        }
215    }
216
217    // Fallback: taskset for CPU pinning on Linux.
218    if let Some(cpus) = bind.cpus.as_deref() {
219        let mut taskset_args = vec![
220            "-c".to_string(),
221            cpus.to_string(),
222            cmd.program.to_string_lossy().into_owned(),
223        ];
224        taskset_args.append(&mut cmd.args);
225        cmd.program = "taskset".into();
226        cmd.arg0 = None;
227        cmd.args = taskset_args;
228    }
229
230    cmd
231}

236    cmd

239/// Convert a platform `std::process::ExitStatus` into our
240/// launcher-neutral [`ProcExitKind`] representation.
241///
242/// This interprets an exit status in the usual POSIX order:
243/// - If the process terminated due to a signal, returns
244///   [`ProcExitKind::Signaled`] with the signal number and whether a
245///   core was produced.
246/// - Otherwise, if the process exited normally, returns
247///   [`ProcExitKind::Exited`] with the integer exit code.
248/// - If neither signal nor code is available (should be rare /
249///   platform-specific), returns [`ProcExitKind::Failed`] with an
250///   "unknown exit status" reason.
251///
252/// Note: on Unix, `signal()` / `core_dumped()` come from
253/// `std::os::unix::process::ExitStatusExt`.
254fn exit_kind_from_status(status: &ExitStatus) -> ProcExitKind {
255    if let Some(sig) = status.signal() {
256        ProcExitKind::Signaled {
257            signal: sig,
258            core_dumped: status.core_dumped(),
259        }
260    } else if let Some(code) = status.code() {
261        ProcExitKind::Exited { code }
262    } else {
263        ProcExitKind::Failed {
264            reason: "unknown exit status".to_string(),
265        }
266    }
267}

269#[async_trait]
270impl ProcLauncher for NativeProcLauncher {
271    /// Launch a bootstrap proc as a native OS child process.
272    ///
273    /// This implementation uses [`tokio::process::Command`] to spawn
274    /// the configured [`BootstrapCommand`], passes the bootstrap
275    /// payload and diagnostic metadata via environment variables, and
276    /// returns a [`LaunchResult`] whose `exit_rx` resolves exactly
277    /// once with the proc's terminal status.
278    ///
279    /// ## What this method does
280    ///
281    /// - **Constructs a Tokio command**.
282    /// - **Injects bootstrap state** via [`BOOTSTRAP_MODE_ENV`] using
283    ///   [`Bootstrap::to_env_safe_string`].
284    /// - **Injects a human-readable name** via [`PROCESS_NAME_ENV`]
285    ///   (see [`super::format_process_name`]).
286    /// - **Optionally injects log forwarding address** via
287    ///   [`BOOTSTRAP_LOG_CHANNEL`] when `opts.log_channel` is `Some`.
288    /// - **Configures stdio** based on `opts.want_stdio`:
289    ///   - `true` → pipe stdout/stderr and return
290    ///     [`StdioHandling::Captured`]
291    ///   - `false` → inherit stdout/stderr and return
292    ///     [`StdioHandling::Inherited`]
293    /// - **Spawns an exit monitor task** that awaits `child.wait()`,
294    ///   maps the OS status into [`ProcExitKind`], removes the PID
295    ///   from `pid_table`, and sends the final [`ProcExitResult`]
296    ///   through `exit_rx`.
297    ///
298    /// ## Notes / invariants
299    ///
300    /// - **Readiness is not handled here.** The proc is considered
301    ///   *ready* only when it invokes the bootstrap callback address
302    ///   carried inside `bootstrap`. This method only starts the
303    ///   process.
304    /// - **Termination is initiation-only.** `terminate`/`kill` only
305    ///   send signals; completion is observed by awaiting `exit_rx`.
306    #[hyperactor::instrument(
307        level = "debug",
308        fields(
309            proc_id = proc_id.to_string(),
310            want_stdio = opts.want_stdio,
311            log_channel_present = opts.log_channel.is_some(),
312            pid = tracing::field::Empty,
313        )
314    )]
315    async fn launch(
316        &self,
317        proc_id: &hyperactor_reference::ProcId,
318        opts: LaunchOptions,
319    ) -> Result<LaunchResult, ProcLauncherError> {
320        // Apply NUMA/CPU binding if requested.
321        let command = match &opts.proc_bind {
322            Some(bind) => wrap_command_for_binding(opts.command.clone(), bind),
323            None => opts.command.clone(),
324        };
325
326        // New Tokio Command from BootstrapCommand (template)
327        let mut cmd = command.new();
328
329        // Bootstrap payload
330        cmd.env(BOOTSTRAP_MODE_ENV, opts.bootstrap_payload);
331
332        // Diagnostics name
333        cmd.env(PROCESS_NAME_ENV, opts.process_name);
334
335        // Manager may decide to create a log-forwarding channel; if
336        // so, native passes it through via BOOTSTRAP_LOG_CHANNEL.
337        if let Some(addr) = &opts.log_channel {
338            cmd.env(BOOTSTRAP_LOG_CHANNEL, addr.to_string());
339        }
340
341        // Stdio behavior gated on 'want_stdio'.
342        if opts.want_stdio {
343            cmd.stdout(Stdio::piped()).stderr(Stdio::piped());
344        } else {
345            cmd.stdout(Stdio::inherit()).stderr(Stdio::inherit());
346        }
347
348        // Put child in its own process group so we can kill the entire
349        // tree with kill(-pgid, ...).
350        //
351        // SAFETY: runs in the child between fork and exec. We must not
352        // allocate or do anything complex here.
353        unsafe {
354            cmd.pre_exec(|| {
355                // setpgid(0, 0) => make this process the leader of a new process group.
356                if libc::setpgid(0, 0) != 0 {
357                    return Err(std::io::Error::last_os_error());
358                }
359                Ok(())
360            });
361        }
362
363        let started_at = std::time::SystemTime::now();
364
365        let mut child = cmd.spawn().map_err(ProcLauncherError::Launch)?;
366        let pid = child.id().expect("spawned child pid unavailable");
367
368        // Record PID in current span and log spawn event.
369        tracing::Span::current().record("pid", pid);
370        tracing::debug!("spawned");
371
372        // Record PID for signaling and Drop cleanup.
373        {
374            let mut table = self.pid_table.lock().expect("pid_table mutex poisoned");
375            table.insert(proc_id.clone(), pid);
376        }
377
378        // Extract stdio handles only if requested (only present if
379        // piped).
380        let stdio = if opts.want_stdio {
381            let stdout = child.stdout.take().expect("stdout piped but missing");
382            let stderr = child.stderr.take().expect("stderr piped but missing");
383            StdioHandling::Captured { stdout, stderr }
384        } else {
385            StdioHandling::Inherited
386        };
387
388        // Terminal status channel.
389        let (exit_tx, exit_rx) = oneshot::channel();
390        let pid_table = Arc::clone(&self.pid_table);
391
392        let proc_id = proc_id.clone();
393        // Propagate the launch span into the exit monitor so that
394        // exit_observed and any subsequent events are not "floating".
395        let launch_span = tracing::Span::current();
396        tokio::spawn(
397            async move {
398                // Exit monitoring runs out-of-band:
399                // `launch()` returns immediately with an `exit_rx`, and
400                // this task is responsible for eventually resolving that
401                // channel exactly once.
402                //
403                // We intentionally keep this task simple and robust: wait
404                // for the child to terminate and translate the OS
405                // `ExitStatus` into the trait-level `ProcExitKind` used
406                // by the manager.
407                let kind = match child.wait().await {
408                    Ok(status) => exit_kind_from_status(&status),
409                    Err(e) => ProcExitKind::Failed {
410                        reason: format!("wait failed: {e}"),
411                    },
412                };
413
414                tracing::debug!(?kind, "exit_observed");
415
416                // Once the process has reached a terminal state (or we
417                // failed to observe it), remove it from the PID table so:
418                // - `terminate`/`kill` become idempotent no-ops for this
419                //   proc_id, and
420                // - Drop-time cleanup doesn't try to SIGKILL something
421                //   that's already done.
422                //
423                {
424                    let mut table = pid_table.lock().expect("pid_table mutex poisoned");
425                    table.remove(&proc_id);
426                }
427
428                // Publish the terminal status. It's okay if the receiver
429                // was dropped (e.g., manager shutdown); we just stop
430                // trying.
431                //
432                // Native launcher doesn't provide stderr tail here: the
433                // manager attaches StreamFwder for `Captured` stdio and
434                // can derive any tailing diagnostics itself.
435                let _ = exit_tx.send(ProcExitResult {
436                    kind,
437                    stderr_tail: None,
438                });
439            }
440            .instrument(launch_span),
441        );
442
443        Ok(LaunchResult {
444            pid: Some(pid),
445            started_at,
446            stdio,
447            exit_rx,
448        })
449    }
450
451    /// Initiate graceful termination of a running proc.
452    ///
453    /// This is a *fire-and-forget* termination request:
454    /// - We send **SIGTERM** to the recorded PID (if any).
455    /// - We then schedule an **escalation to SIGKILL** after
456    ///   `timeout` *iff* the proc still appears present in
457    ///   `pid_table`.
458    /// - We return immediately; callers observe completion by
459    ///   awaiting the `exit_rx` returned from [`Self::launch`].
460    ///
461    /// ## Semantics
462    ///
463    /// - If `proc_id` is not present in `pid_table`, this returns
464    ///   `Ok(())` (idempotent success: already exited or never
465    ///   tracked).
466    /// - If SIGTERM fails because the process is already gone
467    ///  (`ESRCH`), this returns `Ok(())` (idempotent success).
468    /// - Other signal delivery failures are returned as
469    ///   [`ProcLauncherError::Terminate`].
470    ///
471    /// ## Implementation notes
472    ///
473    /// - Escalation is performed by a background task that sleeps for
474    ///   `timeout` using `tokio::time::sleep`, re-checks `pid_table`, and
475    ///   sends SIGKILL if the proc is still present.
476    /// - We rely on the exit-monitor task spawned in `launch` to
477    ///   remove the proc from `pid_table` and resolve the `exit_rx`
478    ///   channel when the proc actually terminates.
479    async fn terminate(
480        &self,
481        proc_id: &hyperactor_reference::ProcId,
482        timeout: Duration,
483    ) -> Result<(), ProcLauncherError> {
484        let pid = {
485            let table = self.pid_table.lock().expect("pid_table mutex poisoned");
486            table.get(proc_id).copied()
487        };
488
489        let Some(pid) = pid else {
490            // Idempotent success: already exited or unknown.
491            return Ok(());
492        };
493
494        let timeout_ms = timeout.as_millis() as u64;
495
496        // Create a termination span so escalation logs are not
497        // floating.
498        let term_span = tracing::debug_span!(
499            "terminate",
500            %proc_id,
501            pid,
502            timeout_ms
503        );
504        // Emit the request event *inside* the terminate span.
505        {
506            let _g = term_span.enter();
507            tracing::debug!("terminate_requested");
508        }
509
510        // Send SIGTERM.
511        Self::send_signal(pid, libc::SIGTERM, ProcLauncherError::Terminate)?;
512
513        //  Escalate to SIGKILL after timeout if still present.
514        let pid_table = Arc::clone(&self.pid_table);
515        let proc_id = proc_id.clone();
516
517        tokio::spawn(
518            async move {
519                tokio::time::sleep(timeout).await;
520
521                let pid = {
522                    let table = pid_table.lock().expect("pid_table mutex poisoned");
523                    table.get(&proc_id).copied()
524                };
525
526                if let Some(pid) = pid {
527                    tracing::info!("terminate timeout; escalating to SIGKILL");
528                    if let Err(e) = Self::send_signal(pid, libc::SIGKILL, ProcLauncherError::Kill) {
529                        tracing::warn!(error = %e, "SIGKILL escalation failed");
530                    }
531                }
532            }
533            .instrument(term_span),
534        );
535
536        Ok(())
537    }
538
539    /// Initiate an immediate force-kill of a running proc.
540    ///
541    /// This is the "hard stop" counterpart to [`Self::terminate`]:
542    /// - We send **SIGKILL** to the recorded PID (if any).
543    /// - We return immediately; callers observe completion by
544    ///   awaiting the `exit_rx` returned from [`Self::launch`].
545    ///
546    /// ## Semantics
547    ///
548    /// - If `proc_id` is not present in `pid_table`, this returns
549    ///   `Ok(())` (idempotent success: already exited or never
550    ///   tracked).
551    /// - If SIGKILL fails because the process is already gone
552    ///   (`ESRCH`), this returns `Ok(())` (idempotent success).
553    /// - Other signal delivery failures are returned as
554    ///   [`ProcLauncherError::Kill`].
555    ///
556    /// ## Implementation notes
557    ///
558    /// - We do not remove `proc_id` from `pid_table` here; the
559    ///   exit-monitor task spawned in `launch` removes it once the
560    ///   child actually exits.
561    async fn kill(&self, proc_id: &hyperactor_reference::ProcId) -> Result<(), ProcLauncherError> {
562        let pid = {
563            let table = self.pid_table.lock().expect("pid_table mutex poisoned");
564            table.get(proc_id).copied()
565        };
566
567        let Some(pid) = pid else {
568            // Idempotent success.
569            return Ok(());
570        };
571
572        let kill_span = tracing::debug_span!("kill", %proc_id, pid);
573        {
574            let _g = kill_span.enter();
575            tracing::debug!("kill_requested");
576
577            // Immediate SIGKILL.
578            Self::send_signal(pid, libc::SIGKILL, ProcLauncherError::Kill)?;
579        }
580
581        Ok(())
582    }
583}

591impl Drop for NativeProcLauncher {
592    // Best-effort safety net: if the manager is being torn down while
593    // some children are still running, we don't want to leak
594    // long-lived procs.
595    //
596    // This is intentionally *coarse*: at Drop time we may not have an
597    // async runtime available to do graceful termination, wait for
598    // exit, or run a timeout/escalation ladder. So we prefer a hard
599    // stop (SIGKILL) to avoid orphaned processes and test flakiness.
600    //
601    // Note that this does **not** remove entries from pid_table; the
602    // whole structure is being dropped anyway. Also, failure is
603    // ignored because Drop must not panic; the most common failure is
604    // ESRCH (already exited).
605    fn drop(&mut self) {
606        // Collect PIDs while holding the lock, then release before killing.
607        // This avoids holding the lock during syscalls.
608        let pids: Vec<(hyperactor_reference::ProcId, u32)> = {
609            let table = self.pid_table.lock().expect("pid_table mutex poisoned");
610            table.iter().map(|(k, v)| (k.clone(), *v)).collect()
611        };
612
613        for (proc_id, pid) in pids {
614            match Self::send_signal(pid, libc::SIGKILL, ProcLauncherError::Kill) {
615                Ok(()) => {
616                    tracing::info!(%proc_id, pid, "drop cleanup: sent SIGKILL");
617                }
618                Err(_e) => {
619                    tracing::warn!(%proc_id, pid, "drop cleanup: SIGKILL failed");
620                }
621            }
622        }
623    }
624}

656    /// Consume captured stdout/stderr pipes from
657    /// `StdioHandling::Captured` and return their contents.
658    async fn read_captured_lines(stdio: StdioHandling) -> (Vec<String>, Vec<u8>) {
659        match stdio {
660            StdioHandling::Captured { stdout, mut stderr } => {
661                let mut out_lines = vec![];
662                let mut out = BufReader::new(stdout).lines();
663                while let Ok(Some(line)) = out.next_line().await {
664                    out_lines.push(line);
665                }
666
667                // Drain stderr just so pipes close cleanly; return
668                // raw bytes if useful.
669                let mut err_bytes = vec![];
670                let _ = stderr.read_to_end(&mut err_bytes).await;
671
672                (out_lines, err_bytes)
673            }
674            other => panic!("expected Captured stdio, got {other:?}"),
675        }
676    }

963    /// `terminate()` escalates to SIGKILL when the child ignores
964    /// SIGTERM.
965    ///
966    /// This test exercises the "graceful then forceful" shutdown
967    /// ladder:
968    ///
969    /// - The child process installs a SIGTERM handler that ignores
970    ///   termination (`SIGTERM=IGN`) and then blocks (sleeps).
971    /// - The parent waits for a small READY handshake on captured
972    ///   stdout to ensure the signal disposition has been installed
973    ///   *before* sending SIGTERM (avoids a race where SIGTERM
974    ///   arrives too early).
975    /// - We call [`ProcLauncher::terminate`] with a short timeout.
976    /// - After the timeout elapses, the launcher should observe that
977    ///   the proc still appears present and escalate by sending
978    ///   SIGKILL.
979    ///
980    /// The assertion is made on the terminal status delivered via
981    /// `exit_rx`: it must be [`ProcExitKind::Signaled`] with `signal
982    /// == SIGKILL`.
983    #[tokio::test]
984    async fn terminate_escalates_to_sigkill_if_child_ignores_sigterm() {
985        // Use an explicit READY handshake so the child installs
986        // SIGTERM=IGN before the parent sends SIGTERM. `exec` ensures
987        // we signal the Python process, not the shell.
988        let launcher = NativeProcLauncher::new();
989        let script = r#"exec python3 -c 'import signal,sys,time; signal.signal(signal.SIGTERM, signal.SIG_IGN); print("READY"); sys.stdout.flush(); time.sleep(30)'"#;
990
991        // v0 bootstrap by default but it doesn't matter here.
992        let bootstrap = Bootstrap::default();
993        let proc_id = hyperactor_reference::ProcId::with_name(any_unix_addr(), "term-escalate");
994        let opts = LaunchOptions {
995            command: with_sh(script),
996            bootstrap_payload: bootstrap.to_env_safe_string().unwrap(),
997            process_name: proc_id.to_string(),
998            // Need captured stdout for the READY handshake.
999            want_stdio: true,
1000            tail_lines: 0,
1001            log_channel: None,
1002            proc_bind: None,
1003        };
1004
1005        let lr = launcher.launch(&proc_id, opts).await.expect("launch");
1006
1007        // Wait until the child has definitely installed SIGTERM=IGN.
1008        // We only need stdout for this; drain stderr best-effort.
1009        let (stdout, mut stderr) = match lr.stdio {
1010            StdioHandling::Captured { stdout, stderr } => (stdout, stderr),
1011            other => panic!("expected Captured stdio, got {other:?}"),
1012        };
1013
1014        let mut out = BufReader::new(stdout);
1015        let mut line = String::new();
1016        tokio::time::timeout(Duration::from_secs(2), out.read_line(&mut line))
1017            .await
1018            .expect("timed out waiting for READY")
1019            .expect("read_line failed");
1020        assert_eq!(line.trim(), "READY", "child did not signal readiness");
1021
1022        // Drain stderr in the background so pipes close cleanly.
1023        tokio::spawn(async move {
1024            let mut _buf = Vec::new();
1025            let _ = stderr.read_to_end(&mut _buf).await;
1026        });
1027
1028        // Now request graceful termination + escalation after a short
1029        // timeout.
1030        launcher
1031            .terminate(&proc_id, Duration::from_millis(100))
1032            .await
1033            .expect("terminate");
1034
1035        // Give enough time for: sleep(timeout) + SIGKILL delivery +
1036        // wait() + oneshot send.
1037        let exit = tokio::time::timeout(Duration::from_secs(5), lr.exit_rx)
1038            .await
1039            .expect("timed out waiting for exit_rx")
1040            .expect("exit_rx dropped");
1041
1042        match exit.kind {
1043            ProcExitKind::Signaled { signal, .. } => {
1044                assert_eq!(
1045                    signal,
1046                    libc::SIGKILL,
1047                    "expected SIGKILL escalation, got {signal}"
1048                );
1049            }
1050            other => panic!("expected Signaled(SIGKILL), got {other:?}"),
1051        }
1052    }

1054    /// Verify that `NativeProcLauncher::Drop` kills child processes.
1055    ///
1056    /// This test exercises the critical cleanup path: when a launcher
1057    /// is dropped without explicit termination, its `Drop` impl
1058    /// should SIGKILL any tracked children to prevent orphans.
1059    ///
1060    /// We test the observable effect: when the child is killed, its
1061    /// stdout closes and we get EOF.
1062    #[tokio::test]
1063    async fn drop_closes_child_stdio() {
1064        // Child prints "READY", then writes dots until killed.
1065        let python_script = r#"
1066import sys, time
1067print("READY"); sys.stdout.flush()
1068while True:
1069    sys.stdout.write("."); sys.stdout.flush()
1070    time.sleep(0.05)
1071"#;
1072
1073        let command = BootstrapCommand {
1074            program: PathBuf::from("python3"),
1075            args: vec!["-u".into(), "-c".into(), python_script.trim().into()],
1076            ..Default::default()
1077        };
1078
1079        let bootstrap = Bootstrap::default();
1080        let proc_id = hyperactor_reference::ProcId::with_name(any_unix_addr(), "drop-cleanup-test");
1081        let opts = LaunchOptions {
1082            command,
1083            bootstrap_payload: bootstrap.to_env_safe_string().unwrap(),
1084            process_name: proc_id.to_string(),
1085            want_stdio: true,
1086            tail_lines: 0,
1087            log_channel: None,
1088            proc_bind: None,
1089        };
1090
1091        // Keep stdout in outer scope so we can read after launcher
1092        // drop.
1093        let mut stdout_reader: BufReader<tokio::process::ChildStdout>;
1094
1095        {
1096            let launcher = NativeProcLauncher::new();
1097            let lr = launcher.launch(&proc_id, opts).await.expect("launch");
1098
1099            let (stdout, _stderr) = match lr.stdio {
1100                StdioHandling::Captured { stdout, stderr } => (stdout, stderr),
1101                other => panic!("expected Captured stdio, got {other:?}"),
1102            };
1103
1104            stdout_reader = BufReader::new(stdout);
1105
1106            // Wait for "READY".
1107            let mut line = String::new();
1108            tokio::time::timeout(Duration::from_secs(2), stdout_reader.read_line(&mut line))
1109                .await
1110                .expect("timeout waiting for READY")
1111                .expect("read_line failed");
1112            assert!(line.trim() == "READY", "expected 'READY', got: {line}");
1113
1114            // Prove child is alive by reading one dot.
1115            let mut b = [0u8; 1];
1116            tokio::time::timeout(Duration::from_secs(2), stdout_reader.read_exact(&mut b))
1117                .await
1118                .expect("timeout waiting for dot")
1119                .expect("read_exact failed");
1120            assert_eq!(b[0], b'.');
1121
1122            // launcher drops here; its Drop should SIGKILL all tracked
1123            // children.
1124        }
1125
1126        // After drop: stdout should hit EOF once the child is killed.
1127        // Drain any remaining output and wait for EOF with a timeout.
1128        let drain_eof = async {
1129            let mut buf = [0u8; 64];
1130            loop {
1131                match stdout_reader.read(&mut buf).await {
1132                    Ok(0) => break,     // EOF - child is dead
1133                    Ok(_n) => continue, // still getting dots, keep draining
1134                    Err(e) => panic!("read failed: {e}"),
1135                }
1136            }
1137        };
1138
1139        tokio::time::timeout(Duration::from_secs(2), drain_eof)
1140            .await
1141            .expect("never observed EOF after launcher drop - child likely still alive");
1142    }
