39/// Render a full frame of the TUI.
40///
41/// Splits the screen into header/body/footer regions and delegates to
42/// the corresponding render helpers.
43pub(crate) fn ui(frame: &mut ratatui::Frame<'_>, app: &App) {
44    let chunks = Layout::default()
45        .direction(Direction::Vertical)
46        .constraints([
47            Constraint::Length(3),
48            Constraint::Min(10),
49            Constraint::Length(2),
50        ])
51        .split(frame.area());
52
53    render_header(frame, chunks[0], app);
54    render_body(frame, chunks[1], app);
55    render_footer(frame, chunks[2], app);
56}
57
58/// Render the main body of the UI.
59///
60/// Splits the screen into a left topology pane and a right detail
61/// pane. When diagnostics is active the topology tree is dimmed
62/// (non-interactive) and the right pane shows the diagnostics view.
63pub(crate) fn render_body(frame: &mut ratatui::Frame<'_>, area: ratatui::layout::Rect, app: &App) {
64    let chunks = Layout::default()
65        .direction(Direction::Horizontal)
66        .constraints([Constraint::Percentage(40), Constraint::Percentage(60)])
67        .split(area);
68
69    render_topology_tree(frame, chunks[0], app);
70    render_detail_pane(frame, chunks[1], app);
71}
