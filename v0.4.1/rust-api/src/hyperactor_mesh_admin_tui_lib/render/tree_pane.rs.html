22/// Render the topology tree (left pane).
23///
24/// Uses `visible_rows()` to display only expanded nodes. Each row
25/// includes indentation/connectors, an expand/collapse glyph for
26/// nodes with children, and color-coding by `NodeType`, with the
27/// selected row highlighted.
28pub(crate) fn render_topology_tree(frame: &mut ratatui::Frame<'_>, area: Rect, app: &App) {
29    let rows = app.visible_rows();
30    let scheme = &app.theme.scheme;
31
32    // When any overlay is active the tree is non-interactive.
33    // Render it uniformly dim so the user can see it is inactive.
34    // This covers both the diagnostics overlay and the py-spy overlay.
35    let pane_inactive = app.overlay.is_some();
36
37    let items: Vec<ListItem> = rows
38        .as_slice()
39        .iter()
40        .enumerate()
41        .map(|(vis_idx, row)| {
42            let node = row.node;
43            let indent = "  ".repeat(row.depth);
44
45            // Tree connector
46            let connector = if row.depth == 0 {
47                ""
48            } else if rows.has_sibling_after(vis_idx, row.depth) {
49                "├─ "
50            } else {
51                "└─ "
52            };
53
54            // Fold indicator for expandable nodes
55            let fold = if node.has_children {
56                if node.expanded { "▼ " } else { "▶ " }
57            } else {
58                "  "
59            };
60
61            // Style precedence: inactive > selected > failed > stopped >
62            // system > node-type.  When pane_inactive the entire pane is dimmed
63            // and the selection/failed/system colours must not bleed through.
64            let style = if pane_inactive {
65                scheme.detail_stopped
66            } else if vis_idx == app.cursor.pos() {
67                scheme.stat_selection.add_modifier(Modifier::BOLD)
68            } else if node.failed {
69                scheme.node_failed
70            } else if node.stopped {
71                scheme.detail_stopped
72            } else if node.is_system {
73                scheme.node_system_actor
74            } else {
75                scheme.node_user_actor
76            };
77
78            // Hide the selection caret while the pane is inactive.
79            let marker = if !pane_inactive && vis_idx == app.cursor.pos() {
80                app.theme.labels.selection_caret
81            } else {
82                "  "
83            };
84
85            ListItem::new(Line::from(Span::styled(
86                format!("{}{}{}{}{}", marker, indent, connector, fold, node.label),
87                style,
88            )))
89        })
90        .collect();
91
92    let block = if pane_inactive {
93        Block::default()
94            .title(Span::styled(
95                app.theme.labels.pane_topology,
96                scheme.detail_stopped,
97            ))
98            .borders(Borders::ALL)
99            .border_style(scheme.detail_stopped)
100    } else {
101        Block::default()
102            .title(app.theme.labels.pane_topology)
103            .borders(Borders::ALL)
104            .border_style(scheme.border)
105    };
106
107    let list = List::new(items)
108        .block(block)
109        .highlight_style(Style::default());
110    let mut list_state = ListState::default()
111        .with_selected(Some(app.cursor.pos()))
112        .with_offset(app.tree_scroll_offset);
113    frame.render_stateful_widget(list, area, &mut list_state);
114}
