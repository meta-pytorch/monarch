45/// Render the contextual details pane (right side).
46///
47/// If a `NodePayload` for the current selection is available in
48/// `app.detail`, dispatches to `render_node_detail` to show a
49/// type-specific view (root/host/proc/actor). Otherwise, shows either
50/// the last fetch error (`app.detail_error`) or a neutral "select a
51/// node" placeholder message.
52pub(crate) fn render_detail_pane(frame: &mut ratatui::Frame<'_>, area: Rect, app: &App) {
53    if let Some(overlay) = &app.overlay {
54        render_overlay(frame, area, overlay, &app.theme.scheme);
55        return;
56    }
57    match &app.detail {
58        Some(payload) => {
59            render_node_detail(frame, area, payload, &app.theme.scheme, &app.theme.labels)
60        }
61        None => {
62            let msg = app
63                .detail_error
64                .as_deref()
65                .unwrap_or("Select a node to view details");
66            let msg_style = if app.detail_error.is_some() {
67                app.theme.scheme.error
68            } else {
69                app.theme.scheme.info
70            };
71            let block = Block::default()
72                .title(app.theme.labels.pane_details)
73                .borders(Borders::ALL)
74                .border_style(app.theme.scheme.border);
75            let p = Paragraph::new(Span::styled(msg, msg_style)).block(block);
76            frame.render_widget(p, area);
77        }
78    }
79}

81/// Render the details view for a resolved node.
82///
83/// This is the main dispatcher for the right-hand pane: it matches on
84/// `payload.properties` and forwards to the appropriate renderer
85/// (`render_root_detail`, `render_host_detail`, `render_proc_detail`,
86/// or `render_actor_detail`) with the relevant fields extracted from
87/// the payload.
88pub(crate) fn render_node_detail(
89    frame: &mut ratatui::Frame<'_>,
90    area: Rect,
91    payload: &NodePayload,
92    scheme: &ColorScheme,
93    labels: &Labels,
94) {
95    match &payload.properties {
96        NodeProperties::Root {
97            num_hosts,
98            started_at,
99            started_by,
100            ..
101        } => {
102            render_root_detail(
103                frame, area, payload, *num_hosts, started_at, started_by, scheme, labels,
104            );
105        }
106        NodeProperties::Host {
107            addr, num_procs, ..
108        } => {
109            render_host_detail(frame, area, payload, addr, *num_procs, scheme, labels);
110        }
111        NodeProperties::Proc {
112            proc_name,
113            num_actors,
114            is_poisoned,
115            failed_actor_count,
116            ..
117        } => {
118            render_proc_detail(
119                frame,
120                area,
121                payload,
122                proc_name,
123                *num_actors,
124                *is_poisoned,
125                *failed_actor_count,
126                scheme,
127                labels,
128            );
129        }
130        NodeProperties::Actor {
131            actor_status,
132            actor_type,
133            messages_processed,
134            created_at,
135            last_message_handler,
136            total_processing_time_us,
137            flight_recorder,
138            failure_info,
139            ..
140        } => {
141            render_actor_detail(
142                frame,
143                area,
144                payload,
145                actor_status,
146                actor_type,
147                *messages_processed,
148                created_at,
149                last_message_handler.as_deref(),
150                *total_processing_time_us,
151                flight_recorder.as_deref(),
152                failure_info.as_ref(),
153                scheme,
154                labels,
155            );
156        }
157        NodeProperties::Error { code, message } => {
158            let text = format!("Error: {} — {}", code, message);
159            let paragraph = Paragraph::new(text)
160                .block(
161                    Block::default()
162                        .borders(Borders::ALL)
163                        .border_style(scheme.border)
164                        .title(labels.pane_error),
165                )
166                .style(scheme.error)
167                .wrap(Wrap { trim: true });
168            frame.render_widget(paragraph, area);
169        }
170    }
171}

175    label: &'a str,
176    value: impl Into<String>,
177    scheme: &ColorScheme,

185/// Render the right-pane detail view for the mesh root node.
186///
187/// Shows a simple summary (host count) and then lists the root's
188/// immediate children (host references) so the user can quickly see
189/// which hosts are currently registered under the mesh.
190fn render_root_detail(
191    frame: &mut ratatui::Frame<'_>,
192    area: Rect,
193    payload: &NodePayload,
194    num_hosts: usize,
195    started_at: &SystemTime,
196    started_by: &str,
197    scheme: &ColorScheme,
198    l: &Labels,
199) {
200    let block = Block::default()
201        .title(l.pane_root_details)
202        .borders(Borders::ALL)
203        .border_style(scheme.border);
204
205    let uptime_str = format_system_time_uptime(started_at);
206
207    let mut lines = vec![
208        detail_line(l.hosts, num_hosts.to_string(), scheme),
209        detail_line(l.started_by, started_by, scheme),
210        detail_line(l.uptime_detail, &uptime_str, scheme),
211        detail_line(l.started_at, format_system_time_local(started_at), scheme),
212        detail_line(
213            l.data_as_of,
214            format_system_time_relative(&payload.as_of),
215            scheme,
216        ),
217        Line::default(),
218    ];
219    for child in &payload.children {
220        let child_str = child.to_string();
221        lines.push(Line::from(vec![
222            Span::styled("  ", Style::default()),
223            Span::styled(child_str, scheme.node_host),
224        ]));
225    }
226
227    let p = Paragraph::new(lines).block(block);
228    frame.render_widget(p, area);
229}

231/// Render the right-pane detail view for a host node.
232///
233/// Displays the host's address and proc count, then lists the host's
234/// proc children using a shortened proc name for readability.
235fn render_host_detail(
236    frame: &mut ratatui::Frame<'_>,
237    area: Rect,
238    payload: &NodePayload,
239    addr: &str,
240    num_procs: usize,
241    scheme: &ColorScheme,
242    l: &Labels,
243) {
244    let block = Block::default()
245        .title(l.pane_host_details)
246        .borders(Borders::ALL)
247        .border_style(scheme.border);
248
249    let mut lines = vec![
250        detail_line(l.address, addr, scheme),
251        detail_line(l.procs, num_procs.to_string(), scheme),
252        detail_line(
253            l.data_as_of,
254            format_system_time_relative(&payload.as_of),
255            scheme,
256        ),
257        Line::default(),
258    ];
259    for child in &payload.children {
260        let child_str = child.to_string();
261        let short = match child {
262            NodeRef::Proc(proc_id) => proc_id.name().to_string(),
263            _ => child_str.clone(),
264        };
265        lines.push(Line::from(vec![
266            Span::styled("  ", Style::default()),
267            Span::styled(short, scheme.node_proc),
268        ]));
269    }
270
271    let p = Paragraph::new(lines).block(block);
272    frame.render_widget(p, area);
273}

275/// Render the right-pane detail view for a proc node.
276///
277/// Shows the proc's full name and actor count, then lists up to the
278/// first 50 child actor references (with an elision line if there are
279/// more) to keep the UI responsive and the pane readable.
280fn render_proc_detail(
281    frame: &mut ratatui::Frame<'_>,
282    area: Rect,
283    payload: &NodePayload,
284    proc_name: &str,
285    num_actors: usize,
286    is_poisoned: bool,
287    failed_actor_count: usize,
288    scheme: &ColorScheme,
289    l: &Labels,
290) {
291    let block = Block::default()
292        .title(l.pane_proc_details)
293        .borders(Borders::ALL)
294        .border_style(scheme.border);
295
296    let mut lines = vec![
297        detail_line(l.name, proc_name, scheme),
298        detail_line(l.actors, num_actors.to_string(), scheme),
299        detail_line(
300            l.data_as_of,
301            format_system_time_relative(&payload.as_of),
302            scheme,
303        ),
304    ];
305
306    if is_poisoned {
307        lines.push(Line::from(vec![
308            Span::styled(l.poisoned, scheme.node_failed),
309            Span::styled(l.yes, scheme.node_failed),
310        ]));
311        lines.push(Line::from(vec![
312            Span::styled(l.failed_actors, scheme.node_failed),
313            Span::styled(failed_actor_count.to_string(), scheme.node_failed),
314        ]));
315    }
316
317    lines.push(Line::default());
318    for (i, actor) in payload.children.iter().enumerate() {
319        if i >= 50 {
320            lines.push(Line::from(Span::styled(
321                format!("  ... and {} more", payload.children.len() - 50),
322                scheme.detail_stopped,
323            )));
324            break;
325        }
326        lines.push(Line::from(vec![
327            Span::styled("  ", Style::default()),
328            Span::raw(actor.to_string()),
329        ]));
330    }
331
332    let p = Paragraph::new(lines).block(block);
333    frame.render_widget(p, area);
334}

336/// Render the right-pane detail view for an actor node, including
337/// flight-recorder events.
338///
339/// The top section summarizes actor status/type, message/processing
340/// stats, creation time, last handler, and child count; the bottom
341/// section parses the optional flight-recorder JSON and displays a
342/// compact, timestamped list of recent events.
343#[allow(clippy::too_many_arguments)]
344fn render_actor_detail(
345    frame: &mut ratatui::Frame<'_>,
346    area: Rect,
347    payload: &NodePayload,
348    actor_status: &str,
349    actor_type: &str,
350    messages_processed: u64,
351    created_at: &Option<SystemTime>,
352    last_message_handler: Option<&str>,
353    total_processing_time_us: u64,
354    flight_recorder_json: Option<&str>,
355    failure_info: Option<&FailureInfo>,
356    scheme: &ColorScheme,
357    l: &Labels,
358) {
359    let info_height = if failure_info.is_some() { 15 } else { 11 };
360    let chunks = Layout::default()
361        .direction(Direction::Vertical)
362        .constraints([Constraint::Length(info_height), Constraint::Min(5)])
363        .split(area);
364
365    // Actor info
366    let info_block = Block::default()
367        .title(l.pane_actor_details)
368        .borders(Borders::ALL)
369        .border_style(scheme.border);
370
371    let status_style = if failure_info.is_some() {
372        scheme.detail_status_failed
373    } else if actor_status == "Running" {
374        scheme.detail_status_ok
375    } else {
376        scheme.detail_status_warn
377    };
378
379    let created_str = created_at
380        .as_ref()
381        .map(|t| format_system_time_iso(t))
382        .unwrap_or_else(|| "-".to_string());
383
384    let mut lines = vec![
385        Line::from(vec![
386            Span::styled(l.status, scheme.detail_label),
387            Span::styled(actor_status, status_style),
388        ]),
389        detail_line(
390            l.data_as_of,
391            format_system_time_relative(&payload.as_of),
392            scheme,
393        ),
394        detail_line(l.actor_type, actor_type, scheme),
395        detail_line(l.messages, messages_processed.to_string(), scheme),
396        detail_line(
397            l.processing_time,
398            humantime::format_duration(std::time::Duration::from_micros(total_processing_time_us))
399                .to_string(),
400            scheme,
401        ),
402        detail_line(l.created, &created_str, scheme),
403        detail_line(l.last_handler, last_message_handler.unwrap_or("-"), scheme),
404        detail_line(l.children, payload.children.len().to_string(), scheme),
405    ];
406
407    if let Some(fi) = failure_info {
408        lines.push(Line::default());
409        lines.push(Line::from(vec![
410            Span::styled(l.error_message, scheme.node_failed),
411            Span::styled(&fi.error_message, scheme.node_failed),
412        ]));
413        let root_cause_display = match &fi.root_cause_name {
414            Some(name) => format!("{} ({})", name, fi.root_cause_actor),
415            None => fi.root_cause_actor.to_string(),
416        };
417        lines.push(Line::from(vec![
418            Span::styled(l.root_cause, scheme.detail_label),
419            Span::raw(root_cause_display),
420        ]));
421        lines.push(detail_line(
422            l.failed_at,
423            format_system_time_iso(&fi.occurred_at),
424            scheme,
425        ));
426        lines.push(Line::from(vec![
427            Span::styled(l.propagated, scheme.detail_label),
428            Span::raw(if fi.is_propagated { l.yes } else { l.no }),
429        ]));
430    }
431
432    let info = Paragraph::new(lines)
433        .block(info_block)
434        .wrap(Wrap { trim: false });
435    frame.render_widget(info, chunks[0]);
436
437    // Flight recorder
438    let recorder_block = Block::default()
439        .title(l.pane_flight_recorder)
440        .borders(Borders::ALL)
441        .border_style(scheme.border);
442
443    let recorded_events: Vec<RecordedEvent> = flight_recorder_json
444        .and_then(|json| serde_json::from_str(json).ok())
445        .unwrap_or_default();
446
447    let events: Vec<Line> = if recorded_events.is_empty() {
448        vec![Line::from(Span::styled("No events", scheme.detail_label))]
449    } else {
450        recorded_events
451            .iter()
452            .take(20)
453            .map(|event| {
454                let level_style = match event.level.as_str() {
455                    "ERROR" => scheme.error,
456                    "WARN" => scheme.detail_status_warn,
457                    "INFO" => scheme.detail_status_ok,
458                    "DEBUG" => scheme.info,
459                    _ => scheme.detail_label,
460                };
461                Line::from(vec![
462                    Span::styled(
463                        format!("{} ", event.level.chars().next().unwrap_or('?')),
464                        level_style,
465                    ),
466                    Span::styled(
467                        format!("{} ", format_local_time(&event.timestamp)),
468                        scheme.detail_label,
469                    ),
470                    Span::raw(format_event_summary(&event.name, &event.fields)),
471                ])
472            })
473            .collect()
474    };
475
476    let recorder = Paragraph::new(events)
477        .block(recorder_block)
478        .wrap(Wrap { trim: true });
479    frame.render_widget(recorder, chunks[1]);
480}

482/// Build an `Overlay` from diagnostics state.
483///
484/// Called by `ActiveJob::build_overlay` for the `Diagnostics` variant.
485/// Takes individual fields rather than `&App` so that the borrow
486/// checker allows `rebuild_overlay` to hold `&self.active_job` (shared)
487/// while writing to `self.overlay`.
488pub(crate) fn build_diag_overlay(
489    results: &[DiagResult],
490    running: bool,
491    completed_at: Option<&str>,
492    theme: &Theme,
493) -> crate::overlay::Overlay {
494    let scheme = &theme.scheme;
495    let labels = &theme.labels;
496    let sep = labels.separator;
497
498    // Title line.
499    let title = if running {
500        Line::from(vec![
501            Span::styled(
502                labels.pane_diagnostics,
503                Style::default().add_modifier(Modifier::BOLD),
504            ),
505            Span::styled(format!("{}{}", sep, labels.diag_running), scheme.info),
506        ])
507    } else if let Some(t) = completed_at {
508        Line::from(vec![
509            Span::styled(
510                labels.pane_diagnostics,
511                Style::default().add_modifier(Modifier::BOLD),
512            ),
513            Span::styled(
514                format!("{}{}", sep, labels.diag_completed_at),
515                scheme.detail_label,
516            ),
517            Span::raw(" "),
518            Span::styled(t.to_string(), scheme.stat_timing),
519            Span::styled(
520                format!("{}{}", sep, labels.diag_static_snapshot),
521                scheme.detail_stopped,
522            ),
523        ])
524    } else {
525        Line::from(Span::raw(labels.pane_diagnostics))
526    };
527
528    // Pinned status line.
529    let status_line = if running {
530        Line::from(vec![
531            Span::styled(labels.diag_running, scheme.info),
532            Span::styled(
533                format!("{}{}", sep, labels.diag_live_updates),
534                scheme.detail_label,
535            ),
536        ])
537    } else if !results.is_empty() {
538        let s = DiagSummary::from_results(results);
539        let admin_status = if s.admin_passed == s.admin_total && s.admin_total > 0 {
540            labels.diag_status_healthy
541        } else {
542            labels.diag_status_failing
543        };
544        let mesh_status = if s.mesh_total == 0 {
545            labels.diag_status_na
546        } else if s.mesh_passed == s.mesh_total {
547            labels.diag_status_healthy
548        } else {
549            labels.diag_status_failing
550        };
551        let (summary, summary_style) = if !s.any_fail {
552            (
553                format!(
554                    "{} {} {}. {} {} {} {}.",
555                    labels.diag_checks_all,
556                    s.total,
557                    labels.diag_checks_passed,
558                    labels.diag_admin_label,
559                    admin_status,
560                    labels.diag_mesh_label,
561                    mesh_status,
562                ),
563                scheme.detail_status_ok,
564            )
565        } else {
566            (
567                format!(
568                    "{}/{} {}. {} {}/{}. {} {}/{}.",
569                    s.passed,
570                    s.total,
571                    labels.diag_checks_passed,
572                    labels.diag_admin_label,
573                    s.admin_passed,
574                    s.admin_total,
575                    labels.diag_mesh_label,
576                    s.mesh_passed,
577                    s.mesh_total,
578                ),
579                scheme.error,
580            )
581        };
582        Line::from(Span::styled(summary, summary_style))
583    } else {
584        Line::default()
585    };
586
587    // Scrollable section results.
588    let sep_style = Style::default().add_modifier(Modifier::BOLD);
589    let mut lines: Vec<Line<'static>> = Vec::new();
590
591    lines.push(Line::from(Span::styled(
592        "── Admin Infra ──────────────────────────────────",
593        sep_style,
594    )));
595    for r in results.iter().filter(|r| r.phase == DiagPhase::AdminInfra) {
596        lines.push(diag_result_line(r, scheme, labels));
597    }
598    lines.push(Line::default());
599    lines.push(Line::from(Span::styled(
600        "── Mesh ─────────────────────────────────────────",
601        sep_style,
602    )));
603    for r in results.iter().filter(|r| r.phase == DiagPhase::Mesh) {
604        lines.push(diag_result_line(r, scheme, labels));
605    }
606
607    crate::overlay::Overlay {
608        title,
609        status_line: Some(status_line),
610        lines,
611        loading: running,
612        scroll: std::cell::Cell::new(0),
613        max_scroll: std::cell::Cell::new(u16::MAX),
614    }
615}

617/// Format one diagnostic probe result as a TUI row.
618fn diag_result_line(r: &DiagResult, scheme: &ColorScheme, labels: &Labels) -> Line<'static> {
619    let (icon, icon_style) = match &r.outcome {
620        DiagOutcome::Pass { .. } => ("✓", scheme.detail_status_ok),
621        DiagOutcome::Slow { .. } => ("⚠", scheme.detail_status_warn),
622        DiagOutcome::Fail { .. } => ("✗", scheme.error),
623    };
624    let timing = match &r.outcome {
625        DiagOutcome::Pass { elapsed_ms } | DiagOutcome::Slow { elapsed_ms } => {
626            format!(" {}ms", elapsed_ms)
627        }
628        DiagOutcome::Fail { elapsed_ms, error } => format!(" {}ms — {}", elapsed_ms, error),
629    };
630    let mut spans = vec![
631        Span::styled(format!(" {} ", icon), icon_style),
632        Span::raw(r.label.clone()),
633        Span::styled(timing, scheme.detail_label),
634    ];
635    if let Some(role) = r.note {
636        let note = match role {
637            DiagNodeRole::AdminServer => labels.diag_note_admin_server,
638            DiagNodeRole::HostAgent => labels.diag_note_host_agent,
639            DiagNodeRole::AdminServiceProc => labels.diag_note_admin_service_proc,
640            DiagNodeRole::LocalClientProc => labels.diag_note_local_client_proc,
641            DiagNodeRole::IntrospectionHandler => labels.diag_note_introspection_handler,
642            DiagNodeRole::ActorLifecycleManager => labels.diag_note_actor_lifecycle_manager,
643            DiagNodeRole::RootClientBridge => labels.diag_note_root_client_bridge,
644            DiagNodeRole::CommActor => labels.diag_note_comm_actor,
645            DiagNodeRole::ProcAgent => labels.diag_note_proc_agent,
646            DiagNodeRole::UserProc => labels.diag_note_user_proc,
647            DiagNodeRole::UserActor => labels.diag_note_user_actor,
648        };
649        spans.push(Span::styled(format!("  — {}", note), scheme.stat_url));
650    }
651    Line::from(spans)
652}

654/// Render a generic scrollable overlay in the detail pane area.
655///
656/// Supports an optional pinned status line (used by diagnostics)
657/// above the scrollable content.
658fn render_overlay(
659    frame: &mut ratatui::Frame<'_>,
660    area: Rect,
661    overlay: &crate::overlay::Overlay,
662    scheme: &ColorScheme,
663) {
664    let block = Block::default()
665        .title(overlay.title.clone())
666        .borders(Borders::ALL)
667        .border_style(scheme.border);
668    let inner = block.inner(area);
669    frame.render_widget(block, area);
670
671    // If there's a pinned status line, split inner into status + scrollable.
672    let content_area = if let Some(status) = &overlay.status_line {
673        let chunks = Layout::default()
674            .direction(Direction::Vertical)
675            .constraints([Constraint::Length(1), Constraint::Min(1)])
676            .split(inner);
677        frame.render_widget(Paragraph::new(status.clone()), chunks[0]);
678        chunks[1]
679    } else {
680        inner
681    };
682
683    // Show "fetching…" only when loading with no lines AND no status line.
684    // When a status_line is present it already communicates the loading state
685    // (e.g. "Running… • fetching stack trace"), so the placeholder is redundant.
686    let lines: Vec<Line<'static>> =
687        if overlay.loading && overlay.lines.is_empty() && overlay.status_line.is_none() {
688            vec![Line::from("fetching…")]
689        } else {
690            overlay.lines.clone()
691        };
692
693    let max_scroll = (lines.len() as u16).saturating_sub(content_area.height);
694    overlay.max_scroll.set(max_scroll);
695    let scroll = overlay.scroll.get().min(max_scroll);
696    overlay.scroll.set(scroll);
697
698    let paragraph = Paragraph::new(lines)
699        .wrap(Wrap { trim: false })
700        .scroll((scroll, 0));
701    frame.render_widget(paragraph, content_area);
702}
