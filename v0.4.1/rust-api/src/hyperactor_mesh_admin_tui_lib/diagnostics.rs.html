9//! Self-diagnostic for the mesh admin TUI.
10//!
11//! Walks the full resolution graph (root → hosts → service proc
12//! actors → first user proc → first user actor) and probes each node
13//! via `GET /v1/{reference}`. Results stream through an
14//! `mpsc::Receiver` so the TUI can render them live as each check
15//! completes.
16//!
17//! # Failure domains
18//!
19//! Checks are tagged with a [`DiagPhase`]:
20//!
21//! - **[`DiagPhase::AdminInfra`]**: root, host agents, service proc,
22//!   `agent[0]`, `mesh_admin[0]`, `mesh_admin_bridge[0]`. These test
23//!   the admin layer itself independent of user workloads.
24//! - **[`DiagPhase::Mesh`]**: first user proc and actor. These test
25//!   whether the user's mesh is healthy.
26//!
27//! If Phase 1 passes but Phase 2 fails → mesh problem.
28//! If Phase 1 fails → admin infra bug.
29

46/// Which layer of the stack a diagnostic check exercises.
47#[derive(Debug, Clone, PartialEq, Serialize)]
48pub(crate) enum DiagPhase {
49    /// Root, host agents, service proc, and its actors — tests admin
50    /// infra independent of user workloads.
51    AdminInfra,
52    /// First user proc and actor — tests whether the mesh itself is
53    /// healthy.
54    Mesh,
55}
56
57/// Outcome of a single diagnostic probe.
58#[derive(Debug, Clone, Serialize)]
59pub(crate) enum DiagOutcome {
60    /// HTTP 200 received within [`SLOW_MS`] milliseconds.
61    Pass { elapsed_ms: u64 },
62    /// HTTP 200 received, but slower than [`SLOW_MS`].
63    Slow { elapsed_ms: u64 },
64    /// HTTP error, non-200 status, or timeout.
65    Fail { elapsed_ms: u64, error: String },
66}
67
68/// The semantic role of a probed node, used to look up a localised
69/// annotation in [`Labels`](crate::theme::Labels).
70#[derive(Debug, Clone, Copy, Serialize)]
71pub(crate) enum DiagNodeRole {
72    /// The admin HTTP server itself — the root of the resolution graph.
73    AdminServer,
74    /// A host agent process — one per machine, manages all procs.
75    HostAgent,
76    /// The system proc that houses the admin actor layer.
77    AdminServiceProc,
78    /// The `mesh_admin` actor that handles all `GET /v1/…` requests.
79    IntrospectionHandler,
80    /// The `agent` actor that manages actor spawn and lifecycle.
81    ActorLifecycleManager,
82    /// The `mesh_admin_bridge` instance that connects admin to the mesh.
83    RootClientBridge,
84    /// The local client proc — in-process, starts empty (LP-1). Gets a
85    /// ProcAgent and root client actor only when activated via
86    /// `this_proc()` (Python/Monarch).
87    LocalClientProc,
88    /// The `comm` actor on a user proc — enables proc-to-proc mesh messaging.
89    CommActor,
90    /// The `proc_agent` actor on a user proc — manages actor spawn and lifecycle.
91    ProcAgent,
92    /// The first non-system proc — confirms user workload is alive.
93    UserProc,
94    /// The first non-system actor in a user proc — reachable through full stack.
95    UserActor,
96}

98/// Result of a single diagnostic probe.
99#[derive(Debug, Clone, Serialize)]
100pub(crate) struct DiagResult {
101    /// Human-readable check name shown in the TUI (e.g. `"agent[0]"`).
102    pub(crate) label: String,
103    /// Exact reference passed to `GET /v1/{reference}` — use this to
104    /// reproduce the probe or identify the failing node directly.
105    pub(crate) reference: String,
106    /// Semantic role used to look up the localised annotation.
107    pub(crate) note: Option<DiagNodeRole>,
108    /// Which failure domain this check belongs to.
109    pub(crate) phase: DiagPhase,
110    /// Probe outcome.
111    pub(crate) outcome: DiagOutcome,
112}
113
114// Response latency above which a pass is reported as slow.
115const SLOW_MS: u64 = 500;
116
117// Per-probe timeout. Set above the server's SINGLE_HOST_TIMEOUT (3 s)
118// so server-side 504s are surfaced as Fail(error) rather than our own
119// timeout.
120const TIMEOUT_MS: u64 = 5000;
121
122/// Aggregated pass/fail counts across both diagnostic phases.
123/// Computed from a completed (or in-progress) result slice.
124#[derive(Debug, Clone)]
125pub(crate) struct DiagSummary {
126    pub(crate) total: usize,
127    /// Probes that returned Pass or Slow.
128    pub(crate) passed: usize,
129    pub(crate) admin_total: usize,
130    pub(crate) admin_passed: usize,
131    pub(crate) mesh_total: usize,
132    pub(crate) mesh_passed: usize,
133    /// True if any probe returned `DiagOutcome::Fail`.
134    pub(crate) any_fail: bool,
135}
136
137impl DiagSummary {
138    pub(crate) fn from_results(results: &[DiagResult]) -> Self {
139        let is_pass =
140            |o: &DiagOutcome| matches!(o, DiagOutcome::Pass { .. } | DiagOutcome::Slow { .. });
141        Self {
142            total: results.len(),
143            passed: results.iter().filter(|r| is_pass(&r.outcome)).count(),
144            admin_total: results
145                .iter()
146                .filter(|r| r.phase == DiagPhase::AdminInfra)
147                .count(),
148            admin_passed: results
149                .iter()
150                .filter(|r| r.phase == DiagPhase::AdminInfra && is_pass(&r.outcome))
151                .count(),
152            mesh_total: results
153                .iter()
154                .filter(|r| r.phase == DiagPhase::Mesh)
155                .count(),
156            mesh_passed: results
157                .iter()
158                .filter(|r| r.phase == DiagPhase::Mesh && is_pass(&r.outcome))
159                .count(),
160            any_fail: results
161                .iter()
162                .any(|r| matches!(r.outcome, DiagOutcome::Fail { .. })),
163        }
164    }
165}

167/// Run the full diagnostic suite against `base_url`.
168///
169/// Returns a channel receiver. The spawned task sends one
170/// [`DiagResult`] as each probe completes (in walk order). The
171/// channel closes when all probes finish, signalling completion.
172pub(crate) fn run_diagnostics(
173    client: reqwest::Client,
174    base_url: String,
175) -> mpsc::Receiver<DiagResult> {
176    let (tx, rx) = mpsc::channel(64);
177    tokio::spawn(async move {
178        walk(&client, &base_url, &tx).await;
179    });
180    rx
181}
182
183/// Send a single `DiagResult` and ignore send errors (TUI exited).
184macro_rules! emit {
185    ($tx:expr, $result:expr) => {
186        let _ = $tx.send($result).await;
187    };
188}

190/// Probe one reference and return both the `DiagResult` and (on
191/// success) the fetched `NodePayload`.
192async fn probe(
193    client: &reqwest::Client,
194    base_url: &str,
195    label: impl Into<String>,
196    reference: &hyperactor_mesh::introspect::NodeRef,
197    phase: DiagPhase,
198) -> (DiagResult, Option<hyperactor_mesh::introspect::NodePayload>) {
199    let label = label.into();
200    let reference_str = reference.to_string();
201    let t0 = Instant::now();
202
203    let result = tokio::time::timeout(
204        Duration::from_millis(TIMEOUT_MS),
205        fetch_node_raw(client, base_url, reference),
206    )
207    .await;
208
209    let elapsed_ms = t0.elapsed().as_millis() as u64;
210
211    let (outcome, payload) = match result {
212        Ok(Ok(p)) if elapsed_ms >= SLOW_MS => (DiagOutcome::Slow { elapsed_ms }, Some(p)),
213        Ok(Ok(p)) => (DiagOutcome::Pass { elapsed_ms }, Some(p)),
214        Ok(Err(e)) => (
215            DiagOutcome::Fail {
216                elapsed_ms,
217                error: e,
218            },
219            None,
220        ),
221        Err(_) => (
222            DiagOutcome::Fail {
223                elapsed_ms,
224                error: format!("timed out after {}ms", TIMEOUT_MS),
225            },
226            None,
227        ),
228    };
229
230    (
231        DiagResult {
232            label,
233            reference: reference_str,
234            note: None,
235            phase,
236            outcome,
237        },
238        payload,
239    )
240}

242/// Derive a short human-readable label from the resolved NodePayload.
243fn label_from_payload(
244    reference: &hyperactor_mesh::introspect::NodeRef,
245    payload: &hyperactor_mesh::introspect::NodePayload,
246) -> String {
247    use hyperactor_mesh::introspect::NodeRef;
248    match &payload.properties {
249        NodeProperties::Root { .. } => "root".to_string(),
250        NodeProperties::Host { addr, .. } => addr.clone(),
251        NodeProperties::Proc { proc_name, .. } => proc_name.clone(),
252        NodeProperties::Actor { .. } => match reference {
253            NodeRef::Actor(actor_id) => {
254                format!("{}[{}]", actor_id.name(), actor_id.pid())
255            }
256            other => other.to_string(),
257        },
258        NodeProperties::Error { message, .. } => message.clone(),
259    }
260}

262/// Classify the operational role of a system proc by name.
263///
264/// Uses naming convention as identity — consistent with
265/// `hyperactor::host` construction. See LP-1.
266fn proc_role(proc_name: &str) -> DiagNodeRole {
267    if proc_name == LOCAL_PROC_NAME {
268        DiagNodeRole::LocalClientProc
269    } else {
270        DiagNodeRole::AdminServiceProc
271    }
272}

274/// Full diagnostic walk. Probes in order and emits results.
275async fn walk(client: &reqwest::Client, base_url: &str, tx: &mpsc::Sender<DiagResult>) {
276    // Phase 1 — Admin Infra
277
278    use hyperactor_mesh::introspect::NodeRef;
279
280    // Root.
281    let root_ref = NodeRef::Root;
282    let (mut result, root_payload) =
283        probe(client, base_url, "root", &root_ref, DiagPhase::AdminInfra).await;
284    result.note = Some(DiagNodeRole::AdminServer);
285    emit!(tx, result);
286    let root_payload = match root_payload {
287        Some(p) => p,
288        None => return,
289    };
290
291    // Root system_children is always empty (procs are never system,
292    // standalone procs removed from root). Filter kept for robustness.
293    let root_system_refs: HashSet<&NodeRef> = match &root_payload.properties {
294        NodeProperties::Root {
295            system_children, ..
296        } => system_children.iter().collect(),
297        _ => HashSet::new(),
298    };
299
300    for host_ref in root_payload
301        .children
302        .iter()
303        .filter(|r| !root_system_refs.contains(r))
304    {
305        // Host agent.
306        let host_label = host_ref.to_string();
307        let (mut r, host_payload) = probe(
308            client,
309            base_url,
310            &host_label,
311            host_ref,
312            DiagPhase::AdminInfra,
313        )
314        .await;
315        if let Some(p) = &host_payload {
316            r.label = label_from_payload(host_ref, p);
317        }
318        r.note = Some(DiagNodeRole::HostAgent);
319        emit!(tx, r);
320
321        let host_payload = match host_payload {
322            Some(p) => p,
323            None => continue,
324        };
325
326        let system_refs: HashSet<&NodeRef> = match &host_payload.properties {
327            NodeProperties::Host {
328                system_children, ..
329            } => system_children.iter().collect(),
330            _ => HashSet::new(),
331        };
332
333        // Service proc(s) and their actors.
334        for proc_ref in host_payload
335            .children
336            .iter()
337            .filter(|r| system_refs.contains(r))
338        {
339            let proc_label = proc_ref.to_string();
340            let (mut r, proc_payload) = probe(
341                client,
342                base_url,
343                &proc_label,
344                proc_ref,
345                DiagPhase::AdminInfra,
346            )
347            .await;
348            if let Some(p) = &proc_payload {
349                r.label = label_from_payload(proc_ref, p);
350            }
351            // Role is derived from proc_name via proc_role(). Only set
352            // when the proc resolved successfully; leave None on fetch
353            // failure so a bad probe isn't mislabelled.
354            r.note = proc_payload.as_ref().and_then(|p| match &p.properties {
355                NodeProperties::Proc { proc_name, .. } => Some(proc_role(proc_name)),
356                _ => None,
357            });
358            emit!(tx, r);
359
360            if let Some(proc_payload) = proc_payload {
361                for actor_ref in &proc_payload.children {
362                    let actor_label = actor_ref.to_string();
363                    let (mut r, payload) = probe(
364                        client,
365                        base_url,
366                        &actor_label,
367                        actor_ref,
368                        DiagPhase::AdminInfra,
369                    )
370                    .await;
371                    // Use fetched label if available; otherwise derive
372                    // from the typed ref directly.
373                    if let Some(p) = &payload {
374                        r.label = format!("  {}", label_from_payload(actor_ref, p));
375                    } else {
376                        let short = match actor_ref {
377                            NodeRef::Actor(id) => format!("{}[{}]", id.name(), id.pid()),
378                            other => other.to_string(),
379                        };
380                        r.label = format!("  {}", short);
381                    }
382                    // Classify actor role from the typed ref.
383                    r.note = match actor_ref {
384                        NodeRef::Actor(actor_id) => {
385                            let actor_name = actor_id.name();
386                            if actor_name.starts_with(MESH_ADMIN_BRIDGE_NAME) {
387                                Some(DiagNodeRole::RootClientBridge)
388                            } else if actor_name.starts_with(MESH_ADMIN_ACTOR_NAME) {
389                                Some(DiagNodeRole::IntrospectionHandler)
390                            } else if actor_name.starts_with(HOST_MESH_AGENT_ACTOR_NAME) {
391                                Some(DiagNodeRole::ActorLifecycleManager)
392                            } else {
393                                None
394                            }
395                        }
396                        _ => None,
397                    };
398                    emit!(tx, r);
399                }
400            }
401        }
402
403        // Phase 2 — Mesh: every user proc, all its system actors, and
404        // the first non-system actor as a representative user actor.
405        for user_proc_ref in host_payload
406            .children
407            .iter()
408            .filter(|r| !system_refs.contains(r))
409        {
410            let proc_label = user_proc_ref.to_string();
411            let (mut r, proc_payload) = probe(
412                client,
413                base_url,
414                &proc_label,
415                user_proc_ref,
416                DiagPhase::Mesh,
417            )
418            .await;
419            if let Some(p) = &proc_payload {
420                r.label = label_from_payload(user_proc_ref, p);
421            }
422            r.note = Some(DiagNodeRole::UserProc);
423            emit!(tx, r);
424
425            if let Some(proc_payload) = proc_payload {
426                let proc_system_refs: HashSet<&NodeRef> = match &proc_payload.properties {
427                    NodeProperties::Proc {
428                        system_children, ..
429                    } => system_children.iter().collect(),
430                    _ => HashSet::new(),
431                };
432
433                // Derive actor role from a typed NodeRef.
434                let actor_role = |nr: &NodeRef| -> Option<DiagNodeRole> {
435                    match nr {
436                        NodeRef::Actor(actor_id) => {
437                            let name = actor_id.name();
438                            if name.starts_with(COMM_ACTOR_NAME) {
439                                Some(DiagNodeRole::CommActor)
440                            } else if name.starts_with(PROC_AGENT_ACTOR_NAME) {
441                                Some(DiagNodeRole::ProcAgent)
442                            } else {
443                                None
444                            }
445                        }
446                        _ => None,
447                    }
448                };
449
450                // Probe every system actor on this user proc.
451                for actor_ref in proc_payload
452                    .children
453                    .iter()
454                    .filter(|r| proc_system_refs.contains(r))
455                {
456                    let alabel = actor_ref.to_string();
457                    let (mut r, payload) =
458                        probe(client, base_url, &alabel, actor_ref, DiagPhase::Mesh).await;
459                    if let Some(p) = &payload {
460                        r.label = format!("  {}", label_from_payload(actor_ref, p));
461                    } else {
462                        r.label = format!("  {}", alabel);
463                    }
464                    r.note = actor_role(actor_ref);
465                    emit!(tx, r);
466                }
467
468                // Probe the first non-system actor as a representative.
469                if let Some(actor_ref) = proc_payload
470                    .children
471                    .iter()
472                    .find(|r| !proc_system_refs.contains(r))
473                {
474                    let alabel = actor_ref.to_string();
475                    let (mut r, payload) =
476                        probe(client, base_url, &alabel, actor_ref, DiagPhase::Mesh).await;
477                    if let Some(p) = &payload {
478                        r.label = format!("  {}", label_from_payload(actor_ref, p));
479                    } else {
480                        r.label = format!("  {}", alabel);
481                    }
482                    r.note = Some(DiagNodeRole::UserActor);
483                    emit!(tx, r);
484                }
485            }
486        }
487    }
488}

490#[cfg(test)]
491mod tests {
492    use super::*;
493
494    fn phase_fail_count(results: &[DiagResult], phase: DiagPhase) -> usize {
495        results
496            .iter()
497            .filter(|r| r.phase == phase && matches!(r.outcome, DiagOutcome::Fail { .. }))
498            .count()
499    }
500
501    // Exercises LP-1 (see hyperactor::host module doc).
502    #[test]
503    fn test_proc_role_classification() {
504        assert!(matches!(
505            proc_role(LOCAL_PROC_NAME),
506            DiagNodeRole::LocalClientProc
507        ));
508        assert!(matches!(
509            proc_role(hyperactor::host::SERVICE_PROC_NAME),
510            DiagNodeRole::AdminServiceProc
511        ));
512        assert!(matches!(
513            proc_role("anything_else"),
514            DiagNodeRole::AdminServiceProc
515        ));
516    }
517
518    // Exercises LP-1 (see hyperactor::host module doc).
519    #[test]
520    fn test_empty_local_proc_does_not_degrade_admin_health() {
521        let results = vec![DiagResult {
522            label: "local".to_string(),
523            reference: "some_ref".to_string(),
524            note: Some(DiagNodeRole::LocalClientProc),
525            phase: DiagPhase::AdminInfra,
526            outcome: DiagOutcome::Pass { elapsed_ms: 1 },
527        }];
528        assert_eq!(phase_fail_count(&results, DiagPhase::AdminInfra), 0);
529    }
