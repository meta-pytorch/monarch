9//! Read-only FUSE filesystem backed by packed metadata and byte chunks.
10//!
11//! Replaces the Python fusepy-based ChunkedFS with a Rust implementation
12//! using the `fuse3` crate (libfuse3, async/tokio). Exposed to Python
13//! via PyO3.
14//!
15//! Supports live refresh: `PyMountHandle::refresh` atomically swaps
16//! the filesystem metadata behind a `RwLock` without unmounting. Chunk
17//! data lives in a separate `Arc<Mutex<Vec<u8>>>` flat buffer that can be
18//! updated in-place via `update_chunk_range` without a full swap.
19//! All FUSE methods acquire a read lock on metadata, so reads are
20//! lock-free relative to each other and only briefly blocked during
21//! the write-lock swap.
22

48enum FsEntry {
49    Dir {
50        attr: FileAttr,
51        children: Vec<String>,
52    },
53    File {
54        attr: FileAttr,
55        global_offset: usize,
56        file_len: usize,
57    },
58    Symlink {
59        attr: FileAttr,
60        link_target: OsString,
61    },
62}
63
64impl FsEntry {
65    fn attr(&self) -> &FileAttr {
66        match self {
67            FsEntry::Dir { attr, .. } => attr,
68            FsEntry::File { attr, .. } => attr,
69            FsEntry::Symlink { attr, .. } => attr,
70        }
71    }
72}

74/// All filesystem state — metadata and chunk data — protected by a single RwLock.
75///
76/// Keeping them together ensures every FUSE read sees a fully consistent
77/// snapshot: no reader can observe new chunk data paired with old file
78/// offsets/sizes (or vice versa), because `refresh` applies both under
79/// one write-lock acquisition.
80struct FsState {
81    metadata: HashMap<OsString, FsEntry>,
82    chunks: Vec<u8>,
83    chunk_size: usize,
84}
85
86impl FsState {
87    fn lookup_entry(&self, path: &OsStr) -> Option<&FsEntry> {
88        self.metadata.get(path)
89    }
90}

92/// Read chunk data from the flat buffer.
93///
94/// Reads `len` bytes starting at `start` from `buf`. Returns an owned
95/// `Bytes` copy — the copy is small (one FUSE read, up to 128KB) and
96/// avoids holding the lock across an await point.
97fn read_chunk_data(
98    buf: &[u8],
99    chunk_size: usize,
100    global_offset: usize,
101    file_len: usize,
102    offset: u64,
103    size: u32,
104) -> Result<Bytes, libc::c_int> {
105    let offset = offset as usize;
106    if offset >= file_len {
107        return Ok(Bytes::new());
108    }
109    let len = std::cmp::min(size as usize, file_len - offset);
110    let start = global_offset + offset;
111    let end = start + len;
112
113    if end > buf.len() {
114        warn!(
115            "read out of bounds: start={start} end={end} buf_len={} \
116             global_offset={global_offset} file_len={file_len} offset={offset}",
117            buf.len()
118        );
119        // Return what we have rather than an error, to be lenient.
120        let available = buf.len().saturating_sub(start);
121        if available == 0 {
122            return Ok(Bytes::new());
123        }
124        return Ok(Bytes::copy_from_slice(&buf[start..start + available]));
125    }
126
127    // Single contiguous read from the flat buffer — no chunk-boundary logic needed.
128    let _ = chunk_size; // kept in FsMetadata for future use
129    Ok(Bytes::copy_from_slice(&buf[start..end]))
130}

132fn join_path(parent: &OsStr, name: &OsStr) -> OsString {
133    let parent_s = parent.to_string_lossy();
134    let name_s = name.to_string_lossy();
135    if parent_s == "/" {
136        OsString::from(format!("/{name_s}"))
137    } else {
138        OsString::from(format!("{parent_s}/{name_s}"))
139    }
140}
141
142fn parent_path(path: &OsStr) -> OsString {
143    let s = path.to_string_lossy();
144    match s.rfind('/') {
145        Some(0) | None => OsString::from("/"),
146        Some(i) => OsString::from(&s[..i]),
147    }
148}

150fn f64_to_system_time(ts: f64) -> SystemTime {
151    if ts >= 0.0 {
152        UNIX_EPOCH + Duration::from_secs_f64(ts)
153    } else {
154        UNIX_EPOCH
155    }

158fn required_key<'py, T: pyo3::FromPyObject<'py>>(
159    dict: &Bound<'py, PyDict>,
160    key: &str,
161) -> PyResult<T> {
162    dict.get_item(key)?
163        .ok_or_else(|| PyRuntimeError::new_err(format!("missing required attr key: {key}")))?
164        .extract()
165}

167fn extract_attr(dict: &Bound<'_, PyDict>, kind: FileType) -> PyResult<FileAttr> {
168    let st_size: u64 = required_key(dict, "st_size")?;
169    Ok(FileAttr {
170        size: st_size,
171        blocks: st_size.div_ceil(512),
172        atime: f64_to_system_time(required_key(dict, "st_atime")?),
173        mtime: f64_to_system_time(required_key(dict, "st_mtime")?),
174        ctime: f64_to_system_time(required_key(dict, "st_ctime")?),
175        kind,
176        perm: (required_key::<u32>(dict, "st_mode")? & 0o7777) as u16,
177        nlink: required_key(dict, "st_nlink")?,
178        uid: required_key(dict, "st_uid")?,
179        gid: required_key(dict, "st_gid")?,
180        rdev: 0,
181        blksize: 4096,
182    })
183}

185fn extract_metadata(dict: &Bound<'_, PyDict>) -> PyResult<HashMap<OsString, FsEntry>> {
186    let mut entries = HashMap::with_capacity(dict.len());
187    for (key, value) in dict.iter() {
188        let path: String = key.extract()?;
189        let entry_dict: &Bound<'_, PyDict> = value.downcast()?;
190        let attr_obj = entry_dict.get_item("attr")?.ok_or_else(|| {
191            PyRuntimeError::new_err(format!("missing 'attr' key for path: {path}"))
192        })?;
193        let attr_dict: &Bound<'_, PyDict> = attr_obj.downcast()?;
194
195        let fs_entry = if let Some(link_target) = entry_dict.get_item("link_target")? {
196            let target: String = link_target.extract()?;
197            FsEntry::Symlink {
198                attr: extract_attr(attr_dict, FileType::Symlink)?,
199                link_target: OsString::from(target),
200            }
201        } else if let Some(children_obj) = entry_dict.get_item("children")? {
202            let children: Vec<String> = children_obj.extract()?;
203            FsEntry::Dir {
204                attr: extract_attr(attr_dict, FileType::Directory)?,
205                children,
206            }
207        } else {
208            let global_offset: usize = entry_dict
209                .get_item("global_offset")?
210                .map(|v| v.extract())
211                .transpose()?
212                .unwrap_or(0);
213            let file_len: usize = entry_dict
214                .get_item("file_len")?
215                .map(|v| v.extract())
216                .transpose()?
217                .unwrap_or(0);
218            FsEntry::File {
219                attr: extract_attr(attr_dict, FileType::RegularFile)?,
220                global_offset,
221                file_len,
222            }
223        };
224        entries.insert(OsString::from(path), fs_entry);
225    }
226    Ok(entries)
227}

229/// Build the flat chunk buffer by concatenating all Python chunk buffers.
230///
231/// Called once at mount time. For subsequent refreshes, `update_chunk_range`
232/// is used to write only dirty blocks in-place.
233fn build_flat_buffer(chunks: Vec<pyo3::buffer::PyBuffer<u8>>) -> Vec<u8> {
234    let total: usize = chunks.iter().map(|b| b.len_bytes()).sum();
235    let mut out = Vec::with_capacity(total);
236    for buf in &chunks {
237        // SAFETY: buf is a live PyBuffer and we hold the GIL.
238        let slice =
239            unsafe { std::slice::from_raw_parts(buf.buf_ptr() as *const u8, buf.len_bytes()) };
240        out.extend_from_slice(slice);
241    }
242    out
243}

252impl PathFilesystem for ChunkedFuseFs {
253    type DirEntryStream<'a> = stream::Iter<std::vec::IntoIter<FuseResult<DirectoryEntry>>>;
254    type DirEntryPlusStream<'a> = stream::Iter<std::vec::IntoIter<FuseResult<DirectoryEntryPlus>>>;
255
256    async fn init(&self, _req: Request) -> FuseResult<ReplyInit> {
257        Ok(ReplyInit {
258            max_write: NonZeroU32::new(16 * 1024).expect("16KB is non-zero"),
259        })
260    }

262    async fn destroy(&self, _req: Request) {}
263
264    async fn lookup(&self, _req: Request, parent: &OsStr, name: &OsStr) -> FuseResult<ReplyEntry> {
265        let path = join_path(parent, name);
266        let state = self.state.read().unwrap();
267        let entry = state.lookup_entry(&path).ok_or_else(Errno::new_not_exist)?;
268        Ok(ReplyEntry {
269            ttl: self.ttl,
270            attr: *entry.attr(),
271        })
272    }

274    async fn getattr(
275        &self,
276        _req: Request,
277        path: Option<&OsStr>,
278        _fh: Option<u64>,
279        _flags: u32,
280    ) -> FuseResult<ReplyAttr> {
281        let path = path.ok_or_else(Errno::new_not_exist)?;
282        let state = self.state.read().unwrap();
283        let entry = state.lookup_entry(path).ok_or_else(Errno::new_not_exist)?;
284        Ok(ReplyAttr {
285            ttl: self.ttl,
286            attr: *entry.attr(),
287        })
288    }
289
290    async fn readlink(&self, _req: Request, path: &OsStr) -> FuseResult<ReplyData> {
291        let state = self.state.read().unwrap();
292        let entry = state.lookup_entry(path).ok_or_else(Errno::new_not_exist)?;
293        match entry {
294            FsEntry::Symlink { link_target, .. } => Ok(ReplyData {
295                data: Bytes::copy_from_slice(link_target.as_encoded_bytes()),
296            }),
297            _ => Err(libc::EINVAL.into()),
298        }
299    }
300
301    async fn open(&self, _req: Request, path: &OsStr, flags: u32) -> FuseResult<ReplyOpen> {
302        let state = self.state.read().unwrap();
303        let entry = state.lookup_entry(path).ok_or_else(Errno::new_not_exist)?;
304        if matches!(entry, FsEntry::Dir { .. }) {
305            return Err(Errno::new_is_dir());
306        }
307        Ok(ReplyOpen { fh: 0, flags })
308    }
309
310    async fn read(
311        &self,
312        _req: Request,
313        path: Option<&OsStr>,
314        _fh: u64,
315        offset: u64,
316        size: u32,
317    ) -> FuseResult<ReplyData> {
318        let path = path.ok_or_else(Errno::new_not_exist)?;
319        // Hold the read lock for both the metadata lookup and the data copy.
320        // This ensures no refresh() can swap metadata or patch chunks between
321        // the two operations, eliminating the TOCTOU race.
322        let state = self.state.read().unwrap();
323        let entry = state.lookup_entry(path).ok_or_else(Errno::new_not_exist)?;
324        match entry {
325            FsEntry::File {
326                global_offset,
327                file_len,
328                ..
329            } => {
330                let result = read_chunk_data(
331                    &state.chunks,
332                    state.chunk_size,
333                    *global_offset,
334                    *file_len,
335                    offset,
336                    size,
337                )
338                .map_err(Errno::from)?;
339                Ok(ReplyData { data: result })
340            }
341            _ => Err(libc::EINVAL.into()),
342        }
343    }

345    async fn opendir(&self, _req: Request, path: &OsStr, flags: u32) -> FuseResult<ReplyOpen> {
346        let state = self.state.read().unwrap();
347        let entry = state.lookup_entry(path).ok_or_else(Errno::new_not_exist)?;
348        if !matches!(entry, FsEntry::Dir { .. }) {
349            return Err(Errno::new_is_not_dir());
350        }
351        Ok(ReplyOpen { fh: 0, flags })
352    }
353
354    async fn readdir<'a>(
355        &'a self,
356        _req: Request,
357        parent: &'a OsStr,
358        _fh: u64,
359        offset: i64,
360    ) -> FuseResult<ReplyDirectory<Self::DirEntryStream<'a>>> {
361        let state = self.state.read().unwrap();
362        let entry = state
363            .lookup_entry(parent)
364            .ok_or_else(Errno::new_not_exist)?;
365        let children = match entry {
366            FsEntry::Dir { children, .. } => children,
367            _ => return Err(Errno::new_is_not_dir()),
368        };
369
370        let offset = offset as u64;
371        let mut entries: Vec<FuseResult<DirectoryEntry>> = Vec::new();
372        let mut idx: u64 = 1;
373
374        if offset < idx {
375            entries.push(Ok(DirectoryEntry {
376                kind: FileType::Directory,
377                name: OsString::from("."),
378                offset: idx as i64,
379            }));
380        }
381        idx += 1;
382
383        if offset < idx {
384            entries.push(Ok(DirectoryEntry {
385                kind: FileType::Directory,
386                name: OsString::from(".."),
387                offset: idx as i64,
388            }));
389        }
390        idx += 1;
391
392        for child_name in children {
393            if offset < idx {
394                let child_path = join_path(parent, OsStr::new(child_name));
395                if let Some(child_entry) = state.lookup_entry(&child_path) {
396                    entries.push(Ok(DirectoryEntry {
397                        kind: child_entry.attr().kind,
398                        name: OsString::from(child_name),
399                        offset: idx as i64,
400                    }));
401                }
402            }
403            idx += 1;
404        }
405
406        Ok(ReplyDirectory {
407            entries: stream::iter(entries),
408        })
409    }

411    async fn readdirplus<'a>(
412        &'a self,
413        _req: Request,
414        parent: &'a OsStr,
415        _fh: u64,
416        offset: u64,
417        _lock_owner: u64,
418    ) -> FuseResult<ReplyDirectoryPlus<Self::DirEntryPlusStream<'a>>> {
419        let state = self.state.read().unwrap();
420        let entry = state
421            .lookup_entry(parent)
422            .ok_or_else(Errno::new_not_exist)?;
423        let children = match entry {
424            FsEntry::Dir { children, .. } => children,
425            _ => return Err(Errno::new_is_not_dir()),
426        };
427
428        let ttl = self.ttl;
429        let mut entries: Vec<FuseResult<DirectoryEntryPlus>> = Vec::new();
430        let mut idx: u64 = 1;
431
432        if offset < idx {
433            entries.push(Ok(DirectoryEntryPlus {
434                kind: FileType::Directory,
435                name: OsString::from("."),
436                offset: idx as i64,
437                attr: *entry.attr(),
438                entry_ttl: ttl,
439                attr_ttl: ttl,
440            }));
441        }
442        idx += 1;
443
444        if offset < idx {
445            let parent_key = parent_path(parent);
446            let dotdot_attr = match state.lookup_entry(&parent_key) {
447                Some(e) => *e.attr(),
448                None => {
449                    warn!(
450                        "readdirplus: parent path {:?} not found in metadata, \
451                         falling back to current dir attrs",
452                        parent_key
453                    );
454                    *entry.attr()
455                }
456            };
457            entries.push(Ok(DirectoryEntryPlus {
458                kind: FileType::Directory,
459                name: OsString::from(".."),
460                offset: idx as i64,
461                attr: dotdot_attr,
462                entry_ttl: ttl,
463                attr_ttl: ttl,
464            }));
465        }
466        idx += 1;
467
468        for child_name in children {
469            if offset < idx {
470                let child_path = join_path(parent, OsStr::new(child_name));
471                if let Some(child_entry) = state.lookup_entry(&child_path) {
472                    entries.push(Ok(DirectoryEntryPlus {
473                        kind: child_entry.attr().kind,
474                        name: OsString::from(child_name),
475                        offset: idx as i64,
476                        attr: *child_entry.attr(),
477                        entry_ttl: ttl,
478                        attr_ttl: ttl,
479                    }));
480                }
481            }
482            idx += 1;
483        }
484
485        Ok(ReplyDirectoryPlus {
486            entries: stream::iter(entries),
487        })
488    }

534/// Handle to a running FUSE mount. Call `unmount()` to stop it.
535#[pyclass(
536    name = "FuseMountHandle",
537    module = "monarch._rust_bindings.monarch_extension.chunked_fuse"
538)]
539struct PyMountHandle {
540    unmount_tx: Option<oneshot::Sender<()>>,
541    mount_point: String,
542    state: Arc<RwLock<FsState>>,
543}
544
545#[pymethods]
546impl PyMountHandle {
547    /// Unmount the FUSE filesystem.
548    ///
549    /// Cleanup of the background FUSE session task is asynchronous: this
550    /// method returns once `fusermount3 -u` completes, but the tokio task
551    /// may still be winding down.
552    fn unmount(&mut self, py: Python<'_>) -> PyResult<()> {
553        if self.unmount_tx.take().is_none() {
554            return Ok(());
555        }
556        let mount_point = self.mount_point.clone();
557        py.detach(|| {
558            match std::process::Command::new("fusermount3")
559                .arg("-u")
560                .arg(&mount_point)
561                .output()
562            {
563                Ok(o) if !o.status.success() => {
564                    let stderr = String::from_utf8_lossy(&o.stderr);
565                    Err(PyRuntimeError::new_err(format!(
566                        "fusermount3 -u {mount_point} failed: {stderr}"
567                    )))
568                }
569                Err(e) => Err(PyRuntimeError::new_err(format!(
570                    "failed to run fusermount3: {e}"
571                ))),
572                _ => Ok(()),
573            }
574        })
575    }
576
577    /// Atomically apply dirty chunk patches and swap filesystem metadata.
578    ///
579    /// Acquires the write lock once, so no FUSE read can observe a mix of
580    /// old metadata with new chunk data (or vice versa).
581    ///
582    /// `chunk_buf` is the full Python mmap (memoryview). `dirty_ranges` is a
583    /// list of `(offset, length)` byte ranges within that buffer to copy into
584    /// the flat chunk store. Pass `[(0, total_size)]` for an initial full copy.
585    ///
586    /// Sleeps 3× the FUSE TTL after the lock is released so kernel
587    /// attr/dentry caches expire before returning.
588    fn refresh(
589        &self,
590        py: Python<'_>,
591        metadata: &Bound<'_, PyDict>,
592        chunk_buf: pyo3::buffer::PyBuffer<u8>,
593        dirty_ranges: Vec<(usize, usize)>,
594        new_total_size: usize,
595        chunk_size: usize,
596    ) -> PyResult<()> {
597        let new_metadata = extract_metadata(metadata)?;
598        // SAFETY: chunk_buf is a live PyBuffer; we hold the GIL throughout
599        // the write-lock scope below, so the buffer remains valid.
600        let src = unsafe {
601            std::slice::from_raw_parts(chunk_buf.buf_ptr() as *const u8, chunk_buf.len_bytes())
602        };
603        {
604            let mut state = self.state.write().unwrap();
605            state.chunks.resize(new_total_size, 0);
606            for (offset, length) in &dirty_ranges {
607                state.chunks[*offset..*offset + *length]
608                    .copy_from_slice(&src[*offset..*offset + *length]);
609            }
610            state.metadata = new_metadata;
611            state.chunk_size = chunk_size;
612        }
613        // Wait for kernel FUSE attr/page caches (TTL=200ms) to expire.
614        let settle = Duration::from_millis(DEFAULT_TTL_MS * 3);
615        py.detach(|| {
616            #[allow(clippy::disallowed_methods)]
617            std::thread::sleep(settle);
618            Ok(())
619        })
620    }
621}

623/// Mount a read-only FUSE filesystem from packed metadata and chunks.
624///
625/// Args:
626///     metadata: dict mapping paths to entry dicts (as produced by
627///         `pack_directory_chunked`).
628///     chunks: list of memoryview/bytes chunks (used only at initial mount;
629///         copied once into a flat Rust-owned buffer).
630///     chunk_size: size of each chunk in bytes.
631///     mount_point: path to mount the filesystem.
632/// Returns a FuseMountHandle. Call handle.unmount() to unmount,
633/// handle.refresh() to atomically swap metadata, or
634/// handle.update_chunk_range() to write dirty blocks in-place.
635#[pyfunction]
636fn mount_chunked_fuse(
637    py: Python<'_>,
638    metadata: &Bound<'_, PyDict>,
639    chunks: Vec<pyo3::buffer::PyBuffer<u8>>,
640    chunk_size: usize,
641    mount_point: String,
642) -> PyResult<PyMountHandle> {
643    let ttl_ms = DEFAULT_TTL_MS;
644    if chunk_size == 0 {
645        return Err(PyRuntimeError::new_err("chunk_size must be > 0"));
646    }
647    if !mount_point.starts_with('/') {
648        return Err(PyRuntimeError::new_err(
649            "mount_point must be an absolute path",
650        ));
651    }
652
653    let metadata = extract_metadata(metadata)?;
654
655    // Build the initial flat buffer by concatenating all chunk buffers.
656    // Python passes [] at mount time; the first refresh() call populates it.
657    let flat_buf = build_flat_buffer(chunks);
658
659    let state = Arc::new(RwLock::new(FsState {
660        metadata,
661        chunks: flat_buf,
662        chunk_size,
663    }));
664
665    let fs = ChunkedFuseFs {
666        state: state.clone(),
667        ttl: Duration::from_millis(ttl_ms),
668    };
669
670    let mount_path = mount_point.clone();
671    let (unmount_tx, unmount_rx) = oneshot::channel::<()>();
672    let (mount_result_tx, mount_result_rx) = oneshot::channel::<Result<(), String>>();
673
674    // The FUSE session task runs on the shared tokio runtime. We do not
675    // retain the JoinHandle because the task's lifetime is governed by
676    // fusermount3 -u (called in unmount()). If the Python interpreter
677    // exits without calling unmount(), the runtime's atexit shutdown will
678    // drop the task. This is acceptable: the kernel automatically cleans
679    // up the FUSE mount when the process exits.
680    let runtime = monarch_hyperactor::runtime::get_tokio_runtime();
681    runtime.spawn(async move {
682        let mut opts = MountOptions::default();
683        opts.read_only(true).force_readdir_plus(true).nonempty(true);
684
685        let mount_result = fuse3::path::Session::new(opts)
686            .mount_with_unprivileged(fs, &mount_path)
687            .await;
688
689        match mount_result {
690            Ok(mount_handle) => {
691                let _ = mount_result_tx.send(Ok(()));
692                // Run the FUSE session until unmount() calls fusermount3 -u,
693                // which causes the session future to complete.
694                if let Err(e) = mount_handle.await {
695                    warn!("fuse session error: {e}");
696                }
697            }
698            Err(e) => {
699                warn!("fuse mount failed: {e}");
700                let _ = mount_result_tx.send(Err(format!("{e}")));
701            }
702        }
703        drop(unmount_rx);
704    });
705
706    // Poll until mount appears in /proc/mounts or the task signals
707    // failure. Releases the GIL while waiting. We use blocking
708    // std::thread::sleep because this is a synchronous PyO3 context.
709    //
710    // NOTE: Polling /proc/mounts has an inherent TOCTOU race: the mount
711    // could theoretically appear and disappear between our check and the
712    // caller's first filesystem access. In practice this doesn't happen
713    // because nothing else unmounts the path, but inotify on /proc/mounts
714    // would be a more robust alternative if needed.
715    #[allow(clippy::disallowed_methods)]
716    py.detach(|| {
717        let start = std::time::Instant::now();
718        let mut mount_result_rx = Some(mount_result_rx);
719        while start.elapsed() < Duration::from_secs(50) {
720            // Check if the mount task reported an early failure.
721            if let Some(rx) = mount_result_rx.as_mut() {
722                match rx.try_recv() {
723                    Ok(Err(e)) => {
724                        return Err(PyRuntimeError::new_err(format!("fuse mount failed: {e}")));
725                    }
726                    Ok(Ok(())) => {
727                        // Mount succeeded; stop checking the channel but
728                        // still wait for /proc/mounts visibility.
729                        mount_result_rx = None;
730                    }
731                    Err(oneshot::error::TryRecvError::Closed) => {
732                        return Err(PyRuntimeError::new_err(
733                            "fuse mount task exited without reporting status \
734                             (possible panic, tokio runtime shutdown, or \
735                             task cancellation)",
736                        ));
737                    }
738                    Err(oneshot::error::TryRecvError::Empty) => {}
739                }
740            }
741
742            // Check /proc/mounts directly (authoritative source).
743            if let Ok(mounts) = std::fs::read_to_string("/proc/mounts") {
744                for line in mounts.lines() {
745                    if let Some(mp) = line.split_whitespace().nth(1) {
746                        if mp == mount_point {
747                            return Ok(());
748                        }
749                    }
750                }
751            }
752
753            std::thread::sleep(Duration::from_millis(100));
754        }
755        Err(PyRuntimeError::new_err("timed out waiting for FUSE mount"))
756    })?;
757
758    Ok(PyMountHandle {
759        unmount_tx: Some(unmount_tx),
760        mount_point,
761        state,
762    })
763}
