40/// A `ServerCertVerifier` that accepts any server certificate.

83/// Build a `rustls::ClientConfig`.
84///
85/// If `ca_path` is provided, server certs are verified against that CA.
86/// If `ca_path` is `None`, any server cert is accepted (for self-signed certs).
87/// If `cert_path` is provided, it's used for client auth; otherwise no client auth.
88fn make_tls_config(
89    cert_path: Option<&str>,
90    ca_path: Option<&str>,
91) -> Result<Arc<rustls::ClientConfig>, String> {
92    let _ = rustls::crypto::ring::default_provider().install_default();
93
94    let builder = rustls::ClientConfig::builder();
95
96    let builder = match ca_path {
97        Some(ca) => {
98            let ca_pem = std::fs::read(ca).map_err(|e| format!("read {ca} failed: {e}"))?;
99            let mut root_store = rustls::RootCertStore::empty();
100            let ca_certs = rustls_pemfile::certs(&mut BufReader::new(&ca_pem[..]))
101                .filter_map(Result::ok)
102                .collect::<Vec<_>>();
103            root_store.add_parsable_certificates(ca_certs);
104            builder.with_root_certificates(root_store)
105        }
106        None => builder
107            .dangerous()
108            .with_custom_certificate_verifier(Arc::new(AcceptAnyCert)),
109    };
110
111    let config = match cert_path {
112        Some(cp) => {
113            let cert_pem = std::fs::read(cp).map_err(|e| format!("read {cp} failed: {e}"))?;
114            let certs = rustls_pemfile::certs(&mut BufReader::new(&cert_pem[..]))
115                .filter_map(Result::ok)
116                .collect::<Vec<_>>();
117            let key = {
118                let mut reader = BufReader::new(&cert_pem[..]);
119                loop {
120                    match rustls_pemfile::read_one(&mut reader) {
121                        Ok(Some(rustls_pemfile::Item::Pkcs1Key(k))) => {
122                            break rustls::pki_types::PrivateKeyDer::Pkcs1(k);
123                        }
124                        Ok(Some(rustls_pemfile::Item::Pkcs8Key(k))) => {
125                            break rustls::pki_types::PrivateKeyDer::Pkcs8(k);
126                        }
127                        Ok(Some(rustls_pemfile::Item::Sec1Key(k))) => {
128                            break rustls::pki_types::PrivateKeyDer::Sec1(k);
129                        }
130                        Ok(Some(_)) => continue,
131                        Ok(None) => return Err(format!("no private key found in {cp}")),
132                        Err(e) => return Err(format!("parse {cp} failed: {e}")),
133                    }
134                }
135            };
136            builder
137                .with_client_auth_cert(certs, key)
138                .map_err(|e| format!("client auth cert failed: {e}"))?
139        }
140        None => builder.with_no_client_auth(),
141    };
142
143    Ok(Arc::new(config))
144}

146/// RAII wrapper for an anonymous mmap buffer with precomputed block hashes.
147#[pyclass(module = "monarch._rust_bindings.monarch_extension.tls_sender")]
148struct PackedBuffer {
149    ptr: *mut libc::c_void,
150    size: usize,
151    #[pyo3(get)]
152    hashes: Vec<String>,
153}
154
155// SAFETY: the mmap region is process-global memory accessible from any thread.
156// No thread-local state; all access is via raw pointer arithmetic on a
157// process-wide anonymous mapping that outlives any thread.
158unsafe impl Send for PackedBuffer {}
159// SAFETY: PackedBuffer is read-only after construction (ptr/size are immutable).
160// Concurrent readers access disjoint slices via offset-based indexing.
161unsafe impl Sync for PackedBuffer {}
162
163impl Drop for PackedBuffer {
164    fn drop(&mut self) {
165        if self.size > 0 {
166            // SAFETY: self.ptr was returned by mmap_anonymous(self.size) and
167            // has not been munmapped yet. Drop runs at most once.
168            unsafe {
169                libc::munmap(self.ptr, self.size);
170            }
171        }
172    }
173}

175/// Send dirty blocks from a buffer over parallel TLS connections.
176///
177/// Each address in `addresses` gets its own TCP+TLS stream. Blocks
178/// are distributed round-robin across streams.
179fn send_blocks_impl(
180    buf_ptr: usize,
181    total_size: usize,
182    dirty_blocks: &[usize],
183    addresses: &[String],
184    cache_path: &str,
185    block_size: usize,
186    tls_hostname: Option<&str>,
187    cert_path: Option<&str>,
188    ca_path: Option<&str>,
189) -> PyResult<()> {
190    let tls_config = make_tls_config(cert_path, ca_path).map_err(PyRuntimeError::new_err)?;
191
192    let num_streams = addresses.len();
193
194    // Partition blocks across streams round-robin.
195    let mut per_stream: Vec<Vec<(usize, usize)>> = vec![Vec::new(); num_streams];
196    for (i, &bi) in dirty_blocks.iter().enumerate() {
197        let offset = bi.checked_mul(block_size).ok_or_else(|| {
198            PyRuntimeError::new_err(format!("block {bi} * {block_size} overflows"))
199        })?;
200        if offset >= total_size {
201            return Err(PyRuntimeError::new_err(format!(
202                "block {bi} offset {offset} >= total_size {total_size}"
203            )));
204        }
205        let size = std::cmp::min(block_size, total_size - offset);
206        per_stream[i % num_streams].push((offset, size));
207    }
208
209    let cache_path_bytes = cache_path.as_bytes();
210
211    thread::scope(|s| {
212        let mut handles = Vec::with_capacity(num_streams);
213
214        for (stream_idx, blocks) in per_stream.into_iter().enumerate() {
215            let addr = &addresses[stream_idx];
216            let config = Arc::clone(&tls_config);
217            let cp_bytes = cache_path_bytes;
218            let tls_name = tls_hostname;
219
220            handles.push(s.spawn(move || -> Result<(), String> {
221                let (host, port_str) = addr
222                    .rsplit_once(':')
223                    .ok_or_else(|| format!("invalid address: {addr}"))?;
224                let port: u16 = port_str
225                    .parse()
226                    .map_err(|e| format!("invalid port in {addr}: {e}"))?;
227
228                let tcp =
229                    TcpStream::connect((host, port)).map_err(|e| format!("connect {addr}: {e}"))?;
230                tcp.set_nodelay(true)
231                    .map_err(|e| format!("set_nodelay: {e}"))?;
232
233                // 4 MB send buffer for high-bandwidth transfers.
234                #[cfg(unix)]
235                {
236                    use std::os::unix::io::AsRawFd;
237                    let bufsize: libc::c_int = 4 * 1024 * 1024;
238                    // SAFETY: setsockopt with SOL_SOCKET/SO_SNDBUF is safe;
239                    // bufsize is a valid c_int on the stack.
240                    unsafe {
241                        libc::setsockopt(
242                            tcp.as_raw_fd(),
243                            libc::SOL_SOCKET,
244                            libc::SO_SNDBUF,
245                            &bufsize as *const _ as *const libc::c_void,
246                            std::mem::size_of::<libc::c_int>() as libc::socklen_t,
247                        );
248                    }
249                }
250
251                let sni_host = tls_name.unwrap_or(host);
252                let server_name = ServerName::try_from(sni_host.to_string())
253                    .unwrap_or_else(|_| ServerName::try_from("localhost".to_string()).unwrap());
254
255                let conn = rustls::ClientConnection::new(config, server_name)
256                    .map_err(|e| format!("TLS handshake: {e}"))?;
257                let mut tls = rustls::StreamOwned::new(conn, tcp);
258
259                // Protocol header: cache_path_len(u32 BE) + cache_path + total_size(u64 BE)
260                let mut header = Vec::with_capacity(4 + cp_bytes.len() + 8);
261                header.extend_from_slice(&(cp_bytes.len() as u32).to_be_bytes());
262                header.extend_from_slice(cp_bytes);
263                header.extend_from_slice(&(total_size as u64).to_be_bytes());
264                tls.write_all(&header)
265                    .map_err(|e| format!("write header: {e}"))?;
266
267                // Send blocks: offset(u64 BE) + size(u64 BE) + data
268                for (offset, size) in &blocks {
269                    let mut block_header = [0u8; 16];
270                    block_header[..8].copy_from_slice(&(*offset as u64).to_be_bytes());
271                    block_header[8..].copy_from_slice(&(*size as u64).to_be_bytes());
272                    tls.write_all(&block_header)
273                        .map_err(|e| format!("write block header: {e}"))?;
274
275                    // SAFETY: offset + size <= total_size, buffer is valid.
276                    let data = unsafe {
277                        std::slice::from_raw_parts((buf_ptr + offset) as *const u8, *size)
278                    };
279                    tls.write_all(data)
280                        .map_err(|e| format!("write block data: {e}"))?;
281                }
282
283                // Done sentinel: offset=0, size=0
284                tls.write_all(&[0u8; 16])
285                    .map_err(|e| format!("write sentinel: {e}"))?;
286                // Flush all buffered TLS application data before closing.
287                tls.flush().map_err(|e| format!("flush: {e}"))?;
288                tls.conn.send_close_notify();
289                // Drain the close_notify record to TCP.
290                while tls.conn.wants_write() {
291                    tls.conn
292                        .write_tls(&mut tls.sock)
293                        .map_err(|e| format!("close_notify: {e}"))?;
294                }
295                // Shut down the write half of TCP to send FIN cleanly.
296                // ENOTCONN means the peer already closed — not an error.
297                match tls.sock.shutdown(std::net::Shutdown::Write) {
298                    Ok(()) => {}
299                    Err(e) if e.raw_os_error() == Some(libc::ENOTCONN) => {}
300                    Err(e) => return Err(format!("tcp shutdown: {e}")),
301                }
302
303                // Drain the receive buffer (e.g. TLS 1.3 NewSessionTicket
304                // messages the server sent after the handshake). If unread
305                // data remains when close() is called, the kernel sends
306                // TCP RST instead of FIN, which can truncate in-flight
307                // application data at the receiver.
308                tls.sock
309                    .set_read_timeout(Some(std::time::Duration::from_millis(200)))
310                    .ok();
311                let mut drain = [0u8; 4096];
312                loop {
313                    match tls.sock.read(&mut drain) {
314                        Ok(0) => break,
315                        Ok(_) => continue,
316                        Err(_) => break,
317                    }
318                }
319
320                Ok(())
321            }));
322        }
323
324        let mut errors = Vec::new();
325        for h in handles {
326            if let Err(e) = h.join().expect("sender thread panicked") {
327                errors.push(e);
328            }
329        }
330        if errors.is_empty() {
331            Ok(())
332        } else {
333            Err(PyRuntimeError::new_err(errors.join("; ")))
334        }
335    })
336}

338#[pymethods]
339impl PackedBuffer {
340    /// Send dirty blocks over parallel TLS connections.
341    ///
342    /// Each address in `addresses` gets its own TCP+TLS stream. Blocks
343    /// are distributed round-robin across streams. The wire protocol
344    /// matches the Python `SenderShardActor` exactly.
345    #[pyo3(signature = (dirty_blocks, addresses, cache_path, hash_block_size=None, tls_hostname=None, cert_path=None, ca_path=None))]
346    fn send_blocks(
347        &self,
348        py: Python<'_>,
349        dirty_blocks: Vec<usize>,
350        addresses: Vec<String>,
351        cache_path: String,
352        hash_block_size: Option<usize>,
353        tls_hostname: Option<String>,
354        cert_path: Option<String>,
355        ca_path: Option<String>,
356    ) -> PyResult<()> {
357        let block_size = hash_block_size.unwrap_or(HASH_BLOCK_SIZE);
358        let buf_ptr = self.ptr as usize;
359        let total_size = self.size;
360
361        py.detach(move || {
362            send_blocks_impl(
363                buf_ptr,
364                total_size,
365                &dirty_blocks,
366                &addresses,
367                &cache_path,
368                block_size,
369                tls_hostname.as_deref(),
370                cert_path.as_deref(),
371                ca_path.as_deref(),
372            )
373        })
374    }
375}

377/// Send dirty blocks from an existing buffer over parallel TLS connections.
378///
379/// Like `PackedBuffer.send_blocks()` but operates on any buffer (e.g. a
380/// memoryview from `pack_directory_chunked`), avoiding a second pack step.
381#[pyfunction]
382#[pyo3(signature = (buffer, total_size, dirty_blocks, addresses, cache_path, hash_block_size=None, tls_hostname=None, cert_path=None, ca_path=None))]
383fn send_blocks_from_buffer(
384    py: Python<'_>,
385    buffer: pyo3::buffer::PyBuffer<u8>,
386    total_size: usize,
387    dirty_blocks: Vec<usize>,
388    addresses: Vec<String>,
389    cache_path: String,
390    hash_block_size: Option<usize>,
391    tls_hostname: Option<String>,
392    cert_path: Option<String>,
393    ca_path: Option<String>,
394) -> PyResult<()> {
395    let block_size = hash_block_size.unwrap_or(HASH_BLOCK_SIZE);
396    let buf_ptr = buffer.buf_ptr() as usize;
397
398    py.detach(move || {
399        send_blocks_impl(
400            buf_ptr,
401            total_size,
402            &dirty_blocks,
403            &addresses,
404            &cache_path,
405            block_size,
406            tls_hostname.as_deref(),
407            cert_path.as_deref(),
408            ca_path.as_deref(),
409        )
410    })
411}

413/// Pack files into anonymous mmap and compute block hashes.
414///
415/// Returns a `PackedBuffer` whose `.hashes` attribute holds hex-encoded
416/// xxh64 digests for each 100 MB block.
417#[pyfunction]
418#[pyo3(signature = (file_list, total_size, hash_block_size=None))]
419fn pack_and_hash(
420    py: Python<'_>,
421    file_list: Vec<(String, usize, usize)>,
422    total_size: usize,
423    hash_block_size: Option<usize>,
424) -> PyResult<PackedBuffer> {
425    let block_size = hash_block_size.unwrap_or(HASH_BLOCK_SIZE);
426
427    if file_list.is_empty() || total_size == 0 {
428        return Ok(PackedBuffer {
429            ptr: std::ptr::null_mut(),
430            size: 0,
431            hashes: Vec::new(),
432        });
433    }
434
435    let files: Vec<FileInfo> = file_list
436        .into_iter()
437        .map(|(path, offset, size)| FileInfo { path, offset, size })
438        .collect();
439
440    let buffer =
441        mmap_anonymous(total_size).map_err(|e| PyOSError::new_err(format!("mmap failed: {e}")))?;
442
443    let buf_ptr = buffer as usize;
444
445    let hashes = py.detach(move || {
446        let nthreads = std::thread::available_parallelism()
447            .map(|n| n.get().min(16))
448            .unwrap_or(1);
449        pack_files_into(buf_ptr, &files, nthreads);
450        compute_block_hashes(buf_ptr, total_size, block_size, nthreads)
451    });
452
453    Ok(PackedBuffer {
454        ptr: buffer,
455        size: total_size,
456        hashes,
457    })
458}

460pub fn register_python_bindings(module: &Bound<'_, PyModule>) -> PyResult<()> {
461    module.add_class::<PackedBuffer>()?;
462    let f = wrap_pyfunction!(pack_and_hash, module)?;
463    f.setattr(
464        "__module__",
465        "monarch._rust_bindings.monarch_extension.tls_sender",
466    )?;
467    module.add_function(f)?;
468    let f2 = wrap_pyfunction!(send_blocks_from_buffer, module)?;
469    f2.setattr(
470        "__module__",
471        "monarch._rust_bindings.monarch_extension.tls_sender",
472    )?;
473    module.add_function(f2)?;
474    Ok(())
