9//! Local file-based trace reading.
10//!
11//! Reads `.pftrace` files from the local filesystem.
12//!
13//! ## Directory Layout
14//!
15//! ```text
16//! {trace_dir}/
17//! ├── executions/
18//! │   ├── {execution_id}/
19//! │   │   ├── {process_name}.pftrace
20//! │   │   └── ...
21//! │   └── latest -> {execution_id}/   # symlink to most recent
22//! ```

41/// Subdirectory name where traces are stored within a trace root.
42const MONARCH_TRACES_DIR: &str = "monarch_traces";

44/// Returns the default trace root directory: `/tmp/{username}/`.
45pub fn default_trace_root() -> PathBuf {
46    let username = std::env::var("USER").unwrap_or_else(|_| "unknown".to_string());
47    PathBuf::from(format!("/tmp/{}", username))
48}
49
50/// Returns the trace directory for a given root: `{root}/monarch_traces/`.
51pub fn trace_dir(root: &Path) -> PathBuf {
52    root.join(MONARCH_TRACES_DIR)
53}

55/// Returns the path to the execution directory.
56///
57/// If `execution_id` is `None`, follows the `latest` symlink.
58pub fn execution_dir(trace_dir: &Path, execution_id: Option<&str>) -> Result<PathBuf> {
59    let executions_dir = trace_dir.join("executions");
60
61    match execution_id {
62        Some(id) => {
63            let dir = executions_dir.join(id);
64            if !dir.exists() {
65                bail!(
66                    "Execution '{}' not found in {}",
67                    id,
68                    executions_dir.display()
69                );
70            }
71            Ok(dir)
72        }
73        None => {
74            // Follow the `latest` symlink
75            let latest_link = executions_dir.join("latest");
76            if !latest_link.exists() {
77                bail!(
78                    "No 'latest' symlink found in {}. Specify --execution-id or run a traced execution first.",
79                    executions_dir.display()
80                );
81            }
82            let target = fs::read_link(&latest_link).with_context(|| {
83                format!(
84                    "Failed to read 'latest' symlink at {}",
85                    latest_link.display()
86                )
87            })?;
88
89            // The symlink is relative to the executions directory
90            let dir = executions_dir.join(&target);
91            if !dir.exists() {
92                bail!(
93                    "'latest' symlink points to '{}' which does not exist",
94                    target.display()
95                );
96            }
97            Ok(dir)
98        }
99    }
100}

102/// Returns the execution ID from the `latest` symlink.
103pub fn get_latest_execution_id(trace_dir: &Path) -> Result<Option<String>> {
104    let latest_link = trace_dir.join("executions").join("latest");
105    if !latest_link.exists() {
106        return Ok(None);
107    }
108    Ok(fs::read_link(&latest_link)?.to_str().map(|s| s.to_string()))
109}

111/// Lists all available execution IDs in the trace directory.
112pub fn list_executions(trace_dir: &Path) -> Result<Vec<String>> {
113    let executions_dir = trace_dir.join("executions");
114    if !executions_dir.exists() {
115        return Ok(Vec::new());
116    }
117
118    let mut executions = Vec::new();
119    for entry in fs::read_dir(&executions_dir)? {
120        let entry = entry?;
121        let name = entry.file_name().to_string_lossy().to_string();
122        // Skip the `latest` symlink
123        if name == "latest" {
124            continue;
125        }
126        if entry.file_type()?.is_dir() {
127            executions.push(name);
128        }
129    }
130
131    // Sort by modification time (newest first)
132    executions.sort_by(|a, b| {
133        let a_time = fs::metadata(executions_dir.join(a))
134            .and_then(|m| m.modified())
135            .unwrap_or(std::time::SystemTime::UNIX_EPOCH);
136
137        let b_time = fs::metadata(executions_dir.join(b))
138            .and_then(|m| m.modified())
139            .unwrap_or(std::time::SystemTime::UNIX_EPOCH);
140
141        b_time.cmp(&a_time)
142    });
143
144    Ok(executions)
145}

147/// Finds all `.pftrace` files in an execution directory.
148fn find_pftrace_files(execution_dir: &Path) -> Result<Vec<PathBuf>> {
149    let mut files = Vec::new();
150
151    for entry in fs::read_dir(execution_dir)? {
152        let entry = entry?;
153        let path = entry.path();
154        if path.extension().is_some_and(|ext| ext == "pftrace") {
155            files.push(path);
156        }
157    }
158
159    if files.is_empty() {
160        bail!("No .pftrace files found in {}", execution_dir.display());
161    }
162
163    tracing::info!("Found {} .pftrace files", files.len());
164    Ok(files)
165}

167/// Reads and merges trace files from an execution directory.
168///
169/// Track UUIDs are offset to avoid collisions between different processes.
170/// Since process_name is globally unique within an execution, we just need
171/// a simple per-file offset that's large enough to not collide with the
172/// max UUID count within a single process.
173pub fn merge_traces_from_dir<S: Sink>(execution_dir: &Path, sink: &mut S) -> Result<()> {
174    for (file_idx, path) in find_pftrace_files(execution_dir)?.iter().enumerate() {
175        // We have to offset UUIDs here since within a single process,
176        // we generate these by just incrementing a counter.
177        let uuid_offset = (file_idx as u64 + 1) * 1_000_000;
178
179        read_and_offset_trace_file(path, uuid_offset, sink)?;
180    }
181
182    Ok(())
183}

185/// Reads a single .pftrace file and sends packets to the sink with offset UUIDs.
186///
187/// The file may contain multiple concatenated `Trace` messages (protobuf containers).
188fn read_and_offset_trace_file<S: Sink>(path: &Path, uuid_offset: u64, sink: &mut S) -> Result<()> {
189    let mut file = fs::File::open(path)?;
190    let mut buffer = Vec::new();
191    file.read_to_end(&mut buffer)?;
192
193    let mut offset = 0;
194    while offset < buffer.len() {
195        match Trace::decode(&buffer[offset..]) {
196            Ok(trace) => {
197                let encoded_len = trace.encoded_len();
198
199                for mut packet in trace.packet {
200                    offset_packet_uuids(&mut packet, uuid_offset);
201                    sink.consume(packet);
202                }
203
204                offset += encoded_len;
205            }
206            Err(e) => {
207                // If we can't decode, try skipping a byte and retrying
208                // This handles potential alignment issues or partial writes
209                tracing::debug!("Decode error at offset {}: {}", offset, e);
210                offset += 1;
211            }
212        }
213    }
214
215    Ok(())
216}
217
218/// Offsets all track UUIDs in a packet by the given amount.
219fn offset_packet_uuids(packet: &mut TracePacket, offset: u64) {
220    if let Some(ref mut data) = packet.data {
221        match data {
222            Data::TrackDescriptor(td) => {
223                if let Some(ref mut uuid) = td.uuid {
224                    *uuid += offset;
225                }
226                if let Some(ref mut parent) = td.parent_uuid {
227                    *parent += offset;
228                }
229            }
230            Data::TrackEvent(te) => {
231                if let Some(ref mut uuid) = te.track_uuid {
232                    *uuid += offset;
233                }
234            }
235            _ => {}
236        }
237    }
238}

240/// A [`Sink`] that writes trace packets to a file, flushing every 100 packets.
241pub struct Collector {
242    trace: Trace,
243    path: String,
244}
245
246impl Collector {
247    pub fn new(path: &Path) -> Self {
248        Self {
249            path: path.to_string_lossy().to_string(),
250            trace: Trace::default(),
251        }
252    }
253
254    pub fn flush(&mut self) -> Result<()> {
255        let tr = std::mem::take(&mut self.trace);
256        let mut file = fs::OpenOptions::new()
257            .append(true)
258            .create(true)
259            .open(&self.path)?;
260
261        let bytes = prost::Message::encode_to_vec(&tr);
262        file.write_all(&bytes)?;
263        file.flush()?;
264        Ok(())
265    }
266}
267
268impl Sink for Collector {
269    fn consume(&mut self, packet: TracePacket) {
270        self.trace.packet.push(packet);
271        if self.trace.packet.len() > 100 {
272            self.flush().expect("failed to flush trace packets");
273        }
274    }
275}

277/// Resolves trace/execution directories, merges all per-process `.pftrace`
278/// files, and writes the unified trace to `output`.
279///
280/// - `trace_root`: root directory containing `monarch_traces/`. If `None`,
281///   defaults to `/tmp/{USER}/`.
282/// - `execution_id`: specific execution to use. If `None`, follows the
283///   `latest` symlink.
284/// - `output`: path to write the merged trace file.
285///
286/// Returns the resolved execution ID.
287pub fn merge_to_file(
288    trace_root: Option<&Path>,
289    execution_id: Option<&str>,
290    output: &Path,
291) -> Result<String> {
292    let root = trace_root
293        .map(PathBuf::from)
294        .unwrap_or_else(default_trace_root);
295    let tdir = trace_dir(&root);
296    info!("Using trace directory: {}", tdir.display());
297
298    let exec_id = match execution_id {
299        Some(id) => id.to_string(),
300        None => match get_latest_execution_id(&tdir)? {
301            Some(id) => {
302                info!("Using latest execution: {}", id);
303                id
304            }
305            None => {
306                bail!(
307                    "No executions found in {}. Run a traced execution first.",
308                    tdir.display()
309                );
310            }
311        },
312    };
313
314    let edir = execution_dir(&tdir, Some(&exec_id))?;
315    info!("Reading from execution: {}", edir.display());
316
317    let mut sink = Collector::new(output);
318    merge_traces_from_dir(&edir, &mut sink)?;
319    sink.flush()?;
320
321    Ok(exec_id)
322}
