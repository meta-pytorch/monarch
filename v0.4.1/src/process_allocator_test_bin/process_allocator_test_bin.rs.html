9use hyperactor::channel::ChannelTransport;
10/// Test binary for ProcessAllocator child process cleanup behavior.
11/// This binary creates a ProcessAllocator and spawns several child processes,
12/// then keeps running until killed. It's designed to test whether child
13/// processes are properly cleaned up when the parent process is killed.
14use hyperactor_mesh::alloc::Alloc;
15use hyperactor_mesh::alloc::AllocConstraints;
16use hyperactor_mesh::alloc::AllocSpec;
17use hyperactor_mesh::alloc::Allocator;
18use hyperactor_mesh::alloc::ProcState;
19use hyperactor_mesh::alloc::ProcessAllocator;
20use ndslice::extent;
21use tokio::process::Command;
22
23fn emit_proc_state(state: &ProcState) {
24    if let Ok(json) = serde_json::to_string(state) {
25        println!("{}", json);
26        // Flush immediately to ensure parent can read events in real-time
27        use std::io::Write;
28        use std::io::{self};
29        io::stdout().flush().unwrap();
30    }
31}
32
33#[tokio::main]
34async fn main() -> Result<(), Box<dyn std::error::Error>> {
35    // Initialize tracing to stderr to avoid interfering with JSON output
36    tracing_subscriber::fmt()
37        .with_writer(std::io::stderr)
38        .init();
39
40    let bootstrap_path = buck_resources::get("monarch/hyperactor_mesh/bootstrap").unwrap();
41    eprintln!("Bootstrap cmd: {:?}", bootstrap_path);
42    let cmd = Command::new(&bootstrap_path);
43    let mut allocator = ProcessAllocator::new(cmd);
44
45    // Create an allocation with 4 child processes
46    let mut alloc = allocator
47        .allocate(AllocSpec {
48            extent: extent! { replica = 4 },
49            constraints: AllocConstraints::default(),
50            proc_name: None,
51            transport: ChannelTransport::Unix,
52            proc_allocation_mode: Default::default(),
53        })
54        .await?;
55
56    while let Some(state) = alloc.next().await {
57        emit_proc_state(&state);
58    }
59    Ok(())
60}
