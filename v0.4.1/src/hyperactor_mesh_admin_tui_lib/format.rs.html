19/// Derive a human-friendly label for a resolved node payload.
20///
21/// Kept as a free function (rather than an inherent `NodePayload`
22/// method) because `NodePayload` lives in
23/// `hyperactor_mesh::mesh_admin`; adding an extension trait here
24/// would be more ceremony than it's worth for a small formatting
25/// helper.
26///
27/// Uses `NodeProperties` to format a concise label for the tree view:
28/// roots show host counts, hosts show proc counts, procs show actor
29/// counts, and actors are rendered as `name[pid]` when the identity
30/// parses as an `ActorId`.
31pub(crate) fn derive_label(payload: &NodePayload) -> String {
32    match &payload.properties {
33        NodeProperties::Root { num_hosts, .. } => format!("Mesh Root ({} hosts)", num_hosts),
34        NodeProperties::Host {
35            addr, num_procs, ..
36        } => {
37            format!("{}  ({} procs)", addr, num_procs)
38        }
39        NodeProperties::Proc {
40            proc_name,
41            num_actors,
42            system_children,
43            stopped_children,
44            stopped_retention_cap,
45            is_poisoned,
46            failed_actor_count,
47            ..
48        } => {
49            let short = hyperactor_reference::ProcId::from_str(proc_name)
50                .map(|pid| pid.name().to_string())
51                .unwrap_or_else(|_| proc_name.clone());
52            let num_system = system_children.len();
53            let num_stopped = stopped_children.len();
54            let num_user = num_actors.saturating_sub(num_system);
55            let total = num_actors + num_stopped;
56            let mut parts = Vec::new();
57            if num_system > 0 {
58                parts.push(format!("{} system", num_system));
59            }
60            if num_user > 0 {
61                parts.push(format!("{} user", num_user));
62            }
63            if num_stopped > 0 {
64                if num_stopped >= *stopped_retention_cap && *stopped_retention_cap > 0 {
65                    parts.push(format!("{} stopped (max retained)", num_stopped));
66                } else {
67                    parts.push(format!("{} stopped", num_stopped));
68                }
69            }
70            let base = if parts.is_empty() {
71                format!("{}  ({} actors)", short, total)
72            } else {
73                format!("{}  ({} actors: {})", short, total, parts.join(", "))
74            };
75            if *is_poisoned {
76                format!("{}  [POISONED: {} failed]", base, failed_actor_count)
77            } else {
78                base
79            }
80        }
81        NodeProperties::Actor { .. } => match &payload.identity {
82            NodeRef::Actor(actor_id) => format!("{}[{}]", actor_id.name(), actor_id.pid()),
83            other => other.to_string(),
84        },
85        NodeProperties::Error { code, message } => {
86            format!("[error] {}: {}", code, message)
87        }
88    }
89}

91/// Derive a display label from a typed node reference without
92/// fetching.
93///
94/// For actor references, format as `name[pid]`; for all others, fall
95/// back to the `Display` representation.
96pub(crate) fn derive_label_from_ref(reference: &NodeRef) -> String {
97    match reference {
98        NodeRef::Actor(actor_id) => format!("{}[{}]", actor_id.name(), actor_id.pid()),
99        other => other.to_string(),
100    }
101}

103/// Produce a compact, human-readable summary string for a recorded
104/// event.
105///
106/// Prefers common message-like fields (`message`, then `msg`),
107/// otherwise renders a useful hint such as `handler: ...`. As a
108/// fallback, formats up to three key/value pairs from the event
109/// fields (using `format_value`) to keep the TUI line short; if
110/// nothing matches, falls back to the event `name`.
111pub(crate) fn format_event_summary(name: &str, fields: &Value) -> String {
112    if let Some(obj) = fields.as_object() {
113        if let Some(msg) = obj.get("message").and_then(|v| v.as_str()) {
114            return msg.to_string();
115        }
116        if let Some(msg) = obj.get("msg").and_then(|v| v.as_str()) {
117            return msg.to_string();
118        }
119        if let Some(handler) = obj.get("handler").and_then(|v| v.as_str()) {
120            return format!("handler: {}", handler);
121        }
122        if !obj.is_empty() {
123            let summary: Vec<String> = obj
124                .iter()
125                .take(3)
126                .map(|(k, v)| format!("{}={}", k, format_value(v)))
127                .collect();
128            if !summary.is_empty() {
129                return summary.join(" ");
130            }
131        }
132    }
133    name.to_string()
134}

136/// Format a JSON value into a short, single-line representation
137/// suitable for the TUI.
138///
139/// Strings/numbers/bools render as-is; `null` renders as `"null"`.
140/// Arrays and objects are summarized by their length/field count
141/// (e.g. `"[3]"`, `"{5}"`) to avoid dumping large payloads into the
142/// event list.
143pub(crate) fn format_value(v: &Value) -> String {
144    match v {
145        Value::String(s) => s.clone(),
146        Value::Number(n) => n.to_string(),
147        Value::Bool(b) => b.to_string(),
148        Value::Null => "null".to_string(),
149        Value::Array(arr) => format!("[{}]", arr.len()),
150        Value::Object(obj) => format!("{{{}}}", obj.len()),
151    }
152}

154/// Convert an ISO 8601 UTC timestamp (e.g.
155/// "2026-02-11T19:11:01.265Z") to a local-timezone HH:MM:SS string.
156/// Falls back to extracting the raw UTC time portion if parsing
157/// fails.
158pub(crate) fn format_local_time(timestamp: &str) -> String {
159    chrono::DateTime::parse_from_rfc3339(timestamp)
160        .map(|dt| {
161            dt.with_timezone(&chrono::Local)
162                .format("%H:%M:%S")
163                .to_string()
164        })
165        .unwrap_or_else(|_| timestamp.get(11..19).unwrap_or(timestamp).to_string())
166}

168/// Convert a `SystemTime` to a `chrono::DateTime<Utc>`.
169fn system_time_to_chrono(t: &SystemTime) -> chrono::DateTime<chrono::Utc> {
170    let dur = t.duration_since(std::time::UNIX_EPOCH).unwrap_or_default();
171    chrono::DateTime::from_timestamp(dur.as_secs() as i64, dur.subsec_nanos()).unwrap_or_default()
172}
173
174/// Format a `SystemTime` as a local-timezone HH:MM:SS string.
175pub(crate) fn format_system_time_local(t: &SystemTime) -> String {
176    system_time_to_chrono(t)
177        .with_timezone(&chrono::Local)
178        .format("%H:%M:%S")
179        .to_string()
180}

182/// Format a `SystemTime` as a human-readable relative time from now
183/// (e.g. "just now", "5s ago", "3m 12s ago").
184pub(crate) fn format_system_time_relative(t: &SystemTime) -> String {
185    match t.elapsed() {
186        Ok(d) if d.as_secs() < 2 => "just now".to_string(),
187        Ok(d) => format!("{} ago", humantime::format_duration(d)),
188        Err(_) => "just now".to_string(),
189    }
190}

192/// Format uptime from a `SystemTime` start point.
193///
194/// Rounds to nearest 30 seconds for cleaner display.
195pub(crate) fn format_system_time_uptime(started_at: &SystemTime) -> String {
196    match started_at.elapsed() {
197        Ok(d) => {
198            let total_secs = d.as_secs();
199            let rounded_secs = ((total_secs + 15) / 30) * 30;
200            humantime::format_duration(Duration::from_secs(rounded_secs)).to_string()
201        }
202        Err(_) => "unknown".to_string(),
203    }
204}
205
206/// Format a `SystemTime` as an ISO-8601 string for display.
207pub(crate) fn format_system_time_iso(t: &SystemTime) -> String {
208    system_time_to_chrono(t).to_rfc3339()
209}
