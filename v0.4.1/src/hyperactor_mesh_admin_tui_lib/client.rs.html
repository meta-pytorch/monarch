9//! TLS-aware `reqwest` client construction for the admin TUI.
10//!
11//! This module builds `(base_url, reqwest::Client)` from a
12//! [`TuiConfig`], choosing HTTP vs HTTPS and configuring certificate
13//! verification when TLS material is available.
14//!
15//! Admin handle resolution (`mast_conda:///` and bare `host:port`) is
16//! performed in the binary via `AdminHandle::parse` before calling
17//! this library. By the time [`build_client`] is called, `addr` is
18//! always a concrete `https://host:port` URL.
19//!
20//! # Address handling
21//!
22//! - `--addr` may be `host:port` (no scheme) or an explicit
23//!   `http://...` / `https://...`.
24//! - If a scheme is provided, it is treated as authoritative.
25//!
26//! # TLS configuration (highest priority first)
27//!
28//! 1. Explicit CLI paths: `--tls-ca` (required to enable TLS), with
29//!    optional `--tls-cert` + `--tls-key` for mutual TLS.
30//! 2. Auto-detection via [`hyperactor::channel::try_tls_pem_bundle`],
31//!    which probes configured paths (OSS) and Meta well-known
32//!    locations.
33//! 3. Fallback to plain HTTP when no usable CA is found.
34//!
35//! **Note:** At Meta (`fbcode_build`), the mesh admin server requires
36//! mutual TLS. If the client cannot load a CA certificate or fails to
37//! parse a client identity, the connection will be rejected at the TLS
38//! handshake. In OSS, the server falls back to plain HTTP when no
39//! certs are available, so the client's HTTP fallback still works.

43/// Read all bytes from a [`Pem`](hyperactor::config::Pem), returning
44/// `None` if it can't be opened/read or if the result is empty.
45fn read_pem(pem: &hyperactor::config::Pem) -> Option<Vec<u8>> {
46    use std::io::Read;
47    let mut buf = Vec::new();
48    pem.reader().ok()?.read_to_end(&mut buf).ok()?;
49    if buf.is_empty() { None } else { Some(buf) }
50}

52/// Split an address into an optional `http`/`https` scheme and the
53/// remaining host part.
54///
55/// If `addr` starts with `https://` or `http://`, returns
56/// `(Some(scheme), rest)`. Otherwise returns `(None, addr)`. The
57/// returned `&str` values are slices of `addr`.
58fn parse_addr(addr: &str) -> (Option<&str>, &str) {
59    if let Some(host) = addr.strip_prefix("https://") {
60        (Some("https"), host)
61    } else if let Some(host) = addr.strip_prefix("http://") {
62        (Some("http"), host)
63    } else {
64        (None, addr)
65    }
66}

68/// Configure TLS on a `reqwest::ClientBuilder` by adding a root CA,
69/// and optionally a client identity (cert + key) for mutual TLS.
70///
71/// Delegates to [`hyperactor_mesh::mesh_admin_client::add_tls`] which
72/// handles the fbcode/OSS native-tls vs rustls identity split.
73fn add_tls(
74    builder: reqwest::ClientBuilder,
75    ca_bytes: &[u8],
76    cert_bytes: Option<Vec<u8>>,
77    key_bytes: Option<Vec<u8>>,
78) -> (reqwest::ClientBuilder, bool) {
79    hyperactor_mesh::mesh_admin_client::add_tls(builder, ca_bytes, cert_bytes, key_bytes)
80}

82/// Configure TLS on a `reqwest::ClientBuilder` using PEM files
83/// supplied via CLI (`--tls-ca`, and optionally `--tls-cert` +
84/// `--tls-key`).
85///
86/// Reads `ca_path` and installs it as the root trust anchor. If the
87/// CA file cannot be read, returns `(builder, false)` with no
88/// changes.
89///
90/// If `cert_path`/`key_path` are provided, attempts to read them and
91/// pass the bytes through to [`add_tls`] to configure an mTLS
92/// identity; failures to read these optional files simply omit the
93/// identity (the CA may still be applied).
94///
95/// Returns `(updated_builder, ca_installed)`.
96fn add_tls_from_paths(
97    builder: reqwest::ClientBuilder,
98    ca_path: &str,
99    cert_path: Option<&str>,
100    key_path: Option<&str>,
101) -> (reqwest::ClientBuilder, bool) {
102    let ca_bytes = match std::fs::read(ca_path) {
103        Ok(b) => b,
104        Err(e) => {
105            eprintln!("TLS: cannot read CA file {}: {}", ca_path, e);
106            return (builder, false);
107        }
108    };
109    let cert_bytes = cert_path.and_then(|p| std::fs::read(p).ok());
110    let key_bytes = key_path.and_then(|p| std::fs::read(p).ok());
111    add_tls(builder, &ca_bytes, cert_bytes, key_bytes)
112}

114/// Configure TLS on a `reqwest::ClientBuilder` using a hyperactor
115/// [`PemBundle`](hyperactor::config::PemBundle).
116///
117/// The bundle provides a CA (`bundle.ca`) and optionally a client
118/// certificate and key (`bundle.cert`, `bundle.key`). This reads the
119/// CA via [`read_pem`] and installs it as a root trust anchor. If the
120/// CA is missing, unreadable, or empty, returns `(builder, false)`
121/// with no changes.
122///
123/// If both client cert and key are readable, they are combined and
124/// passed to [`add_tls`] to configure an mTLS identity; otherwise the
125/// client identity is omitted (CA-only TLS verification still
126/// applies).
127///
128/// Returns `(updated_builder, ca_installed)`.
129fn add_tls_from_bundle(
130    builder: reqwest::ClientBuilder,
131    bundle: &hyperactor::config::PemBundle,
132) -> (reqwest::ClientBuilder, bool) {
133    let ca_bytes = match read_pem(&bundle.ca) {
134        Some(b) => b,
135        None => {
136            eprintln!("TLS: CA not readable from PemBundle");
137            return (builder, false);
138        }
139    };
140    add_tls(
141        builder,
142        &ca_bytes,
143        read_pem(&bundle.cert),
144        read_pem(&bundle.key),
145    )
146}

148/// Build a `reqwest` client and a base URL from CLI arguments.
149///
150/// `args.addr` may be either a bare `host:port` or an explicit URL
151/// (`http://host:port` / `https://host:port`). If an explicit scheme
152/// is provided, that scheme is honored.
153///
154/// TLS configuration is applied in priority order:
155/// 1. If `--tls-ca` is provided, attempt to load the CA (and
156///    optionally `--tls-cert` + `--tls-key` for a client identity)
157///    from those paths.
158/// 2. Otherwise, if no `--tls-ca` was given, try auto-detection via
159///    [`hyperactor::channel::try_tls_pem_bundle`] (OSS config first,
160///    then Meta well-known paths). This runs even when the user
161///    provides an explicit `https://` scheme, so the mTLS client
162///    identity is picked up from well-known paths.
163/// 3. If no CA can be loaded, fall back to plain HTTP.
164///
165/// Returns `(base_url, client)` where `base_url` always includes the
166/// scheme selected (`http://...` or `https://...`).
167pub(crate) fn build_client(config: &TuiConfig) -> (String, reqwest::Client) {
168    let (explicit_scheme, host) = parse_addr(&config.addr);
169
170    let client_timeout =
171        hyperactor_config::global::get(hyperactor_mesh::config::MESH_ADMIN_PYSPY_CLIENT_TIMEOUT);
172    let mut builder = reqwest::Client::builder().timeout(client_timeout);
173    let mut use_tls = explicit_scheme == Some("https");
174
175    // 1. Explicit CLI cert paths.
176    if let Some(ca_path) = &config.tls_ca {
177        let (b, ok) = add_tls_from_paths(
178            builder,
179            ca_path,
180            config.tls_cert.as_deref(),
181            config.tls_key.as_deref(),
182        );
183        builder = b;
184        use_tls = use_tls || ok;
185    }
186
187    // 2. Auto-detect (when no CLI certs were provided).
188    // This runs even with an explicit https:// scheme, so the client
189    // picks up the mTLS identity from Meta well-known paths.
190    if config.tls_ca.is_none() {
191        if let Some(bundle) = hyperactor::channel::try_tls_pem_bundle() {
192            let (b, ok) = add_tls_from_bundle(builder, &bundle);
193            builder = b;
194            use_tls = use_tls || ok;
195        }
196    }
197
198    let scheme = if use_tls { "https" } else { "http" };
199    let base_url = format!("{}://{}", scheme, host);
200    let client = builder.build().unwrap_or_else(|_| reqwest::Client::new());
201
202    (base_url, client)
203}
