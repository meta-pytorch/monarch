16/// Selectable color theme.
17#[derive(Debug, Clone, Copy, Default, ValueEnum)]
18pub enum ThemeName {
19    /// Nord — an arctic, north-bluish color palette.
20    #[default]
21    Nord,
22    /// doom-nord-light — desaturated Nord accents for light backgrounds.
23    DoomNordLight,
24}
25
26impl std::fmt::Display for ThemeName {
27    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
28        match self {
29            ThemeName::Nord => write!(f, "nord"),
30            ThemeName::DoomNordLight => write!(f, "doom-nord-light"),
31        }
32    }
33}

309/// Color scheme for the TUI.
310///
311/// Defines all colors and styles used throughout the interface.
312/// Selectable via `--theme` (see [`ThemeName`]).
313///
314/// ## Semantic Roles
315///
316/// Each field maps to a semantic role, not a specific color. Themes
317/// assign concrete colors to these roles while preserving the
318/// following intent:
319///
320/// - **Topology**: host, proc, actor node types in the tree
321/// - **Selection/focus**: currently highlighted node
322/// - **System state**: config toggles, system/infrastructure actors
323/// - **Timing/temporal**: refresh intervals, durations
324/// - **Secondary**: URLs, labels, borders, muted text
325/// - **Error/warning/success**: status indicators
326/// - **Headers**: app name, info headings
327///
328/// ## Semantic Categories
329///
330/// - **UI chrome**: App name, borders, footer
331/// - **Node types**: Distinct style per tree node kind
332/// - **States**: Error, warning, success, stopped
333/// - **Header stats**: Timing, topology counts, selection, config
334/// - **Detail pane**: Labels, status indicators
335#[derive(Clone, Copy)]
336pub(crate) struct ColorScheme {
337    // UI chrome
338    pub(crate) app_name: Style,
339    pub(crate) border: Style,
340    pub(crate) _border_focused: Style,
341    pub(crate) _footer_help: Style,
342
343    // Node types (tree rendering) — one style per node kind.
344    // Specific colors vary by theme; see default() and nord().
345    pub(crate) node_root: Style,
346    pub(crate) node_host: Style,
347    pub(crate) node_proc: Style,
348    pub(crate) node_actor: Style,
349    pub(crate) node_failed: Style,
350    pub(crate) node_system_actor: Style,
351    pub(crate) node_user_actor: Style,
352
353    // Semantic states
354    pub(crate) error: Style,
355    pub(crate) info: Style,
356
357    // Header stat categories
358    pub(crate) stat_timing: Style, // refresh intervals, durations (temporal)
359    pub(crate) stat_selection: Style, // current selection (focus)
360    pub(crate) stat_system: Style, // config toggles (state)
361    pub(crate) stat_url: Style,    // connection info (secondary)
362    pub(crate) stat_label: Style,  // stat labels/prefixes
363    pub(crate) _stat_value: Style, // stat numeric values
364
365    // Detail pane and misc rendering
366    pub(crate) detail_label: Style, // label text in detail key-value lines
367    pub(crate) detail_stopped: Style, // stopped/failed nodes in tree + detail
368    pub(crate) detail_status_ok: Style, // actor status "Running"
369    pub(crate) detail_status_warn: Style, // actor status non-Running (idle, etc.)
370    pub(crate) detail_status_failed: Style, // actor status "failed:*"
371    pub(crate) footer_help: Style,  // footer help bar text
372    pub(crate) header_class_bracket: Style, // classification brackets in header
373}

502    /// Return the style for a given node type.
503    pub(crate) fn node_style(&self, node_type: NodeType) -> Style {
504        match node_type {
505            NodeType::Root => self.node_root,
506            NodeType::Host => self.node_host,
507            NodeType::Proc => self.node_proc,
508            NodeType::Actor => self.node_actor,
509        }

521impl Theme {

523    pub(crate) fn new(theme_name: ThemeName, lang_name: LangName) -> Self {
524        let scheme = match theme_name {

526            ThemeName::DoomNordLight => ColorScheme::doom_nord_light(),
