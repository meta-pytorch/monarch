9//! OSS binary for merging per-process `.pftrace` files into a single unified trace.
10
11use std::path::PathBuf;
12
13use anyhow::Result;
14use clap::Parser;
15use monarch_perfetto_trace::local;
16use tracing::Level;
17use tracing::info;
18use tracing_subscriber::FmtSubscriber;
19
20/// Merge per-process .pftrace files into a single unified trace.
21#[derive(Parser, Debug)]
22#[clap()]
23struct Cli {
24    /// Execution ID to use. Uses the latest execution if not specified.
25    #[clap(short, long)]
26    execution_id: Option<String>,
27
28    /// Root directory containing monarch_traces/. Defaults to /tmp/{user}/.
29    #[clap(short, long)]
30    trace_dir: Option<PathBuf>,
31
32    /// Output file path. Defaults to ./{execution_id}.pftrace.
33    #[clap(short, long)]
34    output: Option<PathBuf>,
35
36    /// Verbose output.
37    #[clap(short, long)]
38    verbose: bool,
39}
40
41fn main() -> Result<()> {
42    let cli = Cli::parse();
43
44    let level = if cli.verbose {
45        Level::DEBUG
46    } else {
47        Level::INFO
48    };
49    let subscriber = FmtSubscriber::builder().with_max_level(level).finish();
50    tracing::subscriber::set_global_default(subscriber).expect("Failed to set tracing subscriber");
51
52    let output = match cli.output {
53        Some(path) => {
54            local::merge_to_file(cli.trace_dir.as_deref(), cli.execution_id.as_deref(), &path)?;
55            path
56        }
57        None => {
58            // Write to a temp file first, then rename once we know the execution ID.
59            let tmp = PathBuf::from(".expanse_merge.pftrace.tmp");
60            let exec_id =
61                local::merge_to_file(cli.trace_dir.as_deref(), cli.execution_id.as_deref(), &tmp)?;
62            let path = PathBuf::from(format!("{}.pftrace", exec_id));
63            std::fs::rename(&tmp, &path)?;
64            path
65        }
66    };
67
68    info!("Merged trace written to {}", output.display());
69    println!("{}", output.display());
70
71    Ok(())
72}
