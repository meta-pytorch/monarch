9/// This is an "empty shell" bootstrap process,
10/// simply invoking [`hyperactor_mesh::bootstrap_or_die`].
11#[tokio::main]
12async fn main() {
13    // This causes folly to intercept SIGTERM. When run in
14    // '@fbcode//mode/dev-nosan' that translates into SEGFAULTs.
15    hyperactor::initialize_with_current_runtime();
16    // SAFETY: Does not derefrence pointers or rely on undefined
17    // memory. No other threads are likely to be modifying it
18    // concurrently.
19    unsafe {
20        libc::signal(libc::SIGTERM, libc::SIG_DFL);
21    }
22
23    hyperactor_mesh::bootstrap_or_die().await;
24}
