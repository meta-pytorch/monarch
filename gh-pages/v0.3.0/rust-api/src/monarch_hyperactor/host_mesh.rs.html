55pub struct PyBootstrapCommand {

57    pub program: String,
58    #[pyo3(get, set)]
59    pub arg0: Option<String>,
60    #[pyo3(get, set)]
61    pub args: Vec<String>,
62    #[pyo3(get, set)]
63    pub env: HashMap<String, String>,
64}
65
66#[pymethods]
67impl PyBootstrapCommand {
68    #[new]
69    fn new(
70        program: String,
71        arg0: Option<String>,
72        args: Vec<String>,
73        env: HashMap<String, String>,
74    ) -> Self {
75        Self {
76            program,
77            arg0,
78            args,
79            env,
80        }
81    }
82
83    fn __repr__(&self) -> String {
84        format!(
85            "BootstrapCommand(program='{}', args={:?}, env={:?})",
86            self.program, self.args, self.env
87        )
88    }
89}

91impl PyBootstrapCommand {
92    pub fn to_rust(&self) -> BootstrapCommand {
93        BootstrapCommand {
94            program: PathBuf::from(&self.program),
95            arg0: self.arg0.clone(),
96            args: self.args.clone(),
97            env: self.env.clone(),
98        }
99    }
100
101    pub fn default<'py>(py: Python<'py>) -> PyResult<Bound<'py, Self>> {
102        py.import("monarch._src.actor.host_mesh")?
103            .getattr("_bootstrap_cmd")?
104            .call0()?
105            .downcast::<PyBootstrapCommand>()
106            .cloned()
107            .map_err(to_py_error)
108    }
109}

111#[pyclass(
112    name = "HostMesh",
113    module = "monarch._rust_bindings.monarch_hyperactor.host_mesh"
114)]
115pub(crate) enum PyHostMesh {
116    Owned(PyHostMeshImpl),
117    Ref(PyHostMeshRefImpl),
118}
119
120impl PyHostMesh {
121    pub(crate) fn new_owned(inner: HostMesh) -> Self {
122        Self::Owned(PyHostMeshImpl(SharedCell::from(inner)))
123    }
124
125    pub(crate) fn new_ref(inner: HostMeshRef) -> Self {
126        Self::Ref(PyHostMeshRefImpl(inner))
127    }
128
129    fn mesh_ref(&self) -> Result<HostMeshRef, anyhow::Error> {
130        match self {
131            PyHostMesh::Owned(inner) => Ok(inner.0.borrow()?.clone()),
132            PyHostMesh::Ref(inner) => Ok(inner.0.clone()),
133        }
134    }
135}
136
137#[pymethods]
138impl PyHostMesh {
139    #[classmethod]
140    fn allocate_nonblocking(
141        _cls: &Bound<'_, PyType>,
142        instance: &PyInstance,
143        alloc: &mut PyAlloc,
144        name: String,
145        bootstrap_params: Option<PyBootstrapCommand>,
146    ) -> PyResult<PyPythonTask> {
147        let bootstrap_params =
148            bootstrap_params.map_or_else(|| alloc.bootstrap_command.clone(), |b| Some(b.to_rust()));
149        let alloc = match alloc.take() {
150            Some(alloc) => alloc,
151            None => {
152                return Err(PyException::new_err(
153                    "Alloc object already used".to_string(),
154                ));
155            }
156        };
157        let instance = instance.clone();
158        PyPythonTask::new(async move {
159            let mesh = HostMesh::allocate(instance.deref(), alloc, &name, bootstrap_params)
160                .await
161                .map_err(|err| PyException::new_err(err.to_string()))?;
162            Ok(Self::new_owned(mesh))
163        })
164    }
165
166    fn spawn_nonblocking(
167        &self,
168        instance: &PyInstance,
169        name: String,
170        per_host: &PyExtent,
171    ) -> PyResult<PyPythonTask> {
172        let host_mesh = self.mesh_ref()?.clone();
173        let instance = instance.clone();
174        let per_host = per_host.clone().into();
175        let mesh_impl = async move {
176            let proc_mesh = host_mesh
177                .spawn(instance.deref(), &name, per_host)
178                .await
179                .map_err(to_py_error)?;
180            Ok(PyProcMesh::new_owned(proc_mesh))
181        };
182        PyPythonTask::new(mesh_impl)
183    }
184
185    fn sliced(&self, region: &PyRegion) -> PyResult<Self> {
186        Ok(Self::new_ref(
187            self.mesh_ref()?.sliced(region.as_inner().clone()),
188        ))
189    }
190
191    #[getter]
192    fn region(&self) -> PyResult<PyRegion> {
193        Ok(PyRegion::from(self.mesh_ref()?.region()))
194    }
195
196    fn __reduce__<'py>(&self, py: Python<'py>) -> PyResult<(Bound<'py, PyAny>, Bound<'py, PyAny>)> {
197        let bytes = bincode::serialize(&self.mesh_ref()?)
198            .map_err(|e| PyErr::new::<PyValueError, _>(e.to_string()))?;
199        let py_bytes = (PyBytes::new(py, &bytes),).into_bound_py_any(py).unwrap();
200        let from_bytes =
201            PyModule::import(py, "monarch._rust_bindings.monarch_hyperactor.host_mesh")?
202                .getattr("py_host_mesh_from_bytes")?;
203        Ok((from_bytes, py_bytes))
204    }
205
206    fn __eq__(&self, other: &PyHostMesh) -> PyResult<bool> {
207        Ok(self.mesh_ref()? == other.mesh_ref()?)
208    }
209
210    fn shutdown(&self, instance: &PyInstance) -> PyResult<PyPythonTask> {
211        match self {
212            PyHostMesh::Owned(inner) => {
213                let instance = instance.clone();
214                let mesh_borrow = inner.0.clone();
215                let fut = async move {
216                    match mesh_borrow.take().await {
217                        Ok(mut mesh) => {
218                            mesh.shutdown(instance.deref()).await?;
219                            Ok(())
220                        }
221                        Err(_) => {
222                            // Don't return an exception, silently ignore the stop request
223                            // because it was already done.
224                            tracing::info!("shutdown was already called on host mesh");
225                            Ok(())
226                        }
227                    }
228                };
229                PyPythonTask::new(fut)
230            }
231            PyHostMesh::Ref(_) => Err(PyRuntimeError::new_err(
232                "cannot shut down `HostMesh` that is a reference instead of owned",
233            )),
234        }
235    }
236}

258/// Static storage for the root client instance when using host-based bootstrap.
259static ROOT_CLIENT_INSTANCE_FOR_HOST: OnceLock<Instance<PythonActor>> = OnceLock::new();
260
261/// Static storage for the host mesh agent created by bootstrap_host().
262static HOST_MESH_AGENT_FOR_HOST: OnceLock<ActorHandle<HostMeshAgent>> = OnceLock::new();
263
264/// Bootstrap the client host and root client actor.
265///
266/// This creates a proper Host with BootstrapProcManager, spawns the root client
267/// actor on the Host's local_proc.
268///
269/// Returns a tuple of (HostMesh, ProcMesh, PyInstance) where:
270/// - PyHostMesh: the bootstrapped (local) host mesh; and
271/// - PyProcMesh: the local ProcMesh on this HostMesh; and
272/// - PyInstance: the root client actor instance, on the ProcMesh.
273///
274/// The HostMesh is served on the default transport.
275///
276/// This should be called only once, at process initialization
277#[pyfunction]
278fn bootstrap_host(bootstrap_cmd: Option<PyBootstrapCommand>) -> PyResult<PyPythonTask> {
279    let bootstrap_cmd = match bootstrap_cmd {
280        Some(cmd) => cmd.to_rust(),
281        None => BootstrapCommand::current().map_err(|e| PyException::new_err(e.to_string()))?,
282    };
283
284    PyPythonTask::new(async move {
285        let host_mesh_agent = host(
286            default_bind_spec().binding_addr(),
287            Some(bootstrap_cmd),
288            None,
289        )
290        .await
291        .map_err(|e| PyException::new_err(e.to_string()))?;
292
293        // Store the agent for later shutdown
294        HOST_MESH_AGENT_FOR_HOST.set(host_mesh_agent.clone()).ok(); // Ignore error if already set
295
296        let host_mesh_name = hyperactor_mesh::v1::Name::new_reserved("local").unwrap();
297        let host_mesh = HostMeshRef::from_host_agent(host_mesh_name, host_mesh_agent.bind())
298            .map_err(|e| PyException::new_err(e.to_string()))?;
299
300        // We require a temporary instance to make a call to the host/proc agent.
301        let temp_proc = Proc::local();
302        let (temp_instance, _) = temp_proc
303            .instance("temp")
304            .map_err(|e| PyException::new_err(e.to_string()))?;
305
306        let local_proc_agent: hyperactor::ActorHandle<
307            hyperactor_mesh::proc_mesh::mesh_agent::ProcMeshAgent,
308        > = host_mesh_agent
309            .get_local_proc(&temp_instance)
310            .await
311            .map_err(|e| PyException::new_err(e.to_string()))?;
312
313        let proc_mesh_name = hyperactor_mesh::v1::Name::new_reserved("local").unwrap();
314        let proc_mesh = ProcMeshRef::new_singleton(
315            proc_mesh_name,
316            ProcRef::new(
317                local_proc_agent.actor_id().proc_id().clone(),
318                0,
319                local_proc_agent.bind(),
320            ),
321        );
322
323        let local_proc = local_proc_agent
324            .get_proc(&temp_instance)
325            .await
326            .map_err(|e| PyException::new_err(e.to_string()))?;
327
328        let (instance, _handle) = monarch_with_gil(|py| {
329            PythonActor::bootstrap_client_inner(py, local_proc, &ROOT_CLIENT_INSTANCE_FOR_HOST)
330        })
331        .await;
332
333        Ok((
334            PyHostMesh::new_ref(host_mesh),
335            PyProcMesh::new_ref(proc_mesh),
336            PyInstance::from(instance),
337        ))
338    })
339}

341#[pyfunction]
342fn py_host_mesh_from_bytes(bytes: &Bound<'_, PyBytes>) -> PyResult<PyHostMesh> {
343    let r: PyResult<HostMeshRef> = bincode::deserialize(bytes.as_bytes())
344        .map_err(|e| PyErr::new::<PyValueError, _>(e.to_string()));
345    r.map(PyHostMesh::new_ref)
346}

348#[pyfunction]
349fn shutdown_local_host_mesh() -> PyResult<PyPythonTask> {
350    let agent = HOST_MESH_AGENT_FOR_HOST
351        .get()
352        .ok_or_else(|| PyException::new_err("No local host mesh to shutdown"))?
353        .clone();
354
355    PyPythonTask::new(async move {
356        // Create a temporary instance to send the shutdown message
357        let temp_proc = hyperactor::Proc::local();
358        let (instance, _) = temp_proc
359            .instance("shutdown_requester")
360            .map_err(|e| PyException::new_err(e.to_string()))?;
361
362        // Use same defaults as HostMesh::shutdown():
363        // - MESH_TERMINATE_TIMEOUT = 10 seconds
364        // - MESH_TERMINATE_CONCURRENCY = 16
365        agent
366            .shutdown_host(&instance, Duration::from_secs(10), 16)
367            .await
368            .map_err(|e| PyException::new_err(e.to_string()))?;
369
370        Ok(())
371    })
372}

374pub fn register_python_bindings(hyperactor_mod: &Bound<'_, PyModule>) -> PyResult<()> {
375    let f = wrap_pyfunction!(py_host_mesh_from_bytes, hyperactor_mod)?;
376    f.setattr(
377        "__module__",
378        "monarch._rust_bindings.monarch_hyperactor.host_mesh",
379    )?;
380    hyperactor_mod.add_function(f)?;
381
382    let f2 = wrap_pyfunction!(bootstrap_host, hyperactor_mod)?;
383    f2.setattr(
384        "__module__",
385        "monarch._rust_bindings.monarch_hyperactor.host_mesh",
386    )?;
387    hyperactor_mod.add_function(f2)?;
388
389    let f3 = wrap_pyfunction!(shutdown_local_host_mesh, hyperactor_mod)?;
390    f3.setattr(
391        "__module__",
392        "monarch._rust_bindings.monarch_hyperactor.host_mesh",
393    )?;
394    hyperactor_mod.add_function(f3)?;
395
396    hyperactor_mod.add_class::<PyHostMesh>()?;
397    hyperactor_mod.add_class::<PyBootstrapCommand>()?;
398    Ok(())
