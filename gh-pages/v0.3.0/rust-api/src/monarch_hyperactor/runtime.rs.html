34// this must be a RwLock and only return a guard for reading the runtime.
35// Otherwise multiple threads can deadlock fighting for the Runtime object if they hold it
36// while blocking on something.
37static INSTANCE: std::sync::LazyLock<RwLock<Option<tokio::runtime::Runtime>>> =
38    std::sync::LazyLock::new(|| RwLock::new(None));
39
40pub fn get_tokio_runtime<'l>() -> std::sync::MappedRwLockReadGuard<'l, tokio::runtime::Runtime> {
41    // First try to get a read lock and check if runtime exists
42    {
43        let read_guard = INSTANCE.read().unwrap();
44        if read_guard.is_some() {
45            return RwLockReadGuard::map(read_guard, |lock: &Option<tokio::runtime::Runtime>| {
46                lock.as_ref().unwrap()
47            });
48        }
49        // Drop the read lock by letting it go out of scope
50    }
51
52    // Runtime doesn't exist, upgrade to write lock to initialize
53    let mut write_guard = INSTANCE.write().unwrap();
54    if write_guard.is_none() {
55        *write_guard = Some(
56            tokio::runtime::Builder::new_multi_thread()
57                .thread_name_fn(|| {
58                    static ATOMIC_ID: AtomicUsize = AtomicUsize::new(0);
59                    let id = ATOMIC_ID.fetch_add(1, Ordering::SeqCst);
60                    format!("monarch-pytokio-worker-{}", id)
61                })
62                .enable_all()
63                .build()
64                .unwrap(),
65        );
66    }
67
68    // Downgrade write lock to read lock and return the reference
69    let read_guard = std::sync::RwLockWriteGuard::downgrade(write_guard);
70    RwLockReadGuard::map(read_guard, |lock: &Option<tokio::runtime::Runtime>| {
71        lock.as_ref().unwrap()
72    })
73}

75#[pyfunction]
76pub fn shutdown_tokio_runtime(py: Python<'_>) {
77    // Called from Python's atexit, which holds the GIL. Release it so tokio
78    // worker threads can acquire it to complete their Python work.
79    py.allow_threads(|| {
80        if let Some(x) = INSTANCE.write().unwrap().take() {
81            x.shutdown_timeout(Duration::from_secs(1));
82        }
83    });
84}

95/// Stores the native thread ID of the main Python thread.
96/// This is lazily initialized on first call to `is_main_thread`.
97static MAIN_THREAD_NATIVE_ID: OnceLock<i64> = OnceLock::new();
98
99/// Returns the native thread ID of the main Python thread.
100/// On first call, looks it up via `threading.main_thread().native_id`.
101fn get_main_thread_native_id() -> i64 {
102    *MAIN_THREAD_NATIVE_ID.get_or_init(|| {
103        Python::with_gil(|py| {
104            let threading = py.import("threading").expect("failed to import threading");
105            let main_thread = threading
106                .call_method0("main_thread")
107                .expect("failed to get main_thread");
108            main_thread
109                .getattr("native_id")
110                .expect("failed to get native_id")
111                .extract::<i64>()
112                .expect("native_id is not an i64")
113        })
114    })
115}

117/// Returns the current thread's native ID in a cross-platform way.
118#[cfg(target_os = "linux")]
119fn get_current_thread_id() -> i64 {
120    nix::unistd::gettid().as_raw() as i64
121}

123/// Returns the current thread's native ID in a cross-platform way.
124#[cfg(target_os = "macos")]
125fn get_current_thread_id() -> i64 {
126    let mut tid: u64 = 0;
127    // pthread_threadid_np with thread=0 (null pthread_t) gets the current thread's ID.
128    unsafe {
129        let ret = libc::pthread_threadid_np(0, &mut tid);
130        debug_assert_eq!(
131            ret, 0,
132            "pthread_threadid_np failed with error code: {}",
133            ret
134        );
135    }
136    // macOS thread IDs are u64 so we need to convert to i64.
137    debug_assert!(tid <= i64::MAX as u64, "thread ID {} exceeds i64::MAX", tid);
138    tid as i64
139}

141/// Returns the current thread's native ID in a cross-platform way.
142#[cfg(not(any(target_os = "linux", target_os = "macos")))]
143compile_error!("get_current_thread_id is only implemented for Linux and macOS");

145/// Returns true if the current thread is the main Python thread.
146/// Compares the current thread's native ID against the main Python thread's native ID.
147pub fn is_main_thread() -> bool {
148    let current_tid = get_current_thread_id();
149    current_tid == get_main_thread_native_id()
150}

152pub fn initialize(py: Python) -> Result<()> {
153    let atexit = py.import("atexit")?;
154    let shutdown_fn = wrap_pyfunction!(shutdown_tokio_runtime, py)?;
155    atexit.call_method1("register", (shutdown_fn,))?;
156    Ok(())
157}

159/// Block the current thread on a future, but make sure to check for signals
160/// originating from the Python signal handler.
161///
162/// Python's signal handler just sets a flag that it expects the Python
163/// interpreter to handle later via a call to `PyErr_CheckSignals`. When we
164/// enter into potentially long-running native code, we need to make sure to be
165/// checking for signals frequently, otherwise we will ignore them. This will
166/// manifest as `ctrl-C` not doing anything.
167///
168/// One additional wrinkle is that `PyErr_CheckSignals` only works on the main
169/// Python thread; if it's called on any other thread it silently does nothing.
170/// So, we check if we're on the main thread by comparing native thread IDs.
171pub fn signal_safe_block_on<F>(py: Python, future: F) -> PyResult<F::Output>
172where
173    F: Future + Send + 'static,
174    F::Output: Send + 'static,
175{
176    let runtime = get_tokio_runtime();
177    // Release the GIL, otherwise the work in `future` that tries to acquire the
178    // GIL on another thread may deadlock.
179    Python::allow_threads(py, || {
180        if is_main_thread() {
181            // Spawn the future onto the tokio runtime
182            let handle = runtime.spawn(future);
183            // Block the current thread on waiting for *either* the future to
184            // complete or a signal.
185            runtime.block_on(async {
186                tokio::select! {
187                    result = handle => result.map_err(|e| PyRuntimeError::new_err(format!("JoinErr: {:?}", e))),
188                    signal = async {
189                        let sleep_for = std::time::Duration::from_millis(100);
190                        loop {
191                            // Acquiring the GIL in a loop is sad, hopefully once
192                            // every 100ms is fine.
193                            Python::with_gil(|py| py.check_signals())?;
194                            #[allow(clippy::disallowed_methods)]
195                            tokio::time::sleep(sleep_for).await;
196                        }
197                    } => signal
198                }
199            })
200        } else {
201            // If we're not on the main thread, we can just block it. We've
202            // released the GIL, so the Python main thread will continue on, and
203            // `PyErr_CheckSignals` doesn't do anything anyway.
204            Ok(runtime.block_on(future))
205        }
206    })
207}

209/// A test function that sleeps indefinitely in a loop.
210/// This is used for testing signal handling in signal_safe_block_on.
211/// The function will sleep forever until interrupted by a signal.
212#[pyfunction]
213pub fn sleep_indefinitely_for_unit_tests(py: Python) -> PyResult<()> {
214    // Create a future that sleeps indefinitely
215    let future = async {
216        loop {
217            tracing::info!("idef sleeping for 100ms");
218            #[allow(clippy::disallowed_methods)]
219            tokio::time::sleep(Duration::from_millis(100)).await;
220        }
221    };
222
223    // Use signal_safe_block_on to run the future, which should make it
224    // interruptible by signals like SIGINT
225    signal_safe_block_on(py, future)
226}

289/// Global lock to serialize GIL acquisition from Rust threads in async contexts.
290///
291/// Under high concurrency, many async tasks can simultaneously try to acquire the GIL.
292/// Each call blocks the current tokio worker thread, which can cause runtime starvation
293/// and apparent deadlocks (nothing else gets polled).
294///
295/// This wrapper serializes GIL acquisition among callers that opt in, so at most one
296/// tokio task is blocked in `Python::with_gil` at a time, improving fairness under
297/// contention.
298///
299/// Note: this does not globally prevent other sync code from calling `Python::with_gil`
300/// directly. Use `monarch_with_gil` or `monarch_with_gil_blocking` for Python interaction
301/// that occurs on async hot paths.
302static GIL_LOCK: Lazy<tokio::sync::Mutex<()>> = Lazy::new(|| tokio::sync::Mutex::new(()));
303
304// Thread-local depth counter for re-entrant GIL acquisition.
305//
306// This tracks when we're already inside a `monarch_with_gil` or `monarch_with_gil_blocking`
307// call. On re-entry (e.g., when Python calls back into Rust while we're already executing
308// under `Python::with_gil`), we bypass the `GIL_LOCK` to avoid deadlocks.
309//
310// Without this, the following scenario would deadlock:
311// 1. Rust async code calls `monarch_with_gil`, acquires `GIL_LOCK`
312// 2. Inside the closure, Python code is executed
313// 3. Python code calls back into Rust (e.g., via a PyO3 callback)
314// 4. The callback tries to call `monarch_with_gil` again
315// 5. DEADLOCK: waiting for `GIL_LOCK` which is held by the same logical call chain
316thread_local! {
317    static GIL_DEPTH: std::cell::Cell<u32> = const { std::cell::Cell::new(0) };
318}
319
320/// RAII guard that decrements the GIL depth counter when dropped.
321struct GilDepthGuard {
322    prev_depth: u32,
323}
324
325impl Drop for GilDepthGuard {
326    fn drop(&mut self) {
327        GIL_DEPTH.with(|d| d.set(self.prev_depth));
328    }
329}
330
331/// Increments the GIL depth counter and returns a guard that restores it on drop.
332fn increment_gil_depth() -> GilDepthGuard {
333    let prev_depth = GIL_DEPTH.with(|d| {
334        let current = d.get();
335        d.set(current + 1);
336        current
337    });
338    GilDepthGuard { prev_depth }
339}
340
341/// Returns true if we're already inside a `monarch_with_gil` call (re-entrant).
342fn is_reentrant() -> bool {
343    GIL_DEPTH.with(|d| d.get() > 0)
344}

346/// Async wrapper around `Python::with_gil` intended for async call sites.
347///
348/// Why: under high concurrency, many async tasks can simultaneously
349/// try to acquire the GIL. Each call blocks the current tokio worker
350/// thread, which can cause runtime starvation / apparent deadlocks
351/// (nothing else gets polled).
352///
353/// This wrapper serializes GIL acquisition among async callers so at most one tokio
354/// task is blocked in `Python::with_gil` at a time, preventing runtime starvation
355/// under GIL contention.
356///
357/// Note: this does not globally prevent other sync code from calling
358/// `Python::with_gil` directly. Use this wrapper for Python
359/// interaction that occurs on async hot paths.
360///
361/// # Re-entrancy Safety
362///
363/// This function is re-entrant safe. If called while already inside a `monarch_with_gil`
364/// or `monarch_with_gil_blocking` call (e.g., from a Python→Rust callback), it bypasses
365/// the `GIL_LOCK` to avoid deadlocks.
366///
367/// # Example
368/// ```ignore
369/// let result = monarch_with_gil(|py| {
370///     // Do work with Python GIL
371///     Ok(42)
372/// })
373/// .await?;
374/// ```
375pub async fn monarch_with_gil<F, R>(f: F) -> R
376where
377    F: for<'py> FnOnce(Python<'py>) -> R + Send,
378{
379    // If we're already inside a monarch_with_gil call (re-entrant), skip the lock
380    // to avoid deadlock from Python→Rust callbacks
381    if is_reentrant() {
382        let _depth_guard = increment_gil_depth();
383        return Python::with_gil(f);
384    }
385
386    // Not re-entrant: acquire the serialization lock
387    let _lock_guard = GIL_LOCK.lock().await;
388    let _depth_guard = increment_gil_depth();
389    Python::with_gil(f)
390}

392/// Blocking wrapper around `Python::with_gil` for use in synchronous contexts.
393///
394/// Unlike `monarch_with_gil`, this function does NOT use the `GIL_LOCK` async mutex.
395/// Since it is blocking call, it simply acquires the GIL and releases it when the
396/// closure returns.
397///
398/// # Example
399/// ```ignore
400/// let result = monarch_with_gil_blocking(|py| {
401///     // Do work with Python GIL
402///     Ok(42)
403/// })?;
404/// ```
405pub fn monarch_with_gil_blocking<F, R>(f: F) -> R
406where
407    F: for<'py> FnOnce(Python<'py>) -> R + Send,
408{
409    let _depth_guard = increment_gil_depth();
410    Python::with_gil(f)
