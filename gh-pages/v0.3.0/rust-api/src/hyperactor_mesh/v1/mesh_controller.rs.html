57declare_attrs! {
58    /// Time between checks of actor states to create supervision events for
59    /// owners. The longer this is, the longer it will take to detect a failure
60    /// and report it to all subscribers; however, shorter intervals will send
61    /// more frequent messages and heartbeats just to see everything is still running.
62    /// The default is chosen to balance these two objectives.
63    /// This also controls how frequently the healthy heartbeat is sent out to
64    /// subscribers if there are no failures encountered.
65    @meta(CONFIG = ConfigAttr {
66        env_name: Some("HYPERACTOR_MESH_SUPERVISION_POLL_FREQUENCY".to_string()),
67        py_name: None,
68    })
69    pub attr SUPERVISION_POLL_FREQUENCY: Duration = Duration::from_secs(10);
70}

72#[derive(Debug)]
73struct HealthState {
74    /// The status of each actor in the controlled mesh.
75    /// TODO: replace with ValueMesh?
76    statuses: HashMap<Point, resource::Status>,
77    unhealthy_event: Option<Unhealthy>,
78    crashed_ranks: HashMap<usize, ActorSupervisionEvent>,
79    // The unique owner of this actor.
80    owner: Option<PortRef<MeshFailure>>,
81    /// A set of subscribers to send messages to when events are encountered.
82    subscribers: HashSet<PortRef<Option<MeshFailure>>>,
83}
84
85impl HealthState {
86    fn new(
87        statuses: HashMap<Point, resource::Status>,
88        owner: Option<PortRef<MeshFailure>>,
89    ) -> Self {
90        Self {
91            statuses,
92            unhealthy_event: None,
93            crashed_ranks: HashMap::new(),
94            owner,
95            subscribers: HashSet::new(),
96        }
97    }
98}

118/// The implementation of monitoring works as follows:
119/// * ActorMesh and ActorMeshRef subscribe for updates from this controller,
120///   which aggregates events from all owned actors.
121/// * The monitor continuously polls for new events. When new events are
122///   found, it sends messages to all subscribers
123/// * In addition to sending to subscribers, the owner is an automatic subscriber
124///   that also has to handle the events.
125#[hyperactor::export(handlers = [
126    Subscribe,
127    Unsubscribe,
128    resource::CreateOrUpdate<resource::mesh::Spec<()>> { cast = true },
129    resource::GetState<resource::mesh::State<()>> { cast = true },
130    resource::Stop { cast = true },
131])]
132pub struct ActorMeshController<A>
133where
134    A: Referable,
135{
136    mesh: ActorMeshRef<A>,
137    supervision_display_name: String,
138    // Shared health state for the monitor and responding to queries.
139    health_state: HealthState,
140    // The monitor which continuously runs in the background to refresh the state
141    // of actors.
142    // If None, the actor it monitors has already stopped.
143    monitor: Option<()>,
144}

151impl<A: Referable> ActorMeshController<A> {
152    /// Create a new mesh controller based on the provided reference.
153    pub(crate) fn new(
154        mesh: ActorMeshRef<A>,
155        supervision_display_name: Option<String>,
156        port: Option<PortRef<MeshFailure>>,
157        initial_statuses: ValueMesh<resource::Status>,
158    ) -> Self {
159        let supervision_display_name =
160            supervision_display_name.unwrap_or_else(|| mesh.name().to_string());
161        Self {
162            mesh,
163            supervision_display_name,
164            health_state: HealthState::new(initial_statuses.iter().collect(), port),
165            monitor: None,
166        }
167    }
168
169    async fn stop(
170        &self,
171        cx: &impl context::Actor,
172        reason: String,
173    ) -> v1::Result<ValueMesh<resource::Status>> {
174        // Cannot use "ActorMesh::stop" as it tries to message the controller, which is this actor.
175        self.mesh
176            .proc_mesh()
177            .stop_actor_by_name(cx, self.mesh.name().clone(), reason)
178            .await
179    }
180
181    fn self_check_state_message(&self, cx: &Instance<Self>) -> Result<(), ActorError> {
182        // Only schedule a self message if the monitor has not been dropped.
183        if self.monitor.is_some() {
184            cx.self_message_with_delay(
185                CheckState {},
186                hyperactor_config::global::get(SUPERVISION_POLL_FREQUENCY),
187            )
188        } else {
189            Ok(())
190        }
191    }
192}

194declare_attrs! {
195    /// If present in a message header, the message is from an ActorMeshController
196    /// to a subscriber and can be safely dropped if it is returned as undeliverable.
197    pub attr ACTOR_MESH_SUBSCRIBER_MESSAGE: bool;
198}
199
200fn send_subscriber_message(
201    cx: &impl context::Actor,
202    subscriber: &PortRef<Option<MeshFailure>>,
203    message: MeshFailure,
204) {
205    let mut headers = Attrs::new();
206    headers.set(ACTOR_MESH_SUBSCRIBER_MESSAGE, true);
207    if let Err(error) = subscriber.send_with_headers(cx, headers, Some(message.clone())) {
208        tracing::warn!(
209            event = %message,
210            "failed to send supervision event to subscriber {}: {}",
211            subscriber.port_id(),
212            error
213        );
214    } else {
215        tracing::info!(event = %message, "sent supervision failure message to subscriber {}", subscriber.port_id());
216    }
217}

229#[async_trait]
230impl<A: Referable> Actor for ActorMeshController<A> {
231    async fn init(&mut self, this: &Instance<Self>) -> Result<(), anyhow::Error> {
232        // Start the monitor task.
233        // There's a shared monitor for all whole mesh ref. Note that slices do
234        // not share the health state. This is fine because requerying a slice
235        // of a mesh will still return any failed state.
236        self.monitor = Some(());
237        self.self_check_state_message(this)?;
238        let owner = if let Some(owner) = &self.health_state.owner {
239            owner.to_string()
240        } else {
241            String::from("None")
242        };
243        tracing::info!(actor_id = %this.self_id(), %owner, "started mesh controller for {}", self.mesh.name());
244        Ok(())
245    }
246
247    async fn cleanup(
248        &mut self,
249        this: &Instance<Self>,
250        _err: Option<&ActorError>,
251    ) -> Result<(), anyhow::Error> {
252        // If the monitor hasn't been dropped yet, send a stop message to the
253        // proc mesh.
254        if self.monitor.take().is_some() {
255            tracing::info!(actor_id = %this.self_id(), actor_mesh = %self.mesh.name(), "starting cleanup for ActorMeshController, stopping actor mesh");
256            self.stop(this, "actor mesh controller cleanup".to_string())
257                .await?;
258        }
259        Ok(())
260    }
261
262    async fn handle_undeliverable_message(
263        &mut self,
264        cx: &Instance<Self>,
265        mut envelope: Undeliverable<MessageEnvelope>,
266    ) -> Result<(), anyhow::Error> {
267        // Update the destination in case this was a casting message.
268        envelope = update_undeliverable_envelope_for_casting(envelope);
269        if let Some(true) = envelope.0.headers().get(ACTOR_MESH_SUBSCRIBER_MESSAGE) {
270            // Remove from the subscriber list (if it existed) so we don't
271            // send to this subscriber again.
272            // NOTE: The only part of the port that is used for equality checks is
273            // the port id, so create a new one just for the comparison.
274            let dest_port_id = envelope.0.dest().clone();
275            let port = PortRef::<Option<MeshFailure>>::attest(dest_port_id);
276            let did_exist = self.health_state.subscribers.remove(&port);
277            if did_exist {
278                tracing::debug!(
279                    actor_id = %cx.self_id(),
280                    num_subscribers = self.health_state.subscribers.len(),
281                    "ActorMeshController: handle_undeliverable_message: removed subscriber {} from mesh controller",
282                    port.port_id()
283                );
284            }
285            Ok(())
286        } else {
287            handle_undeliverable_message(cx, envelope)
288        }
289    }
290}

292#[async_trait]
293impl<A: Referable> Handler<Subscribe> for ActorMeshController<A> {
294    async fn handle(&mut self, cx: &Context<Self>, message: Subscribe) -> anyhow::Result<()> {
295        // If we can't send a message to a subscriber, the subscriber might be gone.
296        // That shouldn't cause this actor to exit.
297        // This is handled by the handle_undeliverable_message method.
298        match &self.health_state.unhealthy_event {
299            None => {}
300            // For an adverse event like stopped or crashed, send a notification
301            // immediately. This represents an initial bad state, if subscribing
302            // to an already-dead mesh.
303            Some(Unhealthy::StreamClosed(msg)) => {
304                send_subscriber_message(cx, &message.0, msg.clone());
305            }
306            Some(Unhealthy::Crashed(msg)) => {
307                send_subscriber_message(cx, &message.0, msg.clone());
308            }
309        }
310        let port_id = message.0.port_id().clone();
311        if self.health_state.subscribers.insert(message.0) {
312            tracing::debug!(actor_id = %cx.self_id(), num_subscribers = self.health_state.subscribers.len(), "added subscriber {} to mesh controller", port_id);
313        }
314        Ok(())
315    }
316}

343#[async_trait]
344impl<A: Referable> Handler<resource::GetState<resource::mesh::State<()>>>
345    for ActorMeshController<A>
346{
347    async fn handle(
348        &mut self,
349        cx: &Context<Self>,
350        message: resource::GetState<resource::mesh::State<()>>,
351    ) -> anyhow::Result<()> {
352        let status = if let Some(Unhealthy::Crashed(e)) = &self.health_state.unhealthy_event {
353            resource::Status::Failed(e.to_string())
354        } else if let Some(Unhealthy::StreamClosed(_)) = &self.health_state.unhealthy_event {
355            resource::Status::Stopped
356        } else {
357            resource::Status::Running
358        };
359        let statuses = &self.health_state.statuses;
360        let mut statuses = statuses.clone().into_iter().collect::<Vec<_>>();
361        statuses.sort_by_key(|(p, _)| p.rank());
362        let statuses: ValueMesh<resource::Status> =
363            statuses
364                .into_iter()
365                .map(|(_, s)| s)
366                .collect_mesh::<ValueMesh<_>>(self.mesh.region().clone())?;
367        let state = resource::mesh::State {
368            statuses,
369            state: (),
370        };
371        message.reply.send(
372            cx,
373            resource::State {
374                name: message.name,
375                status,
376                state: Some(state),
377            },
378        )?;
379        Ok(())
380    }
381}

383#[async_trait]
384impl<A: Referable> Handler<resource::Stop> for ActorMeshController<A> {
385    async fn handle(&mut self, cx: &Context<Self>, message: resource::Stop) -> anyhow::Result<()> {
386        let mesh = &self.mesh;
387        let mesh_name = mesh.name();
388        tracing::info!(
389            name = "ActorMeshControllerStatus",
390            %mesh_name,
391            reason = %message.reason,
392            "stopping actor mesh"
393        );
394        // Run the drop on the monitor loop. The actors will not change state
395        // after this point, because they will be stopped.
396        // This message is idempotent because multiple stops only send out one
397        // set of messages to subscribers.
398        if self.monitor.take().is_none() {
399            tracing::debug!(actor_id = %cx.self_id(), actor_mesh = %mesh_name, "duplicate stop request, actor mesh is already stopped");
400            return Ok(());
401        }
402        tracing::info!(actor_id = %cx.self_id(), actor_mesh = %mesh_name, "forwarding stop request from ActorMeshController to proc mesh");
403
404        // Let the client know that the controller has stopped. Since the monitor
405        // is cancelled, it will not alert the owner or the subscribers.
406        // We use a placeholder rank to get an actor id, but really there should
407        // be a stop event for every rank in the mesh. Since every rank has the
408        // same owner, we assume the rank doesn't matter, and the owner can just
409        // assume the stop happened on all actors.
410        let rank = 0usize;
411        let event = ActorSupervisionEvent::new(
412            // Use an actor id from the mesh.
413            mesh.get(rank).unwrap().actor_id().clone(),
414            None,
415            ActorStatus::Stopped,
416            None,
417        );
418        let failure_message = MeshFailure {
419            actor_mesh_name: Some(mesh_name.to_string()),
420            // Rank = none means it affects the whole mesh.
421            rank: None,
422            event,
423        };
424        self.health_state.unhealthy_event = Some(Unhealthy::StreamClosed(failure_message.clone()));
425        // We don't send a message to the owner on stops, because only the owner
426        // can request a stop. We just send to subscribers instead, as they did
427        // not request the stop themselves.
428        for subscriber in self.health_state.subscribers.iter() {
429            send_subscriber_message(cx, subscriber, failure_message.clone());
430        }
431
432        // max_rank and extent are only needed for the deprecated RankedValues.
433        // TODO: add cmp::Ord to Point for a max() impl.
434        let max_rank = self.health_state.statuses.keys().map(|p| p.rank()).max();
435        let extent = self
436            .health_state
437            .statuses
438            .keys()
439            .next()
440            .map(|p| p.extent().clone());
441        // Send a stop message to the ProcMeshAgent for these actors.
442        match self.stop(cx, message.reason.clone()).await {
443            Ok(statuses) => {
444                // All stops successful, set actor status on health state.
445                for (rank, status) in statuses.iter() {
446                    self.health_state
447                        .statuses
448                        .entry(rank)
449                        .and_modify(move |s| *s = status);
450                }
451            }
452            // An ActorStopError means some actors didn't reach the stopped state.
453            Err(v1::Error::ActorStopError { statuses }) => {
454                // If there are no states yet, nothing to update.
455                if let Some(max_rank) = max_rank {
456                    let extent = extent.expect("no actors in mesh");
457                    for (rank, status) in statuses.materialized_iter(max_rank).enumerate() {
458                        *self
459                            .health_state
460                            .statuses
461                            .get_mut(&extent.point_of_rank(rank).expect("illegal rank"))
462                            .unwrap() = status.clone();
463                    }
464                }
465            }
466            // Other error types should be reported as supervision errors.
467            Err(e) => {
468                return Err(e.into());
469            }
470        }
471
472        tracing::info!(actor_id = %cx.self_id(), actor_mesh = %mesh_name, "stopped mesh");
473        Ok(())
474    }
475}

477/// Like send_state_change, but when there was no state change that occurred.
478/// Will send a None message to subscribers, and there is no state to change.
479/// Is not sent to the owner, because the owner is only watching for failures.
480/// Should be called once every so often so subscribers can discern the difference
481/// between "no messages because no errors" and "no messages because controller died".
482/// Without sending these hearbeats, subscribers will assume the mesh is dead.
483fn send_heartbeat(cx: &impl context::Actor, health_state: &HealthState) {
484    tracing::debug!(
485        num_subscribers = health_state.subscribers.len(),
486        "sending heartbeat to subscribers",
487    );
488
489    for subscriber in health_state.subscribers.iter() {
490        let mut headers = Attrs::new();
491        headers.set(ACTOR_MESH_SUBSCRIBER_MESSAGE, true);
492        if let Err(e) = subscriber.send_with_headers(cx, headers, None) {
493            tracing::warn!(subscriber = %subscriber.port_id(), "error sending heartbeat message: {:?}", e);
494        }
495    }
496}

498/// Sends a MeshFailure to the owner and subscribers of this mesh,
499/// and changes the health state stored unhealthy_event.
500/// Owners are sent a message only for Failure events, not for Stopped events.
501/// Subscribers are sent both Stopped and Failure events.
502fn send_state_change(
503    cx: &impl context::Actor,
504    rank: usize,
505    event: ActorSupervisionEvent,
506    mesh_name: &Name,
507    is_proc_stopped: bool,
508    health_state: &mut HealthState,
509) {
510    // This does not include the Stopped status, which is a state that occurs when the
511    // user calls stop() on a proc or actor mesh.
512    let is_failed = event.is_error();
513    if is_failed {
514        tracing::warn!(
515            name = "SupervisionEvent",
516            actor_mesh = %mesh_name,
517            %event,
518            "detected supervision error on monitored mesh: name={mesh_name}",
519        );
520    } else {
521        tracing::debug!(
522            name = "SupervisionEvent",
523            actor_mesh = %mesh_name,
524            %event,
525            "detected non-error supervision event on monitored mesh: name={mesh_name}",
526        );
527    }
528
529    let failure_message = MeshFailure {
530        actor_mesh_name: Some(mesh_name.to_string()),
531        rank: Some(rank),
532        event: event.clone(),
533    };
534    health_state.crashed_ranks.insert(rank, event.clone());
535    health_state.unhealthy_event = Some(if is_proc_stopped {
536        Unhealthy::StreamClosed(failure_message.clone())
537    } else {
538        Unhealthy::Crashed(failure_message.clone())
539    });
540    // Send a notification to the owning actor of this mesh, if there is one.
541    // Don't send a message to the owner for non-failure events such as "stopped".
542    // Those events are always initiated by the owner, who don't need to be
543    // told that they were stopped.
544    if is_failed {
545        if let Some(owner) = &health_state.owner {
546            if let Err(error) = owner.send(cx, failure_message.clone()) {
547                tracing::warn!(
548                    name = "SupervisionEvent",
549                    actor_mesh = %mesh_name,
550                    %event,
551                    %error,
552                    "failed to send supervision event to owner {}: {}. dropping event",
553                    owner.port_id(),
554                    error
555                );
556            } else {
557                tracing::info!(actor_mesh = %mesh_name, %event, "sent supervision failure message to owner {}", owner.port_id());
558            }
559        }
560    }
561    // Subscribers get all messages, even for non-failures like Stopped, because
562    // they need to know if the owner stopped the mesh.
563    for subscriber in health_state.subscribers.iter() {
564        send_subscriber_message(cx, subscriber, failure_message.clone());
565    }
566}

568fn actor_state_to_supervision_events(
569    state: resource::State<ActorState>,
570) -> (usize, Vec<ActorSupervisionEvent>) {
571    let (rank, actor_id, events) = match state.state {
572        Some(inner) => (
573            inner.create_rank,
574            Some(inner.actor_id),
575            inner.supervision_events.clone(),
576        ),
577        None => (0, None, vec![]),
578    };
579    let events = match state.status {
580        // If the actor was killed, it might not have a Failed status
581        // or supervision events, and it can't tell us which rank
582        resource::Status::NotExist | resource::Status::Stopped | resource::Status::Timeout(_) => {
583            // it was.
584            if !events.is_empty() {
585                events
586            } else {
587                vec![ActorSupervisionEvent::new(
588                    actor_id.expect("actor_id is None"),
589                    None,
590                    ActorStatus::Stopped,
591                    None,
592                )]
593            }
594        }
595        resource::Status::Failed(_) => events,
596        // All other states are successful.
597        _ => vec![],
598    };
599    (rank, events)
600}

602#[async_trait]
603impl<A: Referable> Handler<CheckState> for ActorMeshController<A> {
604    /// Checks actor states and reschedules as a self-message.
605    ///
606    /// When any actor in this mesh changes state,
607    /// including once for the initial state of all actors, send a message to the
608    /// owners and subscribers of this mesh.
609    /// The receivers will get a MeshFailure. The created rank is
610    /// the original rank of the actor on the mesh, not the rank after
611    /// slicing.
612    ///
613    /// * SUPERVISION_POLL_FREQUENCY controls how frequently to poll.
614    /// * self-messaging stops when self.monitor is set to None.
615    async fn handle(&mut self, cx: &Context<Self>, _: CheckState) -> Result<(), anyhow::Error> {
616        // This implementation polls every "time_between_checks" duration, checking
617        // for changes in the actor states. It can be improved in two ways:
618        // 1. Use accumulation, to get *any* actor with a change in state, not *all*
619        //    actors.
620        // 2. Use a push-based mode instead of polling.
621        // Wait in between checking to avoid using too much network.
622        let mesh = &self.mesh;
623        let supervision_display_name = &self.supervision_display_name;
624        // First check if the proc mesh is dead before trying to query their agents.
625        let proc_states = mesh.proc_mesh().proc_states(cx).await;
626        if let Err(e) = proc_states {
627            send_state_change(
628                cx,
629                0,
630                ActorSupervisionEvent::new(
631                    cx.self_id().clone(),
632                    None,
633                    ActorStatus::generic_failure(format!(
634                        "unable to query for proc states: {:?}",
635                        e
636                    )),
637                    None,
638                ),
639                mesh.name(),
640                false,
641                &mut self.health_state,
642            );
643            self.self_check_state_message(cx)?;
644            return Ok(());
645        }
646        if let Some(proc_states) = proc_states.unwrap() {
647            // Check if the proc mesh is still alive.
648            if let Some((point, state)) = proc_states
649                .iter()
650                .find(|(_rank, state)| state.status.is_terminating())
651            {
652                // TODO: allow "actor supervision event" to be general, and
653                // make the proc failure the cause. It is a hack to try to determine
654                // the correct status based on process exit status.
655                let actor_status = match state.state.and_then(|s| s.proc_status) {
656                    Some(ProcStatus::Stopped { .. })
657                    // SIGTERM
658                    | Some(ProcStatus::Killed { signal: 15, .. })
659                    // Conservatively treat lack of status as stopped
660                    | None => ActorStatus::Stopped,
661
662                    Some(status) => ActorStatus::Failed(ActorErrorKind::Generic(format!(
663                        "process failure: {}",
664                        status
665                    ))),
666                };
667                let display_name = if !point.is_empty() {
668                    let coords_display = point.format_as_dict();
669                    if let Some(pos) = supervision_display_name.rfind('>') {
670                        format!(
671                            "{}{}{}",
672                            &supervision_display_name[..pos],
673                            coords_display,
674                            &supervision_display_name[pos..]
675                        )
676                    } else {
677                        format!("{}{}", supervision_display_name, coords_display)
678                    }
679                } else {
680                    supervision_display_name.clone()
681                };
682                send_state_change(
683                    cx,
684                    point.rank(),
685                    ActorSupervisionEvent::new(
686                        // Attribute this to the monitored actor, even if the underlying
687                        // cause is a proc_failure. We propagate the cause explicitly.
688                        mesh.get(point.rank()).unwrap().actor_id().clone(),
689                        Some(format!("{} was running on a process which", display_name)),
690                        actor_status,
691                        None,
692                    ),
693                    mesh.name(),
694                    true,
695                    &mut self.health_state,
696                );
697                self.self_check_state_message(cx)?;
698                return Ok(());
699            }
700        }
701
702        // Now that we know the proc mesh is alive, check for actor state changes.
703        let events = mesh.actor_states(cx).await;
704        if let Err(e) = events {
705            send_state_change(
706                cx,
707                0,
708                ActorSupervisionEvent::new(
709                    cx.self_id().clone(),
710                    Some(supervision_display_name.clone()),
711                    ActorStatus::generic_failure(format!(
712                        "unable to query for actor states: {:?}",
713                        e
714                    )),
715                    None,
716                ),
717                mesh.name(),
718                false,
719                &mut self.health_state,
720            );
721            self.self_check_state_message(cx)?;
722            return Ok(());
723        }
724        // If there was any state change, we don't need to send a heartbeat.
725        let mut did_send_state_change = false;
726        // True if any rank is in a terminal status. Once that is true, no more
727        // heartbeats are sent.
728        let mut is_terminal = false;
729        // This returned point is the created rank, *not* the rank of
730        // the possibly sliced input mesh.
731        for (point, state) in events.unwrap().iter() {
732            let mut is_new = false;
733            let entry = self
734                .health_state
735                .statuses
736                .entry(point.clone())
737                .or_insert_with(|| {
738                    tracing::debug!(
739                        "PythonActorMeshImpl: received initial state: point={:?}, state={:?}",
740                        point,
741                        state
742                    );
743                    let (_rank, events) = actor_state_to_supervision_events(state.clone());
744                    // Wait for next event if the change in state produced no supervision events.
745                    if !events.is_empty() {
746                        is_new = true;
747                    }
748                    state.status.clone()
749                });
750            // If the status of any rank is terminal, we don't want to send
751            // a heartbeat message.
752            if !is_terminal && entry.is_terminating() {
753                is_terminal = true;
754            }
755            // If this actor is new, or the state changed, send a message to the owner.
756            let (rank, event) = if is_new {
757                let (rank, events) = actor_state_to_supervision_events(state.clone());
758                if events.is_empty() {
759                    continue;
760                }
761                (rank, events[0].clone())
762            } else if *entry != state.status {
763                tracing::debug!(
764                    "PythonActorMeshImpl: received state change event: point={:?}, old_state={:?}, new_state={:?}",
765                    point,
766                    entry,
767                    state
768                );
769                let (rank, events) = actor_state_to_supervision_events(state.clone());
770                if events.is_empty() {
771                    continue;
772                }
773                *entry = state.status;
774                (rank, events[0].clone())
775            } else {
776                continue;
777            };
778            did_send_state_change = true;
779            send_state_change(cx, rank, event, mesh.name(), false, &mut self.health_state);
780        }
781        if !did_send_state_change && !is_terminal {
782            // No state change, but subscribers need to be sent a message
783            // every so often so they know the controller is still alive.
784            // Send a "no state change" message.
785            // Only if the last state for any actor in this mesh is not a terminal state.
786            send_heartbeat(cx, &self.health_state);
787        }
788
789        // If all ranks are in a terminal state, we don't need to continue checking,
790        // as statuses cannot change.
791        // Any new subscribers will get an immediate message saying the mesh is stopped.
792        let all_ranks_terminal = self
793            .health_state
794            .statuses
795            .values()
796            .all(|s| s.is_terminating());
797        if !all_ranks_terminal {
798            // Schedule a self send after a waiting period.
799            self.self_check_state_message(cx)?;
800        } else {
801            // There's no need to send a stop message during cleanup if all the
802            // ranks are already terminal.
803            self.monitor.take();
804        }
805        return Ok(());
806    }
807}

809#[derive(Debug)]
810#[hyperactor::export]
811pub(crate) struct ProcMeshController {
812    mesh: ProcMeshRef,
813}
814
815impl ProcMeshController {
816    /// Create a new proc controller based on the provided reference.
817    pub(crate) fn new(mesh: ProcMeshRef) -> Self {
818        Self { mesh }
819    }
820}
821
822#[async_trait]
823impl Actor for ProcMeshController {
824    async fn cleanup(
825        &mut self,
826        this: &Instance<Self>,
827        _err: Option<&ActorError>,
828    ) -> Result<(), anyhow::Error> {
829        // Cannot use "ProcMesh::stop" as it's only defined on ProcMesh, not ProcMeshRef.
830        let names = self.mesh.proc_ids().collect::<Vec<ProcId>>();
831        let region = self.mesh.region().clone();
832        if let Some(hosts) = self.mesh.hosts() {
833            hosts
834                .stop_proc_mesh(
835                    this,
836                    self.mesh.name(),
837                    names,
838                    region,
839                    "proc mesh controller cleanup".to_string(),
840                )
841                .await
842        } else {
843            Ok(())
844        }
845    }
846}

863    async fn cleanup(
864        &mut self,
865        this: &Instance<Self>,
866        _err: Option<&ActorError>,
867    ) -> Result<(), anyhow::Error> {
868        // Cannot use "HostMesh::shutdown" as it's only defined on HostMesh, not HostMeshRef.
869        for host in self.mesh.values() {
870            if let Err(e) = host.shutdown(this).await {
871                tracing::warn!(host = %host, error = %e, "host shutdown failed");
872            }
873        }
874        Ok(())
875    }
