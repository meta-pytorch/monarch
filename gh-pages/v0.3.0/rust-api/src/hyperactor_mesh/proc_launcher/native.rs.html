9//! Native OS process launcher.
10//!
11//! This module provides [`NativeProcLauncher`], a [`ProcLauncher`]
12//! backend that runs bootstrap procs as ordinary local OS processes
13//! using [`tokio::process::Command`].
14//!
15//! ## Role in the proc-launching architecture
16//!
17//! [`BootstrapProcManager`](crate::bootstrap::BootstrapProcManager)
18//! owns the *lifecycle* model (readiness tracking, supervision,
19//! bookkeeping) and delegates only the *mechanics* of
20//! starting/stopping a proc to a launcher. `NativeProcLauncher` is
21//! the simplest such backend and serves as the reference
22//! implementation for the [`ProcLauncher`] contract.
23//!
24//! In particular:
25//! - **Readiness is callback-driven.** A proc is considered ready
26//!   only when it invokes the bootstrap callback address carried in
27//!   [`Bootstrap`]. This module does not determine readiness.
28//! - **Terminal status is channel-driven.** `launch` returns a
29//!   [`LaunchResult`] whose `exit_rx` resolves exactly once with a
30//!   [`ProcExitResult`]. The native backend implements this by
31//!   awaiting `child.wait()` in a background task and mapping the OS
32//!   [`ExitStatus`] into a [`ProcExitKind`].
33//! - **Termination is initiation-only.** [`ProcLauncher::terminate`]
34//!   and [`ProcLauncher::kill`] send signals and return without
35//!   waiting for exit; completion is observed by awaiting `exit_rx`
36//!   (or a higher-level handle built on it).
37//!
38//! ## Bootstrap + diagnostics plumbing
39//!
40//! The bootstrap payload and a human-readable process name are passed
41//! to the child via environment variables:
42//! - [`BOOTSTRAP_MODE_ENV`] carries an env-safe encoding of [`Bootstrap`].
43//! - [`PROCESS_NAME_ENV`] carries a diagnostic name derived from the
44//!   [`ProcId`] and hostname (useful for logs/`ps`/systemd).
45//! - [`BOOTSTRAP_LOG_CHANNEL`] is optionally set when the manager
46//!   provisions a mesh log-forwarding channel in [`LaunchOptions`].
47//!
48//! ## Stdio behavior
49//!
50//! Stdio configuration is driven by [`LaunchOptions::want_stdio`]:
51//! - When true, stdout/stderr are piped and returned as
52//!   [`StdioHandling::Captured`] so the manager can attach forwarding
53//!   / tailing infrastructure.
54//! - When false, stdio is inherited from the parent process.
55//!
56//! The native launcher itself does not compute stderr tails; the
57//! manager can do that when it attaches log forwarding for captured
58//! pipes.
59//!
60//! ## Termination semantics
61//!
62//! Termination is implemented with POSIX signals:
63//! - `terminate` sends `SIGTERM` and schedules escalation to
64//!   `SIGKILL` after a timeout if the proc still appears present.
65//! - `kill` sends `SIGKILL` immediately.
66//!
67//! A small best-effort PID registry is maintained to support
68//! signaling and to avoid leaking long-lived children if the launcher
69//! is dropped during teardown.

106/// Native OS process launcher.
107///
108/// This launcher runs bootstrap procs as ordinary local OS processes
109/// via `tokio::process::Command`.
110///
111/// It is the "baseline" backend for [`ProcLauncher`]:
112/// - it spawns a child process directly,
113/// - optionally captures stdout/stderr pipes for the manager,
114/// - and reports terminal status by awaiting `child.wait()` and
115///   sending a [`ProcExitResult`] through the `exit_rx` channel.
116///
117/// Termination is implemented by sending POSIX signals (`SIGTERM`
118/// with optional escalation to `SIGKILL`, or immediate `SIGKILL`).
119pub(crate) struct NativeProcLauncher {
120    /// Best-effort PID registry for processes launched by this
121    /// launcher.
122    ///
123    /// The map is used to:
124    /// - look up the OS pid when `terminate`/`kill` are requested,
125    ///   and
126    /// - perform best-effort cleanup on `Drop`.
127    ///
128    /// Entries are inserted immediately after a successful spawn and
129    /// removed by the exit-monitor task after `wait()` completes.
130    /// Missing entries are treated as idempotent success (already
131    /// exited / unknown).
132    pid_table: Arc<Mutex<HashMap<ProcId, u32>>>,
133}
134
135impl NativeProcLauncher {
136    /// Create a new native launcher.
137    /// `command` as the base "bootstrap command".
138    ///
139    /// A fresh `tokio::process::Command` is constructed per launch
140    /// configured with:
141    /// - the bootstrap payload / diagnostics env vars, and
142    /// - stdio behavior as directed by [`LaunchOptions`].
143    ///
144    /// The internal PID table starts empty and is populated as procs
145    /// are spawned.
146    pub fn new() -> Self {
147        Self {
148            pid_table: Arc::new(Mutex::new(HashMap::new())),
149        }
150    }
151
152    /// Send a POSIX signal to `pid`.
153    ///
154    /// Semantics:
155    /// - `Ok(())` if delivered successfully.
156    /// - `Ok(())` if the process is already gone (`ESRCH`).
157    /// - Otherwise returns a launcher error in the provided `kind`.
158    ///
159    /// Note: We signal the process *group* (negative PID) because the
160    /// child is started in its own process group via `setpgid(0, 0)`.
161    /// This ensures we kill the entire process tree, including any
162    /// sub-processes spawned by wrappers (e.g., shell scripts, python
163    /// launchers).
164    fn send_signal(
165        pid: u32,
166        sig: i32,
167        kind: fn(String) -> ProcLauncherError,
168    ) -> Result<(), ProcLauncherError> {
169        // We use negative PID to signal the process group (PGID = pid)
170        // rather than just the leader.
171        let pgid = -(pid as i32);
172        // SAFETY: `libc::kill` is a simple syscall wrapper with no
173        // memory safety concerns. Passing a negative PID signals the
174        // process group, which is valid POSIX semantics.
175        let rc = unsafe { libc::kill(pgid, sig) };
176        if rc == 0 {
177            return Ok(());
178        }
179
180        let e = std::io::Error::last_os_error();
181        if e.raw_os_error() == Some(libc::ESRCH) {
182            Ok(())
183        } else {
184            Err(kind(format!("signal(pgid={pgid}, {sig}) failed: {e}")))
185        }
186    }
187}

189/// Convert a platform `std::process::ExitStatus` into our
190/// launcher-neutral [`ProcExitKind`] representation.
191///
192/// This interprets an exit status in the usual POSIX order:
193/// - If the process terminated due to a signal, returns
194///   [`ProcExitKind::Signaled`] with the signal number and whether a
195///   core was produced.
196/// - Otherwise, if the process exited normally, returns
197///   [`ProcExitKind::Exited`] with the integer exit code.
198/// - If neither signal nor code is available (should be rare /
199///   platform-specific), returns [`ProcExitKind::Failed`] with an
200///   "unknown exit status" reason.
201///
202/// Note: on Unix, `signal()` / `core_dumped()` come from
203/// `std::os::unix::process::ExitStatusExt`.
204fn exit_kind_from_status(status: &ExitStatus) -> ProcExitKind {
205    if let Some(sig) = status.signal() {
206        ProcExitKind::Signaled {
207            signal: sig,
208            core_dumped: status.core_dumped(),
209        }
210    } else if let Some(code) = status.code() {
211        ProcExitKind::Exited { code }
212    } else {
213        ProcExitKind::Failed {
214            reason: "unknown exit status".to_string(),
215        }
216    }
217}

219#[async_trait]
220impl ProcLauncher for NativeProcLauncher {
221    /// Launch a bootstrap proc as a native OS child process.
222    ///
223    /// This implementation uses [`tokio::process::Command`] to spawn
224    /// the configured [`BootstrapCommand`], passes the bootstrap
225    /// payload and diagnostic metadata via environment variables, and
226    /// returns a [`LaunchResult`] whose `exit_rx` resolves exactly
227    /// once with the proc's terminal status.
228    ///
229    /// ## What this method does
230    ///
231    /// - **Constructs a Tokio command**.
232    /// - **Injects bootstrap state** via [`BOOTSTRAP_MODE_ENV`] using
233    ///   [`Bootstrap::to_env_safe_string`].
234    /// - **Injects a human-readable name** via [`PROCESS_NAME_ENV`]
235    ///   (see [`super::format_process_name`]).
236    /// - **Optionally injects log forwarding address** via
237    ///   [`BOOTSTRAP_LOG_CHANNEL`] when `opts.log_channel` is `Some`.
238    /// - **Configures stdio** based on `opts.want_stdio`:
239    ///   - `true` → pipe stdout/stderr and return
240    ///     [`StdioHandling::Captured`]
241    ///   - `false` → inherit stdout/stderr and return
242    ///     [`StdioHandling::Inherited`]
243    /// - **Spawns an exit monitor task** that awaits `child.wait()`,
244    ///   maps the OS status into [`ProcExitKind`], removes the PID
245    ///   from `pid_table`, and sends the final [`ProcExitResult`]
246    ///   through `exit_rx`.
247    ///
248    /// ## Notes / invariants
249    ///
250    /// - **Readiness is not handled here.** The proc is considered
251    ///   *ready* only when it invokes the bootstrap callback address
252    ///   carried inside `bootstrap`. This method only starts the
253    ///   process.
254    /// - **Termination is initiation-only.** `terminate`/`kill` only
255    ///   send signals; completion is observed by awaiting `exit_rx`.
256    #[hyperactor::instrument(
257        level = "debug",
258        fields(
259            proc_id = proc_id.to_string(),
260            want_stdio = opts.want_stdio,
261            log_channel_present = opts.log_channel.is_some(),
262            pid = tracing::field::Empty,
263        )
264    )]
265    async fn launch(
266        &self,
267        proc_id: &ProcId,
268        opts: LaunchOptions,
269    ) -> Result<LaunchResult, ProcLauncherError> {
270        // New Tokio Command from BootstrapCommand (template)
271        let mut cmd = opts.command.new();
272
273        // Bootstrap payload
274        cmd.env(BOOTSTRAP_MODE_ENV, opts.bootstrap_payload);
275
276        // Diagnostics name
277        cmd.env(PROCESS_NAME_ENV, opts.process_name);
278
279        // Manager may decide to create a log-forwarding channel; if
280        // so, native passes it through via BOOTSTRAP_LOG_CHANNEL.
281        if let Some(addr) = &opts.log_channel {
282            cmd.env(BOOTSTRAP_LOG_CHANNEL, addr.to_string());
283        }
284
285        // Stdio behavior gated on 'want_stdio'.
286        if opts.want_stdio {
287            cmd.stdout(Stdio::piped()).stderr(Stdio::piped());
288        } else {
289            cmd.stdout(Stdio::inherit()).stderr(Stdio::inherit());
290        }
291
292        // Put child in its own process group so we can kill the entire
293        // tree with kill(-pgid, ...).
294        //
295        // SAFETY: runs in the child between fork and exec. We must not
296        // allocate or do anything complex here.
297        unsafe {
298            cmd.pre_exec(|| {
299                // setpgid(0, 0) => make this process the leader of a new process group.
300                if libc::setpgid(0, 0) != 0 {
301                    return Err(std::io::Error::last_os_error());
302                }
303                Ok(())
304            });
305        }
306
307        let started_at = hyperactor::clock::RealClock.system_time_now();
308
309        let mut child = cmd.spawn().map_err(ProcLauncherError::Launch)?;
310        let pid = child.id().expect("spawned child pid unavailable");
311
312        // Record PID in current span and log spawn event.
313        tracing::Span::current().record("pid", pid);
314        tracing::debug!("spawned");
315
316        // Record PID for signaling and Drop cleanup.
317        {
318            let mut table = self.pid_table.lock().expect("pid_table mutex poisoned");
319            table.insert(proc_id.clone(), pid);
320        }
321
322        // Extract stdio handles only if requested (only present if
323        // piped).
324        let stdio = if opts.want_stdio {
325            let stdout = child.stdout.take().expect("stdout piped but missing");
326            let stderr = child.stderr.take().expect("stderr piped but missing");
327            StdioHandling::Captured { stdout, stderr }
328        } else {
329            StdioHandling::Inherited
330        };
331
332        // Terminal status channel.
333        let (exit_tx, exit_rx) = oneshot::channel();
334        let pid_table = Arc::clone(&self.pid_table);
335
336        let proc_id = proc_id.clone();
337        // Propagate the launch span into the exit monitor so that
338        // exit_observed and any subsequent events are not "floating".
339        let launch_span = tracing::Span::current();
340        tokio::spawn(
341            async move {
342                // Exit monitoring runs out-of-band:
343                // `launch()` returns immediately with an `exit_rx`, and
344                // this task is responsible for eventually resolving that
345                // channel exactly once.
346                //
347                // We intentionally keep this task simple and robust: wait
348                // for the child to terminate and translate the OS
349                // `ExitStatus` into the trait-level `ProcExitKind` used
350                // by the manager.
351                let kind = match child.wait().await {
352                    Ok(status) => exit_kind_from_status(&status),
353                    Err(e) => ProcExitKind::Failed {
354                        reason: format!("wait failed: {e}"),
355                    },
356                };
357
358                tracing::debug!(?kind, "exit_observed");
359
360                // Once the process has reached a terminal state (or we
361                // failed to observe it), remove it from the PID table so:
362                // - `terminate`/`kill` become idempotent no-ops for this
363                //   proc_id, and
364                // - Drop-time cleanup doesn't try to SIGKILL something
365                //   that's already done.
366                //
367                {
368                    let mut table = pid_table.lock().expect("pid_table mutex poisoned");
369                    table.remove(&proc_id);
370                }
371
372                // Publish the terminal status. It's okay if the receiver
373                // was dropped (e.g., manager shutdown); we just stop
374                // trying.
375                //
376                // Native launcher doesn't provide stderr tail here: the
377                // manager attaches StreamFwder for `Captured` stdio and
378                // can derive any tailing diagnostics itself.
379                let _ = exit_tx.send(ProcExitResult {
380                    kind,
381                    stderr_tail: None,
382                });
383            }
384            .instrument(launch_span),
385        );
386
387        Ok(LaunchResult {
388            pid: Some(pid),
389            started_at,
390            stdio,
391            exit_rx,
392        })
393    }
394
395    /// Initiate graceful termination of a running proc.
396    ///
397    /// This is a *fire-and-forget* termination request:
398    /// - We send **SIGTERM** to the recorded PID (if any).
399    /// - We then schedule an **escalation to SIGKILL** after
400    ///   `timeout` *iff* the proc still appears present in
401    ///   `pid_table`.
402    /// - We return immediately; callers observe completion by
403    ///   awaiting the `exit_rx` returned from [`Self::launch`].
404    ///
405    /// ## Semantics
406    ///
407    /// - If `proc_id` is not present in `pid_table`, this returns
408    ///   `Ok(())` (idempotent success: already exited or never
409    ///   tracked).
410    /// - If SIGTERM fails because the process is already gone
411    ///  (`ESRCH`), this returns `Ok(())` (idempotent success).
412    /// - Other signal delivery failures are returned as
413    ///   [`ProcLauncherError::Terminate`].
414    ///
415    /// ## Implementation notes
416    ///
417    /// - Escalation is performed by a background task that sleeps for
418    ///   `timeout` using [`RealClock`], re-checks `pid_table`, and
419    ///   sends SIGKILL if the proc is still present.
420    /// - We rely on the exit-monitor task spawned in `launch` to
421    ///   remove the proc from `pid_table` and resolve the `exit_rx`
422    ///   channel when the proc actually terminates.
423    async fn terminate(
424        &self,
425        proc_id: &ProcId,
426        timeout: Duration,
427    ) -> Result<(), ProcLauncherError> {
428        let pid = {
429            let table = self.pid_table.lock().expect("pid_table mutex poisoned");
430            table.get(proc_id).copied()
431        };
432
433        let Some(pid) = pid else {
434            // Idempotent success: already exited or unknown.
435            return Ok(());
436        };
437
438        let timeout_ms = timeout.as_millis() as u64;
439
440        // Create a termination span so escalation logs are not
441        // floating.
442        let term_span = tracing::debug_span!(
443            "terminate",
444            %proc_id,
445            pid,
446            timeout_ms
447        );
448        // Emit the request event *inside* the terminate span.
449        {
450            let _g = term_span.enter();
451            tracing::debug!("terminate_requested");
452        }
453
454        // Send SIGTERM.
455        Self::send_signal(pid, libc::SIGTERM, ProcLauncherError::Terminate)?;
456
457        //  Escalate to SIGKILL after timeout if still present.
458        let pid_table = Arc::clone(&self.pid_table);
459        let proc_id = proc_id.clone();
460
461        tokio::spawn(
462            async move {
463                RealClock.sleep(timeout).await;
464
465                let pid = {
466                    let table = pid_table.lock().expect("pid_table mutex poisoned");
467                    table.get(&proc_id).copied()
468                };
469
470                if let Some(pid) = pid {
471                    tracing::info!("terminate timeout; escalating to SIGKILL");
472                    if let Err(e) = Self::send_signal(pid, libc::SIGKILL, ProcLauncherError::Kill) {
473                        tracing::warn!(error = %e, "SIGKILL escalation failed");
474                    }
475                }
476            }
477            .instrument(term_span),
478        );
479
480        Ok(())
481    }
482
483    /// Initiate an immediate force-kill of a running proc.
484    ///
485    /// This is the "hard stop" counterpart to [`Self::terminate`]:
486    /// - We send **SIGKILL** to the recorded PID (if any).
487    /// - We return immediately; callers observe completion by
488    ///   awaiting the `exit_rx` returned from [`Self::launch`].
489    ///
490    /// ## Semantics
491    ///
492    /// - If `proc_id` is not present in `pid_table`, this returns
493    ///   `Ok(())` (idempotent success: already exited or never
494    ///   tracked).
495    /// - If SIGKILL fails because the process is already gone
496    ///   (`ESRCH`), this returns `Ok(())` (idempotent success).
497    /// - Other signal delivery failures are returned as
498    ///   [`ProcLauncherError::Kill`].
499    ///
500    /// ## Implementation notes
501    ///
502    /// - We do not remove `proc_id` from `pid_table` here; the
503    ///   exit-monitor task spawned in `launch` removes it once the
504    ///   child actually exits.
505    async fn kill(&self, proc_id: &ProcId) -> Result<(), ProcLauncherError> {
506        let pid = {
507            let table = self.pid_table.lock().expect("pid_table mutex poisoned");
508            table.get(proc_id).copied()
509        };
510
511        let Some(pid) = pid else {
512            // Idempotent success.
513            return Ok(());
514        };
515
516        let kill_span = tracing::debug_span!("kill", %proc_id, pid);
517        {
518            let _g = kill_span.enter();
519            tracing::debug!("kill_requested");
520
521            // Immediate SIGKILL.
522            Self::send_signal(pid, libc::SIGKILL, ProcLauncherError::Kill)?;
523        }
524
525        Ok(())
526    }
527}

535impl Drop for NativeProcLauncher {
536    // Best-effort safety net: if the manager is being torn down while
537    // some children are still running, we don't want to leak
538    // long-lived procs.
539    //
540    // This is intentionally *coarse*: at Drop time we may not have an
541    // async runtime available to do graceful termination, wait for
542    // exit, or run a timeout/escalation ladder. So we prefer a hard
543    // stop (SIGKILL) to avoid orphaned processes and test flakiness.
544    //
545    // Note that this does **not** remove entries from pid_table; the
546    // whole structure is being dropped anyway. Also, failure is
547    // ignored because Drop must not panic; the most common failure is
548    // ESRCH (already exited).
549    fn drop(&mut self) {
550        // Collect PIDs while holding the lock, then release before killing.
551        // This avoids holding the lock during syscalls.
552        let pids: Vec<(ProcId, u32)> = {
553            let table = self.pid_table.lock().expect("pid_table mutex poisoned");
554            table.iter().map(|(k, v)| (k.clone(), *v)).collect()
555        };
556
557        for (proc_id, pid) in pids {
558            match Self::send_signal(pid, libc::SIGKILL, ProcLauncherError::Kill) {
559                Ok(()) => {
560                    tracing::info!(%proc_id, pid, "drop cleanup: sent SIGKILL");
561                }
562                Err(_e) => {
563                    tracing::warn!(%proc_id, pid, "drop cleanup: SIGKILL failed");
564                }
565            }
566        }
567    }
568}

599    /// Consume captured stdout/stderr pipes from
600    /// `StdioHandling::Captured` and return their contents.
601    async fn read_captured_lines(stdio: StdioHandling) -> (Vec<String>, Vec<u8>) {
602        match stdio {
603            StdioHandling::Captured { stdout, mut stderr } => {
604                let mut out_lines = vec![];
605                let mut out = BufReader::new(stdout).lines();
606                while let Ok(Some(line)) = out.next_line().await {
607                    out_lines.push(line);
608                }
609
610                // Drain stderr just so pipes close cleanly; return
611                // raw bytes if useful.
612                let mut err_bytes = vec![];
613                let _ = stderr.read_to_end(&mut err_bytes).await;
614
615                (out_lines, err_bytes)
616            }
617            other => panic!("expected Captured stdio, got {other:?}"),
618        }
619    }

904    /// `terminate()` escalates to SIGKILL when the child ignores
905    /// SIGTERM.
906    ///
907    /// This test exercises the "graceful then forceful" shutdown
908    /// ladder:
909    ///
910    /// - The child process installs a SIGTERM handler that ignores
911    ///   termination (`SIGTERM=IGN`) and then blocks (sleeps).
912    /// - The parent waits for a small READY handshake on captured
913    ///   stdout to ensure the signal disposition has been installed
914    ///   *before* sending SIGTERM (avoids a race where SIGTERM
915    ///   arrives too early).
916    /// - We call [`ProcLauncher::terminate`] with a short timeout.
917    /// - After the timeout elapses, the launcher should observe that
918    ///   the proc still appears present and escalate by sending
919    ///   SIGKILL.
920    ///
921    /// The assertion is made on the terminal status delivered via
922    /// `exit_rx`: it must be [`ProcExitKind::Signaled`] with `signal
923    /// == SIGKILL`.
924    #[tokio::test]
925    async fn terminate_escalates_to_sigkill_if_child_ignores_sigterm() {
926        // Use an explicit READY handshake so the child installs
927        // SIGTERM=IGN before the parent sends SIGTERM. `exec` ensures
928        // we signal the Python process, not the shell.
929        let launcher = NativeProcLauncher::new();
930        let script = r#"exec python3 -c 'import signal,sys,time; signal.signal(signal.SIGTERM, signal.SIG_IGN); print("READY"); sys.stdout.flush(); time.sleep(30)'"#;
931
932        // v0 bootstrap by default but it doesn't matter here.
933        let bootstrap = Bootstrap::default();
934        let proc_id = ProcId::Direct(any_unix_addr(), "term-escalate".into());
935        let opts = LaunchOptions {
936            command: with_sh(script),
937            bootstrap_payload: bootstrap.to_env_safe_string().unwrap(),
938            process_name: proc_id.to_string(),
939            // Need captured stdout for the READY handshake.
940            want_stdio: true,
941            tail_lines: 0,
942            log_channel: None,
943        };
944
945        let lr = launcher.launch(&proc_id, opts).await.expect("launch");
946
947        // Wait until the child has definitely installed SIGTERM=IGN.
948        // We only need stdout for this; drain stderr best-effort.
949        let (stdout, mut stderr) = match lr.stdio {
950            StdioHandling::Captured { stdout, stderr } => (stdout, stderr),
951            other => panic!("expected Captured stdio, got {other:?}"),
952        };
953
954        let mut out = BufReader::new(stdout);
955        let mut line = String::new();
956        RealClock
957            .timeout(Duration::from_secs(2), out.read_line(&mut line))
958            .await
959            .expect("timed out waiting for READY")
960            .expect("read_line failed");
961        assert_eq!(line.trim(), "READY", "child did not signal readiness");
962
963        // Drain stderr in the background so pipes close cleanly.
964        tokio::spawn(async move {
965            let mut _buf = Vec::new();
966            let _ = stderr.read_to_end(&mut _buf).await;
967        });
968
969        // Now request graceful termination + escalation after a short
970        // timeout.
971        launcher
972            .terminate(&proc_id, Duration::from_millis(100))
973            .await
974            .expect("terminate");
975
976        // Give enough time for: sleep(timeout) + SIGKILL delivery +
977        // wait() + oneshot send.
978        let exit = RealClock
979            .timeout(Duration::from_secs(5), lr.exit_rx)
980            .await
981            .expect("timed out waiting for exit_rx")
982            .expect("exit_rx dropped");
983
984        match exit.kind {
985            ProcExitKind::Signaled { signal, .. } => {
986                assert_eq!(
987                    signal,
988                    libc::SIGKILL,
989                    "expected SIGKILL escalation, got {signal}"
990                );
991            }
992            other => panic!("expected Signaled(SIGKILL), got {other:?}"),
993        }
994    }

996    /// Verify that `NativeProcLauncher::Drop` kills child processes.
997    ///
998    /// This test exercises the critical cleanup path: when a launcher
999    /// is dropped without explicit termination, its `Drop` impl
1000    /// should SIGKILL any tracked children to prevent orphans.
1001    ///
1002    /// We test the observable effect: when the child is killed, its
1003    /// stdout closes and we get EOF.
1004    #[tokio::test]
1005    async fn drop_closes_child_stdio() {
1006        // Child prints "READY", then writes dots until killed.
1007        let python_script = r#"
1008import sys, time
1009print("READY"); sys.stdout.flush()
1010while True:
1011    sys.stdout.write("."); sys.stdout.flush()
1012    time.sleep(0.05)
1013"#;
1014
1015        let command = BootstrapCommand {
1016            program: PathBuf::from("python3"),
1017            args: vec!["-u".into(), "-c".into(), python_script.trim().into()],
1018            ..Default::default()
1019        };
1020
1021        let bootstrap = Bootstrap::default();
1022        let proc_id = ProcId::Direct(any_unix_addr(), "drop-cleanup-test".into());
1023        let opts = LaunchOptions {
1024            command,
1025            bootstrap_payload: bootstrap.to_env_safe_string().unwrap(),
1026            process_name: proc_id.to_string(),
1027            want_stdio: true,
1028            tail_lines: 0,
1029            log_channel: None,
1030        };
1031
1032        // Keep stdout in outer scope so we can read after launcher
1033        // drop.
1034        let mut stdout_reader: BufReader<tokio::process::ChildStdout>;
1035
1036        {
1037            let launcher = NativeProcLauncher::new();
1038            let lr = launcher.launch(&proc_id, opts).await.expect("launch");
1039
1040            let (stdout, _stderr) = match lr.stdio {
1041                StdioHandling::Captured { stdout, stderr } => (stdout, stderr),
1042                other => panic!("expected Captured stdio, got {other:?}"),
1043            };
1044
1045            stdout_reader = BufReader::new(stdout);
1046
1047            // Wait for "READY".
1048            let mut line = String::new();
1049            RealClock
1050                .timeout(Duration::from_secs(2), stdout_reader.read_line(&mut line))
1051                .await
1052                .expect("timeout waiting for READY")
1053                .expect("read_line failed");
1054            assert!(line.trim() == "READY", "expected 'READY', got: {line}");
1055
1056            // Prove child is alive by reading one dot.
1057            let mut b = [0u8; 1];
1058            RealClock
1059                .timeout(Duration::from_secs(2), stdout_reader.read_exact(&mut b))
1060                .await
1061                .expect("timeout waiting for dot")
1062                .expect("read_exact failed");
1063            assert_eq!(b[0], b'.');
1064
1065            // launcher drops here; its Drop should SIGKILL all tracked
1066            // children.
1067        }
1068
1069        // After drop: stdout should hit EOF once the child is killed.
1070        // Drain any remaining output and wait for EOF with a timeout.
1071        let drain_eof = async {
1072            let mut buf = [0u8; 64];
1073            loop {
1074                match stdout_reader.read(&mut buf).await {
1075                    Ok(0) => break,     // EOF - child is dead
1076                    Ok(_n) => continue, // still getting dots, keep draining
1077                    Err(e) => panic!("read failed: {e}"),
1078                }
1079            }
1080        };
1081
1082        RealClock
1083            .timeout(Duration::from_secs(2), drain_eof)
1084            .await
1085            .expect("never observed EOF after launcher drop - child likely still alive");
1086    }
