55fn process_state_span(
56    source: &ChannelAddr,
57    dest: &ChannelAddr,
58    session_id: u64,
59    next: &Next,
60    rcv_raw_frame_count: u64,
61) -> Span {
62    let pending_ack_count = if next.seq > next.ack {
63        next.seq - next.ack - 1
64    } else {
65        0
66    };
67
68    hyperactor_telemetry::context_span!(
69        "net i/o loop",
70        dest = %dest,
71        session_id = session_id,
72        source = %source,
73        next_seq = next.seq,
74        last_ack = next.ack,
75        pending_ack_count = pending_ack_count,
76        rcv_raw_frame_count = rcv_raw_frame_count,
77    )
78}

102#[derive(Debug)]
103enum RejectConn {
104    /// Reject the connection due to the given error.
105    EncounterError(String),
106    /// The server is being closed.
107    ServerClosing,
108    /// Do not reject the connection.
109    No,
110}

112impl<S: AsyncRead + AsyncWrite + Send + 'static + Unpin> ServerConn<S> {
113    async fn handshake<M: RemoteMessage>(&mut self) -> Result<u64, anyhow::Error> {
114        let Some(frame) = self
115            .reader
116            .next()
117            .instrument(hyperactor_telemetry::context_span!("read handshake"))
118            .await?
119        else {
120            anyhow::bail!("end of stream before first frame from {}", self.source);
121        };
122        let message = serde_multipart::Message::from_framed(frame)?;
123        let Frame::Init(session_id) = serde_multipart::deserialize_bincode::<Frame<M>>(message)?
124        else {
125            anyhow::bail!("unexpected initial frame from {}", self.source);
126        };
127        Ok(session_id)
128    }

130    async fn process_step<M: RemoteMessage>(
131        &mut self,
132        session_id: u64,
133        tx: &mpsc::Sender<M>,
134        cancel_token: &CancellationToken,
135        next: &Next,
136        last_ack_time: &mut tokio::time::Instant,
137        rcv_raw_frame_count: &mut u64,
138        ack_time_interval: Duration,
139        ack_msg_interval: u64,
140        log_id: &str,
141    ) -> (Next, Option<(Result<(), anyhow::Error>, RejectConn)>) {
142        let mut next = next.clone();
143        if self.write_state.is_idle()
144            && (next.ack + ack_msg_interval <= next.seq
145                || (next.ack < next.seq && last_ack_time.elapsed() > ack_time_interval))
146        {
147            let Ok(writer) = replace(&mut self.write_state, WriteState::Broken).into_idle() else {
148                panic!("illegal state");
149            };
150            let ack = match serialize_response(NetRxResponse::Ack(next.seq - 1)) {
151                Ok(ack) => ack,
152                Err(err) => {
153                    return (
154                        next,
155                        Some((
156                            Err::<(), anyhow::Error>(err.into())
157                                .context(format!("{log_id}: serializing ack")),
158                            RejectConn::No,
159                        )),
160                    );
161                }
162            };
163            match FrameWrite::new(
164                writer,
165                ack,
166                hyperactor_config::global::get(config::CODEC_MAX_FRAME_LENGTH),
167            ) {
168                Ok(fw) => {
169                    self.write_state = WriteState::Writing(fw, next.seq);
170                }
171                Err((writer, e)) => {
172                    debug_assert_eq!(e.kind(), std::io::ErrorKind::InvalidData);
173                    tracing::error!(
174                        source = %self.source,
175                        dest = %self.dest,
176                        error = %e,
177                        "failed to create ack frame"
178                    );
179                    self.write_state = WriteState::Idle(writer);
180                }
181            }
182        }
183
184        tokio::select! {
185            // Prioritize ack, and then shutdown. Leave read last because
186            // there could be a large volume of messages to read, which
187            // subsequently starves the other select! branches.
188            biased;
189
190            // We have to be careful to manage the ack write state here, so that we do not
191            // write partial acks in the presence of cancellation.
192            ack_result = self.write_state.send().instrument(hyperactor_telemetry::context_span!("write ack")) => {
193                match ack_result {
194                    Ok(acked_seq) => {
195                        *last_ack_time = RealClock.now();
196                        next.ack = acked_seq;
197                    }
198                    Err(err) => {
199                        let v = self.write_state.value();
200                        return (
201                            next,
202                            Some((
203                                Err::<(), anyhow::Error>(err.into())
204                                    .context(format!("{log_id}: acking peer message: {v:?}")),
205                                RejectConn::No,
206                            )),
207                        );
208                    }
209                }
210            },
211            // Have a tick to abort select! call to make sure the ack for the last message can get the chance
212            // to be sent as a result of time interval being reached.
213            _ = RealClock.sleep_until(*last_ack_time + ack_time_interval), if next.ack < next.seq => {},
214
215            _ = cancel_token.cancelled() => return (next, Some((Ok(()), RejectConn::ServerClosing))),
216
217            bytes_result = self.reader.next().instrument(hyperactor_telemetry::context_span!("read bytes")) => {
218                *rcv_raw_frame_count += 1;
219                // First handle transport-level I/O errors, and EOFs.
220                let bytes = match bytes_result {
221                    Ok(Some(bytes)) => bytes,
222                    Ok(None) => {
223                        tracing::debug!(
224                                source = %self.source,
225                                dest = %self.dest,
226                                "received EOF from client"
227                            );
228                        return (next, Some((Ok(()), RejectConn::No)));
229                    }
230                    Err(err) => {
231                        return (
232                            next,
233                            Some((
234                                Err::<(), anyhow::Error>(err.into()).context(format!(
235                                    "{log_id}: reading into Frame with M = {}",
236                                    type_name::<M>(),
237                                )),
238                                RejectConn::No,
239                            )),
240                        )
241                    }
242                };
243
244                // De-frame the multi-part message.
245                let bytes_len = bytes.len();
246                let message = match serde_multipart::Message::from_framed(bytes) {
247                    Ok(message) => message,
248                    Err(err) => {
249                        // Track deframing error for this channel pair
250                        metrics::CHANNEL_ERRORS.add(
251                            1,
252                            hyperactor_telemetry::kv_pairs!(
253                                "source" => self.source.to_string(),
254                                "dest" => self.dest.to_string(),
255                                "session_id" => session_id.to_string(),
256                                "error_type" => metrics::ChannelErrorType::DeframeError.as_str(),
257                            ),
258                        );
259                        return (
260                            next,
261                            Some((
262                                Err::<(), anyhow::Error>(err.into()).context(format!(
263                                    "{log_id}: de-frame message with M = {}",
264                                    type_name::<M>(),
265                                )),
266                                RejectConn::No,
267                            )),
268                        )
269                    }
270                };
271
272                // Finally decode the message. This assembles the M-typed message
273                // from its constituent parts.
274                match serde_multipart::deserialize_bincode(message) {
275                    Ok(Frame::Init(_)) => {
276                        return (
277                            next,
278                            Some((
279                                Err(anyhow::anyhow!("{log_id}: unexpected init frame")),
280                                RejectConn::EncounterError("expect Frame::Message; got Frame::Int".to_string())
281                            )),
282                        )
283                    },
284                    // Ignore retransmits.
285                    Ok(Frame::Message(seq, _)) if seq < next.seq => {
286                        tracing::debug!(
287                            source = %self.source,
288                            dest = %self.dest,
289                            seq = seq,
290                            "ignoring retransmit; retransmit seq: {}; expected next seq: {}",
291                            seq,
292                            next.seq
293                        );
294                    },
295                    // The following segment ensures exactly-once semantics.
296                    // That means No out-of-order delivery and no duplicate delivery.
297                    Ok(Frame::Message(seq, message)) => {
298                        // received seq should be equal to next seq. Else error out!
299                        if seq > next.seq {
300                            let error_msg = format!("out-of-sequence message, expected seq {}, got {}", next.seq, seq);
301                            tracing::error!(
302                                source = %self.source,
303                                dest = %self.dest,
304                                seq = seq,
305                                "{}", error_msg
306                            );
307                            return (
308                                next,
309                                Some((
310                                    Err(anyhow::anyhow!(format!("{log_id}: {error_msg}"))),
311                                    RejectConn::EncounterError(error_msg)
312                                ))
313                            )
314                        }
315                        match self.send_with_buffer_metric(session_id, tx, message)
316                            .instrument(hyperactor_telemetry::context_span!(
317                                "send_with_buffer_metric",
318                                seq = seq,
319                            ))
320                            .await
321                        {
322                            Ok(()) => {
323                                // Track throughput for this channel pair
324                                metrics::CHANNEL_THROUGHPUT_BYTES.add(
325                                    bytes_len as u64,
326                                    hyperactor_telemetry::kv_pairs!(
327                                        "source" => self.source.to_string(),
328                                        "dest" => self.dest.to_string(),
329                                        "session_id" => session_id.to_string(),
330                                    ),
331                                );
332                                metrics::CHANNEL_THROUGHPUT_MESSAGES.add(
333                                    1,
334                                    hyperactor_telemetry::kv_pairs!(
335                                        "source" => self.source.to_string(),
336                                        "dest" => self.dest.to_string(),
337                                        "session_id" => session_id.to_string(),
338                                    ),
339                                );
340                                // In channel's contract, "delivered" means the message
341                                // is sent to the NetRx object. Therefore, we could bump
342                                // `next_seq` as far as the message is put on the mpsc
343                                // channel.
344                                //
345                                // Note that when/how the messages in NetRx are processed
346                                // is not covered by channel's contract. For example,
347                                // the message might never be taken out of netRx, but
348                                // channel still considers those messages delivered.
349                                next.seq = seq+1;
350                            }
351                            Err(err) => {
352                                return (
353                                    next,
354                                    Some((
355                                        Err::<(), anyhow::Error>(err)
356                                            .context(format!("{log_id}: relaying message to mpsc channel")),
357                                        RejectConn::No,
358                                    )),
359                                )
360                            }
361                        }
362                    },
363                    Err(err) => {
364                        // Track deserialization error for this channel pair
365                        metrics::CHANNEL_ERRORS.add(
366                            1,
367                            hyperactor_telemetry::kv_pairs!(
368                                "source" => self.source.to_string(),
369                                "dest" => self.dest.to_string(),
370                                "session_id" => session_id.to_string(),
371                                "error_type" => metrics::ChannelErrorType::DeserializeError.as_str(),
372                            ),
373                        );
374                        return (
375                           next,
376                            Some((
377                                Err::<(), anyhow::Error>(err.into()).context(format!(
378                                    "{log_id}: deserialize message with M = {}",
379                                    type_name::<M>(),
380                                )),
381                                RejectConn::No,
382                            )),
383                        )
384                    }
385                }
386            },
387        }
388
389        (next, None)
390    }

392    /// Handles a server side stream created during the `listen` loop.
393    async fn process<M: RemoteMessage>(
394        &mut self,
395        session_id: u64,
396        tx: mpsc::Sender<M>,
397        cancel_token: CancellationToken,
398        mut next: Next,
399    ) -> (Next, Result<(), anyhow::Error>) {
400        let log_id = format!("session {}.{}<-{}", self.dest, session_id, self.source);
401        let initial_next: Next = next.clone();
402        let mut rcv_raw_frame_count = 0u64;
403        let mut last_ack_time = RealClock.now();
404
405        let ack_time_interval = hyperactor_config::global::get(config::MESSAGE_ACK_TIME_INTERVAL);
406        let ack_msg_interval = hyperactor_config::global::get(config::MESSAGE_ACK_EVERY_N_MESSAGES);
407
408        let (mut final_next, final_result, reject_conn) = loop {
409            let span = process_state_span(
410                &self.source,
411                &self.dest,
412                session_id,
413                &next,
414                rcv_raw_frame_count,
415            );
416
417            let (new_next, break_info) = self
418                .process_step(
419                    session_id,
420                    &tx,
421                    &cancel_token,
422                    &next,
423                    &mut last_ack_time,
424                    &mut rcv_raw_frame_count,
425                    ack_time_interval,
426                    ack_msg_interval,
427                    &log_id,
428                )
429                .instrument(span)
430                .await;
431
432            next = new_next;
433
434            if let Some((result, reject_conn)) = break_info {
435                break (next, result, reject_conn);
436            }
437        };
438
439        // Note:
440        //   1. processed seq/ack is Next-1;
441        //   2. rcv_raw_frame_count contains the last frame which might not be
442        //      desrializable, e.g. EOF, error, etc.
443        let debug_msg = format!(
444            "NetRx::process exited its loop with states: initial Next \
445            was {initial_next}; final Next is {final_next}; since acked: {}sec; \
446            rcv raw frame count is {rcv_raw_frame_count}; final result: {:?}; \
447            reject_conn is {:?}",
448            last_ack_time.elapsed().as_secs(),
449            final_result,
450            reject_conn,
451        );
452        tracing::info!(
453            source = %self.source,
454            dest = %self.dest,
455            session_id,
456            "{}", debug_msg
457        );
458
459        let mut final_ack = final_next.ack;
460        // Flush any ongoing write.
461        if self.write_state.is_writing() {
462            if let Ok(acked_seq) = self.write_state.send().await {
463                if acked_seq > final_ack {
464                    final_ack = acked_seq;
465                }
466            };
467        }
468        // best effort: "flush" any remaining ack before closing this session
469        if self.write_state.is_idle() && final_ack < final_next.seq {
470            let Ok(writer) = replace(&mut self.write_state, WriteState::Broken).into_idle() else {
471                panic!("illegal state");
472            };
473            let result = async {
474                let ack = serialize_response(NetRxResponse::Ack(final_next.seq - 1))
475                    .map_err(anyhow::Error::from)?;
476
477                let max = hyperactor_config::global::get(config::CODEC_MAX_FRAME_LENGTH);
478                let fw =
479                    FrameWrite::new(writer, ack, max).map_err(|(_, e)| anyhow::Error::from(e))?;
480                self.write_state = WriteState::Writing(fw, final_next.seq);
481                self.write_state.send().await.map_err(anyhow::Error::from)
482            };
483
484            match result.await {
485                Ok(acked_seq) => {
486                    final_next.ack = acked_seq;
487                }
488                Err(e) => {
489                    tracing::debug!(
490                        session_id,
491                        source = %self.source,
492                        dest = %self.dest,
493                        error = %e,
494                        "failed to flush acks during cleanup"
495                    );
496                }
497            }
498        }
499
500        if self.write_state.is_idle()
501            && matches!(
502                reject_conn,
503                RejectConn::EncounterError(_) | RejectConn::ServerClosing
504            )
505        {
506            let Ok(writer) = replace(&mut self.write_state, WriteState::Broken).into_idle() else {
507                panic!("illegal state");
508            };
509            let rsp = match reject_conn {
510                RejectConn::EncounterError(reason) => NetRxResponse::Reject(reason),
511                RejectConn::ServerClosing => NetRxResponse::Closed,
512                RejectConn::No => panic!("illegal state"),
513            };
514            if let Ok(data) = serialize_response(rsp) {
515                match FrameWrite::new(
516                    writer,
517                    data,
518                    hyperactor_config::global::get(config::CODEC_MAX_FRAME_LENGTH),
519                ) {
520                    Ok(fw) => {
521                        self.write_state = WriteState::Writing(fw, 0);
522                        let _ = self.write_state.send().await;
523                    }
524                    Err((w, e)) => {
525                        debug_assert_eq!(e.kind(), std::io::ErrorKind::InvalidData);
526                        tracing::debug!(
527                            source = %self.source,
528                            dest = %self.dest,
529                            session_id = session_id,
530                            error = %e,
531                            "failed to create reject frame"
532                        );
533                        self.write_state = WriteState::Idle(w);
534                        // drop the reject; we're closing anyway
535                    }
536                }
537            };
538        }
539
540        if let Some(mut w) = replace(&mut self.write_state, WriteState::Broken).into_writer() {
541            // Try to shutdown the connection gracefully. This is a best effort
542            // operation, and we don't care if it fails.
543            let _ = w.shutdown().await;
544        }
545
546        (final_next, final_result)
547    }
548
549    // NetRx's buffer, i.e. the mpsc channel between NetRx and its
550    // client, should rarely be full for long. But when it is full, it
551    // will block NetRx from taking more messages, sending back ack,
552    // and subsequently lead to uncommon behaviors such as ack
553    // timeout, backpressure on NetTx, etc. In order to aid debugging,
554    // it is important to add a metric measuring full buffer
555    // occurences.
556    async fn send_with_buffer_metric<M: RemoteMessage>(
557        &mut self,
558        session_id: u64,
559        tx: &mpsc::Sender<M>,
560        message: M,
561    ) -> anyhow::Result<()> {
562        let start = RealClock.now();
563        loop {
564            tokio::select! {
565                biased;
566                permit_result = tx.reserve() => {
567                    permit_result?.send(message);
568                    return Ok(())
569                }
570                _ = RealClock.sleep(hyperactor_config::global::get(config::CHANNEL_NET_RX_BUFFER_FULL_CHECK_INTERVAL)) => {
571                    // When buffer is full too long, we log it.
572                    metrics::CHANNEL_NET_RX_BUFFER_FULL.add(
573                        1,
574                        hyperactor_telemetry::kv_pairs!(
575                            "dest" => self.dest.to_string(),
576                            "source" => self.source.to_string(),
577                            "session_id" => session_id.to_string(),
578                        ),
579                    );
580                    // Full buffer should happen rarely. So we also add a log
581                    // here to make debugging easy.
582                     tracing::debug!(
583                        source = %self.source,
584                        dest = %self.dest,
585                        session_id = session_id,
586                        "encountered full mpsc channel for {} secs",
587                        start.elapsed().as_secs(),
588                    );
589                }
590            }
591        }
592    }
593}

595/// Used to bookkeep message processing states.
596#[derive(Clone)]
597struct Next {
598    // The last received message's seq number + 1.
599    seq: u64,
600    // The last acked seq number + 1.
601    ack: u64,
602}
603
604impl fmt::Display for Next {
605    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
606        write!(f, "(seq: {}, ack: {})", self.seq, self.ack)
607    }
608}

610/// Manages persistent sessions, ensuring that only one connection can own
611/// a session at a time, and arranging for session handover.
612#[derive(Clone)]
613pub(super) struct SessionManager {
614    sessions: Arc<DashMap<u64, MVar<Next>>>,
615}
616
617impl SessionManager {
618    pub(super) fn new() -> Self {
619        Self {
620            sessions: Arc::new(DashMap::new()),
621        }
622    }
623
624    pub(super) async fn serve<S, M>(
625        &self,
626        mut conn: ServerConn<S>,
627        tx: mpsc::Sender<M>,
628        cancel_token: CancellationToken,
629    ) -> Result<(), anyhow::Error>
630    where
631        S: AsyncRead + AsyncWrite + Send + 'static + Unpin,
632        M: RemoteMessage,
633    {
634        let session_id = conn
635            .handshake::<M>()
636            .await
637            .context("while serving handshake")?;
638
639        let session_var = match self.sessions.entry(session_id) {
640            Entry::Occupied(entry) => entry.get().clone(),
641            Entry::Vacant(entry) => {
642                // We haven't seen this session before. We begin with seq=0 and ack=0.
643                let var = MVar::full(Next { seq: 0, ack: 0 });
644                entry.insert(var.clone());
645                var
646            }
647        };
648
649        let source = conn.source.clone();
650        let dest = conn.dest.clone();
651
652        let next = session_var.take().await;
653        let (next, res) = conn.process(session_id, tx, cancel_token, next).await;
654        session_var.put(next).await;
655
656        if let Err(ref err) = res {
657            tracing::info!(
658                source = %source,
659                dest = %dest,
660                error = ?err,
661                session_id = session_id,
662                "process encountered an error"
663            );
664        }
665
666        res
667    }
668}

670/// Main listen loop that actually runs the server. The loop will exit when `parent_cancel_token` is
671/// canceled.
672async fn listen<M: RemoteMessage, L: Listener>(
673    mut listener: L,
674    listener_channel_addr: ChannelAddr,
675    tx: mpsc::Sender<M>,
676    parent_cancel_token: CancellationToken,
677    is_tls: bool,
678) -> Result<(), ServerError>
679where
680    L::Addr: Send + Sync + fmt::Debug + 'static + Into<ChannelAddr>,
681    L::Io: Send + Unpin + fmt::Debug + 'static,
682{
683    let mut connections: JoinSet<Result<(), anyhow::Error>> = JoinSet::new();
684
685    // Cancellation token used to cancel our children only.
686    let child_cancel_token = CancellationToken::new();
687
688    let manager = SessionManager::new();
689
690    // Heartbeat timer for server health metrics
691    let heartbeat_interval = hyperactor_config::global::get(config::SERVER_HEARTBEAT_INTERVAL);
692    let mut heartbeat_timer: Interval = tokio::time::interval(heartbeat_interval);
693
694    let result: Result<(), ServerError> = loop {
695        let _ = tracing::info_span!("channel_listen_accept_loop");
696        tokio::select! {
697            result = listener.accept() => {
698                match result {
699                    Ok((stream, addr)) => {
700                        let source : ChannelAddr = addr.into();
701                        tracing::debug!(
702                            source = %source,
703                            dest = %listener_channel_addr,
704                            "new connection accepted"
705                        );
706                        metrics::CHANNEL_CONNECTIONS.add(
707                            1,
708                            hyperactor_telemetry::kv_pairs!(
709                                "transport" => listener_channel_addr.transport().to_string(),
710                                "operation" => "accept"
711                            ),
712                        );
713
714                        let tx = tx.clone();
715                        let child_cancel_token = child_cancel_token.child_token();
716                        let dest  = listener_channel_addr.clone();
717                        let manager = manager.clone();
718                        connections.spawn(async move {
719                            let res = if is_tls {
720                                let conn = ServerConn::new(meta::tls_acceptor(true)?
721                                    .accept(stream)
722                                    .await?, source.clone(), dest.clone());
723                                manager.serve(conn, tx, child_cancel_token).await
724                            } else {
725                                let conn = ServerConn::new(stream, source.clone(), dest.clone());
726                                manager.serve(conn, tx, child_cancel_token).await
727                            };
728
729                            if let Err(ref err) = res {
730                                        metrics::CHANNEL_ERRORS.add(
731                                    1,
732                                    hyperactor_telemetry::kv_pairs!(
733                                        "transport" => dest.transport().to_string(),
734                                        "error" => err.to_string(),
735                                        "error_type" => metrics::ChannelErrorType::ConnectionError.as_str(),
736                                        "dest" => dest.to_string(),
737                                    ),
738                                );
739
740                                // we don't want the health probe TCP connections to be counted as an error.
741                                match source {
742                                    ChannelAddr::Tcp(source_addr) if source_addr.ip().is_loopback() => {},
743                                    _ => {
744                                        tracing::info!(
745                                            source = %source,
746                                            dest = %dest,
747                                            error = ?err,
748                                            "error processing peer connection"
749                                        );
750
751                                    }
752                                }
753                            }
754                            res
755                    });
756                    }
757                    Err(err) => {
758                        metrics::CHANNEL_ERRORS.add(
759                            1,
760                            hyperactor_telemetry::kv_pairs!(
761                                "transport" => listener_channel_addr.transport().to_string(),
762                                "operation" => "accept",
763                                "error" => err.to_string(),
764                                "error_type" => metrics::ChannelErrorType::ConnectionError.as_str(),
765                            ),
766                        );
767
768                        tracing::info!(
769                            dest = %listener_channel_addr,
770                            error = %err,
771                            "accept error"
772                        );
773                    }
774                }
775            }
776
777            _ = heartbeat_timer.tick() => {
778                metrics::SERVER_HEARTBEAT.add(
779                    1,
780                    hyperactor_telemetry::kv_pairs!(
781                        "dest" => listener_channel_addr.to_string()
782                    ),
783                );
784            }
785
786            _ = parent_cancel_token.cancelled() => {
787                tracing::info!(
788                    dest = %listener_channel_addr,
789                    "received parent token cancellation"
790                );
791                break Ok(());
792            }
793
794            result = join_nonempty(&mut connections) => {
795                if let Err(err) = result {
796                    tracing::info!(
797                        dest = %listener_channel_addr,
798                        error = %err,
799                        "connection task join error"
800                    );
801                }
802            }
803        }
804    };
805
806    child_cancel_token.cancel();
807    while connections.join_next().await.is_some() {}
808
809    result
810}
811
812async fn join_nonempty<T: 'static>(set: &mut JoinSet<T>) -> Result<T, JoinError> {
813    match set.join_next().await {
814        None => std::future::pending::<Result<T, JoinError>>().await,
815        Some(result) => result,
816    }
817}

819/// Handle to an underlying server that is actively listening.
820#[derive(Debug)]
821pub struct ServerHandle {
822    /// Join handle of the listening task.
823    join_handle: JoinHandle<Result<(), ServerError>>,
824
825    /// A cancellation token used to indicate that a stop has been
826    /// initiated.
827    cancel_token: CancellationToken,
828
829    /// Address of the channel that was used to create the listener. This can be used to dial the that listener.
830    channel_addr: ChannelAddr,
831}
832
833impl fmt::Display for ServerHandle {
834    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
835        write!(f, "{}", self.channel_addr)
836    }
837}
838
839impl ServerHandle {
840    /// Signal the server to stop. This will stop accepting new
841    /// incoming connection requests, and drain pending operations
842    /// on active connections. After draining is completed, the
843    /// connections are closed.
844    pub(crate) fn stop(&self, reason: &str) {
845        tracing::info!(
846            name = "ChannelServerStatus",
847            dest = %self.channel_addr,
848            status = "Stop::Sent",
849            reason,
850            "sent Stop signal; check server logs for the stop progress"
851        );
852        self.cancel_token.cancel();
853    }
854
855    /// Reference to the channel that was used to start the server.
856    fn local_channel_addr(&self) -> &ChannelAddr {
857        &self.channel_addr
858    }
859}
860
861impl Future for ServerHandle {
862    type Output = <JoinHandle<Result<(), ServerError>> as Future>::Output;
863
864    fn poll(self: Pin<&mut Self>, cx: &mut std::task::Context<'_>) -> Poll<Self::Output> {
865        // SAFETY: This is safe to do because self is pinned.
866        let join_handle_pinned =
867            unsafe { self.map_unchecked_mut(|container| &mut container.join_handle) };
868        join_handle_pinned.poll(cx)
869    }
870}

872/// serve new connections that are accepted from the given listener.
873pub(super) fn serve<M: RemoteMessage, L: Listener + Send + Unpin + 'static>(
874    listener: L,
875    channel_addr: ChannelAddr,
876    is_tls: bool,
877) -> Result<(ChannelAddr, NetRx<M>), ServerError>
878where
879    L::Addr: Sync + Send + fmt::Debug + Into<ChannelAddr>,
880    L::Io: Sync + Send + Unpin + fmt::Debug,
881{
882    metrics::CHANNEL_CONNECTIONS.add(
883        1,
884        hyperactor_telemetry::kv_pairs!(
885            "transport" => channel_addr.transport().to_string(),
886            "operation" => "serve"
887        ),
888    );
889
890    let (tx, rx) = mpsc::channel::<M>(1024);
891    let cancel_token = CancellationToken::new();
892    let join_handle = tokio::spawn(listen(
893        listener,
894        channel_addr.clone(),
895        tx,
896        cancel_token.child_token(),
897        is_tls,
898    ));
899    let server_handle = ServerHandle {
900        join_handle,
901        cancel_token,
902        channel_addr: channel_addr.clone(),
903    };
904
905    Ok((
906        server_handle.local_channel_addr().clone(),
907        NetRx(rx, channel_addr, server_handle),
908    ))
909}
