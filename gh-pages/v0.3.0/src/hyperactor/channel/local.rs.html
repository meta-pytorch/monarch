9//! Local (in-process) channel implementation.

18/// Create a new local channel, returning its two ends.
19pub fn new<M: RemoteMessage>() -> (impl Tx<M>, impl Rx<M>) {
20    let (tx, rx) = mpsc::unbounded_channel::<M>();
21    let (mpsc_tx, status_sender) = MpscTx::new(tx, ChannelAddr::Local(0));
22    let mpsc_rx = MpscRx::new(rx, ChannelAddr::Local(0), status_sender);
23    (mpsc_tx, mpsc_rx)
24}

26// In-process channels, with a shared registry.
27
28struct Ports {
29    ports: HashMap<u64, (mpsc::UnboundedSender<Data>, watch::Receiver<TxStatus>)>,
30    next_port: u64,
31}
32
33impl Ports {
34    fn alloc(&mut self) -> (u64, mpsc::UnboundedReceiver<Data>, watch::Sender<TxStatus>) {
35        let port = self.next_port;
36        self.next_port += 1;
37        let (tx, rx) = mpsc::unbounded_channel::<Data>();
38        let (status_tx, status_rx) = watch::channel(TxStatus::Active);
39        if self.ports.insert(port, (tx.clone(), status_rx)).is_some() {
40            panic!("port reused")
41        }
42        (port, rx, status_tx)
43    }
44
45    fn free(&mut self, port: u64) {
46        self.ports.remove(&port);
47    }
48
49    fn get(&self, port: u64) -> Option<&(mpsc::UnboundedSender<Data>, watch::Receiver<TxStatus>)> {
50        self.ports.get(&port)
51    }
52}
53
54impl Default for Ports {
55    fn default() -> Self {
56        Self {
57            ports: HashMap::new(),
58            next_port: 1,
59        }
60    }
61}

63static PORTS: LazyLock<Mutex<Ports>> = LazyLock::new(|| Mutex::new(Ports::default()));
64
65pub struct LocalTx<M: RemoteMessage> {
66    tx: mpsc::UnboundedSender<Data>,
67    port: u64,
68    status: watch::Receiver<TxStatus>, // Default impl. Always reports `Active`.
69    _phantom: PhantomData<M>,
70}
71
72#[async_trait]
73impl<M: RemoteMessage> Tx<M> for LocalTx<M> {
74    fn do_post(&self, message: M, return_channel: Option<oneshot::Sender<SendError<M>>>) {
75        let data: Data = match bincode::serialize(&message) {
76            Ok(data) => data,
77            Err(err) => {
78                if let Some(return_channel) = return_channel {
79                    return_channel
80                        .send(SendError {
81                            error: err.into(),
82                            message,
83                            reason: None,
84                        })
85                        .unwrap_or_else(|m| tracing::warn!("failed to deliver SendError: {}", m));
86                }
87                return;
88            }
89        };
90        if self.tx.send(data).is_err() {
91            if let Some(return_channel) = return_channel {
92                return_channel
93                    .send(SendError {
94                        error: ChannelError::Closed,
95                        message,
96                        reason: None,
97                    })
98                    .unwrap_or_else(|m| tracing::warn!("failed to deliver SendError: {}", m));
99            }
100        }
101    }
102
103    fn addr(&self) -> ChannelAddr {
104        ChannelAddr::Local(self.port)
105    }
106
107    fn status(&self) -> &watch::Receiver<TxStatus> {
108        &self.status
109    }
110}

112pub struct LocalRx<M: RemoteMessage> {
113    data_rx: mpsc::UnboundedReceiver<Data>,
114    status_tx: watch::Sender<TxStatus>,
115    port: u64,
116    _phantom: PhantomData<M>,
117}
118
119#[async_trait]
120impl<M: RemoteMessage> Rx<M> for LocalRx<M> {
121    async fn recv(&mut self) -> Result<M, ChannelError> {
122        let data = self.data_rx.recv().await.ok_or(ChannelError::Closed)?;
123        bincode::deserialize(&data).map_err(ChannelError::from)
124    }
125
126    fn addr(&self) -> ChannelAddr {
127        ChannelAddr::Local(self.port)
128    }
129}
130
131impl<M: RemoteMessage> Drop for LocalRx<M> {
132    fn drop(&mut self) {
133        let _ = self.status_tx.send(TxStatus::Closed);
134        PORTS.lock().unwrap().free(self.port);
135    }
136}

138/// Dial a local port, returning a Tx for it.
139#[allow(clippy::result_large_err)] // TODO: Consider reducing the size of `ChannelError`.
140pub fn dial<M: RemoteMessage>(port: u64) -> Result<LocalTx<M>, ChannelError> {
141    let ports = PORTS.lock().unwrap();
142    let result = ports.get(port);
143    if let Some((data_tx, status_rx)) = result {
144        Ok(LocalTx {
145            tx: data_tx.clone(),
146            port,
147            status: status_rx.clone(),
148            _phantom: PhantomData,
149        })
150    } else {
151        Err(ChannelError::Closed)
152    }
153}
154
155/// Serve a local port. The server is shut down when the returned Rx is dropped.
156pub fn serve<M: RemoteMessage>() -> (u64, LocalRx<M>) {
157    let (port, data_rx, status_tx) = PORTS.lock().unwrap().alloc();
158    (
159        port,
160        LocalRx {
161            data_rx,
162            status_tx,
163            port,
164            _phantom: PhantomData,
165        },
166    )
167}

169#[cfg(test)]
170mod tests {
171    use std::assert_matches::assert_matches;
172
173    use super::*;
174
175    #[tokio::test]
176    async fn test_local_basic() {
177        let (tx, mut rx) = local::new::<u64>();
178
179        tx.post(123);
180        assert_eq!(rx.recv().await.unwrap(), 123);
181    }
182
183    #[tokio::test]
184    async fn test_local_dial_serve() {
185        let (port, mut rx) = local::serve::<u64>();
186        assert!(port != 0);
187
188        let tx = local::dial::<u64>(port).unwrap();
189
190        tx.post(123);
191        assert_eq!(rx.recv().await.unwrap(), 123);
192
193        drop(rx);
194
195        let (return_tx, return_rx) = oneshot::channel();
196        tx.try_post(123, return_tx);
197        assert_matches!(
198            return_rx.await,
199            Ok(SendError {
200                error: ChannelError::Closed,
201                message: 123,
202                ..
203            })
204        );
205    }
206
207    #[tokio::test]
208    async fn test_local_drop() {
209        let (port, mut rx) = local::serve::<u64>();
210        let tx = local::dial::<u64>(port).unwrap();
211
212        tx.post(123);
213        assert_eq!(rx.recv().await.unwrap(), 123);
214
215        drop(rx);
216
217        assert_matches!(
218            local::dial::<u64>(port).err().unwrap(),
219            ChannelError::Closed
220        );
221    }
