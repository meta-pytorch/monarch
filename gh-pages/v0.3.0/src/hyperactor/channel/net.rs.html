9//! A simple socket channel implementation using a single-stream
10//! framing protocol. Each frame is encoded as an 8-byte
11//! **big-endian** length prefix (u64), followed by exactly that many
12//! bytes of payload.
13//!
14//! Message frames carry a `serde_multipart::Message` (not raw
15//! bincode). In compat mode (current default), this is encoded as a
16//! sentinel `u64::MAX` followed by a single bincode payload. Response frames
17//! are a bincode-serialized NetRxResponse enum, containing either the acked
18//! sequence number, or the Reject value indicating that the server rejected
19//! the connection.
20//!
21//! Message frame (compat/unipart) example:
22//! ```text
23//! +------------------ len: u64 (BE) ------------------+----------------------- data -----------------------+
24//! | \x00\x00\x00\x00\x00\x00\x00\x10                  | \xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF | <bincode bytes> |
25//! |                       16                          |           u64::MAX             |                   |
26//! +---------------------------------------------------+-----------------------------------------------------+
27//! ```
28//!
29//! Response frame (wire format):
30//! ```text
31//! +------------------ len: u64 (BE) ------------------+---------------- data ------------------+
32//! | \x00\x00\x00\x00\x00\x00\x00\x??                  | <bincode acked sequence num or reject> |
33//! +---------------------------------------------------+----------------------------------------+
34//! ```
35//!
36//! I/O is handled by `FrameReader`/`FrameWrite`, which are
37//! cancellation-safe and avoid extra copies. Helper fns
38//! `serialize_response(NetRxResponse) -> Result<Bytes, bincode::Error>`
39//! and `deserialize_response(Bytes) -> Result<NetRxResponse, bincode::Error>`
40//! convert to/from the response payload.
41//!
42//! ### Limits & EOF semantics
43//! * **Max frame size:** frames larger than
44//!   `config::CODEC_MAX_FRAME_LENGTH` are rejected with
45//!   `io::ErrorKind::InvalidData`.
46//! * **EOF handling:** `FrameReader::next()` returns `Ok(None)` only
47//!   when EOF occurs exactly on a frame boundary. If EOF happens
48//!   mid-frame, it returns `Err(io::ErrorKind::UnexpectedEof)`.

107enum NetRxResponse {

115fn serialize_response(response: NetRxResponse) -> Result<Bytes, bincode::Error> {
116    bincode::serialize(&response).map(|bytes| bytes.into())

120    bincode::deserialize(&data)

138        &self.status

140
141    fn do_post(&self, message: M, return_channel: Option<oneshot::Sender<SendError<M>>>) {
142        tracing::trace!(
143            name = "post",
144            dest = %self.dest,
145            "sending message"
146        );
147
148        let return_channel = return_channel.unwrap_or_else(|| oneshot::channel().0);
149        if let Err(mpsc::error::SendError((message, return_channel, _))) =
150            self.sender.send((message, return_channel, RealClock.now()))
151        {
152            let _ = return_channel.send(SendError {
153                error: ChannelError::Closed,
154                message,
155                reason: None,
156            });
157        }
158    }

278    /// Listen and serve connections on this socket address.
279    /// This function panics if thread-local tokio runtime is not set.
280    pub fn serve<M: RemoteMessage>(
281        addr: SocketAddr,
282    ) -> Result<(ChannelAddr, NetRx<M>), ServerError> {
283        let caddr = ChannelAddr::Unix(addr.clone());
284        let maybe_listener = match &addr {
285            SocketAddr::Bound(sock_addr) => StdUnixListener::bind_addr(sock_addr),
286            SocketAddr::Unbound => StdUnixDatagram::unbound()
287                .and_then(|u| u.local_addr())
288                .and_then(|uaddr| StdUnixListener::bind_addr(&uaddr)),
289        };
290        let std_listener =
291            maybe_listener.map_err(|err| ServerError::Listen(ChannelAddr::Unix(addr), err))?;
292
293        std_listener
294            .set_nonblocking(true)
295            .map_err(|err| ServerError::Listen(caddr.clone(), err))?;
296        let local_addr = std_listener
297            .local_addr()
298            .map_err(|err| ServerError::Resolve(caddr.clone(), err))?;
299        let listener: UnixListener = UnixListener::from_std(std_listener)
300            .map_err(|err| ServerError::Io(caddr.clone(), err))?;
301        super::serve(listener, local_addr.into(), false)
302    }

459            // On non-Linux platforms, convert abstract names to filesystem paths
460            let name = name.strip_prefix("@").unwrap_or(name);
461            let path = Self::abstract_to_filesystem_path(name);
462            Self::from_pathname(&path.to_string_lossy())
463        }
464
465        #[cfg(not(target_os = "linux"))]
466        fn abstract_to_filesystem_path(abstract_name: &str) -> std::path::PathBuf {
467            use std::collections::hash_map::DefaultHasher;
468            use std::hash::Hash;
469            use std::hash::Hasher;
470
471            // Generate a stable hash of the abstract name for deterministic paths
472            let mut hasher = DefaultHasher::new();
473            abstract_name.hash(&mut hasher);
474            let hash = hasher.finish();
475
476            // Include process ID to prevent inter-process conflicts
477            let process_id = std::process::id();
478
479            // TODO: we just leak these. Should we do something smarter?
480            std::path::PathBuf::from(format!("/tmp/hyperactor_{}_{:x}", process_id, hash))
481        }
482
483        /// Pathnames may be absolute or relative.
484        pub fn from_pathname(name: &str) -> anyhow::Result<Self> {
485            Ok(Self::new(StdSocketAddr::from_pathname(name)?))
486        }

541    /// Serve the given address. Supports both v4 and v6 address. If port 0 is provided as
542    /// dynamic port will be resolved and is available on the returned ServerHandle.
543    pub fn serve<M: RemoteMessage>(
544        addr: SocketAddr,
545    ) -> Result<(ChannelAddr, NetRx<M>), ServerError> {
546        // Construct our own std TcpListener to avoid having to await, making this function
547        // non-async.
548        let std_listener = std::net::TcpListener::bind(addr)
549            .map_err(|err| ServerError::Listen(ChannelAddr::Tcp(addr), err))?;
550        std_listener
551            .set_nonblocking(true)
552            .map_err(|e| ServerError::Listen(ChannelAddr::Tcp(addr), e))?;
553        let listener = TcpListener::from_std(std_listener)
554            .map_err(|e| ServerError::Listen(ChannelAddr::Tcp(addr), e))?;
555        let local_addr = listener
556            .local_addr()
557            .map_err(|err| ServerError::Resolve(ChannelAddr::Tcp(addr), err))?;
558        super::serve(listener, ChannelAddr::Tcp(local_addr), false)
559    }

591        // Try to parse as a socket address first

593            return Ok(ChannelAddr::MetaTls(MetaTlsAddr::Socket(socket_addr)));
594        }
595
596        // Otherwise, parse as hostname:port

600            Some((hostname, port_str)) => {
601                let Ok(port) = port_str.parse() else {
602                    return Err(ChannelError::InvalidAddress(addr_string.to_string()));
603                };
604                Ok(ChannelAddr::MetaTls(MetaTlsAddr::Host {
605                    hostname: hostname.to_string(),
606                    port,
607                }))
608            }
609            _ => Err(ChannelError::InvalidAddress(addr_string.to_string())),
610        }

645                Some(rustls_pemfile::Item::PKCS8Key(key)) => key,
646                Some(rustls_pemfile::Item::ECKey(key)) => key,

740                hostname: self.hostname.clone(),
741                port: self.port,
742            })
743        }

745        async fn connect(&self) -> Result<Self::Stream, ClientError> {
746            let mut addrs = (self.hostname.as_ref(), self.port)
747                .to_socket_addrs()
748                .map_err(|_| ClientError::Resolve(self.dest()))?;
749            let addr = addrs.next().ok_or(ClientError::Resolve(self.dest()))?;
750            let stream = TcpStream::connect(&addr).await.map_err(|err| {
751                ClientError::Connect(self.dest(), err, format!("cannot connect to {}", addr))
752            })?;
753            // Always disable Nagle algorithm, so it doesn't hurt the latency of small messages.
754            stream.set_nodelay(true).map_err(|err| {
755                ClientError::Connect(
756                    self.dest(),
757                    err,
758                    "cannot disables Nagle algorithm".to_string(),
759                )
760            })?;
761            let (connector, domain_name) = tls_connector_config(&self.hostname).map_err(|err| {
762                ClientError::Connect(
763                    self.dest(),
764                    io::Error::other(err.to_string()),
765                    format!("cannot config tls connector for addr {}", addr),
766                )
767            })?;
768            connector
769                .connect(domain_name.clone(), stream)
770                .await
771                .map_err(|err| {
772                    ClientError::Connect(
773                        self.dest(),
774                        err,
775                        format!("cannot establish TLS connection to {:?}", domain_name),
776                    )
777                })
778        }

791    /// Serve the given address. If port 0 is provided in a Host address,
792    /// a dynamic port will be resolved and is available in the returned ChannelAddr.
793    /// For Host addresses, binds to all resolved socket addresses.
794    pub fn serve<M: RemoteMessage>(
795        addr: MetaTlsAddr,
796    ) -> Result<(ChannelAddr, NetRx<M>), ServerError> {
797        match addr {
798            MetaTlsAddr::Host { hostname, port } => {
799                // Resolve all addresses for the hostname
800                let addrs: Vec<SocketAddr> = (hostname.as_ref(), port)
801                    .to_socket_addrs()
802                    .map_err(|err| {
803                        ServerError::Resolve(
804                            ChannelAddr::MetaTls(MetaTlsAddr::Host {
805                                hostname: hostname.clone(),
806                                port,
807                            }),
808                            err,
809                        )
810                    })?
811                    .collect();
812
813                if addrs.is_empty() {
814                    return Err(ServerError::Resolve(
815                        ChannelAddr::MetaTls(MetaTlsAddr::Host { hostname, port }),
816                        io::Error::other("no available socket addr"),
817                    ));
818                }
819
820                let channel_addr = ChannelAddr::MetaTls(MetaTlsAddr::Host {
821                    hostname: hostname.clone(),
822                    port,
823                });
824
825                // Bind to all resolved addresses
826                let std_listener = std::net::TcpListener::bind(&addrs[..])
827                    .map_err(|err| ServerError::Listen(channel_addr.clone(), err))?;
828                std_listener
829                    .set_nonblocking(true)
830                    .map_err(|e| ServerError::Listen(channel_addr.clone(), e))?;
831                let listener = TcpListener::from_std(std_listener)
832                    .map_err(|e| ServerError::Listen(channel_addr.clone(), e))?;
833
834                let local_addr = listener
835                    .local_addr()
836                    .map_err(|err| ServerError::Resolve(channel_addr, err))?;
837                super::serve(
838                    listener,
839                    ChannelAddr::MetaTls(MetaTlsAddr::Host {
840                        hostname,
841                        port: local_addr.port(),
842                    }),
843                    true,
844                )
845            }
846            MetaTlsAddr::Socket(socket_addr) => {
847                let channel_addr = ChannelAddr::MetaTls(MetaTlsAddr::Socket(socket_addr));
848
849                // Bind directly to the socket address
850                let std_listener = std::net::TcpListener::bind(socket_addr)
851                    .map_err(|err| ServerError::Listen(channel_addr.clone(), err))?;
852                std_listener
853                    .set_nonblocking(true)
854                    .map_err(|e| ServerError::Listen(channel_addr.clone(), e))?;
855                let listener = TcpListener::from_std(std_listener)
856                    .map_err(|e| ServerError::Listen(channel_addr.clone(), e))?;
857
858                let local_addr = listener
859                    .local_addr()
860                    .map_err(|err| ServerError::Resolve(channel_addr, err))?;
861                super::serve(
862                    listener,
863                    ChannelAddr::MetaTls(MetaTlsAddr::Socket(local_addr)),
864                    true,
865                )
866            }
867        }
868    }

1141    #[derive(Clone, Debug, Default)]
1142    struct NetworkFlakiness {
1143        // A tuple of:
1144        //   1. the probability of a network failure when sending a message.
1145        //   2. the max number of disconnections allowed.
1146        //   3. the minimum duration between disconnections.
1147        //
1148        //   2 and 3 are useful to prevent frequent disconnections leading to
1149        //   unacked messages being sent repeatedly.
1150        disconnect_params: Option<(f64, u64, Duration)>,
1151        // The max possible latency when sending a message. The actual latency
1152        // is randomly generated between 0 and max_latency.
1153        latency_range: Option<(Duration, Duration)>,
1154    }
1155
1156    impl NetworkFlakiness {
1157        // Calculate whether to disconnect
1158        async fn should_disconnect(
1159            &self,
1160            rng: &mut impl rand::Rng,
1161            disconnected_count: u64,
1162            prev_diconnected_at: &RwLock<Instant>,
1163        ) -> bool {
1164            let Some((prob, max_disconnects, duration)) = &self.disconnect_params else {
1165                return false;
1166            };
1167
1168            let disconnected_at = prev_diconnected_at.read().unwrap();
1169            if disconnected_at.elapsed() > *duration && disconnected_count < *max_disconnects {
1170                rng.gen_bool(*prob)
1171            } else {
1172                false
1173            }
1174        }
1175    }

1300                mut disconnect_signal: watch::Receiver<()>,
1301                network_flakiness: NetworkFlakiness,
1302                disconnected_count: Arc<AtomicU64>,
1303                prev_diconnected_at: Arc<RwLock<Instant>>,
1304                mut reader: FrameReader<ReadHalf<DuplexStream>>,
1305                mut writer: WriteHalf<DuplexStream>,
1306                // Used by client and server tokio tasks to coordinate
1307                // stopping together.
1308                task_coordination_token: CancellationToken,
1309                debug_log_sampling_rate: Option<u64>,
1310                // Whether the relayed message is from client to
1311                // server.
1312                is_from_client: bool,
1313            ) {
1314                // Used to simulate latency. Briefly, messages are
1315                // buffered in the queue and wait for the expected
1316                // latency elapse.
1317                async fn wait_for_latency_elapse(
1318                    queue: &VecDeque<(Bytes, Instant)>,
1319                    network_flakiness: &NetworkFlakiness,
1320                    rng: &mut impl rand::Rng,
1321                ) {
1322                    if let Some((min, max)) = network_flakiness.latency_range {
1323                        let diff = max.abs_diff(min);
1324                        let factor = rng.gen_range(0.0..=1.0);
1325                        let latency = min + diff.mul_f64(factor);
1326                        RealClock
1327                            .sleep_until(queue.front().unwrap().1 + latency)
1328                            .await;
1329                    }
1330                }
1331
1332                let mut rng = rand::rngs::SmallRng::from_entropy();
1333                let mut queue: VecDeque<(Bytes, Instant)> = VecDeque::new();
1334                let mut send_count = 0u64;
1335
1336                loop {
1337                    tokio::select! {
1338                        read_res = reader.next() => {
1339                            match read_res {
1340                                Ok(Some(data)) => {
1341                                    queue.push_back((data, RealClock.now()));
1342                                }
1343                                Ok(None) | Err(_) => {
1344                                        tracing::debug!("The upstream is closed or dropped. MockLink disconnects");
1345                                        break;
1346                                }
1347                            }
1348                        }
1349                        _ = wait_for_latency_elapse(&queue, &network_flakiness, &mut rng), if !queue.is_empty() => {
1350                            let count = disconnected_count.load(Ordering::Relaxed);
1351                            if network_flakiness.should_disconnect(&mut rng, count, &prev_diconnected_at).await {
1352                                tracing::debug!("MockLink disconnects");
1353                                disconnected_count.fetch_add(1, Ordering::Relaxed);
1354
1355                                metrics::CHANNEL_RECONNECTIONS.add(
1356                                    1,
1357                                    hyperactor_telemetry::kv_pairs!(
1358                                        "transport" => "mock",
1359                                        "reason" => "network_flakiness",
1360                                    ),
1361                                );
1362
1363                                let mut w = prev_diconnected_at.write().unwrap();
1364                                *w = RealClock.now();
1365                                break;
1366                            }
1367                            let data = queue.pop_front().unwrap().0;
1368                            let is_sampled = debug_log_sampling_rate.is_some_and(|sample_rate| send_count % sample_rate == 1);
1369                            if is_sampled {
1370                                if is_from_client {
1371                                    if let Ok(Frame::Message(_seq, _msg)) = bincode::deserialize::<Frame<M>>(&data) {
1372                                        tracing::debug!("MockLink relays a msg from client. msg type: {}", std::any::type_name::<M>());

1374                                } else {
1375                                    let result = deserialize_response(data.clone());
1376                                    if let Ok(NetRxResponse::Ack(seq)) = result {
1377                                        tracing::debug!("MockLink relays an ack from server. seq: {}", seq);
1378                                    }
1379                                }
1380                            }
1381                            let mut fw  = FrameWrite::new(writer, data, hyperactor_config::global::get(config::CODEC_MAX_FRAME_LENGTH)).unwrap();
1382                            if fw.send().await.is_err() {
1383                                break;
1384                            }
1385                            writer = fw.complete();
1386                            send_count += 1;
1387                        }
1388                        _ = task_coordination_token.cancelled() => break,
1389
1390                        changed = disconnect_signal.changed() => {
1391                            tracing::debug!("MockLink disconnects per disconnect_signal {:?}", changed);
1392                            break;
1393                        }
1394                    }
1395                }
1396
1397                task_coordination_token.cancel();
1398            }
1399
1400            let (server, server_relay) = tokio::io::duplex(self.buffer_size);
1401            let (client, client_relay) = tokio::io::duplex(self.buffer_size);
1402
1403            let (server_r, server_writer) = tokio::io::split(server_relay);
1404            let (client_r, client_writer) = tokio::io::split(client_relay);
1405
1406            let max_len = hyperactor_config::global::get(config::CODEC_MAX_FRAME_LENGTH);
1407            let server_reader = FrameReader::new(server_r, max_len);
1408            let client_reader = FrameReader::new(client_r, max_len);
1409
1410            let task_coordination_token = CancellationToken::new();
1411            let _server_relay_task_handle = tokio::spawn(relay_message::<M>(
1412                self.disconnect_signal.subscribe(),
1413                self.network_flakiness.clone(),
1414                self.disconnected_count.clone(),
1415                self.prev_diconnected_at.clone(),
1416                server_reader,
1417                client_writer,
1418                task_coordination_token.clone(),
1419                self.debug_log_sampling_rate.clone(),
1420                /*is_from_client*/ false,

1424                self.network_flakiness.clone(),
