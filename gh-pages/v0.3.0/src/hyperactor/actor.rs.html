62/// An Actor is an independent, asynchronous thread of execution. Each
63/// actor instance has a mailbox, whose messages are delivered through
64/// the method [`Actor::handle`].
65///
66/// Actors communicate with each other by way of message passing.
67/// Actors are assumed to be _deterministic_: that is, the state of an
68/// actor is determined by the set (and order) of messages it receives.
69#[async_trait]
70pub trait Actor: Sized + Send + 'static {
71    /// Initialize the actor, after the runtime has been fully initialized.
72    /// Init thus provides a mechanism by which an actor can reliably and always
73    /// receive some initial event that can be used to kick off further
74    /// (potentially delayed) processing.
75    async fn init(&mut self, _this: &Instance<Self>) -> Result<(), anyhow::Error> {
76        // Default implementation: no init.
77        Ok(())
78    }
79
80    /// Cleanup things used by this actor before shutting down. Notably this function
81    /// is async and allows more complex cleanup. Simpler cleanup can be handled
82    /// by the impl Drop for this Actor.
83    /// If err is not None, it is the error that this actor is failing with. Any
84    /// errors returned by this function will be logged and ignored.
85    /// If err is None, any errors returned by this function will be propagated
86    /// as an ActorError.
87    /// This function is not called if there is a panic in the actor, as the
88    /// actor may be in an indeterminate state. It is also not called if the
89    /// process is killed, there is no atexit handler or signal handler.
90    async fn cleanup(
91        &mut self,
92        _this: &Instance<Self>,
93        _err: Option<&ActorError>,
94    ) -> Result<(), anyhow::Error> {
95        // Default implementation: no cleanup.
96        Ok(())
97    }
98
99    /// Spawn a child actor, given a spawning capability (usually given by [`Instance`]).
100    /// The spawned actor will be supervised by the parent (spawning) actor.
101    fn spawn(self, cx: &impl context::Actor) -> anyhow::Result<ActorHandle<Self>> {
102        cx.instance().spawn(self)
103    }
104
105    /// Spawns this actor in a detached state, handling its messages
106    /// in a background task. The returned handle is used to control
107    /// the actor's lifecycle and to interact with it.
108    ///
109    /// Actors spawned through `spawn_detached` are not attached to a supervision
110    /// hierarchy, and not managed by a [`Proc`].
111    fn spawn_detached(self) -> Result<ActorHandle<Self>, anyhow::Error> {
112        Proc::local().spawn("anon", self)
113    }
114
115    /// This method is used by the runtime to spawn the actor server. It can be
116    /// used by actors that require customized runtime setups
117    /// (e.g., dedicated actor threads), or want to use a custom tokio runtime.
118    #[hyperactor::instrument_infallible]
119    fn spawn_server_task<F>(future: F) -> JoinHandle<F::Output>
120    where
121        F: Future + Send + 'static,
122        F::Output: Send + 'static,
123    {
124        tokio::spawn(future)
125    }
126
127    /// Handle actor supervision event. Return `Ok(true)`` if the event is handled here.
128    async fn handle_supervision_event(
129        &mut self,
130        _this: &Instance<Self>,
131        _event: &ActorSupervisionEvent,
132    ) -> Result<bool, anyhow::Error> {
133        // By default, the supervision event is not handled, caller is expected to bubble it up.
134        Ok(false)
135    }
136
137    /// Default undeliverable message handling behavior.
138    async fn handle_undeliverable_message(
139        &mut self,
140        cx: &Instance<Self>,
141        envelope: Undeliverable<MessageEnvelope>,
142    ) -> Result<(), anyhow::Error> {
143        handle_undeliverable_message(cx, envelope)
144    }
145
146    /// If overridden, we will use this name in place of the
147    /// ActorId for talking about this actor in supervision error
148    /// messages.
149    fn display_name(&self) -> Option<String> {
150        None
151    }
152}
153
154/// Default implementation of [`Actor::handle_undeliverable_message`]. Defined
155/// as a free function so that `Actor` implementations that override
156/// [`Actor::handle_undeliverable_message`] can fallback to this default.
157pub fn handle_undeliverable_message<A: Actor>(
158    cx: &Instance<A>,
159    Undeliverable(envelope): Undeliverable<MessageEnvelope>,
160) -> Result<(), anyhow::Error> {
161    assert_eq!(envelope.sender(), cx.self_id());
162
163    anyhow::bail!(UndeliverableMessageError::DeliveryFailure { envelope });
164}

179/// A Handler allows an actor to handle a specific message type.
180#[async_trait]
181pub trait Handler<M>: Actor {
182    /// Handle the next M-typed message.
183    async fn handle(&mut self, cx: &Context<Self>, message: M) -> Result<(), anyhow::Error>;
184}
185
186/// We provide this handler to indicate that actors can handle the [`Signal`] message.
187/// Its actual handler is implemented by the runtime.
188#[async_trait]
189impl<A: Actor> Handler<Signal> for A {
190    async fn handle(&mut self, _cx: &Context<Self>, _message: Signal) -> Result<(), anyhow::Error> {
191        unimplemented!("signal handler should not be called directly")
192    }
193}

195/// This handler provides a default behavior when a message sent by
196/// the actor to another is returned due to delivery failure.
197#[async_trait]
198impl<A: Actor> Handler<Undeliverable<MessageEnvelope>> for A {
199    async fn handle(
200        &mut self,
201        cx: &Context<Self>,
202        message: Undeliverable<MessageEnvelope>,
203    ) -> Result<(), anyhow::Error> {
204        self.handle_undeliverable_message(cx, message).await
205    }
206}

208/// This handler enables actors to unbind the [IndexedErasedUnbound]
209/// message, and forward the result to corresponding handler.
210#[async_trait]
211impl<A, M> Handler<IndexedErasedUnbound<M>> for A
212where
213    A: Handler<M>,
214    M: Castable,
215{
216    async fn handle(
217        &mut self,
218        cx: &Context<Self>,
219        msg: IndexedErasedUnbound<M>,
220    ) -> anyhow::Result<()> {
221        let message = msg.downcast()?.bind()?;
222        Handler::handle(self, cx, message).await
223    }
224}

226/// An `Actor` that can be spawned remotely.
227///
228/// Bounds explained:
229/// - `Actor`: only actors may be remotely spawned.
230/// - `Referable`: marks the type as eligible for typed remote
231///   references (`ActorRef<A>`); required because remote spawn
232///   ultimately hands back an `ActorId` that higher-level APIs may
233///   re-type as `ActorRef<A>`.
234/// - `Binds<Self>`: lets the runtime wire this actor's message ports
235///   when it is spawned (the blanket impl calls `handle.bind::<Self>()`).
236///
237/// `gspawn` is a type-erased entry point used by the remote
238/// spawn/registry machinery. It takes serialized params and returns
239/// the new actor's `ActorId`; application code shouldn't call it
240/// directly.
241#[async_trait]
242pub trait RemoteSpawn: Actor + Referable + Binds<Self> {
243    /// The type of parameters used to instantiate the actor remotely.
244    type Params: RemoteMessage;
245
246    /// Creates a new actor instance given its instantiation parameters.
247    async fn new(params: Self::Params) -> anyhow::Result<Self>;
248
249    /// A type-erased entry point to spawn this actor. This is
250    /// primarily used by hyperactor's remote actor registration
251    /// mechanism.
252    // TODO: consider making this 'private' -- by moving it into a non-public trait as in [`cap`].
253    fn gspawn(
254        proc: &Proc,
255        name: &str,
256        serialized_params: Data,
257    ) -> Pin<Box<dyn Future<Output = Result<ActorId, anyhow::Error>> + Send>> {
258        let proc = proc.clone();
259        let name = name.to_string();
260        Box::pin(async move {
261            let params = bincode::deserialize(&serialized_params)?;
262            let actor = Self::new(params).await?;
263            let handle = proc.spawn(&name, actor)?;
264            // We return only the ActorId, not a typed ActorRef.
265            // Callers that hold this ID can interact with the actor
266            // only via the serialized/opaque messaging path, which
267            // makes it safe to export across process boundaries.
268            //
269            // Note: the actor itself is still `A`-typed here; we
270            // merely restrict the *capability* we hand out to an
271            // untyped identifier.
272            //
273            // This will be replaced by a proper export/registry
274            // mechanism.
275            Ok(handle.bind::<Self>().actor_id)
276        })
277    }
278
279    /// The type ID of this actor.
280    fn get_type_id() -> TypeId {
281        TypeId::of::<Self>()
282    }
283}

285/// If an actor implements Default, we use this as the
286/// `RemoteSpawn` implementation, too.
287#[async_trait]
288impl<A: Actor + Referable + Binds<Self> + Default> RemoteSpawn for A {
289    type Params = ();
290
291    async fn new(_params: Self::Params) -> anyhow::Result<Self> {
292        Ok(Default::default())
293    }

311/// Errors that occur while serving actors. Each error is associated
312/// with the ID of the actor being served.
313#[derive(Debug)]
314pub struct ActorError {
315    /// The ActorId for the actor that generated this error.
316    pub actor_id: Box<ActorId>,
317    /// The kind of error that occurred.
318    pub kind: Box<ActorErrorKind>,
319}
320
321/// The kinds of actor serving errors.
322#[derive(thiserror::Error, Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
323pub enum ActorErrorKind {
324    /// Generic error with a formatted message.
325    #[error("{0}")]
326    Generic(String),
327
328    /// An error that occurred while trying to handle a supervision event.
329    #[error("{0} while handling {1}")]
330    ErrorDuringHandlingSupervision(String, Box<ActorSupervisionEvent>),
331
332    /// The actor did not attempt to handle
333    #[error("{0}")]
334    UnhandledSupervisionEvent(Box<ActorSupervisionEvent>),
335
336    /// The actor was explicitly aborted with the provided reason.
337    #[error("actor explicitly aborted due to: {0}")]
338    Aborted(String),
339}
340
341impl ActorErrorKind {
342    /// Error while processing actor, i.e., returned by the actor's
343    /// processing method.
344    pub fn processing(err: anyhow::Error) -> Self {
345        // Unbox err from the anyhow err. Check if it is an ActorErrorKind object.
346        // If it is directly use it as the new ActorError's ActorErrorKind.
347        // This lets us directly pass the ActorErrorKind::UnhandledSupervisionEvent
348        // up the handling infrastructure.
349        err.downcast::<ActorErrorKind>()
350            .unwrap_or_else(|err| Self::Generic(err.to_string()))
351    }
352
353    /// Unwound stracktrace of a panic.
354    pub fn panic(err: anyhow::Error) -> Self {
355        Self::Generic(format!("panic: {}", err))
356    }
357
358    /// Error during actor initialization.
359    pub fn init(err: anyhow::Error) -> Self {
360        Self::Generic(format!("initialization error: {}", err))
361    }
362
363    /// Error during actor cleanup.
364    pub fn cleanup(err: anyhow::Error) -> Self {
365        Self::Generic(format!("cleanup error: {}", err))
366    }
367
368    /// An underlying mailbox error.
369    pub fn mailbox(err: MailboxError) -> Self {
370        Self::Generic(err.to_string())
371    }
372
373    /// An underlying mailbox sender error.
374    pub fn mailbox_sender(err: MailboxSenderError) -> Self {
375        Self::Generic(err.to_string())
376    }
377
378    /// An underlying checkpoint error.
379    pub fn checkpoint(err: CheckpointError) -> Self {
380        Self::Generic(format!("checkpoint error: {}", err))
381    }
382
383    /// An underlying message log error.
384    pub fn message_log(err: MessageLogError) -> Self {
385        Self::Generic(format!("message log error: {}", err))
386    }
387
388    /// The actor's state could not be determined.
389    pub fn indeterminate_state() -> Self {
390        Self::Generic("actor is in an indeterminate state".to_string())
391    }
392}

397        Self {
398            actor_id: Box::new(actor_id.clone()),
399            kind: Box::new(kind),
400        }
401    }
402}

435/// A collection of signals to control the behavior of the actor.
436/// Signals are internal runtime control plane messages and should not be
437/// sent outside of the runtime.
438///
439/// These messages are not handled directly by actors; instead, the runtime
440/// handles the various signals.
441#[derive(Clone, Debug, Serialize, Deserialize, typeuri::Named)]
442pub enum Signal {
443    /// Stop the actor, after draining messages.
444    DrainAndStop(String),
445
446    /// Stop the actor immediately.
447    Stop(String),
448
449    /// The direct child with the given PID was stopped.
450    ChildStopped(Index),
451
452    /// Abort the actor. This will exit the actor loop with an error,
453    /// causing a supervision event to propagate up the supervision
454    /// hierarchy.
455    Abort(String),
456}
457wirevalue::register_type!(Signal);
458
459impl fmt::Display for Signal {
460    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
461        match self {
462            Signal::DrainAndStop(reason) => write!(f, "DrainAndStop({})", reason),
463            Signal::Stop(reason) => write!(f, "Stop({})", reason),
464            Signal::ChildStopped(index) => write!(f, "ChildStopped({})", index),
465            Signal::Abort(reason) => write!(f, "Abort({})", reason),
466        }
467    }
468}

470/// The runtime status of an actor.
471#[derive(
472    Debug,
473    Serialize,
474    Deserialize,
475    PartialEq,
476    Eq,
477    Clone,
478    typeuri::Named,
479    EnumAsInner
480)]
481pub enum ActorStatus {
482    /// The actor status is unknown.
483    Unknown,
484    /// The actor was created, but not yet started.
485    Created,
486    /// The actor is initializing. It is not yet ready to receive messages.
487    Initializing,
488    /// The actor is in "client" state: the user is managing the actor's
489    /// mailboxes manually.
490    Client,
491    /// The actor is ready to receive messages, but is currently idle.
492    Idle,
493    /// The actor has been processing a message, beginning at the specified
494    /// instant. The message handler and arm is included.
495    /// TODO: we shoudl use interned representations here, so we don't copy
496    /// strings willy-nilly.
497    Processing(SystemTime, Option<(String, Option<String>)>),
498    /// The actor has been saving its state.
499    Saving(SystemTime),
500    /// The actor has been loading its state.
501    Loading(SystemTime),
502    /// The actor is stopping. It is draining messages.
503    Stopping,
504    /// The actor is stopped. It is no longer processing messages.
505    Stopped,
506    /// The actor failed with the provided actor error.
507    Failed(ActorErrorKind),
508}
509
510impl ActorStatus {
511    /// Tells whether the status is a terminal state.
512    pub fn is_terminal(&self) -> bool {
513        self.is_stopped() || self.is_failed()
514    }
515
516    /// Create a generic failure status with the provided error message.
517    pub fn generic_failure(message: impl Into<String>) -> Self {
518        Self::Failed(ActorErrorKind::Generic(message.into()))
519    }
520
521    fn span_string(&self) -> &'static str {
522        self.arm().unwrap_or_default()
523    }
524}
525
526impl fmt::Display for ActorStatus {
527    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
528        match self {
529            Self::Unknown => write!(f, "unknown"),
530            Self::Created => write!(f, "created"),
531            Self::Initializing => write!(f, "initializing"),
532            Self::Client => write!(f, "client"),
533            Self::Idle => write!(f, "idle"),
534            Self::Processing(instant, None) => {
535                write!(
536                    f,
537                    "processing for {}ms",
538                    RealClock
539                        .system_time_now()
540                        .duration_since(*instant)
541                        .unwrap_or_default()
542                        .as_millis()
543                )
544            }
545            Self::Processing(instant, Some((handler, None))) => {
546                write!(
547                    f,
548                    "{}: processing for {}ms",
549                    handler,
550                    RealClock
551                        .system_time_now()
552                        .duration_since(*instant)
553                        .unwrap_or_default()
554                        .as_millis()
555                )
556            }
557            Self::Processing(instant, Some((handler, Some(arm)))) => {
558                write!(
559                    f,
560                    "{},{}: processing for {}ms",
561                    handler,
562                    arm,
563                    RealClock
564                        .system_time_now()
565                        .duration_since(*instant)
566                        .unwrap_or_default()
567                        .as_millis()
568                )
569            }
570            Self::Saving(instant) => {
571                write!(
572                    f,
573                    "saving for {}ms",
574                    RealClock
575                        .system_time_now()
576                        .duration_since(*instant)
577                        .unwrap_or_default()
578                        .as_millis()
579                )
580            }
581            Self::Loading(instant) => {
582                write!(
583                    f,
584                    "loading for {}ms",
585                    RealClock
586                        .system_time_now()
587                        .duration_since(*instant)
588                        .unwrap_or_default()
589                        .as_millis()
590                )
591            }
592            Self::Stopping => write!(f, "stopping"),
593            Self::Stopped => write!(f, "stopped"),
594            Self::Failed(err) => write!(f, "failed: {}", err),
595        }
596    }
597}

599/// ActorHandles represent a (local) serving actor. It is used to access
600/// its messaging and signal ports, as well as to synchronize with its
601/// lifecycle (e.g., providing joins).  Once dropped, the handle is
602/// detached from the underlying actor instance, and there is no longer
603/// any way to join it.
604///
605/// Correspondingly, [`crate::ActorRef`]s refer to (possibly) remote
606/// actors.
607pub struct ActorHandle<A: Actor> {
608    cell: InstanceCell,
609    ports: Arc<Ports<A>>,
610}
611
612/// A handle to a running (local) actor.
613impl<A: Actor> ActorHandle<A> {
614    pub(crate) fn new(cell: InstanceCell, ports: Arc<Ports<A>>) -> Self {
615        Self { cell, ports }
616    }
617
618    /// The actor's cell. Used primarily for testing.
619    /// TODO: this should not be a public API.
620    pub(crate) fn cell(&self) -> &InstanceCell {
621        &self.cell
622    }
623
624    /// The [`ActorId`] of the actor represented by this handle.
625    pub fn actor_id(&self) -> &ActorId {
626        self.cell.actor_id()
627    }
628
629    /// Signal the actor to drain its current messages and then stop.
630    pub fn drain_and_stop(&self, reason: &str) -> Result<(), ActorError> {
631        tracing::info!("ActorHandle::drain_and_stop called: {}", self.actor_id());
632        self.cell.signal(Signal::DrainAndStop(reason.to_string()))
633    }
634
635    /// A watch that observes the lifecycle state of the actor.
636    pub fn status(&self) -> watch::Receiver<ActorStatus> {
637        self.cell.status().clone()
638    }
639
640    /// Send a message to the actor. Messages sent through the handle
641    /// are always queued in process, and do not require serialization.
642    pub fn send<M: Message>(
643        &self,
644        cx: &impl context::Actor,
645        message: M,
646    ) -> Result<(), MailboxSenderError>
647    where
648        A: Handler<M>,
649    {
650        self.ports.get().send(cx, message)
651    }
652
653    /// Return a port for the provided message type handled by the actor.
654    pub fn port<M: Message>(&self) -> PortHandle<M>
655    where
656        A: Handler<M>,
657    {
658        self.ports.get()
659    }
660
661    /// TEMPORARY: bind...
662    /// TODO: we shoudl also have a default binding(?)
663    pub fn bind<R: Binds<A>>(&self) -> ActorRef<R> {
664        self.cell.bind(self.ports.as_ref())
665    }
666}
667
668/// IntoFuture allows users to await the handle to join it. The future
669/// resolves when the actor itself has stopped processing messages.
670/// The future resolves to the actor's final status.
671impl<A: Actor> IntoFuture for ActorHandle<A> {
672    type Output = ActorStatus;
673    type IntoFuture = BoxFuture<'static, Self::Output>;
674
675    fn into_future(self) -> Self::IntoFuture {
676        let future = async move {
677            let mut status_receiver = self.cell.status().clone();
678            let result = status_receiver.wait_for(ActorStatus::is_terminal).await;
679            match result {
680                Err(_) => ActorStatus::Unknown,
681                Ok(status) => status.clone(),
682            }
683        };
684
685        future.boxed()
686    }
687}

689impl<A: Actor> Debug for ActorHandle<A> {
690    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> Result<(), fmt::Error> {
691        f.debug_struct("ActorHandle").field("cell", &"..").finish()
692    }
693}

695impl<A: Actor> Clone for ActorHandle<A> {
696    fn clone(&self) -> Self {
697        Self {
698            cell: self.cell.clone(),
699            ports: self.ports.clone(),
700        }
701    }

704/// `Referable` is a marker trait for types that can appear as
705/// remote references across process boundaries.
706///
707/// It is not limited to concrete [`Actor`] implementations. For
708/// example, façade types generated by [`behavior!`] implement
709/// `Referable` so that you can hand out restricted or stable APIs
710/// while still using the same remote messaging machinery.
711///
712/// Implementing this trait means the type can be identified (`Named`)
713/// so the runtime knows what it is.
714///
715///  In contrast, [`RemoteSpawn`] is the trait that marks *actors*
716/// that can actually be **spawned remotely**. A behavior may be a
717/// `Referable` but is never a `RemoteSpawn`.
718pub trait Referable: Named {}
719
720/// Binds determines how an actor's ports are bound to a specific
721/// reference type.
722pub trait Binds<A: Actor>: Referable {
723    /// Bind ports in this actor.
724    fn bind(ports: &Ports<A>);
725}
726
727/// Handles is a marker trait specifying that message type [`M`]
728/// is handled by a specific actor type.
729pub trait RemoteHandles<M: RemoteMessage>: Referable {}
730
731/// Check if the actor behaves-as the a given behavior (defined by [`behavior!`]).
732///
733/// ```
734/// # use serde::Serialize;
735/// # use serde::Deserialize;
736/// # use typeuri::Named;
737/// # use hyperactor::Actor;
738///
739/// // First, define a behavior, based on handling a single message type `()`.
740/// hyperactor::behavior!(UnitBehavior, ());
741///
742/// #[derive(Debug, Default)]
743/// struct MyActor;
744///
745/// impl Actor for MyActor {}
746///
747/// #[async_trait::async_trait]
748/// impl hyperactor::Handler<()> for MyActor {
749///     async fn handle(
750///         &mut self,
751///         _cx: &hyperactor::Context<Self>,
752///         _message: (),
753///     ) -> Result<(), anyhow::Error> {
754///         // no-op
755///         Ok(())
756///     }
757/// }
758///
759/// hyperactor::assert_behaves!(MyActor as UnitBehavior);
760/// ```
761#[macro_export]
762macro_rules! assert_behaves {
763    ($ty:ty as $behavior:ty) => {
764        const _: fn() = || {
765            fn check<B: hyperactor::actor::Binds<$ty>>() {}
766            check::<$behavior>();
767        };
768    };
769}
