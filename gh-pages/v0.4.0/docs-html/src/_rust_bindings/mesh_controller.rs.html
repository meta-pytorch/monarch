98impl _Controller {
99    #[new]
100    fn new(py: Python, client: PyInstance, py_proc_mesh: &Bound<'_, PyAny>) -> PyResult<Self> {
101        let proc_mesh = py_proc_mesh.downcast::<PyProcMesh>()?.borrow().mesh_ref()?;
102
103        // Build rank map from proc ids to ranks.
104        let rank_map: HashMap<reference::ProcId, usize> = proc_mesh
105            .iter()
106            .map(|(point, proc)| (proc.proc_id().clone(), point.rank()))
107            .collect();
108
109        let region = Ranked::region(&proc_mesh);
110        let slice = region.slice();
111        if !slice.is_contiguous() || slice.offset() != 0 {
112            return Err(PyValueError::new_err(
113                "NYI: proc mesh for workers must be contiguous and start at offset 0",
114            ));
115        }
116
117        let id = NEXT_ID.fetch_add(1, atomic::Ordering::Relaxed);
118        let controller_handle: Arc<Mutex<ActorHandle<MeshControllerActor>>> =
119            signal_safe_block_on(py, async move {
120                let controller_handle = client.spawn(
121                    MeshControllerActor::new(MeshControllerActorParams {
122                        proc_mesh_ref: proc_mesh,
123                        id,
124                        rank_map,
125                    })
126                    .await,
127                )?;
128                Ok::<_, anyhow::Error>(Arc::new(Mutex::new(controller_handle)))
129            })??;
130
131        Ok(Self {
132            controller_handle,
133            // note that 0 is the _pid_ of the broker, which will be 0 for
134            // top-level spawned actors.
135            // todo: plumb these through as proper actor mesh refs
136            broker_id: (format!("tensor_engine_brokers_{}", id), 0),
137        })
138    }
139
140    #[getter]
141    fn broker_id(&self) -> (String, usize) {
142        self.broker_id.clone()
143    }

145    #[pyo3(signature = (instance, seq, defs, uses, response_port, tracebacks))]
146    fn _node<'py>(
147        &mut self,
148        instance: &PyInstance,
149        seq: u64,
150        defs: Bound<'py, PyAny>,
151        uses: Bound<'py, PyAny>,
152        response_port: Option<(PyPortId, PySlice)>,
153        tracebacks: Py<PyAny>,
154    ) -> PyResult<()> {
155        let response_port: Option<PortInfo> = response_port.map(|(port, ranks)| PortInfo {
156            port: reference::PortRef::attest(port.into()),
157            ranks: ranks.into(),
158        });
159        let msg = ClientToControllerMessage::Node {
160            seq: seq.into(),
161            defs: defs
162                .try_iter()?
163                .map(|x| Ref::from_py_object(&x?))
164                .collect::<PyResult<Vec<Ref>>>()?,
165            uses: uses
166                .try_iter()?
167                .map(|x| Ref::from_py_object(&x?))
168                .collect::<PyResult<Vec<Ref>>>()?,
169            tracebacks,
170            response_port,
171        };
172        self.controller_handle
173            .blocking_lock()
174            .send(instance.deref(), msg)
175            .map_err(to_py_error)
176    }
177
178    fn _drop_refs(&mut self, instance: &PyInstance, refs: Vec<Ref>) -> PyResult<()> {
179        self.controller_handle
180            .blocking_lock()
181            .send(
182                instance.deref(),
183                ClientToControllerMessage::DropRefs { refs },
184            )
185            .map_err(to_py_error)
186    }
187
188    fn _sync_at_exit(&mut self, instance: &PyInstance, port: PyPortId) -> PyResult<()> {
189        self.controller_handle
190            .blocking_lock()
191            .send(
192                instance.deref(),
193                ClientToControllerMessage::SyncAtExit {
194                    port: reference::PortRef::attest(port.into()),
195                },
196            )
197            .map_err(to_py_error)
198    }
199
200    fn _send<'py>(
201        &mut self,
202        instance: &PyInstance,
203        ranks: Bound<'py, PyAny>,
204        message: Bound<'py, PyAny>,
205    ) -> PyResult<()> {
206        let slices = if let Ok(slice) = ranks.extract::<PySlice>() {
207            vec![slice.into()]
208        } else {
209            let slices = ranks.extract::<Vec<PySlice>>()?;
210            slices.iter().map(|x| x.into()).collect::<Vec<Slice>>()
211        };
212        let message: WorkerMessage = convert(message)?;
213        self.controller_handle
214            .blocking_lock()
215            .send(
216                instance.deref(),
217                ClientToControllerMessage::Send { slices, message },
218            )
219            .map_err(to_py_error)
220    }
221
222    fn _drain_and_stop(&mut self, py: Python<'_>, instance: &PyInstance) -> PyResult<()> {
223        let (stop_worker_port, stop_worker_receiver) = instance.open_once_port();
224
225        self.controller_handle
226            .blocking_lock()
227            .send(
228                instance.deref(),
229                ClientToControllerMessage::StopWorkers {
230                    response_port: stop_worker_port,
231                },
232            )
233            .map_err(to_py_error)?;
234        signal_safe_block_on(py, async move { stop_worker_receiver.recv().await })?
235            .map_err(to_py_error)?
236            .map_err(PyRuntimeError::new_err)?;
237
238        // Get status watch before sending drain_and_stop so we can
239        // wait for the actor to fully terminate.
240        let mut status_rx = self.controller_handle.blocking_lock().status();
241
242        self.controller_handle
243            .blocking_lock()
244            .drain_and_stop("mesh controller shutdown")
245            .map_err(to_py_error)?;
246
247        // Wait for the MeshControllerActor to reach terminal status.
248        // This ensures all resources (mailbox, IPC sessions, etc.) are
249        // fully cleaned up before we return, preventing accumulation of
250        // stale state across rapid spawn/exit cycles.
251        signal_safe_block_on(py, async move {
252            let _ = status_rx.wait_for(ActorStatus::is_terminal).await;
253        })?;
254
255        Ok(())
256    }

259/// An invocation tracks a discrete node in the graph of operations executed by
260/// the worker based on instructions from the client.
261/// It is useful for tracking the dependencies of an operation and propagating
262/// failures. In the future this will be used with more data dependency tracking
263/// to support better failure handling.
264
265#[derive(Debug)]
266enum Status {
267    Errored {
268        exception: Arc<PythonMessage>,
269    },
270    Complete {},
271
272    /// When incomplete this holds this list of users of this invocation,
273    /// so a future error can be propagated to them.,
274    Incomplete {
275        users: HashMap<Seq, Arc<sync::Mutex<Invocation>>>,
276        results: Vec<PythonMessage>,
277    },
278}
279
280impl Status {
281    fn incomplete() -> Status {
282        Self::Incomplete {
283            users: HashMap::new(),
284            results: vec![],
285        }
286    }
287}
288#[derive(Debug)]
289struct Invocation {
290    /// The sequence number of the invocation. This should be unique and increasing across all
291    /// invocations.
292    seq: Seq,
293    status: Status,
294    /// Result reported to a future if this invocation was a fetch
295    /// Not all Invocations will be fetched so sometimes a Invocation will complete with
296    /// both result and error == None
297    response_port: Option<PortInfo>,
298    tracebacks: Py<PyAny>,
299}
300
301impl Invocation {
302    fn new(seq: Seq, tracebacks: Py<PyAny>, response_port: Option<PortInfo>) -> Self {
303        Self {
304            seq,
305            status: Status::incomplete(),
306            response_port,
307            tracebacks,
308        }
309    }
310
311    fn add_user(
312        &mut self,
313        sender: &impl context::Actor,
314        unreported_exception: &mut Option<Arc<PythonMessage>>,
315        user: Arc<sync::Mutex<Invocation>>,
316    ) -> Result<(), MailboxSenderError> {
317        match &mut self.status {
318            Status::Complete {} => {}
319            Status::Incomplete { users, .. } => {
320                let seq = user.lock().unwrap().seq;
321                users.insert(seq, user);
322            }
323            Status::Errored { exception } => {
324                user.lock().unwrap().set_exception(
325                    sender,
326                    unreported_exception,
327                    exception.clone(),
328                )?;
329            }
330        }
331        Ok(())
332    }
333
334    /// Invocation results can only go from valid to failed, or be
335    /// set if the invocation result is empty.
336    fn set_result(&mut self, result: PythonMessage) {
337        match &mut self.status {
338            Status::Incomplete { results, .. } => {
339                results.push(result);
340            }
341            Status::Errored { .. } => {}
342            Status::Complete {} => {
343                panic!("setting result on a complete seq");
344            }
345        }
346    }
347
348    fn complete(&mut self, sender: &impl context::Actor) -> Result<(), MailboxSenderError> {
349        let old_status = std::mem::replace(&mut self.status, Status::Complete {});
350        match old_status {
351            Status::Incomplete { results, .. } => match &self.response_port {
352                Some(PortInfo { port, ranks }) => {
353                    assert!(ranks.len() == results.iter().len());
354                    for result in results.into_iter() {
355                        port.send(sender, result)?;
356                    }
357                }
358                None => {}
359            },
360            _ => {
361                self.status = old_status;
362            }
363        }
364        Ok(())
365    }
366
367    /// Changes the status of this invocation to an Errored. If this invocation was
368    /// Incomplete, it may have users that will also become errored. This function
369    /// will return those users so the error can be propagated. It does not autmoatically
370    /// propagate the error to avoid deep recursive invocations.
371    fn set_exception(
372        &mut self,
373        sender: &impl context::Actor,
374        unreported_exception: &mut Option<Arc<PythonMessage>>,
375        exception: Arc<PythonMessage>,
376    ) -> Result<(), MailboxSenderError> {
377        let mut process =
378            |invocation: &mut Invocation, queue: &mut Vec<Arc<sync::Mutex<Invocation>>>| {
379                let err = Status::Errored {
380                    exception: exception.clone(),
381                };
382                let old_status = std::mem::replace(&mut invocation.status, err);
383                match old_status {
384                    Status::Incomplete { users, .. } => {
385                        match &invocation.response_port {
386                            Some(PortInfo { port, ranks }) => {
387                                *unreported_exception = None;
388                                for rank in ranks.iter() {
389                                    let msg = exception.as_ref().clone().into_rank(rank);
390                                    port.send(sender, msg)?;
391                                }
392                            }
393                            None => {}
394                        };
395                        queue.extend(users.into_values());
396                    }
397                    Status::Complete {} => {
398                        panic!("Complete invocation getting an exception set")
399                    }
400                    Status::Errored { .. } => invocation.status = old_status,
401                }
402                Ok(())
403            };
404        let mut queue = vec![];
405        let mut visited = HashSet::new();
406        process(self, &mut queue)?;
407        while let Some(invocation) = queue.pop() {
408            let mut invocation = invocation.lock().unwrap();
409            if !visited.insert(invocation.seq) {
410                continue;
411            };
412            process(invocation.deref_mut(), &mut queue)?;
413        }
414        Ok(())
415    }
416}

418/// The history of invocations sent by the client to be executed on the workers.
419/// This is used to track dependencies between invocations and to propagate exceptions.
420/// It purges history for completed invocations to avoid memory bloat.
421/// TODO: Revisit this setup around purging refs automatically once we start doing
422/// more complex data dependency tracking. We will want to be more aware of things like
423/// borrows, drops etc. directly.
424#[derive(Debug)]
425struct History {
426    /// The first incomplete Seq for each rank. This is used to determine which
427    /// Seqs are no longer relevant and can be purged from the history.
428    first_incomplete_seqs: MinVector<Seq>,
429    /// The minimum incomplete Seq across all ranks.
430    min_incomplete_seq: Seq,
431    /// A map of seq to the invocation that it represents for all seq >= min_incomplete_seq
432    inflight_invocations: HashMap<Seq, Arc<sync::Mutex<Invocation>>>,
433    /// A map of reference to the seq for the invocation that defines it. This is used to
434    /// compute dependencies between invocations.
435    invocation_for_ref: HashMap<Ref, Arc<sync::Mutex<Invocation>>>,
436    // no new sequence numbers should be below this bound. use for
437    // sanity checking.
438    seq_lower_bound: Seq,
439    unreported_exception: Option<Arc<PythonMessage>>,
440    exit_port: Option<reference::PortRef<PythonMessage>>,
441}

443/// A vector that keeps track of the minimum value.
444#[derive(Debug)]
445struct MinVector<T> {
446    data: Vec<T>,
447    value_counts: BTreeMap<T, usize>,
448}
449
450impl<T> MinVector<T>
451where
452    T: Ord + Copy,
453{
454    fn new(data: Vec<T>) -> Self {
455        let mut value_counts = BTreeMap::new();
456        for &value in &data {
457            *value_counts.entry(value).or_insert(0) += 1;
458        }
459        MinVector { data, value_counts }
460    }
461
462    fn set(&mut self, index: usize, value: T) {
463        // Decrease the count of the old value
464        let old_value = self.data[index];
465        if let Some(count) = self.value_counts.get_mut(&old_value) {
466            *count -= 1;
467            if *count == 0 {
468                self.value_counts.remove(&old_value);
469            }
470        }
471        // Update the value in the vector
472        self.data[index] = value;
473
474        // Increase the count of the new value
475        *self.value_counts.entry(value).or_insert(0) += 1;
476    }
477
478    fn min(&self) -> T {
479        *self.value_counts.keys().next().unwrap()
480    }
481}

483impl History {

486            first_incomplete_seqs: MinVector::new(vec![Seq::default(); world_size]),
487            min_incomplete_seq: Seq::default(),

489            inflight_invocations: HashMap::new(),

491            unreported_exception: None,

507    /// Add an invocation to the history.
508    pub fn add_invocation(
509        &mut self,
510        sender: &impl context::Actor,
511        seq: Seq,
512        uses: Vec<Ref>,
513        defs: Vec<Ref>,
514        tracebacks: Py<PyAny>,
515        response_port: Option<PortInfo>,
516    ) -> Result<(), MailboxSenderError> {
517        assert!(
518            seq >= self.seq_lower_bound,
519            "nonmonotonic seq: {:?}; current lower bound: {:?}",
520            seq,
521            self.seq_lower_bound,
522        );
523        self.seq_lower_bound = seq;
524        let invocation = Arc::new(sync::Mutex::new(Invocation::new(
525            seq,
526            tracebacks,
527            response_port,
528        )));
529        self.inflight_invocations.insert(seq, invocation.clone());
530        for ref use_ in uses {
531            let producer = self.invocation_for_ref.get(use_).unwrap();
532            producer.lock().unwrap().add_user(
533                sender,
534                &mut self.unreported_exception,
535                invocation.clone(),
536            )?;
537        }
538
539        for def in defs {
540            self.invocation_for_ref.insert(def, invocation.clone());
541        }
542        Ok(())
543    }

545    /// Propagate worker error to the invocation with the given Seq. This will also propagate
546    /// to all seqs that depend on this seq directly or indirectly.
547    pub async fn propagate_exception(
548        &mut self,
549        sender: &impl context::Actor,
550        seq: Seq,
551        exception: WorkerError,
552        rank: usize,
553    ) -> Result<(), MailboxSenderError> {
554        // TODO: supplement PythonMessage with the stack trace we have in invocation
555        let invocation = self.inflight_invocations.get(&seq).unwrap().clone();
556
557        let python_message = Arc::new(
558            monarch_hyperactor::runtime::monarch_with_gil(|py| {
559                let traceback = invocation
560                    .lock()
561                    .unwrap()
562                    .tracebacks
563                    .bind(py)
564                    .get_item(0)
565                    .unwrap();
566                let remote_exception = py
567                    .import("monarch.mesh_controller")
568                    .unwrap()
569                    .getattr("RemoteException")
570                    .unwrap();
571                let exe = remote_exception
572                    .call1((exception.backtrace, traceback, rank))
573                    .unwrap();
574                let mut state = pickle(py, exe.unbind(), false, false).unwrap();
575                let inner = state.take_inner().unwrap();
576                PythonMessage::new_from_buf(
577                    PythonMessageKind::Exception { rank: Some(rank) },
578                    inner.take_buffer(),
579                )
580            })
581            .await,
582        );
583
584        let mut invocation = invocation.lock().unwrap();
585
586        if let Status::Incomplete { .. } = &invocation.status {
587            self.unreported_exception = Some(python_message.clone());
588        }
589
590        invocation.set_exception(
591            sender,
592            &mut self.unreported_exception,
593            python_message.clone(),
594        )?;
595
596        Ok(())
597    }

599    /// Mark the given rank as completed up to but excluding the given Seq. This will also purge history for
600    /// any Seqs that are no longer relevant (completed on all ranks).
601    pub fn rank_completed(
602        &mut self,
603        sender: &impl context::Actor,
604        rank: usize,
605        seq: Seq,
606    ) -> Result<(), MailboxSenderError> {
607        self.first_incomplete_seqs.set(rank, seq);
608        let prev = self.min_incomplete_seq;
609        self.min_incomplete_seq = self.first_incomplete_seqs.min();
610
611        for i in Seq::iter_between(prev, self.min_incomplete_seq) {
612            if let Some(invocation) = self.inflight_invocations.remove(&i) {
613                let mut invocation = invocation.lock().unwrap();
614                invocation.complete(sender)?;
615            }
616        }
617        if let Some(port) = &self.exit_port {
618            if self.min_incomplete_seq >= self.seq_lower_bound {
619                let result = match &self.unreported_exception {
620                    Some(exception) => exception.as_ref().clone(),
621                    None => {
622                        // the byte string is just a Python None
623                        PythonMessage::new_from_buf(
624                            PythonMessageKind::Result { rank: None },
625                            b"\x80\x04N.".to_vec(),
626                        )
627                    }
628                };
629                port.send(sender, result)?;
630                self.exit_port = None;
631            }
632        }
633        Ok(())
634    }
635
636    pub fn set_result(&mut self, seq: Seq, result: PythonMessage) {
637        let invocation = self.inflight_invocations.get(&seq).unwrap();
638        invocation.lock().unwrap().set_result(result);
639    }
640
641    fn report_exit(&mut self, port: reference::PortRef<PythonMessage>) {
642        self.exit_port = Some(port);
643    }

727
728    async fn handle_debug(
729        &mut self,
730        this: &Context<'_, Self>,
731        debugger_actor_id: reference::ActorId,
732        action: DebuggerAction,
733    ) -> anyhow::Result<()> {
734        if matches!(action, DebuggerAction::Paused()) {
735            self.debugger_paused
736                .push_back(reference::ActorRef::attest(debugger_actor_id));
737        } else {
738            let debugger_actor = self
739                .debugger_active
740                .as_ref()
741                .ok_or_else(|| anyhow::anyhow!("no active debugger"))?;
742            if debugger_actor_id != *debugger_actor.actor_id() {
743                anyhow::bail!("debugger action for wrong actor");
744            }
745            match action {
746                DebuggerAction::Detach() => {
747                    self.debugger_active = None;
748                }
749                DebuggerAction::Read { requested_size } => {
750                    monarch_hyperactor::runtime::monarch_with_gil(|py| {
751                        let read = py
752                            .import("monarch.controller.debugger")
753                            .unwrap()
754                            .getattr("read")
755                            .unwrap();
756                        let bytes: Vec<u8> =
757                            read.call1((requested_size,)).unwrap().extract().unwrap();
758
759                        debugger_actor.send(
760                            this,
761                            DebuggerMessage::Action {
762                                action: DebuggerAction::Write { bytes },
763                            },
764                        )
765                    })
766                    .await?;
767                }
768                DebuggerAction::Write { bytes } => {
769                    monarch_hyperactor::runtime::monarch_with_gil(
770                        |py| -> Result<(), anyhow::Error> {
771                            let write = py
772                                .import("monarch.controller.debugger")
773                                .unwrap()
774                                .getattr("write")
775                                .unwrap();
776                            write.call1((String::from_utf8(bytes)?,)).unwrap();
777                            Ok(())
778                        },
779                    )
780                    .await?;
781                }
782                _ => {
783                    anyhow::bail!("unexpected action: {:?}", action);
784                }
785            }
786        }
787        if self.debugger_active.is_none() {
788            self.debugger_active = self.debugger_paused.pop_front().and_then(|pdb_actor| {
789                pdb_actor
790                    .send(
791                        this,
792                        DebuggerMessage::Action {
793                            action: DebuggerAction::Attach(),
794                        },
795                    )
796                    .map(|_| pdb_actor)
797                    .ok()
798            });
799        }
800        Ok(())
801    }
802}

804#[async_trait]
805impl Actor for MeshControllerActor {
806    async fn init(&mut self, this: &Instance<Self>) -> Result<(), anyhow::Error> {
807        let controller_actor_ref: reference::ActorRef<ControllerActor> = this.bind();
808        let world_size = Ranked::region(&self.proc_mesh_ref).num_ranks();
809        let param = WorkerParams {
810            world_size,
811            // Rank assignment is consistent with proc indices.
812            rank: 0,
813            device_index: Some(0),
814            controller_actor: controller_actor_ref,
815        };
816
817        let workers = self
818            .proc_mesh_ref
819            .spawn_service(this, &format!("tensor_engine_workers_{}", self.id), &param)
820            .await?;
821        workers.cast(this, AssignRankMessage::AssignRank())?;
822
823        self.workers = Some(workers);
824        let brokers = self
825            .proc_mesh_ref
826            .spawn_service(this, &format!("tensor_engine_brokers_{}", self.id), &())
827            .await?;
828        self.brokers = Some(brokers);
829        Ok(())
830    }

833        Some(format!("mesh_controller_{}", self.id))
834    }

852#[async_trait]
853impl Handler<ControllerMessage> for MeshControllerActor {
854    async fn handle(
855        &mut self,
856        this: &Context<Self>,
857        message: ControllerMessage,
858    ) -> anyhow::Result<()> {
859        match message {
860            ControllerMessage::DebuggerMessage {
861                debugger_actor_id,
862                action,
863            } => {
864                self.handle_debug(this, debugger_actor_id, action).await?;
865            }
866            ControllerMessage::Status {
867                seq,
868                worker_actor_id,
869                controller: false,
870            } => {
871                self.history
872                    .rank_completed(this, self.rank_of_worker(&worker_actor_id), seq)?;
873            }
874            ControllerMessage::FetchResult {
875                seq,
876                value: Ok(value),
877            } => {
878                let msg: PythonMessage = value.deserialized().unwrap();
879                self.history.set_result(seq, msg);
880            }
881            ControllerMessage::RemoteFunctionFailed { seq, error } => {
882                let rank = self.rank_of_worker(&error.worker_actor_id);
883                self.history
884                    .propagate_exception(this, seq, error, rank)
885                    .await?;
886            }
887            message => {
888                panic!("unexpected message: {:?}", message);
889            }
890        };
891        Ok(())
892    }
893}

895#[async_trait]
896impl Handler<ClientToControllerMessage> for MeshControllerActor {
897    async fn handle(
898        &mut self,
899        this: &Context<Self>,
900        message: ClientToControllerMessage,
901    ) -> anyhow::Result<()> {
902        match message {
903            ClientToControllerMessage::Send { slices, message } => {
904                let workers = self.workers();
905                let sel = Ranked::region(workers.deref())
906                    .slice()
907                    .reify_slices(slices)?;
908                workers.cast_for_tensor_engine_only_do_not_use(this, sel, message)?;
909            }
910            ClientToControllerMessage::Node {
911                seq,
912                defs,
913                uses,
914                tracebacks,
915                response_port,
916            } => {
917                self.history
918                    .add_invocation(this, seq, uses, defs, tracebacks, response_port)?;
919            }
920            ClientToControllerMessage::DropRefs { refs } => {
921                self.history.drop_refs(refs);
922            }
923            ClientToControllerMessage::SyncAtExit { port } => {
924                self.workers().cast(
925                    this,
926                    WorkerMessage::RequestStatus {
927                        seq: self.history.seq_lower_bound,
928                        controller: false,
929                    },
930                )?;
931                self.history.report_exit(port);
932            }
933            ClientToControllerMessage::StopWorkers { response_port } => {
934                let worker_stop_result = self
935                    .workers_mut()
936                    .stop(this, "client requested stop".to_string())
937                    .await;
938                let broker_stop_result = self
939                    .brokers_mut()
940                    .stop(this, "client requested stop".to_string())
941                    .await;
942                if worker_stop_result.is_ok() && broker_stop_result.is_ok() {
943                    response_port.send(this, Ok(()))?;
944                } else {
945                    response_port.send(this, Err(format!("stopping mesh workers failed: tensor worker result: {:?}, broker result: {:?}", worker_stop_result, broker_stop_result)))?;
946                }
947            }
948        }
949        Ok(())
950    }
951}

953#[async_trait]
954impl Handler<MeshFailure> for MeshControllerActor {
955    async fn handle(&mut self, this: &Context<Self>, message: MeshFailure) -> anyhow::Result<()> {
956        // If an actor spawned by this one fails, we can't handle it. We fail
957        // ourselves with a chained error and bubble up to the next owner.
958        let err = ActorErrorKind::UnhandledSupervisionEvent(Box::new(ActorSupervisionEvent::new(
959            this.self_id().clone(),
960            None,
961            ActorStatus::Failed(ActorErrorKind::UnhandledSupervisionEvent(Box::new(
962                message.event.clone(),
963            ))),
964            None,
965        )));
966        Err(anyhow::Error::new(err))
967    }
