9/// This crate implements scoped lifetime-erasure.
10///
11/// Its intended use is when you hold a reference with a well-defined scope,
12/// but you want to pass it around to APIs where threading through lifetime
13/// parameters is awkward, for example across Rust/Python FFI boundaries, but
14/// you can nevertheless guarantee that the reference is not used after the
15/// the scope exits.
16///
17/// Erased references are safe to use, but incorrect usage causes the relevant
18/// thread to halt, as we cannot safely drop a keepalive associated with an
19/// extant reference.
20///
21/// The crate provides a single function: [`erase`], which returns a [`Keepalive`]
22/// token, with the lifetime of the erased reference, and [`Ref`], the reference
23/// itself. The user must guarantee that the reference does not outlive the the
24/// keepalive token; otherwise the program halts. Because [`Keepalive`] is parameterized
25/// on the reference lifetime, it is guaranteed by the compiler not to outlive it,
26/// and thus transitively we can guarantee that [`Ref`] access is safe.
27///
28/// [`Ref`] *only* provides lifetime erasure, and can be combined with other cell
29/// types to provide interior mutability, dynamic borrow checking, etc.
30///
31/// ## Usage
32///
33/// ```
34/// fn outer(x: &u32) {
35///     let (keepalive, ref_) = erased_lifetime::erase(x);
36///     inner(ref_);
37/// }
38///
39/// fn inner(x: erased_lifetime::Ref<u32>) {
40///     assert_eq!(*x, 42);
41/// }
42///
43/// let data = 42u32;
44/// outer(&data);
45/// ```
46use std::backtrace::Backtrace;
47use std::fmt;
48use std::marker::PhantomData;
49use std::ops::Deref;
50use std::ptr::NonNull;
51use std::sync::Arc;
52use std::sync::atomic::AtomicBool;
53use std::sync::atomic::Ordering;
54
55/// Panic instead of halting. Only used for testing in this crate.
56static PANIC: AtomicBool = AtomicBool::new(false);
57
58/// Erase the lifetime `'a` from the referand. See crate documentation for a discussion
59/// of usage.
60pub fn erase<'a, T: ?Sized>(x: &'a T) -> (Keepalive<'a>, Ref<T>) {
61    let borrowed = Arc::new(AtomicBool::new(true));
62    (
63        Keepalive {
64            borrowed: Arc::clone(&borrowed),
65            _phantom: PhantomData,
66        },
67        Ref {
68            ptr: NonNull::from(x),
69            borrowed,
70        },
71    )
72}
73
74/// A keepalive token for a reference returned from [`erase`].
75pub struct Keepalive<'a> {
76    borrowed: Arc<AtomicBool>,
77    _phantom: PhantomData<&'a ()>,
78}
79
80impl<'a> fmt::Debug for Keepalive<'a> {
81    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
82        f.debug_struct("Keepalive")
83            .field("borrowed", &self.borrowed.load(Ordering::Relaxed))
84            .finish()
85    }
86}
87
88impl Drop for Keepalive<'_> {
89    fn drop(&mut self) {
90        if self.borrowed.load(Ordering::Acquire) {
91            if PANIC.load(Ordering::Relaxed) {
92                panic!("keepalive dropped with extant reference");
93            }
94            // Here, we have to just just block the thread.
95            eprintln!("keepalive dropped with extant reference; continuing would result in UB");
96            eprintln!("{}", Backtrace::force_capture());
97            loop {
98                std::thread::park();
99            }
100        }
101    }
102}
103
104/// A wrapper type for an erased reference.
105pub struct Ref<T: ?Sized> {
106    ptr: NonNull<T>,
107    borrowed: Arc<AtomicBool>,
108}
109
110impl<T: ?Sized> Deref for Ref<T> {
111    type Target = T;
112    fn deref(&self) -> &Self::Target {
113        // SAFETY: The pointer is valid while `Ref` is in scope.
114        unsafe { self.ptr.as_ref() }
115    }
116}
117
118impl<T: ?Sized> Drop for Ref<T> {
119    fn drop(&mut self) {
120        self.borrowed.store(false, Ordering::Release);
121    }
122}
123
124// SAFETY: Ref is Sync if T is Send by Rust rules.
125unsafe impl<T: Sync> Send for Ref<T> {}
126// SAFETY: Ref is Sync if T is Sync by Rust rules.
127unsafe impl<T: Sync> Sync for Ref<T> {}

129#[cfg(test)]
130mod tests {
131    use std::panic::catch_unwind;
132
133    use super::*;
134
135    fn assert_panics<F: FnOnce() -> R + std::panic::UnwindSafe, R>(f: F) -> String {
136        PANIC.store(true, Ordering::Relaxed);
137        match catch_unwind(f) {
138            Ok(_) => panic!("expected panic, but function returned normally"),
139            Err(e) => {
140                if let Some(s) = e.downcast_ref::<&'static str>() {
141                    (*s).to_string()
142                } else if let Some(s) = e.downcast_ref::<String>() {
143                    s.clone()
144                } else {
145                    "<non-string panic payload>".to_string()
146                }
147            }
148        }
149    }
150
151    #[test]
152    fn test_basic() {
153        let data = Box::new(42usize);
154        let x: &usize = &data;
155
156        let (_keepalive, ref_) = erase(x);
157
158        assert_eq!(*ref_, 42);
159
160        // Free of panics or thread parking. Compiler will drop in this order:
161        // drop(ref_);
162        // drop(keepalive);
163    }
164
165    #[test]
166    fn test_early_keepalive_drop() {
167        let data = Box::new(42usize);
168        let x: &usize = &data;
169
170        let (keepalive, ref_) = erase(x);
171
172        assert_eq!(*ref_, 42);
173
174        // Drop before reference.
175        let msg = assert_panics(|| {
176            drop(keepalive);
177        });
178        assert!(
179            msg.contains("keepalive dropped with extant reference"),
180            "{}",
181            msg
182        );
183    }
