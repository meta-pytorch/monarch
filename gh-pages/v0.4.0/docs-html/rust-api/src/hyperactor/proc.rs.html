9//! [`Proc`] is an addressable actor-runtime boundary.
10//!
11//! It owns actor lifecycle (spawn, run, terminate), routes messages
12//! to local actors, forwards messages for remote destinations, and
13//! hosts supervision state.
14//!
15//! It also stores bounded snapshots of terminated actors for
16//! post-mortem introspection.
17//!
18//! ## Client instance invariants (CI-*)
19//!
20//! - **CI-1 (client status):** `IntrospectMessage::Query` on an
21//!   introspectable instance returns `status: "client"` and
22//!   `actor_type: "()"` in attrs.
23//! - **CI-2 (snapshot on drop):** Dropping the returned `Instance<()>`
24//!   transitions its status to terminal, causing the introspect task
25//!   to store a terminated snapshot.
26//!
27//! ## Actor identity invariants (AI-*)
28//!
29//! - **AI-1 (named-child pid):** The pid of a named child must
30//!   remain in the parent's sibling pid domain. The name is
31//!   presentation only; the numeric pid is allocated from the
32//!   parent's counter, preserving supervision linkage.
33//! - **AI-3 (controller ActorId uniqueness):** Callers must ensure
34//!   the name is unique proc-wide. Two children with the same name
35//!   under different parents get distinct pids but the same name
36//!   prefix.

137/// A proc instance is the runtime managing a single proc in Hyperactor.
138/// It is responsible for spawning actors in the proc, multiplexing messages
139/// to/within actors in the proc, and providing fallback routing to external
140/// procs.
141///
142/// Procs are also responsible for maintaining the local supervision hierarchy.
143#[derive(Clone)]
144pub struct Proc {
145    inner: Arc<ProcState>,
146}

156struct ProcState {
157    /// The proc's id. This should be globally unique, but is not (yet)
158    /// for local-only procs.
159    proc_id: reference::ProcId,
160
161    /// A muxer instance that has entries for every actor managed by
162    /// the proc.
163    proc_muxer: MailboxMuxer,
164
165    /// Sender used to forward messages outside of the proc.
166    forwarder: BoxedMailboxSender,

168    /// Per-name atomic index allocator. Used by `allocate_root_id`

174    /// All actor instances in this proc.
175    instances: DashMap<reference::ActorId, WeakInstanceCell>,

177    /// Snapshots of terminated actors for post-mortem introspection.
178    /// Populated by the introspect task just before it exits on

180    /// [`config::TERMINATED_SNAPSHOT_RETENTION`].
181    terminated_snapshots: DashMap<reference::ActorId, crate::introspect::IntrospectResult>,

185    supervision_coordinator_port: OnceLock<PortHandle<ActorSupervisionEvent>>,
186
187    /// The actor ID of the supervision coordinator, if it lives on this proc.
188    /// Used to ensure the coordinator is shut down last during proc teardown.
189    supervision_coordinator_actor_id: OnceLock<reference::ActorId>,

275    /// Handle a supervision event received by the proc. Attempt to forward it to the
276    /// supervision coordinator port if one is set, otherwise crash the process.
277    pub fn handle_unhandled_supervision_event(
278        &self,
279        cx: &impl context::Actor,
280        event: ActorSupervisionEvent,
281    ) {
282        let result = match self.state().supervision_coordinator_port.get() {
283            Some(port) => port.send(cx, event.clone()).map_err(anyhow::Error::from),
284            None => {
285                if !event.is_error() {
286                    // Normal lifecycle events (e.g. clean stop) without a coordinator
287                    // are silently dropped.
288                    return;
289                }
290                Err(anyhow::anyhow!(
291                    "coordinator port is not set for proc {}",
292                    self.proc_id(),
293                ))
294            }
295        };
296        if let Err(err) = result {
297            if !event.is_error() {
298                // Normal lifecycle events that fail to send (e.g. coordinator
299                // mailbox already closed during shutdown) are silently dropped.
300                tracing::debug!(
301                    "proc {}: dropping non-error supervision event {}: {:?}",
302                    self.proc_id(),
303                    event,
304                    err
305                );
306                return;
307            }
308            tracing::error!(
309                "proc {}: could not propagate supervision event {} due to error: {:?}: crashing",
310                self.proc_id(),
311                event,
312                err
313            );
314
315            std::process::exit(1);
316        }
317    }

452    /// Create and return an actor instance, its handle, and its
453    /// receivers. This allows actors to be "inverted": the caller can
454    /// use the returned [`Instance`] to send and receive messages,
455    /// launch child actors, etc. The actor itself does not handle any
456    /// messages unless driven by the caller.
457    pub fn actor_instance<A: Actor>(&self, name: &str) -> Result<ActorInstance<A>, anyhow::Error> {
458        let actor_id = self.allocate_root_id(name)?;
459        let span = tracing::debug_span!(
460            "actor_instance",
461            actor_name = name,
462            actor_type = std::any::type_name::<A>(),
463            actor_id = actor_id.to_string(),
464        );
465        let _guard = span.enter();
466        let (instance, receivers) = Instance::new(self.clone(), actor_id.clone(), false, None);
467        let handle = ActorHandle::new(instance.inner.cell.clone(), instance.inner.ports.clone());
468        instance.change_status(ActorStatus::Client);
469
470        let introspect_cell = instance.inner.cell.clone();
471        let introspect_mailbox = instance.inner.mailbox.clone();
472        tokio::spawn(crate::introspect::serve_introspect(
473            introspect_cell,
474            introspect_mailbox,
475            receivers.introspect,
476        ));
477
478        let (signal_rx, supervision_rx) = receivers.actor_loop.unwrap();
479        Ok(ActorInstance {
480            instance,
481            handle,
482            supervision: supervision_rx,
483            signal: signal_rx,
484            work: receivers.work,
485        })
486    }

488    /// Traverse all actor trees in this proc, starting from root actors (pid=0).
489    ///
490    /// **Caution:** This holds DashMap shard read locks while doing
491    /// `Weak::upgrade()` and recursively walking the actor tree per
492    /// entry. Under rapid actor churn, this causes convoy starvation
493    /// with concurrent `insert`/`remove` operations. Prefer
494    /// `all_instance_keys()` with point lookups if you only need
495    /// actor IDs. Currently unused in production code.
496    pub fn traverse<F>(&self, f: &mut F)
497    where
498        F: FnMut(&InstanceCell, usize),
499    {
500        for entry in self.state().instances.iter() {
501            if entry.key().pid() == 0 {
502                if let Some(cell) = entry.value().upgrade() {
503                    cell.traverse(f);
504                }
505            }
506        }
507    }

517    /// Returns the ActorIds of all root actors (pid=0) in this proc.
518    ///

527            .instances
528            .iter()
529            .filter(|entry| entry.key().pid() == 0)
530            .map(|entry| entry.key().clone())

534    /// Returns the ActorIds of all live actors in this proc, including
535    /// dynamically spawned children.
536    ///
537    /// An actor is considered live if its weak reference is
538    /// upgradeable and its status is not terminal. This excludes
539    /// actors whose `InstanceCell` has been dropped and actors that
540    /// have stopped or failed but whose Arc is still held (e.g. by
541    /// the introspect task during teardown).
542    pub fn all_actor_ids(&self) -> Vec<reference::ActorId> {
543        self.state()
544            .instances
545            .iter()
546            .filter(|entry| {
547                entry
548                    .value()
549                    .upgrade()
550                    .is_some_and(|cell| !cell.status().borrow().is_terminal())
551            })
552            .map(|entry| entry.key().clone())
553            .collect()
554    }

557    /// values. Each shard read lock is held only long enough to clone

566    /// subsequent per-actor lookups).

568        self.state()
569            .instances
570            .iter()
571            .map(|entry| entry.key().clone())
572            .collect()
573    }

643    /// Call `abort` on the `JoinHandle` associated with the given
644    /// root actor. If successful return `Some(root.clone())` else
645    /// `None`.
646    pub fn abort_root_actor(
647        &self,
648        root: &reference::ActorId,
649        this_handle: Option<&JoinHandle<()>>,
650    ) -> Option<impl Future<Output = reference::ActorId>> {
651        self.state()
652            .instances
653            .get(root)
654            .into_iter()
655            .flat_map(|e| e.upgrade())
656            .map(|cell| {
657                let r1 = root.clone();
658                let r2 = root.clone();
659                // If abort_root_actor was called from inside an actor task, we don't want to abort that actor's task yet.
660                let skip_abort = this_handle.is_some_and(|this_h| {
661                    cell.inner
662                        .actor_task_handle
663                        .get()
664                        .is_some_and(|other_h| std::ptr::eq(this_h, other_h))
665                });
666                // `Instance::start()` is infallible and should
667                // complete quickly, so calling `wait()` on `actor_task_handle`
668                // should be safe (i.e., not hang forever).
669                async move {
670                    tokio::task::spawn_blocking(move || {
671                        if !skip_abort {
672                            let h = cell.inner.actor_task_handle.wait();
673                            tracing::debug!("{}: aborting {:?}", r1, h);
674                            h.abort();
675                        }
676                    })
677                    .await
678                    .unwrap();
679                    r2
680                }
681            })
682            .next()
683    }

685    /// Signals to a root actor to stop,
686    /// returning a status observer if successful.
687    pub fn stop_actor(
688        &self,
689        actor_id: &reference::ActorId,
690        reason: String,
691    ) -> Option<watch::Receiver<ActorStatus>> {
692        if let Some(entry) = self.state().instances.get(actor_id) {
693            match entry.value().upgrade() {
694                None => None, // the actor's cell has been dropped
695                Some(cell) => {
696                    tracing::info!("sending stop signal to {}", cell.actor_id());
697                    if let Err(err) = cell.signal(Signal::DrainAndStop(reason)) {
698                        tracing::error!(
699                            "{}: failed to send stop signal to pid {}: {:?}",
700                            self.proc_id(),
701                            cell.pid(),
702                            err
703                        );
704                        None
705                    } else {
706                        Some(cell.status().clone())
707                    }
708                }
709            }
710        } else {
711            tracing::error!("no actor {} found in {}", actor_id, self.proc_id());
712            None
713        }
714    }

720    /// If `cx` is specified, it means this method was called from inside an actor

725        timeout: Duration,
726        cx: Option<&Context<'_, A>>,
727        reason: &str,
728    ) -> Result<(Vec<reference::ActorId>, Vec<reference::ActorId>), anyhow::Error> {

733    /// Stop the proc. Returns a pair of:
734    /// - the actors observed to stop;
735    /// - the actors not observed to stop when timeout.
736    ///
737    /// If `cx` is specified, it means this method was called from inside an actor
738    /// in which case we shouldn't wait for it to stop and need to delay aborting
739    /// its task.
740    /// If except_current is true, don't stop the actor represented by "cx" at
741    /// all.
742    #[hyperactor::instrument]
743    pub async fn destroy_and_wait_except_current<A: Actor>(
744        &mut self,
745        timeout: Duration,
746        cx: Option<&Context<'_, A>>,
747        except_current: bool,
748        reason: &str,
749    ) -> Result<(Vec<reference::ActorId>, Vec<reference::ActorId>), anyhow::Error> {
750        tracing::debug!("{}: proc stopping", self.proc_id());
751
752        let (this_handle, this_actor_id) = cx.map_or((None, None), |cx| {
753            (
754                Some(cx.actor_task_handle().expect("cannot call destroy_and_wait from inside an actor unless actor has finished starting")),
755                Some(cx.self_id())
756            )
757        });
758
759        let coordinator_id = self.supervision_coordinator_actor_id().cloned();
760
761        // Phase 1: stop all root actors except the supervision coordinator
762        // (which must stay alive to receive stop events from the others).
763        let mut statuses = HashMap::new();
764        for actor_id in self
765            .state()
766            .instances
767            .iter()
768            .filter(|entry| entry.key().pid() == 0)
769            .map(|entry| entry.key().clone())
770            .collect::<Vec<_>>()
771        {
772            if coordinator_id.as_ref() == Some(&actor_id) {
773                continue;
774            }
775            if let Some(status) = self.stop_actor(&actor_id, reason.to_string()) {
776                statuses.insert(actor_id, status);
777            }
778        }
779        tracing::debug!("{}: non-coordinator actors stopped", self.proc_id());
780
781        let waits: Vec<_> = statuses
782            .iter_mut()
783            .filter(|(actor_id, _)| Some(*actor_id) != this_actor_id)
784            .map(|(actor_id, root)| {
785                let actor_id = actor_id.clone();
786                async move {
787                    tokio::time::timeout(
788                        timeout,
789                        root.wait_for(|state: &ActorStatus| state.is_terminal()),
790                    )
791                    .await
792                    .ok()
793                    .map(|_| actor_id)
794                }
795            })
796            .collect();
797
798        let results = futures::future::join_all(waits).await;
799        let mut stopped_actors: Vec<_> = results
800            .iter()
801            .filter_map(|actor_id| actor_id.as_ref())
802            .cloned()
803            .collect();
804        let aborted_actors: Vec<_> = statuses
805            .iter()
806            .filter(|(actor_id, _)| !stopped_actors.contains(actor_id))
807            .map(|(actor_id, _)| {
808                let f = self.abort_root_actor(actor_id, this_handle);
809                async move {
810                    let _ = if let Some(f) = f { Some(f.await) } else { None };
811                    // If `is_none(&_)` then the associated actor's
812                    // instance cell was already dropped when we went
813                    // to call `abort()` on the cell's task handle.
814
815                    actor_id.clone()
816                }
817            })
818            .collect();
819        let mut aborted_actors = futures::future::join_all(aborted_actors).await;
820
821        // Phase 2: now that all other actors have stopped, stop the
822        // supervision coordinator so it had a chance to receive all
823        // supervision events.
824        if let Some(ref coord_id) = coordinator_id
825            && this_actor_id != Some(coord_id)
826        {
827            if let Some(mut status) = self.stop_actor(coord_id, reason.to_string()) {
828                let stopped = tokio::time::timeout(
829                    timeout,
830                    status.wait_for(|s: &ActorStatus| s.is_terminal()),
831                )
832                .await
833                .is_ok();
834                if stopped {
835                    stopped_actors.push(coord_id.clone());
836                } else {
837                    if let Some(f) = self.abort_root_actor(coord_id, this_handle) {
838                        f.await;
839                    }
840                    aborted_actors.push(coord_id.clone());
841                }
842            }
843        }
844
845        // Flush the forwarder so that any messages posted during
846        // teardown (e.g. supervision events) are wire-delivered
847        // before we tear down the proc's networking. The flush is
848        // best-effort: if the remote side has already torn down its
849        // networking, acks may never arrive and flush would hang
850        // indefinitely, so we bound it with a configurable timeout.
851        let flush_timeout = hyperactor_config::global::get(crate::config::FORWARDER_FLUSH_TIMEOUT);
852        match tokio::time::timeout(flush_timeout, self.state().forwarder.flush()).await {
853            Ok(Err(err)) => {
854                tracing::warn!(
855                    "{}: forwarder flush failed during proc exit: {:?}",
856                    self.proc_id(),
857                    err
858                );
859            }
860            Err(_elapsed) => {
861                tracing::warn!(
862                    "{}: forwarder flush timed out during proc exit",
863                    self.proc_id(),
864                );
865            }
866            Ok(Ok(())) => {}
867        }
868
869        if let Some(this_handle) = this_handle
870            && let Some(this_actor_id) = this_actor_id
871            && !except_current
872        {
873            tracing::debug!("{}: aborting (delayed) {:?}", this_actor_id, this_handle);
874            this_handle.abort()
875        };
876
877        tracing::info!(
878            "destroy_and_wait: {} actors stopped, {} actors aborted",
879            stopped_actors.len(),
880            aborted_actors.len()
881        );
882        Ok((stopped_actors, aborted_actors))
883    }

885    /// Resolve an actor reference to a **live** actor on this proc.
886    ///
887    /// Returns `None` if:
888    /// - the actor was never spawned here,
889    /// - the actor's `InstanceCell` has been dropped, or
890    /// - the actor's status is terminal (stopped or failed).
891    ///
892    /// The terminal-status check guards a race window: the introspect
893    /// task (`serve_introspect`) holds a strong `InstanceCell` Arc
894    /// and drops it only after observing terminal status. Between the
895    /// actor reaching terminal and the introspect task reacting,
896    /// `upgrade()` on the weak ref succeeds even though the actor is
897    /// dead. The `is_terminal()` check closes that window. Once the
898    /// introspect task exits, the Arc is dropped and `upgrade()`
899    /// returns `None` on its own.
900    ///
901    /// Bounds:
902    /// - `R: Actor` — must be a real actor that can live in this
903    ///   proc.
904    /// - `R: Referable` — required because the input is an
905    ///   `ActorRef<R>`.
906    pub fn resolve_actor_ref<R: Actor + Referable>(
907        &self,
908        actor_ref: &reference::ActorRef<R>,
909    ) -> Option<ActorHandle<R>> {
910        let cell = self.inner.instances.get(actor_ref.actor_id())?.upgrade()?;
911        // An actor whose status is terminal has stopped processing
912        // messages even if its InstanceCell Arc is still alive (e.g.
913        // held by the introspect task during teardown).
914        if cell.status().borrow().is_terminal() {
915            return None;
916        }
917        cell.downcast_handle()
918    }

920    /// Create a root allocation in the proc.
921    fn allocate_root_id(&self, name: &str) -> Result<reference::ActorId, anyhow::Error> {
922        let name = name.to_string();
923        match self.state().roots.entry(name.to_string()) {
924            Entry::Vacant(entry) => {
925                entry.insert(AtomicUsize::new(1));
926            }
927            Entry::Occupied(_) => {
928                anyhow::bail!("an actor with name '{}' has already been spawned", name)
929            }
930        }
931        Ok(reference::ActorId::new(
932            self.state().proc_id.clone(),
933            name.to_string(),
934            0,
935        ))
936    }

1181/// Receivers created by [`Instance::new`] that must be threaded to
1182/// their respective consumers (actor loop, introspect task, etc.).
1183///
1184/// # Invariant
1185///
1186/// See S10 in `introspect` module doc.
1187pub struct InstanceReceivers<A: Actor> {
1188    /// Signal and supervision receivers for the actor loop. `None`
1189    /// for detached/client instances that don't run an actor loop.
1190    actor_loop: Option<(PortReceiver<Signal>, PortReceiver<ActorSupervisionEvent>)>,
1191    /// Work queue for dispatching messages to actor handlers.
1192    work: mpsc::UnboundedReceiver<WorkCell<A>>,
1193    /// Introspect message receiver for the dedicated introspect task.
1194    introspect: PortReceiver<IntrospectMessage>,
1195}

1197impl<A: Actor> Instance<A> {
1198    /// Create a new actor instance in Created state.
1199    fn new(
1200        proc: Proc,
1201        actor_id: reference::ActorId,
1202        detached: bool,
1203        parent: Option<InstanceCell>,
1204    ) -> (Self, InstanceReceivers<A>) {
1205        // Set up messaging
1206        let mailbox = Mailbox::new(actor_id.clone(), BoxedMailboxSender::new(proc.downgrade()));
1207        let (work_tx, work_rx) = ordered_channel(
1208            actor_id.to_string(),
1209            hyperactor_config::global::get(config::ENABLE_DEST_ACTOR_REORDERING_BUFFER),
1210        );
1211        let ports: Arc<Ports<A>> = Arc::new(Ports::new(mailbox.clone(), work_tx));
1212        proc.state().proc_muxer.bind_mailbox(mailbox.clone());
1213        let (status_tx, status_rx) = watch::channel(ActorStatus::Created);
1214
1215        let actor_type = match TypeInfo::of::<A>() {
1216            Some(info) => ActorType::Named(info),
1217            None => ActorType::Anonymous(std::any::type_name::<A>()),
1218        };
1219        let actor_loop_ports = if detached {
1220            None
1221        } else {
1222            let (signal_port, signal_receiver) = ports.open_message_port().unwrap();
1223            let (supervision_port, supervision_receiver) = mailbox.open_port();
1224            Some((
1225                (signal_port, supervision_port),
1226                (signal_receiver, supervision_receiver),
1227            ))
1228        };
1229
1230        let (actor_loop, actor_loop_receivers) = actor_loop_ports.unzip();
1231
1232        // Introspect port: a separate channel handled by a dedicated
1233        // tokio task (not the actor's message loop). Pre-registered
1234        // so Ports::get finds the Occupied entry and skips WorkCell
1235        // creation. bind_actor_port() registers in the mailbox
1236        // dispatch table at IntrospectMessage::port().
1237        //
1238        // Exercises S3, S4, S9 (see introspect module doc).
1239        let (introspect_port, introspect_receiver) =
1240            ports.open_message_port::<IntrospectMessage>().unwrap();
1241        introspect_port.bind_actor_port();
1242
1243        let cell = InstanceCell::new(
1244            actor_id,
1245            actor_type,
1246            proc.clone(),
1247            actor_loop,
1248            status_rx,
1249            parent,
1250            ports.clone(),
1251        );
1252        let instance_id = Uuid::now_v7();
1253        let inner = Arc::new(InstanceState {
1254            proc,
1255            cell,
1256            mailbox,
1257            ports,
1258            status_tx,
1259            sequencer: Sequencer::new(instance_id),
1260            id: instance_id,
1261            instance_locals: ActorLocalStorage::new(),
1262        });
1263        (
1264            Self { inner },
1265            InstanceReceivers {
1266                actor_loop: actor_loop_receivers,
1267                work: work_rx,
1268                introspect: introspect_receiver,
1269            },
1270        )
1271    }
1272
1273    /// Notify subscribers of a change in the actors status and bump counters with the duration which
1274    /// the last status was active for.
1275    #[track_caller]
1276    fn change_status(&self, new: ActorStatus) {
1277        let old = self.inner.status_tx.send_replace(new.clone());
1278        // 2 cases are allowed:
1279        // * non-terminal -> non-terminal
1280        // * non-terminal -> terminal
1281        // terminal -> terminal is not allowed unless it is the same status (no-op).
1282        // terminal -> non-terminal is never allowed.
1283        assert!(
1284            !old.is_terminal() && !new.is_terminal()
1285                || !old.is_terminal() && new.is_terminal()
1286                || old == new,
1287            "actor changing status illegally, only allow non-terminal -> non-terminal \
1288            and non-terminal -> terminal statuses. actor_id={}, prev_status={}, status={}",
1289            self.self_id(),
1290            old,
1291            new
1292        );
1293        // Actor status changes between Idle and Processing when handling every
1294        // message. It creates too many logs if we want to log these 2 states.
1295        // Also, sometimes the actor transitions from Processing -> Processing.
1296        // Therefore we skip the status changes between them.
1297        if !((old.is_idle() && new.is_processing())
1298            || (old.is_processing() && new.is_idle())
1299            || old == new)
1300        {
1301            let new_status = new.arm().unwrap_or("unknown");
1302            let change_reason = match &new {
1303                ActorStatus::Failed(reason) => reason.to_string(),
1304                ActorStatus::Stopped(reason) => reason.clone(),
1305                _ => "".to_string(),
1306            };
1307            tracing::info!(
1308                name = "ActorStatus",
1309                actor_id = %self.self_id(),
1310                actor_name = self.self_id().name(),
1311                status = new_status,
1312                prev_status = old.arm().unwrap_or("unknown"),
1313                caller = %Location::caller(),
1314                change_reason,
1315            );
1316            let actor_id = hash_to_u64(self.self_id());
1317            notify_actor_status_changed(ActorStatusEvent {
1318                id: generate_actor_status_event_id(actor_id),
1319                timestamp: std::time::SystemTime::now(),
1320                actor_id,
1321                new_status: new_status.to_string(),
1322                reason: if change_reason.is_empty() {
1323                    None
1324                } else {
1325                    Some(change_reason)
1326                },
1327            });
1328        }
1329    }

1586    async fn serve(
1587        mut self,
1588        mut actor: A,
1589        actor_loop_receivers: (PortReceiver<Signal>, PortReceiver<ActorSupervisionEvent>),
1590        mut work_rx: mpsc::UnboundedReceiver<WorkCell<A>>,
1591    ) {
1592        let result = self
1593            .run_actor_tree(&mut actor, actor_loop_receivers, &mut work_rx)
1594            .await;
1595
1596        assert!(self.is_stopping());
1597        let event = match result {
1598            Ok(stop_reason) => {
1599                let status = ActorStatus::Stopped(stop_reason);
1600                self.mailbox().close(status.clone());
1601                let event = ActorSupervisionEvent::new(
1602                    self.inner.cell.actor_id().clone(),
1603                    actor.display_name(),
1604                    status.clone(),
1605                    None,
1606                );
1607                // FI-1: store supervision_event BEFORE change_status.
1608                *self.inner.cell.inner.supervision_event.lock().unwrap() = Some(event.clone());
1609                self.change_status(status);
1610                Some(event)
1611            }
1612            Err(err) => {
1613                match *err.kind {
1614                    ActorErrorKind::UnhandledSupervisionEvent(box event) => {
1615                        // We use the event's actor_status as this actor's terminal status.
1616                        assert!(event.actor_status.is_terminal());
1617                        self.mailbox().close(event.actor_status.clone());
1618                        // FI-1: store supervision_event BEFORE change_status.
1619                        *self.inner.cell.inner.supervision_event.lock().unwrap() =
1620                            Some(event.clone());
1621                        self.change_status(event.actor_status.clone());
1622                        Some(event)
1623                    }
1624                    _ => {
1625                        let error_kind = ActorErrorKind::Generic(err.kind.to_string());
1626                        let status = ActorStatus::Failed(error_kind);
1627                        self.mailbox().close(status.clone());
1628                        let event = ActorSupervisionEvent::new(
1629                            self.inner.cell.actor_id().clone(),
1630                            actor.display_name(),
1631                            status.clone(),
1632                            None,
1633                        );
1634                        // FI-1: store supervision_event BEFORE change_status.
1635                        *self.inner.cell.inner.supervision_event.lock().unwrap() =
1636                            Some(event.clone());
1637                        self.change_status(status);
1638                        Some(event)
1639                    }
1640                }
1641            }
1642        };
1643
1644        if let Some(parent) = self.inner.cell.maybe_unlink_parent() {
1645            if let Some(event) = event {
1646                // Parent exists, failure should be propagated to the parent.
1647                parent.send_supervision_event_or_crash(&self, event);
1648            }
1649            // TODO: we should get rid of this signal, and use *only* supervision events for
1650            // the purpose of conveying lifecycle changes
1651            if let Err(err) = parent.signal(Signal::ChildStopped(self.inner.cell.pid())) {
1652                tracing::error!(
1653                    "{}: failed to send stop message to parent pid {}: {:?}",
1654                    self.self_id(),
1655                    parent.pid(),
1656                    err
1657                );
1658            }
1659        } else {
1660            // Failure happened to the root actor or orphaned child actors.
1661            // In either case, the failure should be propagated to proc.
1662            //
1663            // Note that orphaned actor is unexpected and would only happen if
1664            // there is a bug.
1665            if let Some(event) = event {
1666                self.inner
1667                    .proc
1668                    .handle_unhandled_supervision_event(&self, event);
1669            }
1670        }
1671    }

1673    /// Runs the actor, and manages its supervision tree. When the function returns,
1674    /// the whole tree rooted at this actor has stopped. On success, returns the reason
1675    /// why the actor stopped. On failure, returns the error that caused the failure.
1676    async fn run_actor_tree(
1677        &mut self,
1678        actor: &mut A,
1679        mut actor_loop_receivers: (PortReceiver<Signal>, PortReceiver<ActorSupervisionEvent>),
1680        work_rx: &mut mpsc::UnboundedReceiver<WorkCell<A>>,
1681    ) -> Result<String, ActorError> {
1682        // It is okay to catch all panics here, because we are in a tokio task,
1683        // and tokio will catch the panic anyway:
1684        // https://docs.rs/tokio/latest/tokio/task/struct.JoinError.html#method.is_panic
1685        // What we do here is just to catch it early so we can handle it.
1686
1687        let mut did_panic = false;
1688        let result = match AssertUnwindSafe(self.run(actor, &mut actor_loop_receivers, work_rx))
1689            .catch_unwind()
1690            .await
1691        {
1692            Ok(result) => result,
1693            Err(_) => {
1694                did_panic = true;
1695                let panic_info = panic_handler::take_panic_info()
1696                    .map(|info| info.to_string())
1697                    .unwrap_or_else(|e| format!("Cannot take backtrace due to: {:?}", e));
1698                Err(ActorError::new(
1699                    self.self_id(),
1700                    ActorErrorKind::panic(anyhow::anyhow!(panic_info)),
1701                ))
1702            }
1703        };
1704
1705        assert!(!self.is_terminal());
1706        self.change_status(ActorStatus::Stopping);
1707        if let Err(err) = &result {
1708            tracing::error!("{}: actor failure: {}", self.self_id(), err);
1709        }
1710
1711        // After this point, we know we won't spawn any more children,
1712        // so we can safely read the current child keys.
1713        let mut to_unlink = Vec::new();
1714        for child in self.inner.cell.child_iter() {
1715            if let Err(err) = child
1716                .value()
1717                .signal(Signal::Stop("parent stopping".to_string()))
1718            {
1719                tracing::error!(
1720                    "{}: failed to send stop signal to child pid {}: {:?}",
1721                    self.self_id(),
1722                    child.key(),
1723                    err
1724                );
1725                to_unlink.push(child.value().clone());
1726            }
1727        }
1728        // Manually unlink children that have already been stopped.
1729        for child in to_unlink {
1730            self.inner.cell.unlink(&child);
1731        }
1732
1733        let (mut signal_receiver, _) = actor_loop_receivers;
1734        while self.inner.cell.child_count() > 0 {
1735            match tokio::time::timeout(Duration::from_millis(500), signal_receiver.recv()).await {
1736                Ok(signal) => {
1737                    if let Signal::ChildStopped(pid) = signal? {
1738                        assert!(self.inner.cell.get_child(pid).is_none());
1739                    }
1740                }
1741                Err(_) => {
1742                    tracing::warn!(
1743                        "timeout waiting for ChildStopped signal from child on actor: {}, ignoring",
1744                        self.self_id()
1745                    );
1746                    // No more waiting to receive messages. Unlink all remaining
1747                    // children.
1748                    self.inner.cell.unlink_all();
1749                    break;
1750                }
1751            }
1752        }
1753        // Run the actor cleanup function before the actor stops to delete
1754        // resources. If it times out, continue with stopping the actor.
1755        // Don't call it if there was a panic, because the actor may
1756        // be in an invalid state and unable to access anything, for example
1757        // the GIL.
1758        let cleanup_result = if !did_panic {
1759            let cleanup_timeout = hyperactor_config::global::get(config::CLEANUP_TIMEOUT);
1760            match tokio::time::timeout(cleanup_timeout, actor.cleanup(self, result.as_ref().err()))
1761                .await
1762            {
1763                Ok(Ok(x)) => Ok(x),
1764                Ok(Err(e)) => Err(ActorError::new(self.self_id(), ActorErrorKind::cleanup(e))),
1765                Err(e) => Err(ActorError::new(
1766                    self.self_id(),
1767                    ActorErrorKind::cleanup(e.into()),
1768                )),
1769            }
1770        } else {
1771            Ok(())
1772        };
1773        if let Err(ref actor_err) = result {
1774            // The original result error takes precedence over the cleanup error,
1775            // so make sure the cleanup error is still logged in that case.
1776            if let Err(ref err) = cleanup_result {
1777                tracing::warn!(
1778                    cleanup_err = %err,
1779                    %actor_err,
1780                    "ignoring cleanup error after actor error",
1781                );
1782            }
1783        }
1784        // If the original exit was not an error, let cleanup errors be
1785        // surfaced.
1786        result.and_then(|reason| cleanup_result.map(|_| reason))
1787    }

1789    /// Initialize and run the actor until it fails or is stopped. On success,
1790    /// returns the reason why the actor stopped. On failure, returns the error
1791    /// that caused the failure.
1792    async fn run(
1793        &mut self,
1794        actor: &mut A,
1795        actor_loop_receivers: &mut (PortReceiver<Signal>, PortReceiver<ActorSupervisionEvent>),
1796        work_rx: &mut mpsc::UnboundedReceiver<WorkCell<A>>,
1797    ) -> Result<String, ActorError> {
1798        let (signal_receiver, supervision_event_receiver) = actor_loop_receivers;
1799
1800        self.change_status(ActorStatus::Initializing);
1801        actor
1802            .init(self)
1803            .await
1804            .map_err(|err| ActorError::new(self.self_id(), ActorErrorKind::init(err)))?;
1805        let need_drain;
1806        let stop_reason;
1807        'messages: loop {
1808            self.change_status(ActorStatus::Idle);
1809            let metric_pairs =
1810                hyperactor_telemetry::kv_pairs!("actor_id" => self.self_id().to_string());
1811            tokio::select! {
1812                work = work_rx.recv() => {
1813                    ACTOR_MESSAGES_RECEIVED.add(1, metric_pairs);
1814                    ACTOR_MESSAGE_QUEUE_SIZE.add(-1, metric_pairs);
1815                    let _ = ACTOR_MESSAGE_HANDLER_DURATION.start(metric_pairs);
1816                    let work = work.expect("inconsistent work queue state");
1817                    if let Err(err) = work.handle(actor, self).await {
1818                        for supervision_event in supervision_event_receiver.drain() {
1819                            self.handle_supervision_event(actor, supervision_event).await?;
1820                        }
1821                        let kind = ActorErrorKind::processing(err);
1822                        return Err(ActorError {
1823                            actor_id: Box::new(self.self_id().clone()),
1824                            kind: Box::new(kind),
1825                        });
1826                    }
1827                }
1828                signal = signal_receiver.recv() => {
1829                    let signal = signal.map_err(ActorError::from);
1830                    tracing::debug!("Received signal {signal:?}");
1831                    match signal? {
1832                        Signal::Stop(reason) => {
1833                            need_drain = false;
1834                            stop_reason = reason;
1835                            break 'messages;
1836                        },
1837                        Signal::DrainAndStop(reason) => {
1838                            need_drain = true;
1839                            stop_reason = reason;
1840                            break 'messages;
1841                        },
1842                        Signal::ChildStopped(pid) => {
1843                            assert!(self.inner.cell.get_child(pid).is_none());
1844                        },
1845                        Signal::Abort(reason) => {
1846                            return Err(ActorError { actor_id: Box::new(self.self_id().clone()), kind: Box::new(ActorErrorKind::Aborted(reason)) });
1847                        }
1848                    }
1849                }
1850                Ok(supervision_event) = supervision_event_receiver.recv() => {
1851                    self.handle_supervision_event(actor, supervision_event).await?;
1852                }
1853            }
1854            self.inner
1855                .cell
1856                .inner
1857                .num_processed_messages
1858                .fetch_add(1, Ordering::SeqCst);
1859        }
1860
1861        if need_drain {
1862            let mut n = 0;
1863            while let Ok(work) = work_rx.try_recv() {
1864                if let Err(err) = work.handle(actor, self).await {
1865                    return Err(ActorError::new(
1866                        self.self_id(),
1867                        ActorErrorKind::processing(err),
1868                    ));
1869                }
1870                n += 1;
1871            }
1872            tracing::debug!("drained {} messages", n);
1873        }
1874        tracing::debug!(
1875            actor_id = %self.self_id(),
1876            reason = stop_reason,
1877            "exited actor loop",
1878        );
1879        Ok(stop_reason)
1880    }

1882    /// Handle a supervision event using the provided actor.
1883    pub async fn handle_supervision_event(
1884        &self,
1885        actor: &mut A,
1886        supervision_event: ActorSupervisionEvent,
1887    ) -> Result<(), ActorError> {
1888        // Handle the supervision event with the current actor.
1889        match actor
1890            .handle_supervision_event(self, &supervision_event)
1891            .await
1892        {
1893            Ok(true) => {
1894                // The supervision event was handled by this actor, nothing more to do.
1895                Ok(())
1896            }
1897            Ok(false) => {
1898                let kind = ActorErrorKind::UnhandledSupervisionEvent(Box::new(supervision_event));
1899                Err(ActorError::new(self.self_id(), kind))
1900            }
1901            Err(err) => {
1902                // The actor failed to handle the supervision event, it should die.
1903                // Create a new supervision event for this failure and propagate it.
1904                let kind = ActorErrorKind::ErrorDuringHandlingSupervision(
1905                    err.to_string(),
1906                    Box::new(supervision_event),
1907                );
1908                Err(ActorError::new(self.self_id(), kind))
1909            }
1910        }
1911    }

1913    async unsafe fn handle_message<M: Message>(
1914        &self,
1915        actor: &mut A,
1916        type_info: Option<&'static TypeInfo>,
1917        headers: Flattrs,
1918        message: M,
1919    ) -> Result<(), anyhow::Error>
1920    where
1921        A: Handler<M>,
1922    {
1923        // Build HandlerInfo from TypeInfo (zero-copy) or fall back to type_name.
1924        let handler_info = match type_info {
1925            Some(info) => {
1926                // SAFETY: The caller promises to pass the correct type info.
1927                let arm = unsafe { info.arm_unchecked(&message as *const M as *const ()) };
1928                HandlerInfo::from_static(info.typename(), arm)
1929            }
1930            None => {
1931                // Fall back to std::any::type_name (also static, zero-copy).
1932                HandlerInfo::from_static(std::any::type_name::<M>(), None)
1933            }
1934        };
1935        // Use a helper function for a better instrument log.
1936        self.handle_message_with_handler_info(actor, handler_info, headers, message)
1937            .await
1938    }
1939
1940    // Skip serializing all fields except HandlerInfo which includes the typename.
1941    #[tracing::instrument(level = "debug", name = "handle_message", skip_all, fields(actor_id = %self.self_id(), message_type = %handler_info))]
1942    async fn handle_message_with_handler_info<M: Message>(
1943        &self,
1944        actor: &mut A,
1945        handler_info: HandlerInfo,
1946        headers: Flattrs,
1947        message: M,
1948    ) -> Result<(), anyhow::Error>
1949    where
1950        A: Handler<M>,
1951    {
1952        let now = std::time::SystemTime::now();
1953        let handler_info = Some(handler_info);
1954        self.change_status(ActorStatus::Processing(now, handler_info.clone()));
1955        crate::mailbox::headers::log_message_latency_if_sampling(
1956            &headers,
1957            self.self_id().to_string(),
1958        );
1959
1960        let message_id = headers.get(crate::mailbox::headers::TELEMETRY_MESSAGE_ID);
1961
1962        if let Some(message_id) = message_id {
1963            let from_actor_id = headers
1964                .get(crate::mailbox::headers::SENDER_ACTOR_ID_HASH)
1965                .unwrap_or(0);
1966            let to_actor_id = hash_to_u64(self.self_id());
1967            let port_id = headers.get(crate::mailbox::headers::TELEMETRY_PORT_ID);
1968
1969            notify_message(hyperactor_telemetry::MessageEvent {
1970                timestamp: now,
1971                id: message_id,
1972                from_actor_id,
1973                to_actor_id,
1974                // TODO: populate endpoint
1975                endpoint: None,
1976                port_id,
1977            });
1978
1979            notify_message_status(hyperactor_telemetry::MessageStatusEvent {
1980                timestamp: now,
1981                id: hyperactor_telemetry::generate_status_event_id(message_id),
1982                message_id,
1983                status: "active".to_string(),
1984            });
1985        }
1986
1987        // Record the message handler being invoked.
1988        *self.inner.cell.inner.last_message_handler.write().unwrap() = handler_info;
1989
1990        let context = Context::new(self, headers);
1991        // Pass a reference to the context to the handler, so that deref
1992        // coercion allows the `this` argument to be treated exactly like
1993        // &Instance<A>.
1994        let start = Instant::now();
1995        let result = actor
1996            .handle(&context, message)
1997            .instrument(self.inner.cell.inner.recording.span())
1998            .await;
1999        let elapsed_us = start.elapsed().as_micros() as u64;
2000        self.inner
2001            .cell
2002            .inner
2003            .total_processing_time_us
2004            .fetch_add(elapsed_us, Ordering::SeqCst);
2005
2006        if let Some(message_id) = message_id {
2007            notify_message_status(hyperactor_telemetry::MessageStatusEvent {
2008                timestamp: std::time::SystemTime::now(),
2009                id: hyperactor_telemetry::generate_status_event_id(message_id),
2010                message_id,
2011                status: "complete".to_string(),
2012            });
2013        }
2014
2015        result
2016    }

2300/// Select which terminated snapshots to evict when the retention cap
2301/// is exceeded.
2302///
2303/// Each entry is `(actor_id, Option<occurred_at>)` where `Some` means
2304/// the actor has `failure_info` (i.e. it failed), and `None` means a
2305/// clean stop.
2306///
2307/// Eviction priority:
2308/// 1. Cleanly-stopped actors are evicted first (arbitrary order).
2309/// 2. If more evictions are needed, failed actors are evicted
2310///    newest-first (descending `occurred_at`), preserving the
2311///    earliest failures which are closest to the root cause.
2312fn select_eviction_candidates(
2313    entries: &[(reference::ActorId, Option<String>)],
2314    excess: usize,
2315) -> Vec<reference::ActorId> {
2316    let mut clean: Vec<&reference::ActorId> = Vec::new();
2317    let mut failed: Vec<(&reference::ActorId, &str)> = Vec::new();
2318    for (id, occurred_at) in entries {
2319        match occurred_at {
2320            Some(ts) => failed.push((id, ts.as_str())),
2321            None => clean.push(id),
2322        }
2323    }
2324
2325    let mut to_remove: Vec<reference::ActorId> = Vec::new();
2326    let mut remaining = excess;
2327
2328    // Evict cleanly-stopped first.
2329    for id in clean {
2330        if remaining == 0 {
2331            break;
2332        }
2333        to_remove.push(id.clone());
2334        remaining -= 1;
2335    }
2336
2337    // If still over cap, evict most-recent failures first.
2338    if remaining > 0 {
2339        failed.sort_by(|a, b| b.1.cmp(a.1));
2340        for (id, _) in failed.into_iter().take(remaining) {
2341            to_remove.push(id.clone());
2342        }
2343    }
2344
2345    to_remove
2346}

2645    /// Store a post-mortem snapshot for this actor in the proc's
2646    /// `terminated_snapshots` map. Called by the introspect task
2647    /// just before exiting on terminal status.
2648    ///
2649    /// Eviction policy when the retention cap is exceeded:
2650    /// 1. Evict cleanly-stopped actors first (no `failure_info`).
2651    /// 2. When only failed actors remain, evict the most recent
2652    ///    (by `occurred_at`), preserving the earliest failures
2653    ///    which are closest to the root cause.
2654    pub fn store_terminated_snapshot(&self, payload: crate::introspect::IntrospectResult) {
2655        let snapshots = &self.inner.proc.inner.terminated_snapshots;
2656        snapshots.insert(self.actor_id().clone(), payload);
2657        let max = hyperactor_config::global::get(crate::config::TERMINATED_SNAPSHOT_RETENTION);
2658        let excess = snapshots.len().saturating_sub(max);
2659        if excess > 0 {
2660            // Build entries for the eviction selector.
2661            let entries: Vec<_> = snapshots
2662                .iter()
2663                .map(|entry| {
2664                    let occurred_at =
2665                        serde_json::from_str::<hyperactor_config::Attrs>(&entry.value().attrs)
2666                            .ok()
2667                            .and_then(|attrs| {
2668                                // Presence of FAILURE_ERROR_MESSAGE means the actor failed.
2669                                attrs
2670                                    .get(crate::introspect::FAILURE_ERROR_MESSAGE)
2671                                    .cloned()?;
2672                                // Extract occurred_at timestamp for sorting.
2673                                attrs
2674                                    .get(crate::introspect::FAILURE_OCCURRED_AT)
2675                                    .map(|t| humantime::format_rfc3339(*t).to_string())
2676                            });
2677                    (entry.key().clone(), occurred_at)
2678                })
2679                .collect();
2680
2681            for key in select_eviction_candidates(&entries, excess) {
2682                snapshots.remove(&key);
2683            }
2684        }
2685    }

2730    where
2731        F: FnMut(&InstanceCell, usize),
2732    {
2733        f(self, depth);

2735        let mut children: Vec<_> = self.child_iter().map(|r| r.value().clone()).collect();
2736        children.sort_by_key(|c| c.pid());
2737        for child in children {
