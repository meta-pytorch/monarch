87/// An ActorMesh is a collection of ranked A-typed actors.
88///
89/// Bound note: `A: Referable` because the mesh stores/returns
90/// `hyperactor_reference::ActorRef<A>`, which is only defined for `A: Referable`.
91#[derive(Debug)]
92pub struct ActorMesh<A: Referable> {
93    proc_mesh: ProcMeshRef,
94    name: Name,
95    current_ref: ActorMeshRef<A>,
96    /// If present, this is the controller for the mesh. The controller ensures
97    /// the mesh is stopped when the actor owning it is stopped, and can provide

100    /// such as ProcAgent or CommActor.
101    controller: Option<hyperactor_reference::ActorRef<ActorMeshController<A>>>,
102}

139    /// Stop actors on this mesh across all procs.
140    pub async fn stop(&mut self, cx: &impl context::Actor, reason: String) -> crate::Result<()> {
141        // Remove the controller as an optimization so all future meshes
142        // created from this one (such as slices) know they are already stopped.
143        // Refs and slices on other machines will still be able to query the
144        // controller and will be sent a notification about this stop by the controller
145        // itself.
146        if let Some(controller) = self.controller.take() {
147            // Send a stop to the controller so it stops monitoring the actors.
148            controller
149                .send(
150                    cx,
151                    resource::Stop {
152                        name: self.name.clone(),
153                        reason,
154                    },
155                )
156                .map_err(|e| {
157                    crate::Error::SendingError(controller.actor_id().clone(), Box::new(e))
158                })?;
159            let region = ndslice::view::Ranked::region(&self.current_ref);
160            let num_ranks = region.num_ranks();
161            // Wait for the controller to report all actors have stopped.
162            let (port, mut rx) = cx.mailbox().open_port();
163
164            controller
165                .send(
166                    cx,
167                    resource::GetState::<resource::mesh::State<()>> {
168                        name: self.name.clone(),
169                        reply: port.bind(),
170                    },
171                )
172                .map_err(|e| {
173                    crate::Error::SendingError(controller.actor_id().clone(), Box::new(e))
174                })?;
175
176            let statuses = rx.recv().await?;
177            if let Some(state) = &statuses.state {
178                // Check that all actors are in a terminating state (Stopping
179                // or beyond). The actual wait for full cleanup (terminal)
180                // happens in _drain_and_stop via the controller's status watch.
181                let all_stopped = state.statuses.values().all(|s| s.is_terminating());
182                if all_stopped {
183                    Ok(())
184                } else {
185                    let legacy = mesh_to_rankedvalues_with_default(
186                        &state.statuses,
187                        resource::Status::NotExist,
188                        resource::Status::is_not_exist,
189                        num_ranks,
190                    );
191                    Err(Error::ActorStopError { statuses: legacy })
192                }
193            } else {
194                Err(Error::Other(anyhow::anyhow!(
195                    "non-existent state in GetState reply from controller: {}",
196                    controller.actor_id()
197                )))
198            }?;
199            // Update health state with the new statuses.
200            let mut entry = self.health_state.entry(cx).or_default();
201            let health_state = entry.get_mut();
202            health_state.unhealthy_event = Some(Unhealthy::StreamClosed(MeshFailure {
203                actor_mesh_name: Some(self.name().to_string()),
204                event: ActorSupervisionEvent::new(
205                    // Use an actor id from the mesh.
206                    ndslice::view::Ranked::get(&self.current_ref, 0)
207                        .unwrap()
208                        .actor_id()
209                        .clone(),
210                    None,
211                    ActorStatus::Stopped("mesh stopped".to_string()),
212                    None,
213                ),
214                crashed_ranks: vec![],
215            }));
216        }
217        // Also take the controller from the ref, since that is used for
218        // some operations.
219        self.current_ref.controller.take();
220        Ok(())
221    }
222}

313/// Turn the single-owner PortReceiver into a watch receiver, which can be
314/// cloned and subscribed to. Requires a default message to pre-populate with.
315/// Option can be used as M to provide a default of None.
316fn into_watch<M: Send + Sync + Clone + Default + 'static>(
317    mut rx: PortReceiver<M>,
318) -> watch::Receiver<MessageOrFailure<M>> {
319    let (sender, receiver) = watch::channel(MessageOrFailure::<M>::default());
320    // Apply a watchdog timeout to the supervision stream. If no
321    // supervision message (healthy or unhealthy) is observed within
322    // this window, we assume the controller is unreachable and
323    // surface a terminal failure on the watch channel. This is a
324    // watchdog against indefinite silence, not a message-delivery
325    // guarantee, and may conservatively treat a quiet but healthy
326    // controller as failed.
327    let timeout = hyperactor_config::global::get(SUPERVISION_WATCHDOG_TIMEOUT);
328    let poll_frequency = hyperactor_config::global::get(SUPERVISION_POLL_FREQUENCY);
329    let get_actor_state_max_idle = hyperactor_config::global::get(GET_ACTOR_STATE_MAX_IDLE);
330    let get_proc_state_max_idle = hyperactor_config::global::get(GET_PROC_STATE_MAX_IDLE);
331    let total_time = poll_frequency + get_actor_state_max_idle + get_proc_state_max_idle;
332    if timeout < total_time {
333        tracing::warn!(
334            "HYPERACTOR_MESH_SUPERVISION_WATCHDOG_TIMEOUT={} is too short. It should be >= {} (SUPERVISION_POLL_FREQUENCY={} + GET_ACTOR_STATE_MAX_IDLE={} + GET_PROC_STATE_MAX_IDLE={})",
335            humantime::format_duration(timeout),
336            humantime::format_duration(total_time),
337            humantime::format_duration(poll_frequency),
338            humantime::format_duration(get_actor_state_max_idle),
339            humantime::format_duration(get_proc_state_max_idle),
340        );
341    }
342    tokio::spawn(async move {
343        loop {
344            let message = match tokio::time::timeout(timeout, rx.recv()).await {
345                Ok(Ok(msg)) => MessageOrFailure::Message(msg),
346                Ok(Err(e)) => MessageOrFailure::Failure(e.to_string()),
347                Err(_) => MessageOrFailure::Timeout,
348            };
349            let is_failure = matches!(
350                message,
351                MessageOrFailure::Failure(_) | MessageOrFailure::Timeout
352            );
353            if sender.send(message).is_err() {
354                // After a sending error, exit the task.
355                break;
356            }
357            if is_failure {
358                // No need to keep polling if we've received an error or timeout.
359                break;
360            }
361        }
362    });
363    receiver
364}

366/// A reference to a stable snapshot of an [`ActorMesh`].
367pub struct ActorMeshRef<A: Referable> {
368    proc_mesh: ProcMeshRef,
369    name: Name,
370    /// Reference to a remote controller actor living on the proc that spawned
371    /// the actors in this ref. If None, the actor mesh was already stopped, or
372    /// this is a mesh ref to a "system actor" which has no controller and should
373    /// not be stopped. If Some, the actor mesh may still be stopped, and the
374    /// next_supervision_event function can be used to alert that the mesh has
375    /// stopped.
376    controller: Option<hyperactor_reference::ActorRef<ActorMeshController<A>>>,
377
378    /// Recorded health issues with the mesh, to quickly consult before sending
379    /// out any casted messages. This is a locally updated copy of the authoritative
380    /// state stored on the ActorMeshController.
381    health_state: ActorLocal<HealthState>,
382    /// Shared cloneable receiver for supervision events, used by next_supervision_event.
383    /// Needs tokio mutex because it is held across an await point.
384    /// Should not be shared across actors because each actor context needs its
385    /// own subscriber.
386    receiver: ActorLocal<
387        Arc<
388            tokio::sync::Mutex<(
389                hyperactor_reference::PortRef<Option<MeshFailure>>,
390                watch::Receiver<MessageOrFailure<Option<MeshFailure>>>,
391            )>,
392        >,
393    >,
394    /// Lazily allocated collection of pages:
395    /// - The outer `OnceCell` defers creating the vector until first
396    ///   use.
397    /// - The `Vec` holds slots for multiple pages.
398    /// - Each slot is itself a `OnceCell<Box<Page<A>>>`, so that each
399    ///   page can be initialized on demand.
400    /// - A `Page<A>` is a boxed slice of `OnceCell<hyperactor_reference::ActorRef<A>>`,
401    ///   i.e. the actual storage for actor references within that
402    ///   page.
403    pages: OnceCell<Vec<OnceCell<Box<Page<A>>>>>,
404    // Page size knob (not serialize; defaults after deserialize).
405    page_size: usize,
406}

425        &self,
426        cx: &impl context::Actor,
427        sel: Selection,
428        message: M,
429    ) -> crate::Result<()>
430    where
431        A: RemoteHandles<M> + RemoteHandles<IndexedErasedUnbound<M>>,

433    {

437    #[allow(clippy::result_large_err)]
438    fn cast_with_selection<M>(
439        &self,
440        cx: &impl context::Actor,
441        sel: Selection,
442        message: M,
443    ) -> crate::Result<()>
444    where
445        A: RemoteHandles<M> + RemoteHandles<IndexedErasedUnbound<M>>,
446        M: Castable + RemoteMessage + Clone, // Clone is required until we are fully onto comm actor
447    {
448        // First check if the mesh is already dead before sending out any messages
449        // to a possibly undeliverable actor.
450        {
451            let health_state = self.health_state.entry(cx).or_default();
452            let health_state = health_state.get();
453            match &health_state.unhealthy_event {
454                Some(Unhealthy::StreamClosed(failure)) => {
455                    return Err(crate::Error::Supervision(Box::new(failure.clone())));
456                }
457                Some(Unhealthy::Crashed(failure)) => {
458                    return Err(crate::Error::Supervision(Box::new(failure.clone())));
459                }
460                None => {
461                    // If crashed ranks has any entries, then unhealthy_event should be set.
462                    // This is because all slices get a distinct health state.
463                    assert!(health_state.crashed_ranks.is_empty());
464                }
465            }
466        }
467
468        hyperactor_telemetry::notify_sent_message(hyperactor_telemetry::SentMessageEvent {
469            timestamp: std::time::SystemTime::now(),
470            sender_actor_id: hyperactor_telemetry::hash_to_u64(cx.mailbox().actor_id()),
471            actor_mesh_id: hyperactor_telemetry::hash_to_u64(&self.name.to_string()),
472            view_json: serde_json::to_string(view::Ranked::region(self)).unwrap_or_default(),
473            shape_json: {
474                let shape: ndslice::Shape = view::Ranked::region(self).into();
475                serde_json::to_string(&shape).unwrap_or_default()
476            },
477        });
478
479        // Now that we know these ranks are active, send out the actual messages.
480        if let Some(root_comm_actor) = self.proc_mesh.root_comm_actor() {
481            self.cast_v0(cx, message, sel, root_comm_actor)
482        } else {
483            for (point, actor) in self.iter() {
484                let create_rank = point.rank();
485                let mut headers = Flattrs::new();
486                multicast::set_cast_info_on_headers(
487                    &mut headers,
488                    point,
489                    cx.instance().self_id().clone(),
490                );
491
492                // Make sure that we re-bind ranks, as these may be used for
493                // bootstrapping comm actors.
494                let mut unbound = Unbound::try_from_message(message.clone())
495                    .map_err(|e| Error::CastingError(self.name.clone(), e))?;
496                unbound
497                    .visit_mut::<resource::Rank>(|resource::Rank(rank)| {
498                        *rank = Some(create_rank);
499                        Ok(())
500                    })
501                    .map_err(|e| Error::CastingError(self.name.clone(), e))?;
502                let rebound_message = unbound
503                    .bind()
504                    .map_err(|e| Error::CastingError(self.name.clone(), e))?;
505                actor
506                    .send_with_headers(cx, headers, rebound_message)
507                    .map_err(|e| Error::SendingError(actor.actor_id().clone(), Box::new(e)))?;
508            }
509            Ok(())
510        }
511    }

627    fn ensure_pages(&self) -> &Vec<OnceCell<Box<Page<A>>>> {
628        let n = self.len().div_ceil(self.page_size); // ⌈len / page_size⌉

633    fn materialize(&self, rank: usize) -> Option<&hyperactor_reference::ActorRef<A>> {
634        let len = self.len();
635        if rank >= len {
636            return None;
637        }
638        let p = self.page_size;
639        let page_ix = rank / p;
640        let local_ix = rank % p;
641
642        let pages = self.ensure_pages();
643        let page = pages[page_ix].get_or_init(|| {
644            // Last page may be partial.
645            let base = page_ix * p;
646            let remaining = len - base;
647            let page_len = remaining.min(p);
648            Box::new(Page::<A>::new(page_len))
649        });
650
651        Some(page.slots[local_ix].get_or_init(|| {
652            // AM-1: see module doc.
653            //   - ranks are contiguous [0, self.len()) with no gaps
654            //     or reordering
655            //   - for every rank r, `proc_mesh.get(r)` is Some(..)
656            // Therefore we can index `proc_mesh` with `rank`
657            // directly.
658            debug_assert!(rank < self.len(), "rank must be within [0, len)");
659            debug_assert!(
660                ndslice::view::Ranked::get(&self.proc_mesh, rank).is_some(),
661                "proc_mesh must be dense/aligned with this view"
662            );
663            let proc_ref =
664                ndslice::view::Ranked::get(&self.proc_mesh, rank).expect("rank in-bounds");
665            proc_ref.attest(&self.name)
666        }))
667    }

670        controller: &hyperactor_reference::ActorRef<ActorMeshController<A>>,
671        cx: &impl context::Actor,
672    ) -> (

674        watch::Receiver<MessageOrFailure<Option<MeshFailure>>>,
675    ) {

684    /// Returns the next supervision event occurring on this mesh. Await this
685    /// simultaneously with the return result of a message (such as awaiting a reply after a cast)
686    /// to get back a message that indicates the actor that failed, instead of
687    /// waiting forever for a reply.
688    /// If there are multiple simultaneous awaits of next_supervision_event,
689    /// all of them will receive the same event.
690    pub async fn next_supervision_event(
691        &self,
692        cx: &impl context::Actor,
693    ) -> Result<MeshFailure, anyhow::Error> {
694        let controller = if let Some(c) = self.controller() {
695            c
696        } else {
697            let health_state = self.health_state.entry(cx).or_default();
698            let health_state = health_state.get();
699            return match &health_state.unhealthy_event {
700                Some(Unhealthy::StreamClosed(f)) => Ok(f.clone()),
701                Some(Unhealthy::Crashed(f)) => Ok(f.clone()),
702                None => Err(anyhow::anyhow!(
703                    "unexpected healthy state while controller is gone"
704                )),
705            };
706        };
707        let rx = {
708            // Make sure to create only one PortReceiver per context.
709            let entry = self.receiver.entry(cx).or_insert_with(|| {
710                Arc::new(tokio::sync::Mutex::new(Self::init_supervision_receiver(
711                    controller, cx,
712                )))
713            });
714            // Need to clone so the lifetime is disconnected from entry, which
715            // isn't Send so can't be held across an await point.
716            Arc::clone(entry.get())
717        };
718        let message = {
719            let mut rx = rx.lock().await;
720            let subscriber_port = rx.0.clone();
721            let message =
722                rx.1.wait_for(|message| {
723                    // Filter out messages that do not apply to these ranks. This
724                    // is relevant for slices since we get messages back for the
725                    // whole mesh.
726                    if let MessageOrFailure::Message(message) = message {
727                        if let Some(message) = &message {
728                            let region = ndslice::view::Ranked::region(self).slice();
729                            if message.crashed_ranks.is_empty() {
730                                // Whole-mesh event (e.g. mesh stop).
731                                true
732                            } else {
733                                // Accept if any crashed rank overlaps with
734                                // this slice's region.
735                                message.crashed_ranks.iter().any(|r| region.contains(*r))
736                            }
737                        } else {
738                            // Filter out messages that are not failures. These are used
739                            // to ensure the controller is still reachable, but are not
740                            // otherwise interesting.
741                            false
742                        }
743                    } else {
744                        // either failure case is interesting
745                        true
746                    }
747                })
748                .await?;
749            let message = message.clone();
750            let is_failure = matches!(
751                message,
752                MessageOrFailure::Failure(_) | MessageOrFailure::Timeout
753            );
754            if is_failure {
755                // In failure cases, the receiver is dropped, so we can unsubscribe
756                // from the controller. The controller can detect this
757                // on its own, but an explicit unsubscribe prevents error logs
758                // about this receiver being unreachable.
759                let mut port = controller.port();
760                // We don't care if the controller is unreachable for an unsubscribe.
761                port.return_undeliverable(false);
762                let _ = port.send(cx, Unsubscribe(subscriber_port));
763            }
764            // If we successfully got a message back, we can't unsubscribe because
765            // the receiver might be shared with other calls to next_supervision_event,
766            // or with clones of this ActorMeshRef.
767            match message {
768                MessageOrFailure::Message(message) => Ok::<MeshFailure, anyhow::Error>(
769                    message.expect("filter excludes any None messages"),
770                ),
771                MessageOrFailure::Failure(failure) => Err(anyhow::anyhow!("{}", failure)),
772                MessageOrFailure::Timeout => {
773                    // Treat timeout from controller as a supervision failure,
774                    // the controller is unreachable.
775                    Ok(MeshFailure {
776                        actor_mesh_name: Some(self.name().to_string()),
777                        event: ActorSupervisionEvent::new(
778                            controller.actor_id().clone(),
779                            None,
780                            ActorStatus::generic_failure(format!(
781                                "timed out reaching controller {} for mesh {}. Assuming controller's proc is dead",
782                                controller.actor_id(),
783                                self.name()
784                            )),
785                            None,
786                        ),
787                        crashed_ranks: vec![],
788                    })
789                }
790            }?
791        };
792        // Update the health state now that we have received a message.
793        let event = &message.event;
794        // Make sure not to hold this lock across an await point.
795        let mut entry = self.health_state.entry(cx).or_default();
796        let health_state = entry.get_mut();
797        if let ActorStatus::Failed(_) = event.actor_status {
798            for &rank in &message.crashed_ranks {
799                health_state.crashed_ranks.insert(rank, event.clone());
800            }
801        }
802        health_state.unhealthy_event = match &event.actor_status {
803            ActorStatus::Failed(_) => Some(Unhealthy::Crashed(message.clone())),
804            ActorStatus::Stopped(_) => Some(Unhealthy::StreamClosed(message.clone())),
805            _ => None,
806        };
807        Ok(message)
808    }

1473    /// forcibly aborted. This is the V1 equivalent of
1474    /// hyperactor_multiprocess/src/proc_actor.rs::test_stop_timeout.

1476    #[cfg(fbcode_build)]
1477    async fn test_actor_mesh_stop_timeout() {
1478        hyperactor_telemetry::initialize_logging_for_test();
1479
1480        // Override ACTOR_SPAWN_MAX_IDLE to make test fast and
1481        // deterministic. ACTOR_SPAWN_MAX_IDLE is the maximum idle
1482        // time between status updates during mesh operations
1483        // (spawn/stop). When stop() is called, it waits for actors to
1484        // report they've stopped. If actors don't respond within this
1485        // timeout, they're forcibly aborted via JoinHandle::abort().
1486        // We set this to 1 second (instead of default 30s) so hung
1487        // actors (sleeping 5s in this test) get aborted quickly,
1488        // making the test fast.
1489        let config = hyperactor_config::global::lock();
1490        let _guard = config.override_key(ACTOR_SPAWN_MAX_IDLE, std::time::Duration::from_secs(1));
1491
1492        let instance = testing::instance();
1493
1494        // Create proc mesh with 2 procs
1495        let mut hm = testing::host_mesh(2).await;
1496        let proc_mesh = hm
1497            .spawn(instance, "test", Extent::unity(), None)
1498            .await
1499            .unwrap();
1500
1501        // Spawn SleepActors across the mesh that will block longer
1502        // than timeout
1503        let mut sleep_mesh: ActorMesh<testactor::SleepActor> =
1504            proc_mesh.spawn(instance, "sleepers", &()).await.unwrap();
1505
1506        // Send each actor a message to sleep for 5 seconds (longer
1507        // than 1-second timeout)
1508        for actor_ref in sleep_mesh.values() {
1509            actor_ref
1510                .send(instance, std::time::Duration::from_secs(5))
1511                .unwrap();
1512        }
1513
1514        // Give actors time to start sleeping
1515        tokio::time::sleep(std::time::Duration::from_millis(200)).await;
1516
1517        // Count how many actors we spawned (for verification later)
1518        let expected_actors = sleep_mesh.values().count();
1519
1520        // Now stop the mesh - actors won't respond in time, should be
1521        // aborted. Time this operation to verify abort behavior.
1522        let stop_start = tokio::time::Instant::now();
1523        let result = sleep_mesh.stop(instance, "test stop".to_string()).await;
1524        let stop_duration = tokio::time::Instant::now().duration_since(stop_start);
1525
1526        // Stop will return an error because actors didn't stop within
1527        // the timeout. This is expected - the actors were forcibly
1528        // aborted, and V1 reports this as an error.
1529        match result {
1530            Ok(_) => {
1531                // It's possible actors stopped in time, but unlikely
1532                // given 5-second sleep vs 1-second timeout
1533                tracing::warn!("Actors stopped gracefully (unexpected but ok)");
1534            }
1535            Err(ref e) => {
1536                // Expected: timeout error indicating actors were aborted
1537                let err_str = format!("{:?}", e);
1538                assert!(
1539                    err_str.contains("Timeout"),
1540                    "Expected Timeout error, got: {:?}",
1541                    e
1542                );
1543                tracing::info!(
1544                    "Stop timed out as expected for {} actors, they were aborted",
1545                    expected_actors
1546                );
1547            }
1548        }
1549
1550        // Verify that stop completed quickly (~1-2 seconds for
1551        // timeout + abort) rather than waiting the full 5 seconds for
1552        // actors to finish sleeping. This proves actors were aborted,
1553        // not waited for.
1554        assert!(
1555            stop_duration < std::time::Duration::from_secs(3),
1556            "Stop took {:?}, expected < 3s (actors should have been aborted, not waited for)",
1557            stop_duration
1558        );
1559        assert!(
1560            stop_duration >= std::time::Duration::from_millis(900),
1561            "Stop took {:?}, expected >= 900ms (should have waited for timeout)",
1562            stop_duration
1563        );
1564
1565        let _ = hm.shutdown(instance).await;
1566    }
