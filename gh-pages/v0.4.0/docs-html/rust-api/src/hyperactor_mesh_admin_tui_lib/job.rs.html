25/// An in-progress overlay-producing async job.
26///
27/// Exactly one variant is live at a time. `App` holds
28/// `active_job: Option<ActiveJob>` which is always `Some` iff
29/// `overlay` is `Some` (TUI-21). Enforced by `set_job`/`dismiss_job`.
30pub(crate) enum ActiveJob {
31    /// A streaming diagnostic run. `running` flips to `false` and
32    /// `rx` to `None` when the mpsc sender closes.
33    Diagnostics {
34        results: Vec<DiagResult>,
35        running: bool,
36        rx: Option<mpsc::Receiver<DiagResult>>,
37        completed_at: Option<String>,
38    },
39    /// A single py-spy HTTP fetch.
40    ///
41    /// Use `rx.is_some()` — not `lines.is_empty()` — to distinguish
42    /// "loading" from "completed with no output". An empty `lines`
43    /// with `rx == None` means the fetch completed but returned nothing.
44    PySpy {
45        /// `Some` while the HTTP fetch is in flight; `None` after the
46        /// oneshot fires.
47        rx: Option<oneshot::Receiver<Vec<Line<'static>>>>,
48        short: String,
49        /// Populated when the oneshot result arrives.
50        lines: Vec<Line<'static>>,
51        /// Set to a formatted timestamp when the result arrives.
52        completed_at: Option<String>,
53    },
54    /// A single config dump HTTP fetch.
55    ///
56    /// Same rx/lines semantics as PySpy: `rx.is_some()` → loading.
57    Config {
58        rx: Option<oneshot::Receiver<Vec<Line<'static>>>>,
59        short: String,
60        lines: Vec<Line<'static>>,
61        completed_at: Option<String>,
62    },
63}

65impl ActiveJob {
66    /// Build the display overlay for this job's current state.
67    ///
68    /// Takes `(&self, &Theme)` not `(&self, &App)` because
69    /// `rebuild_overlay` borrows `self.active_job` (shared) while
70    /// holding `&mut self` for the overlay assignment. The borrow
71    /// checker rejects `&App` here.
72    pub(crate) fn build_overlay(&self, theme: &Theme) -> Overlay {
73        match self {
74            ActiveJob::Diagnostics {
75                results,
76                running,
77                completed_at,
78                ..
79            } => build_diag_overlay(results, *running, completed_at.as_deref(), theme),
80            ActiveJob::PySpy {
81                rx,
82                short,
83                lines,
84                completed_at,
85            } => {
86                let scheme = &theme.scheme;
87                let labels = &theme.labels;
88                let sep = labels.separator;
89                // Use rx.is_some() — not lines.is_empty() — to detect
90                // loading state. An empty lines with rx == None means
91                // "completed with no output".
92                let loading = rx.is_some();
93
94                let title = if loading {
95                    Line::from(vec![
96                        Span::styled(
97                            format!("py-spy: {short}"),
98                            Style::default().add_modifier(Modifier::BOLD),
99                        ),
100                        Span::styled(format!("{sep}{}", labels.diag_running), scheme.info),
101                    ])
102                } else if let Some(ts) = completed_at {
103                    Line::from(vec![
104                        Span::styled(
105                            format!("py-spy: {short}"),
106                            Style::default().add_modifier(Modifier::BOLD),
107                        ),
108                        Span::styled(
109                            format!("{sep}{}", labels.diag_completed_at),
110                            scheme.detail_label,
111                        ),
112                        Span::raw(" "),
113                        Span::styled(ts.clone(), scheme.stat_timing),
114                    ])
115                } else {
116                    Line::from(Span::styled(
117                        format!("py-spy: {short}"),
118                        Style::default().add_modifier(Modifier::BOLD),
119                    ))
120                };
121
122                let status_line = if loading {
123                    Some(Line::from(vec![
124                        Span::styled(labels.diag_running, scheme.info),
125                        Span::styled(format!("{sep}fetching stack trace"), scheme.detail_label),
126                    ]))
127                } else {
128                    None
129                };
130
131                Overlay {
132                    title,
133                    status_line,
134                    lines: lines.clone(),
135                    loading,
136                    scroll: Cell::new(0),
137                    max_scroll: Cell::new(u16::MAX),
138                }
139            }
140            ActiveJob::Config {
141                rx,
142                short,
143                lines,
144                completed_at,
145            } => {
146                let scheme = &theme.scheme;
147                let labels = &theme.labels;
148                let sep = labels.separator;
149                let loading = rx.is_some();
150
151                let title = if loading {
152                    Line::from(vec![
153                        Span::styled(
154                            format!("config: {short}"),
155                            Style::default().add_modifier(Modifier::BOLD),
156                        ),
157                        Span::styled(format!("{sep}{}", labels.diag_running), scheme.info),
158                    ])
159                } else if let Some(ts) = completed_at {
160                    Line::from(vec![
161                        Span::styled(
162                            format!("config: {short}"),
163                            Style::default().add_modifier(Modifier::BOLD),
164                        ),
165                        Span::styled(
166                            format!("{sep}{}", labels.diag_completed_at),
167                            scheme.detail_label,
168                        ),
169                        Span::raw(" "),
170                        Span::styled(ts.clone(), scheme.stat_timing),
171                    ])
172                } else {
173                    Line::from(Span::styled(
174                        format!("config: {short}"),
175                        Style::default().add_modifier(Modifier::BOLD),
176                    ))
177                };
178
179                let status_line = if loading {
180                    Some(Line::from(vec![
181                        Span::styled(labels.diag_running, scheme.info),
182                        Span::styled(
183                            format!("{sep}fetching config snapshot"),
184                            scheme.detail_label,
185                        ),
186                    ]))
187                } else {
188                    None
189                };
190
191                Overlay {
192                    title,
193                    status_line,
194                    lines: lines.clone(),
195                    loading,
196                    scroll: Cell::new(0),
197                    max_scroll: Cell::new(u16::MAX),
198                }
199            }
200        }
201    }

203    /// Return the footer help text for the current overlay state.
204    /// Falls back to the default topology help text when no job is active.
205    pub(crate) fn footer_text<'a>(job: &Option<Self>, labels: &'a Labels) -> &'a str {
206        match job {
207            Some(ActiveJob::Diagnostics { running: true, .. }) => {
208                labels.footer_diag_running_help_text
209            }
210            Some(ActiveJob::Diagnostics { .. }) => labels.footer_diag_completed_help_text,
211            Some(ActiveJob::PySpy { .. }) => labels.footer_pyspy_help_text,
212            Some(ActiveJob::Config { .. }) => labels.footer_config_help_text,
213            None => labels.footer_help_text,
214        }
215    }

217    /// Process an event from the async receiver, updating job state.
218    /// Caller always rebuilds the overlay afterward.
219    pub(crate) fn on_event(&mut self, event: ActiveJobEvent) {
220        match event {
221            ActiveJobEvent::DiagResult(Some(r)) => {
222                if let ActiveJob::Diagnostics { results, .. } = self {
223                    results.push(r);
224                } else {
225                    debug_assert!(false, "DiagResult delivered to non-Diagnostics job");
226                }
227            }
228            ActiveJobEvent::DiagResult(None) => {
229                if let ActiveJob::Diagnostics {
230                    running,
231                    rx,
232                    completed_at,
233                    ..
234                } = self
235                {
236                    *running = false;
237                    *rx = None;
238                    *completed_at = Some(Local::now().format("%H:%M:%S").to_string());
239                } else {
240                    debug_assert!(false, "DiagResult(None) delivered to non-Diagnostics job");
241                }
242            }
243            ActiveJobEvent::PySpyResult(new_lines) => {
244                if let ActiveJob::PySpy {
245                    rx,
246                    lines,
247                    completed_at,
248                    ..
249                } = self
250                {
251                    *rx = None;
252                    *lines = new_lines;
253                    *completed_at = Some(Local::now().format("%H:%M:%S").to_string());
254                } else {
255                    debug_assert!(false, "PySpyResult delivered to non-PySpy job");
256                }
257            }
258            ActiveJobEvent::ConfigResult(new_lines) => {
259                if let ActiveJob::Config {
260                    rx,
261                    lines,
262                    completed_at,
263                    ..
264                } = self
265                {
266                    *rx = None;
267                    *lines = new_lines;
268                    *completed_at = Some(Local::now().format("%H:%M:%S").to_string());
269                } else {
270                    debug_assert!(false, "ConfigResult delivered to non-Config job");
271                }
272            }
273        }
274    }
