63    #[pyo3(get, set)]
64    pub env: HashMap<String, String>,
65}

67#[pymethods]
68impl PyBootstrapCommand {
69    #[new]
70    fn new(
71        program: String,
72        arg0: Option<String>,
73        args: Vec<String>,
74        env: HashMap<String, String>,
75    ) -> Self {
76        Self {
77            program,
78            arg0,
79            args,
80            env,
81        }
82    }
83
84    fn __repr__(&self) -> String {
85        format!(
86            "BootstrapCommand(program='{}', args={:?}, env={:?})",
87            self.program, self.args, self.env
88        )
89    }

92impl PyBootstrapCommand {
93    pub fn to_rust(&self) -> BootstrapCommand {
94        BootstrapCommand {
95            program: PathBuf::from(&self.program),
96            arg0: self.arg0.clone(),
97            args: self.args.clone(),
98            env: self.env.clone(),
99        }
100    }
101
102    pub fn default<'py>(py: Python<'py>) -> PyResult<Bound<'py, Self>> {
103        py.import("monarch._src.actor.host_mesh")?
104            .getattr("_bootstrap_cmd")?
105            .call0()?
106            .downcast::<PyBootstrapCommand>()
107            .cloned()
108            .map_err(to_py_error)
109    }

112#[pyclass(
113    name = "HostMesh",
114    module = "monarch._rust_bindings.monarch_hyperactor.host_mesh"
115)]
116pub(crate) enum PyHostMesh {
117    Owned(PyHostMeshImpl),
118    Ref(PyHostMeshRefImpl),
119}
120
121impl PyHostMesh {
122    pub(crate) fn new_owned(inner: HostMesh) -> Self {
123        Self::Owned(PyHostMeshImpl(SharedCell::from(inner)))
124    }
125
126    pub(crate) fn new_ref(inner: HostMeshRef) -> Self {
127        Self::Ref(PyHostMeshRefImpl(inner))
128    }
129
130    fn mesh_ref(&self) -> Result<HostMeshRef, anyhow::Error> {
131        match self {
132            PyHostMesh::Owned(inner) => Ok(inner.0.borrow()?.clone()),
133            PyHostMesh::Ref(inner) => Ok(inner.0.clone()),
134        }
135    }
136}

138#[pymethods]
139impl PyHostMesh {
140    #[classmethod]
141    fn allocate_nonblocking(
142        _cls: &Bound<'_, PyType>,
143        instance: &PyInstance,
144        alloc: &mut PyAlloc,
145        name: String,
146        bootstrap_params: Option<PyBootstrapCommand>,
147    ) -> PyResult<PyPythonTask> {
148        let bootstrap_params =
149            bootstrap_params.map_or_else(|| alloc.bootstrap_command.clone(), |b| Some(b.to_rust()));
150        let alloc = match alloc.take() {
151            Some(alloc) => alloc,
152            None => {
153                return Err(PyException::new_err(
154                    "Alloc object already used".to_string(),
155                ));
156            }
157        };
158        let instance = instance.clone();
159        PyPythonTask::new(async move {
160            let mesh = HostMesh::allocate(instance.deref(), alloc, &name, bootstrap_params)
161                .await
162                .map_err(|err| PyException::new_err(err.to_string()))?;
163            Ok(Self::new_owned(mesh))
164        })
165    }
166
167    #[pyo3(signature = (instance, name, per_host, proc_bind = None))]
168    fn spawn_nonblocking(
169        &self,
170        instance: &PyInstance,
171        name: String,
172        per_host: &PyExtent,
173        proc_bind: Option<Vec<HashMap<String, String>>>,
174    ) -> PyResult<PyPythonTask> {
175        let host_mesh = self.mesh_ref()?.clone();
176        let instance = instance.clone();
177        let per_host = per_host.clone().into();
178        let proc_bind = proc_bind.map(|v| v.into_iter().map(ProcBind::from).collect());
179        let mesh_impl = async move {
180            let proc_mesh = host_mesh
181                .spawn(instance.deref(), &name, per_host, proc_bind)
182                .await
183                .map_err(to_py_error)?;
184            Ok(PyProcMesh::new_owned(proc_mesh))
185        };
186        PyPythonTask::new(mesh_impl)
187    }
188
189    fn with_bootstrap(&self, bootstrap_command: &PyBootstrapCommand) -> PyResult<Self> {
190        Ok(Self::new_ref(
191            self.mesh_ref()?.with_bootstrap(bootstrap_command.to_rust()),
192        ))
193    }
194
195    /// Spawn a MeshAdminAgent on the head host's system proc and
196    /// return its HTTP address as a string.
197    ///
198    /// When `admin_addr` is provided (as a `"host:port"` string), the
199    /// HTTP server binds to that address; otherwise it reads
200    /// `MESH_ADMIN_ADDR` from config.
201    fn _spawn_admin(
202        &self,
203        instance: &PyInstance,
204        admin_addr: Option<String>,
205    ) -> PyResult<PyPythonTask> {
206        let admin_addr = admin_addr
207            .map(|s| {
208                s.parse::<std::net::SocketAddr>()
209                    .map_err(|e| PyException::new_err(format!("invalid admin_addr '{}': {}", s, e)))
210            })
211            .transpose()?;
212        let host_mesh = self.mesh_ref()?.clone();
213        let instance = instance.clone();
214        PyPythonTask::new(async move {
215            let addr = host_mesh
216                .spawn_admin(instance.deref(), admin_addr)
217                .await
218                .map_err(|e| PyException::new_err(e.to_string()))?;
219            Ok(addr)
220        })
221    }
222
223    fn sliced(&self, region: &PyRegion) -> PyResult<Self> {
224        Ok(Self::new_ref(
225            self.mesh_ref()?.sliced(region.as_inner().clone()),
226        ))
227    }
228
229    #[getter]
230    fn region(&self) -> PyResult<PyRegion> {
231        Ok(PyRegion::from(self.mesh_ref()?.region()))
232    }
233
234    fn __reduce__<'py>(&self, py: Python<'py>) -> PyResult<(Bound<'py, PyAny>, Bound<'py, PyAny>)> {
235        let bytes = bincode::serialize(&self.mesh_ref()?)
236            .map_err(|e| PyErr::new::<PyValueError, _>(e.to_string()))?;
237        let py_bytes = (PyBytes::new(py, &bytes),).into_bound_py_any(py).unwrap();
238        let from_bytes =
239            PyModule::import(py, "monarch._rust_bindings.monarch_hyperactor.host_mesh")?
240                .getattr("py_host_mesh_from_bytes")?;
241        Ok((from_bytes, py_bytes))
242    }
243
244    fn __eq__(&self, other: &PyHostMesh) -> PyResult<bool> {
245        Ok(self.mesh_ref()? == other.mesh_ref()?)
246    }
247
248    fn shutdown(&self, instance: &PyInstance) -> PyResult<PyPythonTask> {
249        match self {
250            PyHostMesh::Owned(inner) => {
251                let instance = instance.clone();
252                let mesh_borrow = inner.0.clone();
253                let fut = async move {
254                    match mesh_borrow.take().await {
255                        Ok(mut mesh) => {
256                            mesh.shutdown(instance.deref()).await?;
257                            Ok(())
258                        }
259                        Err(_) => {
260                            // Don't return an exception, silently ignore the stop request
261                            // because it was already done.
262                            tracing::info!("shutdown was already called on host mesh");
263                            Ok(())
264                        }
265                    }
266                };
267                PyPythonTask::new(fut)
268            }
269            PyHostMesh::Ref(_) => Err(PyRuntimeError::new_err(
270                "cannot shut down `HostMesh` that is a reference instead of owned",
271            )),
272        }
273    }
274
275    fn stop(&self, instance: &PyInstance) -> PyResult<PyPythonTask> {
276        match self {
277            PyHostMesh::Owned(inner) => {
278                let instance = instance.clone();
279                let mesh_borrow = inner.0.clone();
280                let fut = async move {
281                    match mesh_borrow.take().await {
282                        Ok(mut mesh) => {
283                            mesh.stop(instance.deref()).await?;
284                            Ok(())
285                        }
286                        Err(_) => {
287                            tracing::info!("stop was already called on host mesh");
288                            Ok(())
289                        }
290                    }
291                };
292                PyPythonTask::new(fut)
293            }
294            PyHostMesh::Ref(_) => Err(PyRuntimeError::new_err(
295                "cannot stop `HostMesh` that is a reference instead of owned",
296            )),
297        }
298    }
299}

321/// Static storage for the root client instance when using host-based bootstrap.
322static ROOT_CLIENT_INSTANCE_FOR_HOST: OnceLock<Instance<PythonActor>> = OnceLock::new();
323
324/// Static storage for the host mesh agent created by bootstrap_host().
325static HOST_MESH_AGENT_FOR_HOST: OnceLock<ActorHandle<HostAgent>> = OnceLock::new();
326
327/// Bootstrap the client host and root client actor.
328///
329/// This creates a proper Host with BootstrapProcManager, spawns the root client
330/// actor on the Host's local_proc.
331///
332/// Returns a tuple of (HostMesh, ProcMesh, PyInstance) where:
333/// - PyHostMesh: the bootstrapped (local) host mesh; and
334/// - PyProcMesh: the local ProcMesh on this HostMesh; and
335/// - PyInstance: the root client actor instance, on the ProcMesh.
336///
337/// The HostMesh is served on the default transport.
338///
339/// This should be called only once, at process initialization
340#[pyfunction]
341fn bootstrap_host(bootstrap_cmd: Option<PyBootstrapCommand>) -> PyResult<PyPythonTask> {
342    let bootstrap_cmd = match bootstrap_cmd {
343        Some(cmd) => cmd.to_rust(),
344        None => BootstrapCommand::current().map_err(|e| PyException::new_err(e.to_string()))?,
345    };
346
347    PyPythonTask::new(async move {
348        let (host_mesh_agent, _shutdown) = host(
349            default_bind_spec().binding_addr(),
350            Some(bootstrap_cmd),
351            None,
352            false,
353        )
354        .await
355        .map_err(|e| PyException::new_err(e.to_string()))?;
356
357        // Store the agent for later shutdown
358        HOST_MESH_AGENT_FOR_HOST.set(host_mesh_agent.clone()).ok(); // Ignore error if already set
359
360        let host_mesh_name = hyperactor_mesh::Name::new_reserved("local").unwrap();
361        let host_mesh = HostMeshRef::from_host_agent(host_mesh_name, host_mesh_agent.bind())
362            .map_err(|e| PyException::new_err(e.to_string()))?;
363
364        // Register C so MeshAdminAgent can discover it ("A/C
365        // invariant" - hyperactor_mesh/src/mesh_admin.rs).
366        hyperactor_mesh::global_context::register_client_host(host_mesh.clone());
367
368        // We require a temporary instance to make a call to the host/proc agent.
369        let temp_proc = Proc::local();
370        let (temp_instance, _) = temp_proc
371            .instance("temp")
372            .map_err(|e| PyException::new_err(e.to_string()))?;
373
374        let local_proc_agent: hyperactor::ActorHandle<hyperactor_mesh::proc_agent::ProcAgent> =
375            host_mesh_agent
376                .get_local_proc(&temp_instance)
377                .await
378                .map_err(|e| PyException::new_err(e.to_string()))?;
379
380        let proc_mesh_name = hyperactor_mesh::Name::new_reserved("local").unwrap();
381        let proc_mesh = ProcMeshRef::new_singleton(
382            proc_mesh_name,
383            ProcRef::new(
384                local_proc_agent.actor_id().proc_id().clone(),
385                0,
386                local_proc_agent.bind(),
387            ),
388        );
389
390        let local_proc = local_proc_agent
391            .get_proc(&temp_instance)
392            .await
393            .map_err(|e| PyException::new_err(e.to_string()))?;
394
395        let (instance, _handle) = monarch_with_gil(|py| {
396            PythonActor::bootstrap_client_inner(py, local_proc, &ROOT_CLIENT_INSTANCE_FOR_HOST)
397        })
398        .await;
399
400        // Notify telemetry of the bootstrap host mesh, proc mesh, and client actor.
401        {
402            let now = std::time::SystemTime::now();
403
404            let host_name_str = host_mesh.name().to_string();
405            let host_mesh_id = hyperactor_telemetry::hash_to_u64(&host_name_str);
406            hyperactor_telemetry::notify_mesh_created(hyperactor_telemetry::MeshEvent {
407                id: host_mesh_id,
408                timestamp: now,
409                class: "Host".to_string(),
410                given_name: host_mesh.name().name().to_string(),
411                full_name: host_name_str,
412                shape_json: host_mesh.region().extent().to_string(),
413                parent_mesh_id: None,
414                parent_view_json: None,
415            });
416
417            let host_agent_id = host_mesh_agent.actor_id();
418            hyperactor_telemetry::notify_actor_created(hyperactor_telemetry::ActorEvent {
419                id: hyperactor_telemetry::hash_to_u64(host_agent_id),
420                timestamp: now,
421                mesh_id: host_mesh_id,
422                rank: 0,
423                full_name: host_agent_id.to_string(),
424                display_name: None,
425            });
426
427            let proc_name_str = proc_mesh.name().to_string();
428            let proc_mesh_id = hyperactor_telemetry::hash_to_u64(&proc_name_str);
429            hyperactor_telemetry::notify_mesh_created(hyperactor_telemetry::MeshEvent {
430                id: proc_mesh_id,
431                timestamp: now,
432                class: "Proc".to_string(),
433                given_name: proc_mesh.name().name().to_string(),
434                full_name: proc_name_str,
435                shape_json: proc_mesh.region().extent().to_string(),
436                parent_mesh_id: Some(host_mesh_id),
437                parent_view_json: None,
438            });
439
440            let proc_agent_id = local_proc_agent.actor_id();
441            hyperactor_telemetry::notify_actor_created(hyperactor_telemetry::ActorEvent {
442                id: hyperactor_telemetry::hash_to_u64(proc_agent_id),
443                timestamp: now,
444                mesh_id: proc_mesh_id,
445                rank: 0,
446                full_name: proc_agent_id.to_string(),
447                display_name: None,
448            });
449
450            let client_mesh_name = format!("{}/client", proc_mesh.name());
451            let client_mesh_id = hyperactor_telemetry::hash_to_u64(&client_mesh_name);
452            hyperactor_telemetry::notify_mesh_created(hyperactor_telemetry::MeshEvent {
453                id: client_mesh_id,
454                timestamp: now,
455                class: <PythonActor as typeuri::Named>::typename().to_string(),
456                given_name: "client".to_string(),
457                full_name: client_mesh_name,
458                shape_json: proc_mesh.region().extent().to_string(),
459                parent_mesh_id: Some(proc_mesh_id),
460                parent_view_json: None,
461            });
462
463            hyperactor_telemetry::notify_actor_created(hyperactor_telemetry::ActorEvent {
464                id: hyperactor_telemetry::hash_to_u64(instance.self_id()),
465                timestamp: now,
466                mesh_id: client_mesh_id,
467                rank: 0,
468                full_name: instance.self_id().to_string(),
469                display_name: Some("<root>".to_string()),
470            });
471        }
472
473        Ok((
474            PyHostMesh::new_ref(host_mesh),
475            PyProcMesh::new_ref(proc_mesh),
476            PyInstance::from(instance),
477        ))
478    })
479}

481#[pyfunction]
482fn py_host_mesh_from_bytes(bytes: &Bound<'_, PyBytes>) -> PyResult<PyHostMesh> {
483    let r: PyResult<HostMeshRef> = bincode::deserialize(bytes.as_bytes())
484        .map_err(|e| PyErr::new::<PyValueError, _>(e.to_string()));
485    r.map(PyHostMesh::new_ref)
486}

488#[pyfunction]
489fn shutdown_local_host_mesh() -> PyResult<PyPythonTask> {
490    let agent = HOST_MESH_AGENT_FOR_HOST
491        .get()
492        .ok_or_else(|| PyException::new_err("No local host mesh to shutdown"))?
493        .clone();
494
495    PyPythonTask::new(async move {
496        // Create a temporary instance to send the shutdown message
497        let temp_proc = hyperactor::Proc::local();
498        let (instance, _) = temp_proc
499            .instance("shutdown_requester")
500            .map_err(|e| PyException::new_err(e.to_string()))?;
501
502        tracing::info!(
503            "sending shutdown_host request to agent {}",
504            agent.actor_id()
505        );
506        // Use same defaults as HostMesh::shutdown():
507        // - MESH_TERMINATE_TIMEOUT = 10 seconds
508        // - MESH_TERMINATE_CONCURRENCY = 16
509
510        let (port, _) = instance.open_port();
511        let mut port = port.bind();
512        // We don't need the ack, and this temporary proc doesn't have a mailbox
513        // receiver set up anyways. Just ignore the message.
514        port.return_undeliverable(false);
515        agent
516            .send(
517                &instance,
518                ShutdownHost {
519                    timeout: Duration::from_secs(10),
520                    max_in_flight: 16,
521                    ack: port,
522                },
523            )
524            .map_err(|e| PyException::new_err(e.to_string()))?;
525
526        Ok(())
527    })
528}

530pub fn register_python_bindings(hyperactor_mod: &Bound<'_, PyModule>) -> PyResult<()> {
531    let f = wrap_pyfunction!(py_host_mesh_from_bytes, hyperactor_mod)?;
532    f.setattr(
533        "__module__",
534        "monarch._rust_bindings.monarch_hyperactor.host_mesh",
535    )?;
536    hyperactor_mod.add_function(f)?;
537
538    let f2 = wrap_pyfunction!(bootstrap_host, hyperactor_mod)?;
539    f2.setattr(
540        "__module__",
541        "monarch._rust_bindings.monarch_hyperactor.host_mesh",
542    )?;
543    hyperactor_mod.add_function(f2)?;
544
545    let f3 = wrap_pyfunction!(shutdown_local_host_mesh, hyperactor_mod)?;
546    f3.setattr(
547        "__module__",
548        "monarch._rust_bindings.monarch_hyperactor.host_mesh",
549    )?;
550    hyperactor_mod.add_function(f3)?;
551
552    hyperactor_mod.add_class::<PyHostMesh>()?;
553    hyperactor_mod.add_class::<PyBootstrapCommand>()?;
554    Ok(())
