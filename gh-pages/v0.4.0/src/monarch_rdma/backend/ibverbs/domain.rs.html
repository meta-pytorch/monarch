9//! ibverbs RDMA domain.
10//!
11//! An [`IbvDomain`] manages the device context and protection domain (PD)
12//! required for RDMA operations. It provides the foundation for creating
13//! queue pairs and establishing connections between RDMA devices.
14
15use std::ffi::CStr;
16use std::io::Error;
17use std::result::Result;
18
19use super::primitives::IbvDevice;
20
21/// Manages RDMA resources including context and protection domain.
22///
23/// # Fields
24///
25/// * `context`: A pointer to the RDMA device context.
26/// * `pd`: A pointer to the protection domain, which provides isolation between
27///   different connections.
28#[derive(Clone)]
29pub struct IbvDomain {
30    pub context: *mut rdmaxcel_sys::ibv_context,
31    pub pd: *mut rdmaxcel_sys::ibv_pd,
32}
33
34impl std::fmt::Debug for IbvDomain {
35    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
36        f.debug_struct("IbvDomain")
37            .field("context", &format!("{:p}", self.context))
38            .field("pd", &format!("{:p}", self.pd))
39            .finish()
40    }
41}
42
43// SAFETY:
44// IbvDomain is `Send` because the raw pointers to ibverbs structs can be
45// accessed from any thread, and it is safe to drop `IbvDomain` (and run the
46// ibverbs destructors) from any thread.
47unsafe impl Send for IbvDomain {}
48
49// SAFETY:
50// IbvDomain is `Sync` because the underlying ibverbs APIs are thread-safe.
51unsafe impl Sync for IbvDomain {}
52
53impl Drop for IbvDomain {
54    fn drop(&mut self) {
55        unsafe {
56            rdmaxcel_sys::ibv_dealloc_pd(self.pd);
57        }
58    }
59}
60
61impl IbvDomain {
62    /// Creates a new IbvDomain for the given device.
63    ///
64    /// Initializes the RDMA device context and creates a protection domain.
65    ///
66    /// SAFETY:
67    /// Our memory region (MR) registration uses implicit ODP for RDMA access, which maps large virtual
68    /// address ranges without explicit pinning. This is convenient, but it broadens the memory footprint
69    /// exposed to the NIC and introduces a security liability.
70    ///
71    /// We currently assume a trusted, single-environment and are not enforcing finer-grained memory isolation
72    /// at this layer. We plan to investigate mitigations - such as memory windows or tighter registration
73    /// boundaries in future follow-ups.
74    ///
75    /// # Errors
76    ///
77    /// Returns errors if no RDMA devices are found, the specified device cannot be found,
78    /// device context creation fails, or protection domain allocation fails.
79    pub fn new(device: IbvDevice) -> Result<Self, anyhow::Error> {
80        tracing::debug!("creating IbvDomain for device {}", device.name());
81        unsafe {
82            let device_name = device.name();
83            let mut num_devices = 0i32;
84            let devices = rdmaxcel_sys::ibv_get_device_list(&mut num_devices as *mut _);
85
86            if devices.is_null() || num_devices == 0 {
87                return Err(anyhow::anyhow!("no RDMA devices found"));
88            }
89
90            let mut device_ptr = std::ptr::null_mut();
91            for i in 0..num_devices {
92                let dev = *devices.offset(i as isize);
93                let dev_name =
94                    CStr::from_ptr(rdmaxcel_sys::ibv_get_device_name(dev)).to_string_lossy();
95
96                if dev_name == *device_name {
97                    device_ptr = dev;
98                    break;
99                }
100            }
101
102            if device_ptr.is_null() {
103                rdmaxcel_sys::ibv_free_device_list(devices);
104                return Err(anyhow::anyhow!("device '{}' not found", device_name));
105            }
106            tracing::info!("using RDMA device: {}", device_name);
107
108            let context = rdmaxcel_sys::ibv_open_device(device_ptr);
109            if context.is_null() {
110                rdmaxcel_sys::ibv_free_device_list(devices);
111                let os_error = Error::last_os_error();
112                return Err(anyhow::anyhow!("failed to create context: {}", os_error));
113            }
114
115            let pd = rdmaxcel_sys::ibv_alloc_pd(context);
116            if pd.is_null() {
117                rdmaxcel_sys::ibv_close_device(context);
118                rdmaxcel_sys::ibv_free_device_list(devices);
119                let os_error = Error::last_os_error();
120                return Err(anyhow::anyhow!(
121                    "failed to create protection domain (PD): {}",
122                    os_error
123                ));
124            }
125
126            rdmaxcel_sys::ibv_free_device_list(devices);
127
128            let domain = IbvDomain { context, pd };
129
130            Ok(domain)
131        }
132    }
133}
