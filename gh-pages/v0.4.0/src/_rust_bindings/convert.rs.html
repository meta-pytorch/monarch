40struct MessageParser<'a> {
41    current: Bound<'a, PyAny>,
42}
43
44fn create_function(obj: Bound<'_, PyAny>) -> PyResult<ResolvableFunction> {
45    let cloudpickle = obj
46        .py()
47        .import("monarch.common.function")?
48        .getattr("ResolvableFromCloudpickle")?;
49    if obj.is_instance(&cloudpickle)? {
50        Ok(ResolvableFunction::Cloudpickle(Cloudpickle::new(
51            obj.getattr("data")?.extract()?,
52        )))
53    } else {
54        Ok(ResolvableFunction::FunctionPath(FunctionPath::new(
55            obj.getattr("path")?.extract()?,
56        )))
57    }
58}
59fn create_ref(obj: Bound<'_, PyAny>) -> PyResult<Ref> {
60    let r = obj.getattr("ref")?;
61    let v: u64 = r.extract()?;
62    Ok(v.into())
63}

65impl<'a> MessageParser<'a> {
66    fn new(current: Bound<'a, PyAny>) -> Self {
67        Self { current }
68    }
69
70    fn attr(&self, name: &str) -> PyResult<Bound<'a, PyAny>> {
71        self.current.getattr(name)
72    }
73    #[allow(non_snake_case)]
74    fn parseStreamRef(&self, name: &str) -> PyResult<StreamRef> {
75        let r = self.attr(name)?.getattr("ref")?;
76        let id: u64 = r.extract()?;
77        Ok(StreamRef { id })
78    }
79    #[allow(non_snake_case)]
80    fn parseStreamCreationMode(&self, name: &str) -> PyResult<StreamCreationMode> {
81        let default: bool = self.parse(name)?;
82        Ok(if default {
83            worker::StreamCreationMode::UseDefaultStream
84        } else {
85            worker::StreamCreationMode::CreateNewStream
86        })
87    }
88    #[allow(non_snake_case)]
89    fn parseSeq(&self, name: &str) -> PyResult<Seq> {
90        let v: u64 = self.attr(name)?.extract()?;
91        Ok(v.into())
92    }
93    #[allow(non_snake_case)]
94    fn parseFlatReferences(&self, name: &str) -> PyResult<Vec<Option<Ref>>> {
95        let tree_flatten = self
96            .current
97            .py()
98            .import("torch.utils._pytree")?
99            .getattr("tree_flatten")?;
100        let output_tuple: (Bound<'a, PyAny>, Bound<'a, PyAny>) =
101            tree_flatten.call1((self.attr(name)?,))?.extract()?;
102        let referenceable = self
103            .current
104            .py()
105            .import("monarch.common.reference")?
106            .getattr("Referenceable")?;
107        let mut flat: Vec<Option<Ref>> = vec![];
108        for x in output_tuple.0.try_iter()? {
109            let v: Bound<'a, PyAny> = x?;
110            if v.is_instance(&referenceable)? {
111                flat.push(Some(create_ref(v)?));
112            } else {
113                let r: Option<Ref> = v.extract().ok();
114                flat.push(r);
115            }
116        }
117        Ok(flat)
118    }
119    #[allow(non_snake_case)]
120    fn parseRef(&self, name: &str) -> PyResult<Ref> {
121        create_ref(self.attr(name)?)
122    }
123    #[allow(non_snake_case)]
124    fn parseOptionalRef(&self, name: &str) -> PyResult<Option<Ref>> {
125        let obj = self.attr(name)?;
126        if obj.is_none() {
127            Ok(None)
128        } else {
129            Ok(Some(create_ref(obj)?))
130        }
131    }
132
133    fn parse<T: pyo3::conversion::FromPyObject<'a>>(&self, name: &str) -> PyResult<T> {
134        self.attr(name)?.extract()
135    }
136
137    #[allow(non_snake_case)]
138    fn parseRefList(&self, name: &str) -> PyResult<Vec<Ref>> {
139        self.attr(name)?
140            .try_iter()?
141            .map(|x| {
142                let v = x?;
143                let vr: PyResult<u64> = v.extract();
144                if let Ok(v) = vr {
145                    Ok(v.into())
146                } else {
147                    create_ref(v)
148                }
149            })
150            .collect()
151    }
152
153    #[allow(non_snake_case)]
154    fn parseFunction(&self, name: &str) -> PyResult<ResolvableFunction> {
155        create_function(self.attr(name)?)
156    }
157    #[allow(non_snake_case)]
158    fn parseOptionalFunction(&self, name: &str) -> PyResult<Option<ResolvableFunction>> {
159        let f = self.attr(name)?;
160        if f.is_none() {
161            Ok(None)
162        } else {
163            Ok(Some(create_function(f)?))
164        }
165    }
166    #[allow(non_snake_case)]
167    fn parseNDSlice(&self, name: &str) -> PyResult<Slice> {
168        let slice: PySlice = self.parse(name)?;
169        Ok(slice.into())
170    }
171    #[allow(non_snake_case)]
172    fn parseFactory(&self, name: &str) -> PyResult<Factory> {
173        Factory::from_py(self.attr(name)?)
174    }
175    #[allow(non_snake_case)]
176    fn parseWorkerMessageList(&self, name: &str) -> PyResult<Vec<WorkerMessage>> {
177        self.attr(name)?.try_iter()?.map(|x| convert(x?)).collect()
178    }
179    fn parse_error_reason(
180        &self,
181        name: &str,
182    ) -> PyResult<Option<(Option<reference::ActorId>, String)>> {
183        let err = self.attr(name)?;
184        if err.is_none() {
185            return Ok(None);
186        }
187        if let Ok(actor_source_id) = err.getattr("source_actor_id") {
188            let actor_id: PyActorId = actor_source_id.extract()?;
189            return Ok(Some((
190                Some(actor_id.into()),
191                err.getattr("message")?.extract()?,
192            )));
193        }
194        let msg: String = err.to_string();
195        Ok(Some((None, msg)))
196    }
197}

199type FnType = for<'py> fn(MessageParser<'py>) -> PyResult<WorkerMessage>;
200static CONVERT_MAP: OnceLock<HashMap<u64, FnType>> = OnceLock::new();
201
202fn create_map(py: Python) -> HashMap<u64, FnType> {
203    let messages = py
204        .import("monarch.common.messages")
205        .expect("import monarch.common.messages");
206    let mut m: HashMap<u64, FnType> = HashMap::new();
207    let key = |name: &str| {
208        messages
209            .getattr(name)
210            .expect("lookup message type")
211            .as_ptr() as u64
212    };
213    m.insert(key("BackendNetworkInit"), |_p| {
214        Ok(WorkerMessage::BackendNetworkInit(
215            UniqueId::new().map_err(|err| PyRuntimeError::new_err(err.to_string()))?,
216        ))
217    });
218    m.insert(key("BackendNetworkPointToPointInit"), |p| {
219        Ok(WorkerMessage::BackendNetworkPointToPointInit {
220            from_stream: p.parseStreamRef("from_stream")?,
221            to_stream: p.parseStreamRef("to_stream")?,
222        })
223    });
224    m.insert(key("CallFunction"), |p| {
225        let function = p.parseFunction("function")?;
226        let args: Bound<'_, PyTuple> = p.parse("args")?;
227        let kwargs: Bound<'_, PyDict> = p.parse("kwargs")?;
228
229        let args_kwargs = ArgsKwargs::from_python(args.into_any(), kwargs.into_any())?;
230        Ok(WorkerMessage::CallFunction(CallFunctionParams {
231            seq: p.parseSeq("ident")?,
232            results: p.parseFlatReferences("result")?,
233            mutates: p.parseRefList("mutates")?,
234            function,
235            args_kwargs,
236            stream: p.parseStreamRef("stream")?,
237            remote_process_groups: p.parseRefList("remote_process_groups")?,
238        }))
239    });
240    m.insert(key("CreateStream"), |p| {
241        Ok(WorkerMessage::CreateStream {
242            id: p.parseStreamRef("result")?,
243            stream_creation: p.parseStreamCreationMode("default")?,
244        })
245    });
246    m.insert(key("CreateDeviceMesh"), |p| {
247        Ok(WorkerMessage::CreateDeviceMesh {
248            result: p.parseRef("result")?,
249            names: p.parse("names")?,
250            ranks: p.parseNDSlice("ranks")?,
251        })
252    });
253    m.insert(key("CreateRemoteProcessGroup"), |p| {
254        Ok(WorkerMessage::CreateRemoteProcessGroup {
255            result: p.parseRef("result")?,
256            device_mesh: p.parseRef("device_mesh")?,
257            dims: p.parse("dims")?,
258        })
259    });
260
261    m.insert(key("BorrowCreate"), |p| {
262        Ok(WorkerMessage::BorrowCreate {
263            result: p.parseRef("result")?,
264            borrow: p.parse("borrow")?,
265            tensor: p.parseRef("tensor")?,
266            from_stream: p.parseStreamRef("from_stream")?,
267            to_stream: p.parseStreamRef("to_stream")?,
268        })
269    });
270    m.insert(key("BorrowFirstUse"), |p| {
271        Ok(WorkerMessage::BorrowFirstUse {
272            borrow: p.parse("borrow")?,
273        })
274    });
275    m.insert(key("BorrowLastUse"), |p| {
276        Ok(WorkerMessage::BorrowLastUse {
277            borrow: p.parse("borrow")?,
278        })
279    });
280    m.insert(key("BorrowDrop"), |p| {
281        Ok(WorkerMessage::BorrowDrop {
282            borrow: p.parse("borrow")?,
283        })
284    });
285    m.insert(key("DeleteRefs"), |p| {
286        Ok(WorkerMessage::DeleteRefs(p.parseRefList("refs")?))
287    });
288    m.insert(key("RequestStatus"), |p| {
289        Ok(WorkerMessage::RequestStatus {
290            seq: p.parseSeq("ident")?,
291            controller: p.parse("controller")?,
292        })
293    });
294    m.insert(key("Reduce"), |p| {
295        let reduction: String = p.parse("reduction")?;
296        let reduction = match reduction.as_str() {
297            "sum" => Reduction::ReduceOp(ReduceOp::Sum),
298            "prod" => Reduction::ReduceOp(ReduceOp::Prod),
299            "stack" => Reduction::Stack,
300            "avg" => Reduction::ReduceOp(ReduceOp::Avg),
301            "min" => Reduction::ReduceOp(ReduceOp::Min),
302            "max" => Reduction::ReduceOp(ReduceOp::Max),
303            _ => {
304                return Err(PyRuntimeError::new_err(format!(
305                    "Unsupported reduction {}",
306                    reduction
307                )));
308            }
309        };
310
311        Ok(WorkerMessage::Reduce {
312            result: p.parseRef("result")?,
313            tensor: p.parseRef("local_tensor")?,
314            factory: p.parseFactory("factory")?,
315            mesh: p.parseRef("source_mesh")?,
316            stream: p.parseStreamRef("stream")?,
317            dims: p.parse("dims")?,
318            reduction,
319            scatter: p.parse("scatter")?,
320            in_place: p.parse("inplace")?,
321            out: p.parseOptionalRef("out")?,
322        })
323    });
324    m.insert(key("SendTensor"), |p| {
325        Ok(WorkerMessage::SendTensor {
326            result: p.parseRef("result")?,
327            from_ranks: p.parseNDSlice("from_ranks")?,
328            to_ranks: p.parseNDSlice("to_ranks")?,
329            tensor: p.parseRef("tensor")?,
330            factory: p.parseFactory("factory")?,
331            from_stream: p.parseStreamRef("from_stream")?,
332            to_stream: p.parseStreamRef("to_stream")?,
333        })
334    });
335    m.insert(key("SendValue"), |p|  {
336            let function = p.parseOptionalFunction("function")?;
337            let args: Bound<'_, PyTuple> = p.parse("args")?;
338            let kwargs: Bound<'_, PyDict> = p.parse("kwargs")?;
339
340            if function.is_none() && (args.len() != 1 || !kwargs.is_empty()) {
341                return Err(PyValueError::new_err(
342                    "SendValue with no function must have exactly one argument and no keyword arguments",
343                ));
344            }
345            let args_kwargs = ArgsKwargs::from_python(args.into_any(), kwargs.into_any())?;
346            Ok(WorkerMessage::SendValue {
347                seq: p.parseSeq("ident")?,
348                destination: p.parseOptionalRef("destination")?,
349                mutates: p.parseRefList("mutates")?,
350                function,
351                args_kwargs,
352                stream: p.parseStreamRef("stream")?,
353            })
354        });
355
356    m.insert(key("SplitComm"), |p| {
357        Ok(WorkerMessage::SplitComm {
358            dims: p.parse("dims")?,
359            device_mesh: p.parseRef("device_mesh")?,
360            stream: p.parseStreamRef("stream")?,
361        })
362    });
363    m.insert(key("SplitCommForProcessGroup"), |p| {
364        Ok(WorkerMessage::SplitCommForProcessGroup {
365            remote_process_group: p.parseRef("remote_process_group")?,
366            stream: p.parseStreamRef("stream")?,
367        })
368    });
369    m.insert(key("DefineRecording"), |p| {
370        Ok(WorkerMessage::DefineRecording {
371            result: p.parseRef("result")?,
372            nresults: p.parse("nresults")?,
373            nformals: p.parse("nformals")?,
374            commands: p.parseWorkerMessageList("commands")?,
375            ntotal_messages: p.parse("ntotal_messages")?,
376            index: p.parse("message_index")?,
377        })
378    });
379    m.insert(key("RecordingFormal"), |p| {
380        Ok(WorkerMessage::RecordingFormal {
381            result: p.parseRef("result")?,
382            argument_index: p.parse("argument_index")?,
383            stream: p.parseStreamRef("stream")?,
384        })
385    });
386    m.insert(key("RecordingResult"), |p| {
387        Ok(WorkerMessage::RecordingResult {
388            result: p.parseRef("input")?,
389            output_index: p.parse("output_index")?,
390            stream: p.parseStreamRef("stream")?,
391        })
392    });
393    m.insert(key("CallRecording"), |p| {
394        Ok(WorkerMessage::CallRecording {
395            seq: p.parseSeq("ident")?,
396            recording: p.parseRef("recording")?,
397            results: p.parseRefList("results")?,
398            actuals: p.parseRefList("actuals")?,
399        })
400    });
401    m.insert(key("PipeRecv"), |p| {
402        Ok(WorkerMessage::PipeRecv {
403            seq: p.parseSeq("ident")?,
404            results: p.parseFlatReferences("result")?,
405            pipe: p.parseRef("pipe")?,
406            stream: p.parseStreamRef("stream")?,
407        })
408    });
409    m.insert(key("Exit"), |p| {
410        Ok(WorkerMessage::Exit {
411            error: p.parse_error_reason("error")?,
412        })
413    });
414    m.insert(key("CommandGroup"), |p| {
415        Ok(WorkerMessage::CommandGroup(
416            p.parseWorkerMessageList("commands")?,
417        ))
418    });
419    m.insert(key("SendResultOfActorCall"), |p| {
420        Ok(WorkerMessage::SendResultOfActorCall(
421            worker::ActorCallParams {
422                seq: p.parseSeq("seq")?,
423                broker_id: p.parse("broker_id")?,
424                local_state: p.parseRefList("local_state")?,
425                mutates: p.parseRefList("mutates")?,
426                stream: p.parseStreamRef("stream")?,
427            },
428        ))
429    });
430    m.insert(key("CallActorMethod"), |p| {
431        Ok(WorkerMessage::CallActorMethod(worker::ActorMethodParams {
432            call: worker::ActorCallParams {
433                seq: p.parseSeq("seq")?,
434                broker_id: p.parse("broker_id")?,
435                local_state: p.parseRefList("local_state")?,
436                mutates: p.parseRefList("mutates")?,
437                stream: p.parseStreamRef("stream")?,
438            },
439            results: p.parseFlatReferences("result")?,
440        }))
441    });
442    m
443}

445pub fn convert<'py>(m: Bound<'py, PyAny>) -> PyResult<WorkerMessage> {
446    let converter = {
447        let typ = m.get_type().as_ptr() as u64;
448        CONVERT_MAP.get_or_init(|| create_map(m.py()))[&typ]
449    };
450    converter(MessageParser::new(m))
451}
