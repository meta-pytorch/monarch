38/// Segment scanner callback that uses PyTorch's memory snapshot API.
39///
40/// This function calls torch.cuda.memory._snapshot() to get CUDA memory segments
41/// and fills the provided buffer with segment information.
42///
43/// # Safety
44/// This function is called from C code as a callback.
45unsafe extern "C" fn pytorch_segment_scanner(
46    segments_out: *mut monarch_rdma::rdmaxcel_sys::rdmaxcel_scanned_segment_t,
47    max_segments: usize,
48) -> usize {
49    // Acquire the GIL to call Python code
50    // Note: We use Python::with_gil here instead of monarch_with_gil_blocking because
51    // the raw pointer segments_out is not Sync and monarch_with_gil_blocking requires Send.
52    let result = Python::with_gil(|py| -> PyResult<usize> {
53        // Check if torch is already imported - don't import it ourselves
54        let sys = py.import("sys")?;
55        let modules = sys.getattr("modules")?;
56
57        // Try to get torch from sys.modules
58        let torch = match modules.get_item("torch") {
59            Ok(torch_module) => torch_module,
60            Err(_) => {
61                // torch not imported yet, return 0 segments
62                return Ok(0);
63            }
64        };
65
66        // Check if CUDA is available
67        let cuda_available: bool = torch
68            .getattr("cuda")?
69            .getattr("is_available")?
70            .call0()?
71            .extract()?;
72
73        if !cuda_available {
74            return Ok(0);
75        }
76
77        // Call torch.cuda.memory._snapshot()
78        let snapshot = torch
79            .getattr("cuda")?
80            .getattr("memory")?
81            .getattr("_snapshot")?
82            .call0()?;
83
84        // Get the segments list from the snapshot dict
85        let segments = snapshot.get_item("segments")?;
86        let segments_list: Vec<Bound<'_, PyAny>> = segments.extract()?;
87
88        let num_segments = segments_list.len();
89
90        // Fill the output buffer with as many segments as will fit
91        let segments_to_write = num_segments.min(max_segments);
92
93        for (i, segment) in segments_list.iter().take(segments_to_write).enumerate() {
94            // Extract fields from the segment dict
95            let address: u64 = segment.get_item("address")?.extract()?;
96            let total_size: usize = segment.get_item("total_size")?.extract()?;
97            let device: i32 = segment.get_item("device")?.extract()?;
98            let is_expandable: bool = segment.get_item("is_expandable")?.extract()?;
99
100            // Write to the output buffer - only the fields the scanner needs to provide
101            let seg_info = &mut *segments_out.add(i);
102            seg_info.address = address as usize;
103            seg_info.size = total_size;
104            seg_info.device = device;
105            seg_info.is_expandable = if is_expandable { 1 } else { 0 };
106        }
107
108        // Return total number of segments found (may be > max_segments)
109        Ok(num_segments)
110    });
111
112    match result {
113        Ok(count) => count,
114        Err(e) => {
115            // Log the specific error for debugging
116            eprintln!("[monarch_rdma] pytorch_segment_scanner failed: {}", e);
117            0
118        }
119    }
120}

122fn setup_rdma_context(
123    rdma_buffer: &PyRdmaBuffer,
124    local_proc_id: String,
125) -> (ActorRef<RdmaManagerActor>, RdmaBuffer) {
126    let proc_id: ProcId = local_proc_id.parse().unwrap();
127    // TODO: find some better way to look this up, or else formally define "service names"
128    let local_owner_id = ActorId(proc_id, "rdma_manager".to_string(), 0);
129    let local_owner_ref: ActorRef<RdmaManagerActor> = ActorRef::attest(local_owner_id);
130    let buffer = rdma_buffer.buffer.clone();
131    (local_owner_ref, buffer)
132}

141async fn create_rdma_buffer(
142    addr: usize,
143    size: usize,
144    proc_id: ProcId,
145    client: PyInstance,
146) -> PyResult<PyRdmaBuffer> {
147    // Get the owning RdmaManagerActor's ActorRef
148    // TODO: find some better way to look this up, or else formally define "service names"
149    let owner_id = ActorId(proc_id, "rdma_manager".to_string(), 0);
150    let owner_ref: ActorRef<RdmaManagerActor> = ActorRef::attest(owner_id);
151
152    // Create the RdmaBuffer
153    let buffer = owner_ref
154        .request_buffer_deprecated(client.deref(), addr, size)
155        .await?;
156    Ok(PyRdmaBuffer { buffer, owner_ref })
157}

159#[pymethods]
160impl PyRdmaBuffer {
161    #[classmethod]
162    fn create_rdma_buffer_nonblocking<'py>(
163        _cls: &Bound<'_, PyType>,
164        _py: Python<'py>,
165        addr: usize,
166        size: usize,
167        proc_id: String,
168        client: PyInstance,
169    ) -> PyResult<PyPythonTask> {
170        if !rdma_supported() {
171            return Err(PyException::new_err("RDMA is not supported on this system"));
172        }
173        PyPythonTask::new(create_rdma_buffer(
174            addr,
175            size,
176            proc_id.parse().unwrap(),
177            client,
178        ))
179    }
180
181    #[classmethod]
182    fn create_rdma_buffer_blocking<'py>(
183        _cls: &Bound<'_, PyType>,
184        py: Python<'py>,
185        addr: usize,
186        size: usize,
187        proc_id: String,
188        client: PyInstance,
189    ) -> PyResult<PyRdmaBuffer> {
190        if !rdma_supported() {
191            return Err(PyException::new_err("RDMA is not supported on this system"));
192        }
193        signal_safe_block_on(
194            py,
195            create_rdma_buffer(addr, size, proc_id.parse().unwrap(), client),
196        )?
197    }
198
199    #[classmethod]
200    fn rdma_supported<'py>(_cls: &Bound<'_, PyType>, _py: Python<'py>) -> bool {
201        rdma_supported()
202    }
203
204    #[pyo3(name = "__repr__")]
205    fn repr(&self) -> String {
206        format!("<RdmaBuffer'{:?}'>", self.buffer)
207    }
208
209    /// Reads data from the local buffer and places it into this remote RDMA buffer.
210    ///
211    /// This operation appears as "read_into" from the caller's perspective (reading from local memory
212    /// into the remote buffer), but internally it's implemented as a "write_from" operation on the
213    /// local buffer since the data flows from the local buffer to the remote one.
214    ///
215    /// # Arguments
216    /// * `addr` - The address of the local buffer to read from
217    /// * `size` - The size of the data to transfer
218    /// * `local_proc_id` - The process ID where the local buffer resides
219    /// * `client` - The actor who does the reading.
220    /// * `timeout` - Maximum time in milliseconds to wait for the operation
221    #[pyo3(signature = (addr, size, local_proc_id, client, timeout))]
222    fn read_into<'py>(
223        &self,
224        _py: Python<'py>,
225        addr: usize,
226        size: usize,
227        local_proc_id: String,
228        client: PyInstance,
229        timeout: u64,
230    ) -> PyResult<PyPythonTask> {
231        let (local_owner_ref, buffer) = setup_rdma_context(self, local_proc_id);
232        PyPythonTask::new(async move {
233            let local_buffer = local_owner_ref
234                .request_buffer_deprecated(client.deref(), addr, size)
235                .await?;
236            local_buffer
237                .write_from(client.deref(), buffer, timeout)
238                .await
239                .map_err(|e| PyException::new_err(format!("failed to read into buffer: {}", e)))?;
240            local_owner_ref
241                .release_buffer_deprecated(client.deref(), local_buffer)
242                .await?;
243            Ok(())
244        })
245    }
246
247    /// Writes data from this remote RDMA buffer into a local buffer.
248    ///
249    /// This operation appears as "write_from" from the caller's perspective (writing from the remote
250    /// buffer into local memory), but internally it's implemented as a "read_into" operation on the
251    /// local buffer since the data flows from the remote buffer to the local one.
252    ///
253    /// # Arguments
254    /// * `addr` - The address of the local buffer to write to
255    /// * `size` - The size of the data to transfer
256    /// * `local_proc_id` - The process ID where the local buffer resides
257    /// * `client` - The actor who does the writing
258    /// * `timeout` - Maximum time in milliseconds to wait for the operation
259    #[pyo3(signature = (addr, size, local_proc_id, client, timeout))]
260    fn write_from<'py>(
261        &self,
262        _py: Python<'py>,
263        addr: usize,
264        size: usize,
265        local_proc_id: String,
266        client: PyInstance,
267        timeout: u64,
268    ) -> PyResult<PyPythonTask> {
269        let (local_owner_ref, buffer) = setup_rdma_context(self, local_proc_id);
270        PyPythonTask::new(async move {
271            let local_buffer = local_owner_ref
272                .request_buffer_deprecated(client.deref(), addr, size)
273                .await?;
274            local_buffer
275                .read_into(client.deref(), buffer, timeout)
276                .await
277                .map_err(|e| PyException::new_err(format!("failed to write from buffer: {}", e)))?;
278            local_owner_ref
279                .release_buffer_deprecated(client.deref(), local_buffer)
280                .await?;
281            Ok(())
282        })
283    }
284
285    fn size(&self) -> usize {
286        self.buffer.size
287    }
288
289    fn __reduce__(&self) -> PyResult<(PyObject, PyObject)> {
290        monarch_with_gil_blocking(|py| {
291            let ctor = py.get_type::<PyRdmaBuffer>().into_py_any(py)?;
292            let json = serde_json::to_string(self).map_err(|e| {
293                PyErr::new::<PyValueError, _>(format!("Serialization failed: {}", e))
294            })?;
295
296            let args = PyTuple::new(py, [json])?.into_py_any(py)?;
297            Ok((ctor, args))
298        })
299    }
300
301    #[new]
302    fn new_from_json(json: &str) -> PyResult<Self> {
303        let deserialized: PyRdmaBuffer = serde_json::from_str(json)
304            .map_err(|e| PyErr::new::<PyValueError, _>(format!("Deserialization failed: {}", e)))?;
305        Ok(deserialized)
306    }
307
308    fn drop<'py>(
309        &self,
310        _py: Python<'py>,
311        local_proc_id: String,
312        client: PyInstance,
313    ) -> PyResult<PyPythonTask> {
314        let (_local_owner_ref, buffer) = setup_rdma_context(self, local_proc_id);
315        PyPythonTask::new(async move {
316            buffer
317                .drop_buffer(client.deref())
318                .await
319                .map_err(|e| PyException::new_err(format!("Failed to drop buffer: {}", e)))?;
320            Ok(())
321        })
322    }
323
324    fn owner_actor_id(&self) -> String {
325        self.owner_ref.actor_id().to_string()
326    }

329#[pyclass(name = "_RdmaManager", module = "monarch._rust_bindings.rdma")]
330pub struct PyRdmaManager {
331    #[allow(dead_code)] // field never read
332    inner: SharedCell<RootActorMesh<'static, RdmaManagerActor>>,
333    device: String,
334}
335
336#[pymethods]
337impl PyRdmaManager {
338    #[pyo3(name = "__repr__")]
339    fn repr(&self) -> String {
340        format!("<RdmaManager(device='{}')>", self.device)
341    }
342
343    #[getter]
344    fn device(&self) -> &str {
345        &self.device
346    }
347    /// Creates an RDMA manager actor on the given ProcMesh (async version).
348    /// Returns the actor mesh if RDMA is supported, None otherwise.
349    #[classmethod]
350    fn create_rdma_manager_nonblocking(
351        _cls: &Bound<'_, PyType>,
352        proc_mesh: &Bound<'_, PyAny>,
353        client: PyInstance,
354    ) -> PyResult<PyPythonTask> {
355        tracing::debug!("spawning RDMA manager on target proc_mesh nodes");
356
357        let proc_mesh = proc_mesh.downcast::<PyProcMesh>()?.borrow().mesh_ref()?;
358        PyPythonTask::new(async move {
359            let actor_mesh: hyperactor_mesh::v1::ActorMesh<RdmaManagerActor> = proc_mesh
360                // Pass None to use default config - RdmaManagerActor will use default IbverbsConfig
361                // TODO - make IbverbsConfig configurable
362                .spawn_service(client.deref(), "rdma_manager", &None)
363                .await
364                .map_err(|err| PyException::new_err(err.to_string()))?;
365
366            let actor_mesh = RootActorMesh::from(actor_mesh);
367            let actor_mesh = SharedCell::from(actor_mesh);
368
369            Ok(Some(PyRdmaManager {
370                inner: actor_mesh,
371                device: "remote_rdma_device".to_string(),
372            }))
373        })
374    }
375}

377pub fn register_python_bindings(module: &Bound<'_, PyModule>) -> PyResult<()> {
378    // Register the PyTorch segment scanner callback.
379    // This calls torch.cuda.memory._snapshot() to get CUDA memory segments.
380    register_segment_scanner(Some(pytorch_segment_scanner));
381
382    module.add_class::<PyRdmaBuffer>()?;
383    module.add_class::<PyRdmaManager>()?;
384    Ok(())
