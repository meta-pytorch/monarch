74/// Error type for build utilities

76pub enum BuildError {
77    CudaNotFound,
78    PythonNotFound,
79    CommandFailed(String),
80    PathNotFound(String),
81}

83impl std::fmt::Display for BuildError {
84    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
85        match self {
86            BuildError::CudaNotFound => write!(f, "CUDA installation not found"),
87            BuildError::PythonNotFound => write!(f, "Python interpreter not found"),
88            BuildError::CommandFailed(cmd) => write!(f, "Command failed: {}", cmd),
89            BuildError::PathNotFound(path) => write!(f, "Path not found: {}", path),
90        }
91    }
92}

102/// Find CUDA home directory using various heuristics
103///
104/// This function attempts to locate CUDA installation through:
105/// 1. CUDA_HOME environment variable
106/// 2. CUDA_PATH environment variable
107/// 3. Finding nvcc in PATH and deriving cuda home
108/// 4. Platform-specific default locations
109pub fn find_cuda_home() -> Option<String> {
110    // Guess #1: Environment variables
111    let mut cuda_home = get_env_var_with_rerun("CUDA_HOME")
112        .ok()
113        .or_else(|| get_env_var_with_rerun("CUDA_PATH").ok());
114
115    if cuda_home.is_none() {
116        // Guess #2: Find nvcc in PATH
117        if let Ok(nvcc_path) = which("nvcc") {
118            // Get parent directory twice (nvcc is in CUDA_HOME/bin)
119            if let Some(cuda_dir) = nvcc_path.parent().and_then(|p| p.parent()) {
120                cuda_home = Some(cuda_dir.to_string_lossy().into_owned());
121            }
122        } else {
123            // Guess #3: Platform-specific defaults
124            if cfg!(windows) {
125                let pattern = r"C:\Program Files\NVIDIA GPU Computing Toolkit\CUDA\v*.*";
126                let cuda_homes: Vec<_> = glob(pattern).unwrap().filter_map(Result::ok).collect();
127                if !cuda_homes.is_empty() {
128                    cuda_home = Some(cuda_homes[0].to_string_lossy().into_owned());
129                }
130            } else {
131                // Unix-like systems
132                let cuda_candidate = "/usr/local/cuda";
133                if Path::new(cuda_candidate).exists() {
134                    cuda_home = Some(cuda_candidate.to_string());
135                }
136            }
137        }
138    }
139
140    cuda_home
141}

143/// Discover CUDA configuration including home, include dirs, and lib dirs
144pub fn discover_cuda_config() -> Result<CudaConfig, BuildError> {
145    let cuda_home_path = PathBuf::from(find_cuda_home().ok_or(BuildError::CudaNotFound)?);
146
147    let mut config = CudaConfig {
148        cuda_home: cuda_home_path.clone(),
149        include_dirs: Vec::new(),
150        lib_dirs: Vec::new(),
151    };
152
153    // Add standard include directories
154    // Check both old-style (include) and new-style (targets/x86_64-linux/include) CUDA installations
155    for include_subdir in &["include", "targets/x86_64-linux/include"] {
156        let include_dir = cuda_home_path.join(include_subdir);
157        if include_dir.exists() {
158            config.include_dirs.push(include_dir);
159        }
160    }
161
162    // Add standard library directories
163    // Check both old-style (lib64, lib) and new-style (targets/x86_64-linux/lib) CUDA installations
164    for lib_subdir in &["lib64", "lib", "lib/x64", "targets/x86_64-linux/lib"] {
165        let lib_dir = cuda_home_path.join(lib_subdir);
166        if lib_dir.exists() {
167            config.lib_dirs.push(lib_dir);
168            break; // Use first found
169        }
170    }
171
172    Ok(config)
173}

175/// Validate CUDA installation exists and is complete
176pub fn validate_cuda_installation() -> Result<String, BuildError> {
177    let cuda_config = discover_cuda_config()?;
178    let cuda_home_str = cuda_config.cuda_home.to_string_lossy().to_string();
179
180    // Verify CUDA include directory exists
181    let cuda_include_path = cuda_config.cuda_home.join("include");
182    if !cuda_include_path.exists() {
183        return Err(BuildError::PathNotFound(format!(
184            "CUDA include directory at {}",
185            cuda_include_path.display()
186        )));
187    }
188
189    Ok(cuda_home_str)
190}

192/// Get CUDA library directory
193///
194/// Searches for the directory containing libcudart_static.a in the CUDA installation.
195/// Panics with a helpful error message if not found.
196pub fn get_cuda_lib_dir() -> String {
197    // Check if user explicitly set CUDA_LIB_DIR and verify it contains the library
198    if let Ok(cuda_lib_dir) = env::var("CUDA_LIB_DIR") {
199        let lib_path = PathBuf::from(&cuda_lib_dir);
200        let cudart_static = lib_path.join("libcudart_static.a");
201        if !cudart_static.exists() {
202            panic!(
203                "CUDA_LIB_DIR is set to '{}' but libcudart_static.a not found at {}",
204                cuda_lib_dir,
205                cudart_static.display()
206            );
207        }
208        return cuda_lib_dir;
209    }
210
211    // Try to deduce from CUDA configuration
212    let cuda_config = match discover_cuda_config() {
213        Ok(config) => config,
214        Err(_) => {
215            eprintln!("Error: CUDA installation not found!");
216            eprintln!("Please ensure CUDA is installed and one of the following is true:");
217            eprintln!(
218                "  1. Set CUDA_HOME environment variable to your CUDA installation directory"
219            );
220            eprintln!(
221                "  2. Set CUDA_PATH environment variable to your CUDA installation directory"
222            );
223            eprintln!("  3. Ensure 'nvcc' is in your PATH");
224            eprintln!("  4. Install CUDA to the default location (/usr/local/cuda on Linux)");
225            eprintln!();
226            eprintln!("Example: export CUDA_HOME=/usr/local/cuda-12.0");
227            panic!("CUDA installation not found");
228        }
229    };
230
231    let cuda_home = &cuda_config.cuda_home;
232    // Check both old-style and new-style CUDA library paths
233    // Look for the actual cudart_static library file to ensure we find the right directory
234    let libs = &["lib64", "lib", "targets/x86_64-linux/lib"];
235    for lib_subdir in libs {
236        let lib_path = cuda_home.join(lib_subdir);
237        let cudart_static = lib_path.join("libcudart_static.a");
238        if cudart_static.exists() {
239            return lib_path.to_string_lossy().to_string();
240        }
241    }
242
243    panic!(
244        "CUDA library directories {:#?} under {} do not contain libcudart_static.a",
245        libs,
246        cuda_home.display()
247    );
248}

257/// Discover Python environment directories with specific interpreter
258pub fn python_env_dirs_with_interpreter(interpreter: &str) -> Result<PythonConfig, BuildError> {
259    let output = std::process::Command::new(interpreter)
260        .arg("-c")
261        .arg(PYTHON_PRINT_DIRS)
262        .output()
263        .map_err(|_| BuildError::CommandFailed(format!("running {}", interpreter)))?;
264
265    if !output.status.success() {
266        return Err(BuildError::CommandFailed(format!(
267            "{} exited with error",
268            interpreter
269        )));
270    }
271
272    let mut include_dir = None;
273    let mut lib_dir = None;
274
275    for line in String::from_utf8_lossy(&output.stdout).lines() {
276        if let Some(path) = line.strip_prefix("PYTHON_INCLUDE_DIR: ") {
277            include_dir = Some(path.to_string());
278        }
279        if let Some(path) = line.strip_prefix("PYTHON_LIB_DIR: ") {
280            lib_dir = Some(path.to_string());
281        }
282    }
283
284    Ok(PythonConfig {
285        include_dir,
286        lib_dir,
287    })
288}

311/// Emit cargo directives to statically link libstdc++
312///
313/// This finds the GCC library path containing libstdc++.a and emits the
314/// appropriate cargo directives to link it statically. This avoids runtime
315/// dependency on system libstdc++.so which can cause GLIBCXX version conflicts.
316///
317/// Uses the `cc` crate to detect the C++ compiler, ensuring we use the same
318/// compiler that `cc::Build` and `cxx_build` would use.
319pub fn link_libstdcpp_static() {
320    // Use cc crate to get the C++ compiler, same as cc::Build and cxx_build use
321    let compiler = cc::Build::new().cpp(true).get_compiler();
322    let gcc_lib_path = std::process::Command::new(compiler.path())
323        .args(["-print-file-name=libstdc++.a"])
324        .output()
325        .ok()
326        .and_then(|output| {
327            if output.status.success() {
328                String::from_utf8(output.stdout).ok().and_then(|s| {
329                    let path = PathBuf::from(s.trim());
330                    path.parent().and_then(|p| {
331                        // parent() returns Some("") for relative paths with one component:
332                        // https://doc.rust-lang.org/stable/std/path/struct.PathBuf.html#method.parent
333                        if p == Path::new("") {
334                            None
335                        } else {
336                            Some(p.to_path_buf())
337                        }
338                    })
339                })
340            } else {
341                None
342            }
343        });
344    if let Some(gcc_lib_path) = gcc_lib_path {
345        if !gcc_lib_path.as_os_str().is_empty() {
346            println!("cargo:rustc-link-search=native={}", gcc_lib_path.display());
347        }
348    }
349    println!("cargo:rustc-link-lib=static=stdc++");
350}

352/// Configuration for rdma-core static libraries from monarch_cpp_static_libs.
353///
354/// Use `CppStaticLibsConfig::from_env()` to get the paths, then use the include
355/// directory for bindgen/cc, and call `emit_link_directives()` to link.
356pub struct CppStaticLibsConfig {
357    /// Include directory for rdma-core headers
358    pub rdma_include_dir: String,
359    /// Paths to static libraries (.a files) to link
360    pub rdma_static_libraries: Vec<String>,
361}
362
363impl CppStaticLibsConfig {
364    /// Load configuration from DEP_* environment variables set by monarch_cpp_static_libs.
365    ///
366    /// The monarch_cpp_static_libs crate must be listed as a build-dependency.
367    pub fn from_env() -> Self {
368        let rdma_include_dir = std::env::var("DEP_MONARCH_CPP_STATIC_LIBS_RDMA_INCLUDE_DIR")
369            .expect("DEP_MONARCH_CPP_STATIC_LIBS_RDMA_INCLUDE_DIR not set - add monarch_cpp_static_libs as build-dependency");
370
371        let rdma_static_libraries = std::env::var("DEP_MONARCH_CPP_STATIC_LIBS_RDMA_STATIC_LIBRARIES")
372            .expect("DEP_MONARCH_CPP_STATIC_LIBS_RDMA_STATIC_LIBRARIES not set - add monarch_cpp_static_libs as build-dependency")
373            .split(';')
374            .map(|s| s.to_string())
375            .collect();
376
377        Self {
378            rdma_include_dir,
379            rdma_static_libraries,
380        }
381    }
382
383    /// Emit all cargo link directives for static linking of rdma-core.
384    ///
385    /// This uses direct paths to the .a files we built, avoiding any path ordering
386    /// issues where the linker might find system libraries or libraries built with
387    /// different flags (e.g., ENABLE_RESOLVE_NEIGH=1).
388    pub fn emit_link_directives(&self) {
389        for lib_path in &self.rdma_static_libraries {
390            println!("cargo::rustc-link-arg={}", lib_path);
391        }
392    }
393}

395/// Convenience function to set up rdma-core static linking.
396///
397/// Returns the config with include paths, and emits all link directives.
398/// The monarch_cpp_static_libs crate must be listed as a build-dependency.
399///
400/// Example:
401/// ```ignore
402/// let config = build_utils::setup_cpp_static_libs();
403/// // Use config.rdma_include_dir for bindgen/cc
404/// ```
405pub fn setup_cpp_static_libs() -> CppStaticLibsConfig {
406    let config = CppStaticLibsConfig::from_env();
407    config.emit_link_directives();
408    config
409}

411/// Set rpath for Python library directory
412///
413/// This emits cargo directives to add the Python library directory to the rpath,
414/// allowing binaries to find libpython at runtime. This is safe to call
415/// unconditionally - when building extension modules with pyo3's extension-module
416/// feature, the rpath won't cause issues since libpython won't be linked anyway.
417///
418/// Note: This function only works in OSS builds (Cargo). In BUCK builds, it's a no-op
419/// since BUCK handles library paths differently.
420pub fn set_python_rpath() {
421    // pyo3-build-config is only available in OSS builds (Cargo), not in BUCK builds
422    #[cfg(not(fbcode_build))]
423    {
424        let python_config = pyo3_build_config::get();
425
426        if let Some(lib_dir) = &python_config.lib_dir {
427            println!("cargo::rustc-link-arg=-Wl,-rpath,{}", lib_dir);
428        }
429    }
430}
