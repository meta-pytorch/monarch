33/// Type alias for trace event fields
34/// We expect that most trace events have fewer than 4 fields.
35pub(crate) type TraceFields = SmallVec<[(&'static str, FieldValue); 4]>;
36
37#[inline]
38pub(crate) fn get_field<'a>(fields: &'a TraceFields, key: &str) -> Option<&'a FieldValue> {
39    fields.iter().find(|(k, _)| *k == key).map(|(_, v)| v)
40}

42/// Unified representation of a trace event captured from the tracing layer.
43/// This is captured once on the application thread, then sent to the background
44/// worker for fan-out to multiple exporters.
45#[derive(Debug, Clone)]
46pub enum TraceEvent {
47    /// A new span was created (on_new_span)
48    NewSpan {
49        id: u64,
50        name: &'static str,
51        target: &'static str,
52        level: tracing::Level,
53        fields: TraceFields,
54        timestamp: SystemTime,
55        parent_id: Option<u64>,
56        thread_name: &'static str,
57        file: Option<&'static str>,
58        line: Option<u32>,
59    },
60    /// A span was entered (on_enter)
61    SpanEnter { id: u64, timestamp: SystemTime },
62    /// A span was exited (on_exit)
63    SpanExit { id: u64, timestamp: SystemTime },
64    /// A span was closed (dropped)
65    SpanClose { id: u64, timestamp: SystemTime },
66    /// A tracing event occurred (e.g., tracing::info!())
67    Event {
68        name: &'static str,
69        target: &'static str,
70        level: tracing::Level,
71        fields: TraceFields,
72        timestamp: SystemTime,
73        parent_span: Option<u64>,
74        thread_id: &'static str,
75        thread_name: &'static str,
76        module_path: Option<&'static str>,
77        file: Option<&'static str>,
78        line: Option<u32>,
79    },
80}

82/// Simplified field value representation for trace events
83#[derive(Debug, Clone)]
84pub enum FieldValue {
85    Bool(bool),
86    I64(i64),
87    U64(u64),
88    F64(f64),
89    Str(String),
90    Debug(String),

93/// Trait for sinks that receive trace events from the dispatcher.
94/// Implementations run on the background worker thread and can perform
95/// expensive I/O operations without blocking the application.
96pub trait TraceEventSink: Send + 'static {
97    /// Consume a single event. Called on background thread.
98    fn consume(&mut self, event: &TraceEvent) -> Result<(), anyhow::Error>;
99
100    /// Optional target/level filter for this sink.
101    ///
102    /// The worker loop automatically applies this filter before calling `consume()`,
103    /// so sinks don't need to check target/level in their consume implementation.
104    /// Only `NewSpan` and `Event` are filtered by target/level; other event types
105    /// are always passed through.
106    ///
107    /// # Returns
108    /// - `None` - No filtering, all events are consumed (default)
109    /// - `Some(Targets)` - Only consume events matching the target/level filter
110    ///
111    /// # Example
112    /// ```ignore
113    /// fn target_filter(&self) -> Option<&Targets> {
114    ///     Some(Targets::new()
115    ///         .with_target("opentelemetry", LevelFilter::OFF)
116    ///         .with_default(LevelFilter::DEBUG))
117    /// }
118    /// ```
119    fn target_filter(&self) -> Option<&Targets> {
120        None
121    }
122
123    /// Flush any buffered events to the backend.
124    /// Called periodically and on shutdown.
125    fn flush(&mut self) -> Result<(), anyhow::Error>;
126
127    /// Optional: return name for debugging/logging
128    fn name(&self) -> &str {
129        std::any::type_name::<Self>()
130    }
131}

133thread_local! {
134    /// Cached thread info (thread_name, thread_id) for minimal overhead.
135    /// Strings are leaked once per thread to get &'static str - threads are long-lived so this is fine.
136    /// Uses Cell since (&'static str, &'static str) is Copy.
137    static CACHED_THREAD_INFO: Cell<Option<(&'static str, &'static str)>> = const { Cell::new(None) };
138}
139
140#[inline(always)]
141fn get_thread_info() -> (&'static str, &'static str) {
142    CACHED_THREAD_INFO.with(|cache| {
143        if let Some(info) = cache.get() {
144            return info;
145        }
146
147        let thread_name: &'static str = Box::leak(
148            std::thread::current()
149                .name()
150                .unwrap_or("")
151                .to_string()
152                .into_boxed_str(),
153        );
154
155        #[cfg(target_os = "linux")]
156        let thread_id: &'static str = {
157            // SAFETY: syscall(SYS_gettid) is always safe to call - it's a read-only
158            // syscall that returns the current thread's kernel thread ID (TID).
159            // The cast to u64 is safe because gettid() returns a positive pid_t.
160            let tid = unsafe { libc::syscall(libc::SYS_gettid) as u64 };
161            Box::leak(tid.to_string().into_boxed_str())
162        };
163        #[cfg(not(target_os = "linux"))]
164        let thread_id: &'static str = {
165            let tid = std::thread::current().id();
166            // SAFETY: ThreadId is a newtype wrapper around a u64 counter.
167            // This transmute relies on the internal representation of ThreadId,
168            // which is stable in practice but not guaranteed by Rust's API.
169            // On non-Linux platforms this is a best-effort approximation.
170            // See: https://doc.rust-lang.org/std/thread/struct.ThreadId.html
171            let tid_num = unsafe { std::mem::transmute::<std::thread::ThreadId, u64>(tid) };
172            Box::leak(tid_num.to_string().into_boxed_str())
173        };
174
175        cache.set(Some((thread_name, thread_id)));
176        (thread_name, thread_id)
177    })
178}

186/// The trace event dispatcher that captures events once and dispatches to multiple sinks
187/// on a background thread.
188pub struct TraceEventDispatcher {
189    sender: Option<mpsc::SyncSender<TraceEvent>>,
190    /// Separate channel so we are always notified of when the main queue is full and events are being dropped.
191    dropped_sender: Option<mpsc::Sender<TraceEvent>>,
192    _worker_handle: WorkerHandle,
193    max_level: Option<LevelFilter>,
194    dropped_events: Arc<AtomicU64>,
195}
196
197struct WorkerHandle {
198    join_handle: Option<JoinHandle<()>>,
199}
200
201impl TraceEventDispatcher {
202    /// Create a new trace event dispatcher with the given sinks.
203    /// Uses a bounded channel (capacity QUEUE_CAPACITY) to ensure telemetry never blocks
204    /// the application. Events are dropped with a warning if the queue is full.
205    /// A separate unbounded priority channel guarantees delivery of critical events
206    /// like drop notifications (safe because drop events are rate-limited).
207    ///
208    /// Takes the global control receiver for dynamic sink registration. Sinks registered
209    /// via `register_sink()` before or after this call will be added to the dispatcher.
210    ///
211    /// # Arguments
212    /// * `sinks` - List of sinks to dispatch events to.
213    pub(crate) fn new(sinks: Vec<Box<dyn TraceEventSink>>) -> Self {
214        let max_level = Self::derive_max_level(&sinks);
215
216        let (sender, receiver) = mpsc::sync_channel(QUEUE_CAPACITY);
217        let (dropped_sender, dropped_receiver) = mpsc::channel();
218        // Take the global control receiver - sinks registered via register_sink() will be received here
219        let control_receiver = crate::take_sink_control_receiver();
220        let dropped_events = Arc::new(AtomicU64::new(0));
221        let dropped_events_worker = Arc::clone(&dropped_events);
222
223        let worker_handle = std::thread::Builder::new()
224            .name("telemetry-worker".into())
225            .spawn(move || {
226                worker_loop(
227                    receiver,
228                    dropped_receiver,
229                    control_receiver,
230                    sinks,
231                    dropped_events_worker,
232                );
233            })
234            .expect("failed to spawn telemetry worker thread");
235
236        Self {
237            sender: Some(sender),
238            dropped_sender: Some(dropped_sender),
239            _worker_handle: WorkerHandle {
240                join_handle: Some(worker_handle),
241            },
242            max_level,
243            dropped_events,
244        }
245    }
246
247    fn derive_max_level(sinks: &[Box<dyn TraceEventSink>]) -> Option<LevelFilter> {
248        let mut max_level: Option<LevelFilter> = None;
249
250        for sink in sinks {
251            let sink_max = match sink.target_filter() {
252                None => LevelFilter::TRACE,
253                Some(targets) => {
254                    let levels = [
255                        (tracing::Level::TRACE, LevelFilter::TRACE),
256                        (tracing::Level::DEBUG, LevelFilter::DEBUG),
257                        (tracing::Level::INFO, LevelFilter::INFO),
258                        (tracing::Level::WARN, LevelFilter::WARN),
259                        (tracing::Level::ERROR, LevelFilter::ERROR),
260                    ];
261                    let mut result = LevelFilter::OFF;
262                    for (level, filter) in levels {
263                        if targets.would_enable("", &level) {
264                            result = filter;
265                            break;
266                        }
267                    }
268                    result
269                }
270            };
271
272            max_level = Some(match max_level {
273                None => sink_max,
274                Some(current) => std::cmp::max(current, sink_max),
275            });
276        }
277
278        max_level
279    }
280
281    fn send_event(&self, event: TraceEvent) {
282        if let Some(sender) = &self.sender {
283            if let Err(mpsc::TrySendError::Full(_)) = sender.try_send(event) {
284                let dropped = self.dropped_events.fetch_add(1, Ordering::Relaxed) + 1;
285
286                if dropped == 1 || dropped.is_multiple_of(1000) {
287                    eprintln!(
288                        "[telemetry]: {}  events and log lines dropped que to full queue (capacity: {})",
289                        dropped, QUEUE_CAPACITY
290                    );
291                    self.send_drop_event(dropped);
292                }
293            }
294        }
295    }
296
297    fn send_drop_event(&self, total_dropped: u64) {
298        if let Some(dropped_sender) = &self.dropped_sender {
299            let (thread_name, thread_id) = get_thread_info();
300
301            let mut fields = TraceFields::new();
302            fields.push((
303                "message",
304                FieldValue::Str(format!(
305                    "Telemetry events and log lines dropped due to full queue (capacity: {}). Worker may be falling behind.",
306                    QUEUE_CAPACITY
307                )),
308            ));
309            fields.push(("dropped_count", FieldValue::U64(total_dropped)));
310
311            // We want to just directly construct and send a `TraceEvent::Event` here so we don't need to
312            // reason very hard about whether or not we are creating a DoS loop
313            let drop_event = TraceEvent::Event {
314                name: "dropped events",
315                target: module_path!(),
316                level: tracing::Level::ERROR,
317                fields,
318                timestamp: SystemTime::now(),
319                parent_span: None,
320                thread_id,
321                thread_name,
322                module_path: Some(module_path!()),
323                file: Some(file!()),
324                line: Some(line!()),
325            };
326
327            if dropped_sender.send(drop_event).is_err() {
328                // Last resort
329                eprintln!(
330                    "[telemetry] CRITICAL: {} events and log lines dropped and unable to log to telemetry \
331                     (worker thread may have died). Telemetry system offline.",
332                    total_dropped
333                );
334            }
335        }
336    }
337}

339impl Drop for TraceEventDispatcher {
340    fn drop(&mut self) {
341        // Explicitly drop both senders to close the channels.
342        // The next field to be dropped is `worker_handle` which
343        // will run its own drop impl to join the thread and flush
344        drop(self.sender.take());
345        drop(self.dropped_sender.take());
346    }
347}

349impl<S> Layer<S> for TraceEventDispatcher
350where
351    S: Subscriber + for<'a> LookupSpan<'a>,
352{
353    fn on_new_span(&self, attrs: &tracing::span::Attributes<'_>, id: &Id, ctx: Context<'_, S>) {
354        let metadata = attrs.metadata();
355        let mut fields = TraceFields::new();
356
357        let mut visitor = FieldVisitor(&mut fields);
358        attrs.record(&mut visitor);
359
360        let parent_id = if let Some(parent) = attrs.parent() {
361            Some(parent.into_u64())
362        } else {
363            ctx.current_span().id().map(|id| id.into_u64())
364        };
365
366        let (thread_name, _) = get_thread_info();
367
368        let event = TraceEvent::NewSpan {
369            id: id.into_u64(),
370            name: metadata.name(),
371            target: metadata.target(),
372            level: *metadata.level(),
373            fields,
374            timestamp: SystemTime::now(),
375            parent_id,
376            thread_name,
377            file: metadata.file(),
378            line: metadata.line(),
379        };
380
381        self.send_event(event);
382    }
383
384    fn on_enter(&self, id: &Id, _ctx: Context<'_, S>) {
385        let event = TraceEvent::SpanEnter {
386            id: id.into_u64(),
387            timestamp: SystemTime::now(),
388        };
389
390        self.send_event(event);
391    }
392
393    fn on_exit(&self, id: &Id, _ctx: Context<'_, S>) {
394        let event = TraceEvent::SpanExit {
395            id: id.into_u64(),
396            timestamp: SystemTime::now(),
397        };
398
399        self.send_event(event);
400    }
401
402    fn on_event(&self, event: &tracing::Event<'_>, ctx: Context<'_, S>) {
403        let metadata = event.metadata();
404        let mut fields = TraceFields::new();
405        let mut visitor = FieldVisitor(&mut fields);
406        event.record(&mut visitor);
407
408        let parent_span = ctx.event_span(event).map(|span| span.id().into_u64());
409
410        let (thread_name, thread_id) = get_thread_info();
411
412        let trace_event = TraceEvent::Event {
413            name: metadata.name(),
414            target: metadata.target(),
415            level: *metadata.level(),
416            fields,
417            timestamp: SystemTime::now(),
418            parent_span,
419            thread_id,
420            thread_name,
421            module_path: metadata.module_path(),
422            file: metadata.file(),
423            line: metadata.line(),
424        };
425
426        self.send_event(trace_event);
427    }
428
429    fn on_close(&self, id: Id, _ctx: Context<'_, S>) {
430        let event = TraceEvent::SpanClose {
431            id: id.into_u64(),
432            timestamp: SystemTime::now(),
433        };
434
435        self.send_event(event);
436    }
437
438    fn max_level_hint(&self) -> Option<LevelFilter> {
439        self.max_level
440    }
441}

470    }

473/// Background worker loop that receives events from both regular and priority channels,
474/// and dispatches them to sinks. Priority events are processed first.
475/// Runs until both senders are dropped.
476fn worker_loop(
477    receiver: mpsc::Receiver<TraceEvent>,
478    dropped_receiver: mpsc::Receiver<TraceEvent>,
479    control_receiver: Option<mpsc::Receiver<DispatcherControl>>,
480    mut sinks: Vec<Box<dyn TraceEventSink>>,
481    dropped_events: Arc<AtomicU64>,
482) {
483    const FLUSH_INTERVAL: Duration = Duration::from_millis(100);
484    const FLUSH_EVENT_COUNT: usize = 1000;
485    let mut last_flush = std::time::Instant::now();
486    let mut events_since_flush = 0;
487
488    fn flush_sinks(sinks: &mut [Box<dyn TraceEventSink>]) {
489        for sink in sinks {
490            if let Err(e) = sink.flush() {
491                eprintln!("[telemetry] sink {} failed to flush: {}", sink.name(), e);
492            }
493        }
494    }
495
496    fn dispatch_to_sinks(sinks: &mut [Box<dyn TraceEventSink>], event: TraceEvent) {
497        for sink in sinks {
498            if match &event {
499                TraceEvent::NewSpan { target, level, .. }
500                | TraceEvent::Event { target, level, .. } => match sink.target_filter() {
501                    Some(targets) => targets.would_enable(target, level),
502                    None => true,
503                },
504                _ => true,
505            } {
506                if let Err(e) = sink.consume(&event) {
507                    eprintln!(
508                        "[telemetry] sink {} failed to consume event: {}",
509                        sink.name(),
510                        e
511                    );
512                }
513            }
514        }
515    }
516
517    loop {
518        while let Ok(event) = dropped_receiver.try_recv() {
519            dispatch_to_sinks(&mut sinks, event);
520            events_since_flush += 1;
521        }
522
523        // Process any pending control messages (e.g., adding new sinks)
524        if let Some(ref ctrl_rx) = control_receiver {
525            while let Ok(control) = ctrl_rx.try_recv() {
526                match control {
527                    DispatcherControl::AddSink(sink) => {
528                        sinks.push(sink);
529                    }
530                }
531            }
532        }
533
534        match receiver.recv_timeout(FLUSH_INTERVAL) {
535            Ok(event) => {
536                dispatch_to_sinks(&mut sinks, event);
537                events_since_flush += 1;
538
539                if events_since_flush >= FLUSH_EVENT_COUNT || last_flush.elapsed() >= FLUSH_INTERVAL
540                {
541                    flush_sinks(&mut sinks);
542                    last_flush = std::time::Instant::now();
543                    events_since_flush = 0;
544                }
545            }
546            Err(mpsc::RecvTimeoutError::Timeout) => {
547                flush_sinks(&mut sinks);
548                last_flush = std::time::Instant::now();
549                events_since_flush = 0;
550            }
551            Err(mpsc::RecvTimeoutError::Disconnected) => {
552                break;
553            }
554        }
555    }
556
557    while let Ok(event) = dropped_receiver.try_recv() {
558        dispatch_to_sinks(&mut sinks, event);
559    }
560    while let Ok(event) = receiver.try_recv() {
561        dispatch_to_sinks(&mut sinks, event);
562    }
563
564    flush_sinks(&mut sinks);
565
566    let total_dropped = dropped_events.load(Ordering::Relaxed);
567    if total_dropped > 0 {
568        eprintln!(
569            "[telemetry] Telemetry worker shutting down. Total events dropped during session: {}",
570            total_dropped
571        );
572    }
573}

575impl Drop for WorkerHandle {
576    fn drop(&mut self) {
577        if let Some(handle) = self.join_handle.take() {
578            if let Err(e) = handle.join() {
579                eprintln!("[telemetry] worker thread panicked: {:?}", e);
580            }
581        }
582    }
