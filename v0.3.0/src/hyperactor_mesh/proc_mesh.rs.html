125/// Single, process-wide supervision sink storage.
126///
127/// This is a pragmatic "good enough for now" global used to route
128/// undeliverables observed by the process-global root client (c.f.
129/// [`global_root_client`])to the *currently active* `ProcMesh`. Newer
130/// meshes override older ones ("last sink wins").
131static GLOBAL_SUPERVISION_SINK: OnceLock<RwLock<Option<PortHandle<ActorSupervisionEvent>>>> =
132    OnceLock::new();
133
134/// Returns the lazily-initialized container that holds the current
135/// process-global supervision sink.
136///
137/// Internal helper: callers should use `set_global_supervision_sink`
138/// and `get_global_supervision_sink` instead.
139fn sink_cell() -> &'static RwLock<Option<PortHandle<ActorSupervisionEvent>>> {
140    GLOBAL_SUPERVISION_SINK.get_or_init(|| RwLock::new(None))
141}
142
143/// Install (or replace) the process-global supervision sink.
144///
145/// This function enforces "last sink wins" semantics: if a sink was
146/// already installed, it is replaced and the previous sink is
147/// returned. Called from `ProcMesh::allocate_boxed`, after creating
148/// the mesh's supervision port.
149///
150/// Returns:
151/// - `Some(prev)` if a prior sink was installed, allowing the caller
152///   to log/inspect it if desired;
153/// - `None` if this is the first sink.
154///
155/// Thread-safety: takes a write lock briefly to swap the handle.
156pub(crate) fn set_global_supervision_sink(
157    sink: PortHandle<ActorSupervisionEvent>,
158) -> Option<PortHandle<ActorSupervisionEvent>> {
159    let cell = sink_cell();
160    let mut guard = cell.write().unwrap();
161    let prev = guard.take();
162    *guard = Some(sink);
163    prev
164}
165
166/// Get a clone of the current process-global supervision sink, if
167/// any.
168///
169/// This is used by the process-global root client [c.f.
170/// `global_root_client`] to forward undeliverables once a mesh has
171/// installed its sink. If no sink has been installed yet, returns
172/// `None` and callers should defer/ignore forwarding until one
173/// appears.
174///
175/// Thread-safety: takes a read lock briefly; cloning the `PortHandle`
176/// is cheap.
177pub(crate) fn get_global_supervision_sink() -> Option<PortHandle<ActorSupervisionEvent>> {
178    sink_cell().read().unwrap().clone()
179}

182pub struct GlobalClientActor {
183    signal_rx: PortReceiver<Signal>,
184    supervision_rx: PortReceiver<ActorSupervisionEvent>,
185    work_rx: mpsc::UnboundedReceiver<WorkCell<Self>>,
186}
187
188impl GlobalClientActor {
189    fn run(mut self, instance: &'static Instance<Self>) -> JoinHandle<()> {
190        tokio::spawn(async move {
191            let err = 'messages: loop {
192                tokio::select! {
193                    work = self.work_rx.recv() => {
194                        let work = work.expect("inconsistent work queue state");
195                        if let Err(err) = work.handle(&mut self, instance).await {
196                            for supervision_event in self.supervision_rx.drain() {
197                                if let Err(err) = instance.handle_supervision_event(&mut self, supervision_event).await {
198                                    break 'messages err;
199                                }
200                            }
201                            let kind = ActorErrorKind::processing(err);
202                            break ActorError {
203                                actor_id: Box::new(instance.self_id().clone()),
204                                kind: Box::new(kind),
205                            };
206                        }
207                    }
208                    _ = self.signal_rx.recv() => {
209                        // TODO: do we need any signal handling for the root client?
210                    }
211                    Ok(supervision_event) = self.supervision_rx.recv() => {
212                        if let Err(err) = instance.handle_supervision_event(&mut self, supervision_event).await {
213                            break err;
214                        }
215                    }
216                };
217            };
218            let event = match *err.kind {
219                ActorErrorKind::UnhandledSupervisionEvent(event) => *event,
220                _ => {
221                    let status = ActorStatus::generic_failure(err.kind.to_string());
222                    ActorSupervisionEvent::new(
223                        instance.self_id().clone(),
224                        Some("testclient".into()),
225                        status,
226                        None,
227                    )
228                }
229            };
230            instance
231                .proc()
232                .handle_unhandled_supervision_event(instance, event);
233        })
234    }
235}

247fn fresh_instance() -> (
248    &'static Instance<GlobalClientActor>,
249    &'static ActorHandle<GlobalClientActor>,
250) {
251    static INSTANCE: OnceLock<(Instance<GlobalClientActor>, ActorHandle<GlobalClientActor>)> =
252        OnceLock::new();
253    let client_proc = Proc::direct_with_default(
254        default_bind_spec().binding_addr(),
255        "mesh_root_client_proc".into(),
256        router::global().clone().boxed(),
257    )
258    .unwrap();
259
260    // Make this proc reachable through the global router, so that we can use the
261    // same client in both direct-addressed and ranked-addressed modes.
262    router::global().bind(client_proc.proc_id().clone().into(), client_proc.clone());
263
264    // The work_rx messages loop is ignored. v0 will support Handler<MeshFailure>,
265    // but it doesn't actually handle the messages.
266    // This is fine because v0 doesn't use this supervision mechanism anyway.
267    let (client, handle, supervision_rx, signal_rx, work_rx) = client_proc
268        .actor_instance::<GlobalClientActor>("client")
269        .expect("root instance create");
270
271    // Bind the global root client's undeliverable port and
272    // forward any undeliverable messages to the currently active
273    // supervision sink.
274    //
275    // The resolver (`get_global_supervision_sink`) is passed as a
276    // function pointer, so each time an undeliverable is
277    // processed, we look up the *latest* sink. This allows the
278    // root client to seamlessly track whichever ProcMesh most
279    // recently installed a supervision sink (e.g., the
280    // application mesh instead of an internal controller mesh).
281    //
282    // The hook logs each undeliverable, along with whether a sink
283    // was present at the time of receipt, which helps diagnose
284    // lost or misrouted events.
285    let (_undeliverable_tx, undeliverable_rx) =
286        client.open_port::<Undeliverable<MessageEnvelope>>();
287    hyperactor::mailbox::supervise_undeliverable_messages_with(
288        undeliverable_rx,
289        crate::proc_mesh::get_global_supervision_sink,
290        |env| {
291            let sink_present = crate::proc_mesh::get_global_supervision_sink().is_some();
292            tracing::info!(
293                actor_id = %env.dest().actor_id(),
294                "global root client undeliverable observed with headers {:?} {}", env.headers(), sink_present
295            );
296        },
297    );
298
299    // Use the OnceLock to get a 'static lifetime for the instance.
300    INSTANCE
301        .set((client, handle))
302        .map_err(|_| "already initialized root client instance")
303        .unwrap();
304    let (instance, handle) = INSTANCE.get().unwrap();
305    let client = GlobalClientActor {
306        signal_rx,
307        supervision_rx,
308        work_rx,
309    };
310    client.run(instance);
311    (instance, handle)
312}

314/// Context use by root client to send messages.
315/// This mailbox allows us to open ports before we know which proc the

317pub fn global_root_client() -> &'static Instance<GlobalClientActor> {
318    static GLOBAL_INSTANCE: OnceLock<(
319        &'static Instance<GlobalClientActor>,
320        &'static ActorHandle<GlobalClientActor>,
321    )> = OnceLock::new();
322    GLOBAL_INSTANCE.get_or_init(fresh_instance).0

366impl ProcMesh {
367    #[hyperactor::instrument(fields(name = "ProcMesh::allocate"))]
368    pub async fn allocate(
369        alloc: impl Alloc + Send + Sync + 'static,
370    ) -> Result<Self, AllocatorError> {
371        ProcMesh::allocate_boxed(Box::new(alloc)).await
372    }
373
374    /// Allocate a new ProcMesh from the provided allocator. Allocate returns
375    /// after the mesh has been successfully (and fully) allocated, returning
376    /// early on any allocation failure.
377    #[track_caller]
378    pub fn allocate_boxed(
379        alloc: Box<dyn Alloc + Send + Sync>,
380    ) -> impl std::future::Future<Output = Result<Self, AllocatorError>> {
381        Self::allocate_boxed_inner(alloc, Location::caller())
382    }
383
384    fn alloc_counter() -> &'static AtomicUsize {
385        static C: OnceLock<AtomicUsize> = OnceLock::new();
386        C.get_or_init(|| AtomicUsize::new(0))
387    }
388
389    #[hyperactor::instrument]
390    #[hyperactor::observe_result("ProcMesh")]
391    async fn allocate_boxed_inner(
392        mut alloc: Box<dyn Alloc + Send + Sync>,
393        loc: &'static Location<'static>,
394    ) -> Result<Self, AllocatorError> {
395        let alloc_id = Self::alloc_counter().fetch_add(1, Ordering::Relaxed) + 1;
396        let world = alloc.world_id().name().to_string();
397        tracing::info!(
398            name = "ProcMesh::Allocate::Attempt",
399            %world,
400            alloc_id,
401            caller = %format!("{}:{}", loc.file(), loc.line()),
402            shape = ?alloc.shape(),
403            "allocating proc mesh"
404        );
405
406        // 1. Initialize the alloc, producing the initial set of ranked procs:
407        let running = alloc
408            .initialize()
409            .instrument(span!(
410                Level::INFO,
411                "ProcMesh::Allocate::Initialize",
412                alloc_id
413            ))
414            .await?;
415
416        // 2. Set up routing to the initialized procs; these require dialing.
417        // let router = DialMailboxRouter::new();
418        let router = DialMailboxRouter::new_with_default(router::global().boxed());
419        for AllocatedProc { proc_id, addr, .. } in running.iter() {
420            if proc_id.is_direct() {
421                continue;
422            }
423            router.bind(proc_id.clone().into(), addr.clone());
424        }
425
426        // 3. Set up a client proc for the mesh itself, so that we can attach ourselves
427        //    to it, and communicate with the agents. We wire it into the same router as
428        //    everything else, so now the whole mesh should be able to communicate.
429        let client_proc_id =
430            ProcId::Ranked(WorldId(format!("{}_client", alloc.world_id().name())), 0);
431        let (client_proc_addr, client_rx) = channel::serve(ChannelAddr::any(alloc.transport()))
432            .map_err(|err| AllocatorError::Other(err.into()))?;
433        tracing::info!(
434            name = "ProcMesh::Allocate::ChannelServe",
435            alloc_id = alloc_id,
436            "client proc started listening on addr: {client_proc_addr}"
437        );
438        let client_proc = Proc::new(
439            client_proc_id.clone(),
440            BoxedMailboxSender::new(router.clone()),
441        );
442        client_proc.clone().serve(client_rx);
443        router.bind(client_proc_id.clone().into(), client_proc_addr.clone());
444
445        // 4. Bind the dial router to the global router, so that everything is
446        //    connected to a single root.
447        router::global().bind_dial_router(&router);
448
449        let (supervisor, _supervisor_handle) = client_proc.instance("supervisor")?;
450        let (supervision_port, supervision_events) =
451            supervisor.open_port::<ActorSupervisionEvent>();
452
453        // 5. Install this mesh’s supervision sink.
454        //
455        // We intentionally use "last sink wins": if multiple
456        // ProcMeshes exist in the process (e.g., a hidden
457        // controller_controller mesh and the app/test mesh), the most
458        // recently allocated mesh’s sink replaces the prior global
459        // sink.
460        //
461        // Scope: this only affects undeliverables that arrive on the
462        // `global_root_client()` undeliverable port. Per-mesh client
463        // bindings (set up below) are unaffected and continue to
464        // forward their own undeliverables to this mesh’s
465        // `supervision_port`.
466        //
467        // NOTE: This is a pragmatic stopgap to restore correct
468        // routing with multiple meshes in-process. If/when we move to
469        // per-world root clients, this override can be removed.
470        let _prev = set_global_supervision_sink(supervision_port.clone());
471
472        // Wire this mesh’s *own* client mailbox to supervision.
473        //
474        // Attach a client mailbox for this `ProcMesh`, bind its
475        // undeliverable port, and forward those undeliverables as
476        // `ActorSupervisionEvent` records into this mesh's
477        // supervision_port.
478        //
479        // Scope: covers undeliverables observed on this mesh's client
480        // mailbox only. It does not affect other meshes or the
481        // `global_root_client()`.
482        let (client, _handle) = client_proc.instance("client")?;
483        // Bind an undeliverable message port in the client.
484        let (_undeliverable_messages, client_undeliverable_receiver) =
485            client.bind_actor_port::<Undeliverable<MessageEnvelope>>();
486        hyperactor::mailbox::supervise_undeliverable_messages(
487            supervision_port.clone(),
488            client_undeliverable_receiver,
489            |env| {
490                tracing::info!(actor=%env.dest().actor_id(), "per-mesh client undeliverable observed");
491            },
492        );
493
494        // Ensure that the router is served so that agents may reach us.
495        let (router_channel_addr, router_rx) = channel::serve(alloc.client_router_addr())
496            .map_err(|e| AllocatorError::Other(e.into()))?;
497        router.serve(router_rx);
498        tracing::info!("router channel started listening on addr: {router_channel_addr}");
499
500        // 6. Configure the mesh agents. This transmits the address book to all agents,
501        //    so that they can resolve and route traffic to all nodes in the mesh.
502        let address_book: HashMap<_, _> = running
503            .iter()
504            .map(
505                |AllocatedProc {
506                     addr, mesh_agent, ..
507                 }| { (mesh_agent.actor_id().proc_id().clone(), addr.clone()) },
508            )
509            .collect();
510
511        let (config_handle, mut config_receiver) = client.open_port();
512        for (rank, AllocatedProc { mesh_agent, .. }) in running.iter().enumerate() {
513            mesh_agent
514                .configure(
515                    &client,
516                    rank,
517                    router_channel_addr.clone(),
518                    Some(supervision_port.bind()),
519                    address_book.clone(),
520                    config_handle.bind(),
521                    false,
522                )
523                .await?;
524        }
525        let mut completed = Ranks::new(running.len());
526        while !completed.is_full() {
527            let rank = config_receiver
528                .recv()
529                .await
530                .map_err(|err| AllocatorError::Other(err.into()))?;
531            if completed.insert(rank, rank).is_some() {
532                tracing::warn!("multiple completions received for rank {}", rank);
533            }
534        }
535
536        // For reasons I fail to fully understand, the below call fails
537        // when invoked from `pyo3_async_runtimes::tokio::future_into_py`
538        // when using a closure. It appears to be some subtle failure of
539        // the compiler to unify lifetimes. If we use a function instead,
540        // it does better.
541        //
542        // Interestingly, this only appears to fail in *specific* caller
543        // contexts (e.g., https://fburl.com/code/evfgtfx1), and the error
544        // is reported there as "implementation of `std::ops::FnOnce` is not general enough",
545        // suggesting some failure of modularity in the compiler's lifetime
546        // unification!
547        //
548        // Baffling and unsettling.
549        fn project_mesh_agent_ref(allocated_proc: &AllocatedProc) -> ActorRef<ProcMeshAgent> {
550            allocated_proc.mesh_agent.clone()
551        }
552
553        // 7. Start comm actors and set them up to communicate via the same address book.
554
555        // Spawn a comm actor on each proc, so that they can be used
556        // to perform tree distribution and accumulation.
557        let comm_actors = Self::spawn_on_procs::<CommActor>(
558            &client,
559            running.iter().map(project_mesh_agent_ref),
560            "comm",
561            &Default::default(),
562        )
563        .await?;
564        let address_book: HashMap<_, _> = comm_actors.iter().cloned().enumerate().collect();
565        // Now that we have all of the spawned comm actors, kick them all into
566        // mesh mode.
567        for (rank, comm_actor) in comm_actors.iter().enumerate() {
568            comm_actor
569                .send(&client, CommMeshConfig::new(rank, address_book.clone()))
570                .map_err(anyhow::Error::from)?;
571        }
572
573        let shape = alloc.shape().clone();
574        let world_id = alloc.world_id().clone();
575        metrics::PROC_MESH_ALLOCATION.add(
576            running.len() as u64,
577            hyperactor_telemetry::kv_pairs!("alloc_id" => alloc_id.to_string()),
578        );
579
580        Ok(Self {
581            inner: ProcMeshKind::V0 {
582                event_state: Some(EventState {
583                    alloc,
584                    supervision_events,
585                }),
586                actor_event_router: Arc::new(DashMap::new()),
587                shape,
588                ranks: running
589                    .into_iter()
590                    .map(
591                        |AllocatedProc {
592                             create_key,
593                             proc_id,
594                             addr,
595                             mesh_agent,
596                         }| (create_key, proc_id, addr, mesh_agent),
597                    )
598                    .collect(),
599                client_proc,
600                client,
601                comm_actors,
602                world_id,
603            },
604            shape: OnceLock::new(),
605        })
606    }
607
608    /// Bounds:
609    /// - `A: Actor` - we actually spawn this concrete actor on each
610    ///   proc.
611    /// - `A: Referable` - required because we return
612    ///   `Vec<ActorRef<A>>`, and `ActorRef` is only defined for `A:
613    ///   Referable`.
614    /// - `A::Params: RemoteMessage` - params must serialize for
615    ///   cross-proc spawn.
616    async fn spawn_on_procs<A: RemoteSpawn>(
617        cx: &impl context::Actor,
618        agents: impl IntoIterator<Item = ActorRef<ProcMeshAgent>> + '_,
619        actor_name: &str,
620        params: &A::Params,
621    ) -> Result<Vec<ActorRef<A>>, anyhow::Error>
622    where
623        A::Params: RemoteMessage,
624    {
625        let remote = Remote::collect();
626        let actor_type = remote
627            .name_of::<A>()
628            .ok_or(anyhow::anyhow!("actor not registered"))?
629            .to_string();
630
631        let (completed_handle, mut completed_receiver) = mailbox::open_port(cx);
632        let mut n = 0;
633        for agent in agents {
634            agent
635                .gspawn(
636                    cx,
637                    actor_type.clone(),
638                    actor_name.to_string(),
639                    bincode::serialize(params)?,
640                    completed_handle.bind(),
641                )
642                .await?;
643            n += 1;
644        }
645        let mut completed = Ranks::new(n);
646        while !completed.is_full() {
647            let result = completed_receiver.recv().await?;
648            match result {
649                GspawnResult::Success { rank, actor_id } => {
650                    if completed.insert(rank, actor_id).is_some() {
651                        tracing::warn!("multiple completions received for rank {}", rank);
652                    }
653                }
654                GspawnResult::Error(error_msg) => {
655                    metrics::PROC_MESH_ACTOR_FAILURES.add(
656                        1,
657                        hyperactor_telemetry::kv_pairs!(
658                            "actor_name" => actor_name.to_string(),
659                            "error" => error_msg.clone(),
660                        ),
661                    );
662
663                    anyhow::bail!("gspawn failed: {}", error_msg);
664                }
665            }
666        }
667
668        // `Ranks` really should have some way to convert into a "completed" rank
669        // in a one-shot way; the API here is too awkward otherwise.
670        Ok(completed
671            .into_iter()
672            .map(Option::unwrap)
673            .map(ActorRef::attest)
674            .collect())
675    }

677    fn agents(&self) -> Box<dyn Iterator<Item = ActorRef<ProcMeshAgent>> + '_ + Send> {
678        match &self.inner {
679            ProcMeshKind::V0 { ranks, .. } => {
680                Box::new(ranks.iter().map(|(_, _, _, agent)| agent.clone()))
681            }
682            ProcMeshKind::V1(proc_mesh) => Box::new(
683                proc_mesh
684                    .agent_mesh()
685                    .iter()
686                    .map(|(_point, agent)| agent.clone())
687                    // We need to collect here so that we can return an iterator
688                    // that fully owns the data and does not reference temporary
689                    // values.
690                    //
691                    // Because this is a shim that we expect to be short-lived,
692                    // we'll leave this hack as is; a proper solution here would
693                    // be to have implement an owning iterator (into_iter) for views.
694                    .collect::<Vec<_>>()
695                    .into_iter(),
696            ),
697        }
698    }

708    /// Spawn an `ActorMesh` by launching the same actor type on all
709    /// agents, using the **same** parameters instance for every
710    /// actor.
711    ///
712    /// - `actor_name`: Name for all spawned actors.
713    /// - `params`: Reference to the parameter struct, reused for all
714    ///   actors.
715    ///
716    /// Bounds:
717    /// - `A: Actor` — we actually spawn this type on each agent.
718    /// - `A: Referable` — we return a `RootActorMesh<'_, A>` that
719    ///   contains `ActorRef<A>`s; those exist only for `A:
720    ///   Referable`.
721    /// - `A::Params: RemoteMessage` — params must be serializable to
722    ///   cross proc boundaries when launching each actor.
723    pub async fn spawn<A: RemoteSpawn, C: context::Actor>(
724        &self,
725        cx: &C,
726        actor_name: &str,
727        params: &A::Params,
728    ) -> Result<RootActorMesh<'_, A>, anyhow::Error>
729    where
730        A::Params: RemoteMessage,
731        C::A: Handler<MeshFailure>,
732    {
733        match &self.inner {
734            ProcMeshKind::V0 {
735                actor_event_router,
736                client,
737                ..
738            } => {
739                let (tx, rx) = mpsc::unbounded_channel::<ActorSupervisionEvent>();
740                {
741                    // Instantiate supervision routing BEFORE spawning the actor mesh.
742                    actor_event_router.insert(actor_name.to_string(), tx);
743                    tracing::info!(
744                        name = "router_insert",
745                        actor_name = %actor_name,
746                        "the length of the router is {}", actor_event_router.len(),
747                    );
748                }
749                let root_mesh = RootActorMesh::new(
750                    self,
751                    actor_name.to_string(),
752                    rx,
753                    Self::spawn_on_procs::<A>(client, self.agents(), actor_name, params).await?,
754                );
755                Ok(root_mesh)
756            }
757            ProcMeshKind::V1(proc_mesh) => {
758                let actor_mesh = proc_mesh.spawn(cx, actor_name, params).await?;
759                Ok(RootActorMesh::new_v1(actor_mesh.detach()))
760            }
761        }
762    }

824    /// Send stop actors message to all mesh agents for a specific mesh name
825    #[hyperactor::observe_result("ProcMesh")]
826    pub async fn stop_actor_by_name(
827        &self,
828        cx: &impl context::Actor,
829        mesh_name: &str,
830        reason: &str,
831    ) -> Result<(), anyhow::Error> {
832        match &self.inner {
833            ProcMeshKind::V0 { client, .. } => {
834                let timeout =
835                    hyperactor_config::global::get(hyperactor::config::STOP_ACTOR_TIMEOUT);
836                let results = join_all(self.agents().map(|agent| async move {
837                    let actor_id =
838                        ActorId(agent.actor_id().proc_id().clone(), mesh_name.to_string(), 0);
839                    (
840                        actor_id.clone(),
841                        agent
842                            .clone()
843                            .stop_actor(
844                                client,
845                                actor_id,
846                                timeout.as_millis() as u64,
847                                reason.to_string(),
848                            )
849                            .await,
850                    )
851                }))
852                .await;
853
854                for (actor_id, result) in results {
855                    match result {
856                        Ok(StopActorResult::Timeout) => {
857                            tracing::warn!("timed out while stopping actor {}", actor_id);
858                        }
859                        Ok(StopActorResult::NotFound) => {
860                            tracing::warn!("no actor {} on proc {}", actor_id, actor_id.proc_id());
861                        }
862                        Ok(StopActorResult::Success) => {
863                            tracing::info!("stopped actor {} with reason: {}", actor_id, reason);
864                        }
865                        Err(e) => {
866                            tracing::warn!("error stopping actor {}: {}", actor_id, e);
867                        }
868                    }
869                }
870                Ok(())
871            }
872            ProcMeshKind::V1(proc_mesh) => {
873                proc_mesh
874                    .stop_actor_by_name(cx, Name::new_reserved(mesh_name)?, reason.to_string())
875                    .await?;
876                Ok(())
877            }
878        }
879    }

882/// Proc lifecycle events.

884pub enum ProcEvent {
885    /// The proc of the given rank was stopped with the provided reason.
886    Stopped(usize, ProcStopReason),
887    /// The proc crashed, with the provided "reason". This is reserved for
888    /// unhandled supervision events.
889    Crashed(usize, String),
890}

892#[derive(Debug, Clone, AsRefStr)]
893pub enum SupervisionEventState {
894    SupervisionEventForward,
895    SupervisionEventForwardFailed,
896    SupervisionEventReceived,
897    SupervisionEventTransmitFailed,
898}

900impl fmt::Display for ProcEvent {
901    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
902        match self {
903            ProcEvent::Stopped(rank, reason) => {

905            }
906            ProcEvent::Crashed(rank, reason) => {
907                write!(f, "Proc at rank {} crashed: {}", rank, reason)
908            }
909        }

915/// An event stream of [`ProcEvent`]
916// TODO: consider using streams for this.
917pub struct ProcEvents {
918    event_state: EventState,
919    // Proc id to its rank and create key.
920    ranks: HashMap<ProcId, (usize, ShortUuid)>,
921    actor_event_router: ActorEventRouter,
922}
923
924impl ProcEvents {
925    /// Get the next lifecycle event. The stream is closed when this method
926    /// returns `None`.
927    pub async fn next(&mut self) -> Option<ProcEvent> {
928        loop {
929            tokio::select! {
930                result = self.event_state.alloc.next() => {
931                    tracing::debug!(name = "ProcEventReceived", "received ProcEvent alloc update: {result:?}");
932                    // Don't disable the outer branch on None: this is always terminal.
933                    let Some(alloc_event) = result else {
934                        self.actor_event_router.clear();
935                        break None;
936                    };
937
938                    let ProcState::Stopped { create_key, reason } = alloc_event else {
939                        // Ignore non-stopped events for now.
940                        continue;
941                    };
942
943                    let Some((proc_id, (rank, _create_key))) = self.ranks.iter().find(|(_proc_id, (_rank, key))| key == &create_key) else {
944                        tracing::warn!("received stop event for unmapped proc {}", create_key);
945                        continue;
946                    };
947
948                    metrics::PROC_MESH_PROC_STOPPED.add(
949                        1,
950                        hyperactor_telemetry::kv_pairs!(
951                            "create_key" => create_key.to_string(),
952                            "rank" => rank.to_string(),
953                            "reason" => reason.to_string(),
954                        ),
955                    );
956
957                    // Need to send this event to actor meshes to notify them of the proc's death.
958                    // TODO(albertli): only send this event to all root actor meshes if any of them use this proc.
959                    for entry in self.actor_event_router.iter() {
960                        // Make a dummy actor supervision event, all actors on the proc are affected if a proc stops.
961                        // TODO(T231868026): find a better way to represent all actors in a proc for supervision event
962                        let event = ActorSupervisionEvent::new(
963                            proc_id.actor_id("any", 0),
964                            None,
965                            ActorStatus::generic_failure(format!("proc {} is stopped", proc_id)),
966                            None,
967                        );
968                        tracing::debug!(name = "SupervisionEvent", %event);
969                        if entry.value().send(event.clone()).is_err() {
970                            tracing::warn!(
971                                name = SupervisionEventState::SupervisionEventTransmitFailed.as_ref(),
972                                "unable to transmit supervision event to actor {}", entry.key()
973                            );
974                        }
975                    }
976
977                    let event = ProcEvent::Stopped(*rank, reason.clone());
978                    tracing::debug!(name = "SupervisionEvent", %event);
979
980                    break Some(ProcEvent::Stopped(*rank, reason));
981                }
982
983                // Supervision events for this ProcMesh, delivered on
984                // the client's "supervisor" port. Some failures are
985                // observed while messages are routed through the
986                // comm-actor tree; in those cases the event's
987                // `actor_id` points at a comm actor rather than the
988                // logical actor-mesh. When the `CAST_ACTOR_MESH_ID`
989                // header is present, we normalize the event by
990                // rewriting `actor_id` to a synthetic mesh-level id
991                // so that routing reaches the correct `ActorMesh`
992                // subscribers.
993                Ok(event) = self.event_state.supervision_events.recv() => {
994                    let had_headers = event.message_headers.is_some();
995                    tracing::info!(
996                        name = SupervisionEventState::SupervisionEventReceived.as_ref(),
997                        actor_id = %event.actor_id,
998                        actor_name = %event.actor_id.name(),
999                        status = %event.actor_status,
1000                        "proc supervision: event received with {had_headers} headers"
1001                    );
1002                    tracing::debug!(
1003                        name = "SupervisionEvent",
1004                        %event,
1005                        "proc supervision: full event");
1006
1007                    // Normalize events that came via the comm tree.
1008                    let event = update_event_actor_id(event);
1009
1010                    // Forward the supervision event to the ActorMesh (keyed by its mesh name)
1011                    // that registered for events in this ProcMesh. The routing table
1012                    // (actor_event_router) is keyed by ActorMeshName, which we obtain from
1013                    // actor_id.name(). If no matching mesh is found, log the current table
1014                    // to aid diagnosis.
1015                    let actor_id = event.actor_id.clone();
1016                    let actor_status = event.actor_status.clone();
1017                    let reason = event.to_string();
1018                    if let Some(tx) = self.actor_event_router.get(actor_id.name()) {
1019                        tracing::info!(
1020                            name = SupervisionEventState::SupervisionEventForwardFailed.as_ref(),
1021                            actor_id = %actor_id,
1022                            actor_name = actor_id.name(),
1023                            status = %actor_status,
1024                            "proc supervision: delivering event to registered ActorMesh"
1025                        );
1026                        if tx.send(event).is_err() {
1027                            tracing::warn!(
1028                                name = SupervisionEventState::SupervisionEventForwardFailed.as_ref(),
1029                                actor_id = %actor_id,
1030                                "proc supervision: registered ActorMesh dropped receiver; unable to deliver"
1031                            );
1032                        }
1033                    } else {
1034                        let registered_meshes: Vec<_> = self.actor_event_router.iter().map(|e| e.key().clone()).collect();
1035                        tracing::warn!(
1036                            name = SupervisionEventState::SupervisionEventForwardFailed.as_ref(),
1037                            actor_id = %actor_id,
1038                            "proc supervision: no ActorMesh registered for this actor {:?}", registered_meshes,
1039                        );
1040                    }
1041                    // Ensure we have a known rank for the proc
1042                    // containing this actor. If we don't, we can't
1043                    // attribute the failure to a known process.
1044                    let Some((rank, _)) = self.ranks.get(actor_id.proc_id()) else {
1045                        tracing::warn!(
1046                            actor_id = %actor_id,
1047                            "proc supervision: actor belongs to an unmapped proc; dropping event"
1048                        );
1049                        continue;
1050                    };
1051
1052                    metrics::PROC_MESH_ACTOR_FAILURES.add(
1053                        1,
1054                        hyperactor_telemetry::kv_pairs!(
1055                            "actor_id" => actor_id.to_string(),
1056                            "rank" => rank.to_string(),
1057                            "status" => actor_status.to_string(),
1058                        ),
1059                    );
1060
1061                    // Send this event to Python proc mesh to keep its
1062                    // health status up to date.
1063                    break Some(ProcEvent::Crashed(*rank, reason))
1064                }
1065            }
1066        }
1067    }
