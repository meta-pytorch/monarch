57/// A message buffered due to out-of-order delivery.
58#[derive(Debug)]
59struct Buffered {
60    /// Sequence number of this message.
61    seq: usize,
62    /// Whether to deliver this message to this comm-actors actors.
63    deliver_here: bool,
64    /// Peer comm actors to forward message to.
65    next_steps: HashMap<usize, Vec<RoutingFrame>>,
66    /// The message to deliver.
67    message: CastMessageEnvelope,
68}

70/// Bookkeeping to handle sequence numbers and in-order delivery for messages
71/// sent to and through this comm actor.
72#[derive(Debug, Default)]
73struct ReceiveState {
74    /// The sequence of the last received message.
75    seq: usize,
76    /// A buffer storing messages we received out-of-order, indexed by the seq
77    /// that should precede it.
78    buffer: HashMap<usize, Buffered>,
79    /// A map of the last sequence number we sent to next steps, indexed by rank.
80    last_seqs: HashMap<usize, usize>,
81}

114impl CommMeshConfig {
115    /// Create a new mesh configuration with the given rank and peer mapping.
116    pub fn new(rank: usize, peers: HashMap<usize, ActorRef<CommActor>>) -> Self {
117        Self { rank, peers }
118    }

120    /// Return the peer comm actor for the given rank.
121    fn peer_for_rank(&self, rank: usize) -> Result<ActorRef<CommActor>> {
122        self.peers
123            .get(&rank)
124            .cloned()
125            .ok_or_else(|| anyhow::anyhow!("no peer for rank {}", rank))
126    }

129    fn self_rank(&self) -> usize {
130        self.rank
131    }

134#[async_trait]
135impl Actor for CommActor {
136    // This is an override of the default actor behavior.
137    async fn handle_undeliverable_message(
138        &mut self,
139        cx: &Instance<Self>,
140        undelivered: hyperactor::mailbox::Undeliverable<hyperactor::mailbox::MessageEnvelope>,
141    ) -> Result<(), anyhow::Error> {
142        let Undeliverable(mut message_envelope) = undelivered;
143
144        // 1. Case delivery failure at a "forwarding" step.
145        if let Ok(ForwardMessage { message, .. }) =
146            message_envelope.deserialized::<ForwardMessage>()
147        {
148            let sender = message.sender();
149            let return_port = PortRef::attest_message_port(sender);
150            message_envelope.set_error(DeliveryError::Multicast(format!(
151                "comm actor {} failed to forward the cast message; returning to origin {}",
152                cx.self_id(),
153                return_port.port_id(),
154            )));
155
156            // Needed so that the receiver of the undeliverable message can easily find the
157            // original sender of the cast message.
158            message_envelope
159                .headers_mut()
160                .set(CAST_ORIGINATING_SENDER, sender.clone());
161
162            return_port
163                .send(cx, Undeliverable(message_envelope.clone()))
164                .map_err(|err| {
165                    let error = DeliveryError::BrokenLink(format!(
166                        "error occured when returning ForwardMessage to the original \
167                        sender's port {}; error is: {}",
168                        return_port.port_id(),
169                        err,
170                    ));
171                    message_envelope.set_error(error);
172                    UndeliverableMessageError::ReturnFailure {
173                        envelope: message_envelope,
174                    }
175                })?;
176            return Ok(());
177        }
178
179        // 2. Case delivery failure at a "deliver here" step.
180        if let Some(sender) = message_envelope.headers().get(CAST_ORIGINATING_SENDER) {
181            let return_port = PortRef::attest_message_port(sender);
182            message_envelope.set_error(DeliveryError::Multicast(format!(
183                "comm actor {} failed to deliver the cast message to the dest \
184                actor; returning to origin {}",
185                cx.self_id(),
186                return_port.port_id(),
187            )));
188            return_port
189                .send(cx, Undeliverable(message_envelope.clone()))
190                .map_err(|err| {
191                    let error = DeliveryError::BrokenLink(format!(
192                        "error occured when returning cast message to the origin \
193                        sender {}; error is: {}",
194                        return_port.port_id(),
195                        err,
196                    ));
197                    message_envelope.set_error(error);
198                    UndeliverableMessageError::ReturnFailure {
199                        envelope: message_envelope,
200                    }
201                })?;
202            return Ok(());
203        }
204
205        // 3. A return of an undeliverable message was itself returned.
206        UndeliverableMailboxSender
207            .post(message_envelope, /*unused */ monitored_return_handle());
208        Ok(())
209    }
210}

212impl CommActor {
213    /// Forward the message to the comm actor on the given peer rank.
214    fn forward<M: RemoteMessage>(
215        cx: &Context<Self>,
216        config: &CommMeshConfig,
217        rank: usize,
218        message: M,
219    ) -> Result<()>
220    where
221        CommActor: hyperactor::RemoteHandles<M>,
222    {
223        let child = config.peer_for_rank(rank)?;
224        // TEMPORARY: until dropping v0 support
225        if let Some(cast_actor_mesh_id) = cx.headers().get(CAST_ACTOR_MESH_ID) {
226            let mut headers = Attrs::new();
227            headers.set(CAST_ACTOR_MESH_ID, cast_actor_mesh_id.clone());
228            child.send_with_headers(cx, headers, message)?;
229        } else {
230            child.send(cx, message)?;
231        }
232        Ok(())
233    }
234
235    fn handle_message(
236        cx: &Context<Self>,
237        config: &CommMeshConfig,
238        deliver_here: bool,
239        next_steps: HashMap<usize, Vec<RoutingFrame>>,
240        sender: ActorId,
241        mut message: CastMessageEnvelope,
242        seq: usize,
243        last_seqs: &mut HashMap<usize, usize>,
244    ) -> Result<()> {
245        split_ports(cx, message.data_mut(), deliver_here, &next_steps)?;
246
247        // Deliver message here, if necessary.
248        if deliver_here {
249            // We should not copy cx.headers() because it contains auto-generated
250            // headers from mailbox. We want fresh headers only containing
251            // user-provided headers.
252            let headers = message.headers().clone();
253            Self::deliver_to_dest(cx, headers, &mut message, config)?;
254        }
255
256        // Forward to peers.
257        next_steps
258            .into_iter()
259            .map(|(peer, dests)| {
260                let last_seq = last_seqs.entry(peer).or_default();
261                Self::forward(
262                    cx,
263                    config,
264                    peer,
265                    ForwardMessage {
266                        dests,
267                        sender: sender.clone(),
268                        message: message.clone(),
269                        seq,
270                        last_seq: *last_seq,
271                    },
272                )?;
273                *last_seq = seq;
274                Ok(())
275            })
276            .collect::<Result<Vec<_>>>()?;
277
278        Ok(())
279    }
280
281    fn deliver_to_dest<M: CastEnvelope>(
282        cx: &Context<Self>,
283        mut headers: Attrs,
284        message: &mut M,
285        config: &CommMeshConfig,
286    ) -> anyhow::Result<()> {
287        let cast_point = message.cast_point(config)?;
288        // Replace ranks with self ranks.
289        replace_with_self_ranks(&cast_point, message.data_mut())?;
290
291        set_cast_info_on_headers(&mut headers, cast_point, message.sender().clone());
292        cx.post(
293            cx.self_id()
294                .proc_id()
295                .actor_id(message.dest_port().actor_name(), 0)
296                .port_id(message.dest_port().port()),
297            headers,
298            wirevalue::Any::serialize(message.data())?,
299        );
300
301        Ok(())
302    }
303}

305// Split ports, if any, and update message with new ports. In this
306// way, children actors will reply to this comm actor's ports, instead
307// of to the original ports provided by parent.
308fn split_ports(
309    cx: &Context<CommActor>,
310    data: &mut ErasedUnbound,
311    deliver_here: bool,
312    next_steps: &HashMap<usize, Vec<RoutingFrame>>,
313) -> anyhow::Result<()> {
314    // Split ports, if any, and update message with new ports. In this
315    // way, children actors will reply to this comm actor's ports, instead
316    // of to the original ports provided by parent.
317    data.visit_mut::<UnboundPort>(
318        |UnboundPort(port_id, reducer_spec, return_undeliverable, kind)| {
319            let reducer_mode = match kind {
320                UnboundPortKind::Streaming(opts) => {
321                    ReducerMode::Streaming(opts.clone().unwrap_or_default())
322                }
323                UnboundPortKind::Once if reducer_spec.is_none() => {
324                    // We can only split OncePorts that have reducers.
325                    // Pass this through -- if it is used multiple times,
326                    // it will cause a delivery error downstream.
327                    // However we should reconsider this behavior
328                    // as it its semantics will now differ between
329                    // unicast and broadcast messages.
330                    return Ok(());
331                }
332                UnboundPortKind::Once => {
333                    // Compute peer count for OncePort splitting. This is the number of
334                    // destinations the message will be delivered to, so that the split
335                    // port can correctly accumulate responses.
336                    let peer_count = next_steps.len() + if deliver_here { 1 } else { 0 };
337                    ReducerMode::Once(peer_count)
338                }
339            };
340
341            let split = port_id.split(
342                cx,
343                reducer_spec.clone(),
344                reducer_mode,
345                *return_undeliverable,
346            )?;
347
348            #[cfg(test)]
349            tests::collect_split_port(port_id, &split, deliver_here);
350
351            *port_id = split;
352            Ok(())
353        },
354    )
355}

357fn replace_with_self_ranks(cast_point: &Point, data: &mut ErasedUnbound) -> anyhow::Result<()> {
358    data.visit_mut::<resource::Rank>(|resource::Rank(rank)| {
359        *rank = Some(cast_point.rank());
360        Ok(())
361    })

372// TODO(T218630526): reliable casting for mutable topology
373#[async_trait]
374impl Handler<CastMessage> for CommActor {
375    #[hyperactor::instrument]
376    async fn handle(&mut self, cx: &Context<Self>, cast_message: CastMessage) -> Result<()> {
377        // Always forward the message to the root rank of the slice, casting starts from there.
378        let slice = cast_message.dest.slice.clone();
379        let selection = cast_message.dest.selection.clone();
380        let frame = RoutingFrame::root(selection, slice);
381        let rank = frame.slice.location(&frame.here)?;
382        let seq = self
383            .send_seq
384            .entry(cast_message.message.stream_key())
385            .or_default();
386        let last_seq = *seq;
387        *seq += 1;
388
389        let fwd_message = ForwardMessage {
390            dests: vec![frame],
391            sender: cx.self_id().clone(),
392            message: cast_message.message,
393            seq: *seq,
394            last_seq,
395        };
396
397        let config = self
398            .mesh_config
399            .as_ref()
400            .ok_or_else(|| anyhow::anyhow!("CommMeshConfig has not been set yet"))?;
401
402        // Optimization: if forwarding to ourselves, handle inline instead of
403        // going through the message queue
404        if config.self_rank() == rank {
405            Handler::<ForwardMessage>::handle(self, cx, fwd_message).await?;
406        } else {
407            Self::forward(cx, config, rank, fwd_message)?;
408        }
409        Ok(())
410    }
411}

413#[async_trait]
414impl Handler<ForwardMessage> for CommActor {
415    #[hyperactor::instrument]
416    async fn handle(&mut self, cx: &Context<Self>, fwd_message: ForwardMessage) -> Result<()> {
417        let ForwardMessage {
418            sender,
419            dests,
420            message,
421            seq,
422            last_seq,
423        } = fwd_message;
424
425        let config = self
426            .mesh_config
427            .as_ref()
428            .ok_or_else(|| anyhow::anyhow!("CommMeshConfig has not been set yet"))?;
429
430        // Resolve/dedup routing frames.
431        let rank = config.self_rank();
432        let (deliver_here, next_steps) =
433            ndslice::selection::routing::resolve_routing(rank, dests, &mut |_| {
434                panic!("Choice encountered in CommActor routing")
435            })?;
436
437        let recv_state = self.recv_state.entry(message.stream_key()).or_default();
438        match recv_state.seq.cmp(&last_seq) {
439            // We got the expected next message to deliver to this host.
440            Ordering::Equal => {
441                // We got an in-order operation, so handle it now.
442                Self::handle_message(
443                    cx,
444                    config,
445                    deliver_here,
446                    next_steps,
447                    sender.clone(),
448                    message,
449                    seq,
450                    &mut recv_state.last_seqs,
451                )?;
452                recv_state.seq = seq;
453
454                // Also deliver any pending operations from the recv buffer that
455                // were received out-of-order that are now unblocked.
456                while let Some(Buffered {
457                    seq,
458                    deliver_here,
459                    next_steps,
460                    message,
461                }) = recv_state.buffer.remove(&recv_state.seq)
462                {
463                    Self::handle_message(
464                        cx,
465                        config,
466                        deliver_here,
467                        next_steps,
468                        sender.clone(),
469                        message,
470                        seq,
471                        &mut recv_state.last_seqs,
472                    )?;
473                    recv_state.seq = seq;
474                }
475            }
476            // We got an out-of-order operation, so buffer it for now, until we
477            // recieved the onces sequenced before it.
478            Ordering::Less => {
479                tracing::warn!(
480                    "buffering out-of-order message with seq {} (last {}), expected {}: {:?}",
481                    seq,
482                    last_seq,
483                    recv_state.seq,
484                    message
485                );
486                recv_state.buffer.insert(
487                    last_seq,
488                    Buffered {
489                        seq,
490                        deliver_here,
491                        next_steps,
492                        message,
493                    },
494                );
495            }
496            // We already got this message -- just drop it.
497            Ordering::Greater => {
498                tracing::warn!("received duplicate message with seq {}: {:?}", seq, message);
499            }
500        }
501
502        Ok(())
503    }
504}

677    // A representation of a tree.
678    //   * Map's keys are the tree's leafs;
679    //   * Map's values are the path from the root to that leaf.
680    #[derive(PartialEq)]
681    struct PathToLeaves<T>(BTreeMap<T, Vec<T>>);
682
683    // Add a custom Debug trait impl so the result from assert_eq! is readable.
684    impl<T: Display> Debug for PathToLeaves<T> {
685        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
686            fn vec_to_string<T: Display>(v: &[T]) -> String {
687                v.iter()
688                    .map(ToString::to_string)
689                    .collect::<Vec<String>>()
690                    .join(", ")
691            }
692
693            for (src, path) in &self.0 {
694                write!(f, "{} -> {}\n", src, vec_to_string(path))?;
695            }
696            Ok(())
697        }
698    }

700    fn build_paths<T: Clone + Eq + Hash + Ord>(edges: &[Edge<T>]) -> PathToLeaves<T> {
701        let mut child_parent_map = HashMap::new();
702        let mut all_nodes = HashSet::new();
703        let mut parents = HashSet::new();
704        let mut children = HashSet::new();
705        let mut dests = HashSet::new();
706
707        // Build parent map and track all nodes and children
708        for Edge { from, to, is_leaf } in edges {
709            child_parent_map.insert(to.clone(), from.clone());
710            all_nodes.insert(from.clone());
711            all_nodes.insert(to.clone());
712            parents.insert(from.clone());
713            children.insert(to.clone());
714            if *is_leaf {
715                dests.insert(to.clone());
716            }
717        }
718
719        // For each leaf, reconstruct path back to root
720        let mut result = BTreeMap::new();
721        for dest in dests {
722            let mut path = vec![dest.clone()];
723            let mut current = dest.clone();
724            while let Some(parent) = child_parent_map.get(&current) {
725                path.push(parent.clone());
726                current = parent.clone();
727            }
728            path.reverse();
729            result.insert(dest, path);
730        }
731
732        PathToLeaves(result)
733    }

766    //  Given a port tree,
767    //     * remove the client port, i.e. the 1st element of the path;
768    //     * verify all remaining ports are comm actor ports;
769    //     * remove the actor information and return a rank-based tree representation.
770    //
771    //  The rank-based tree representation is what [collect_commactor_routing_tree] returns.
772    //  This conversion enables us to compare the path against [collect_commactor_routing_tree]'s result.
773    //
774    //      For example, for a 2x2 slice, the port tree could look like:
775    //      dest[0].comm[0][1028] -> [client[0].client_user[0][1025], dest[0].comm[0][1028]]
776    //      dest[1].comm[0][1028] -> [client[0].client_user[0][1025], dest[0].comm[0][1028], dest[1].comm[0][1028]]
777    //      dest[2].comm[0][1028] -> [client[0].client_user[0][1025], dest[0].comm[0][1028], dest[2].comm[0][1028]]
778    //      dest[3].comm[0][1028] -> [client[0].client_user[0][1025], dest[0].comm[0][1028], dest[2].comm[0][1028], dest[3].comm[0][1028]]
779    //
780    //     The result should be:
781    //     0 -> 0
782    //     1 -> 0, 1
783    //     2 -> 0, 2
784    //     3 -> 0, 2, 3
785    fn get_ranks(paths: PathToLeaves<PortId>, client_reply: &PortId) -> PathToLeaves<Index> {
786        let ranks = paths
787            .0
788            .into_iter()
789            .map(|(dst, mut path)| {
790                let first = path.remove(0);
791                // The first PortId is the client's reply port.
792                assert_eq!(&first, client_reply);
793                // Other ports's actor ID must be dest[?].comm[0], where ? is
794                // the rank we want to extract here.
795                assert!(dst.actor_id().name().contains("comm"));
796                let actor_path = path
797                    .into_iter()
798                    .map(|p| {
799                        assert!(p.actor_id().name().contains("comm"));
800                        p.actor_id().rank()
801                    })
802                    .collect();
803                (dst.into_actor_id().rank(), actor_path)
804            })
805            .collect();
806        PathToLeaves(ranks)
807    }

835    // Verify the split port paths are the same as the casting paths.
836    fn verify_split_port_paths(
837        selection: &Selection,
838        extent: &Extent,
839        reply_port_ref1: &PortRef<u64>,
840        reply_port_ref2: &PortRef<MyReply>,
841    ) {
842        // Get the paths used in casting
843        let sel_paths = PathToLeaves(
844            collect_commactor_routing_tree(selection, &extent.to_slice())
845                .delivered
846                .into_iter()
847                .collect(),
848        );
849
850        // Get the split port paths collected in SPLIT_PORT_TREE during casting
851        let (reply1_paths, reply2_paths) = {
852            let tree = SPLIT_PORT_TREE.get().unwrap();
853            let edges = tree.lock().unwrap();
854            let (reply1, reply2): (BTreeMap<_, _>, BTreeMap<_, _>) = build_paths(&edges)
855                .0
856                .into_iter()
857                .partition(|(_dst, path)| &path[0] == reply_port_ref1.port_id());
858            (
859                get_ranks(PathToLeaves(reply1), reply_port_ref1.port_id()),
860                get_ranks(PathToLeaves(reply2), reply_port_ref2.port_id()),
861            )
862        };
863
864        // split port paths should be the same as casting paths
865        assert_eq!(sel_paths, reply1_paths);
866        assert_eq!(sel_paths, reply2_paths);
867    }

956    async fn execute_cast_and_reply(
957        ranks: Vec<ActorRef<TestActor>>,
958        instance: &impl context::Actor,
959        mut reply1_rx: PortReceiver<u64>,
960        mut reply2_rx: PortReceiver<MyReply>,
961        reply_tos: Vec<(PortRef<u64>, PortRef<MyReply>)>,
962    ) {
963        // Reply from each dest actor. The replies should be received by client.
964        {
965            for (dest_actor, (reply_to1, reply_to2)) in ranks.iter().zip(reply_tos.iter()) {
966                let rank = dest_actor.actor_id().rank() as u64;
967                reply_to1.send(instance, rank).unwrap();
968                let my_reply = MyReply {
969                    sender: dest_actor.actor_id().clone(),
970                    value: rank,
971                };
972                reply_to2.send(instance, my_reply.clone()).unwrap();
973
974                assert_eq!(reply1_rx.recv().await.unwrap(), rank);
975                assert_eq!(reply2_rx.recv().await.unwrap(), my_reply);
976            }
977        }
978
979        tracing::info!("the 1st updates from all dest actors were receivered by client");
980
981        // Now send multiple replies from the dest actors. They should all be
982        // received by client. Replies sent from the same dest actor should
983        // be received in the same order as they were sent out.
984        {
985            let n = 100;
986            let mut expected2: HashMap<usize, Vec<MyReply>> = hashmap! {};
987            for (dest_actor, (_reply_to1, reply_to2)) in ranks.iter().zip(reply_tos.iter()) {
988                let rank = dest_actor.actor_id().rank();
989                let mut sent2 = vec![];
990                for i in 0..n {
991                    let value = (rank * 100 + i) as u64;
992                    let my_reply = MyReply {
993                        sender: dest_actor.actor_id().clone(),
994                        value,
995                    };
996                    reply_to2.send(instance, my_reply.clone()).unwrap();
997                    sent2.push(my_reply);
998                }
999                assert!(
1000                    expected2.insert(rank, sent2).is_none(),
1001                    "duplicate rank {rank} in map"
1002                );
1003            }
1004
1005            let mut received2: HashMap<usize, Vec<MyReply>> = hashmap! {};
1006
1007            for _ in 0..(n * ranks.len()) {
1008                let my_reply = reply2_rx.recv().await.unwrap();
1009                received2
1010                    .entry(my_reply.sender.rank())
1011                    .or_default()
1012                    .push(my_reply);
1013            }
1014            assert_eq!(received2, expected2);
1015        }
1016    }
