33#[pyfunction]
34#[pyo3(signature = ())]
35pub fn bootstrap_main(py: Python) -> PyResult<Bound<PyAny>> {
36    // SAFETY: this is a correct use of this function.
37    let _ = unsafe {
38        fbinit::perform_init();
39    };
40
41    hyperactor::internal_macro_support::tracing::debug!("entering async bootstrap");
42    crate::runtime::future_into_py::<_, ()>(py, async move {
43        // SAFETY:
44        // - Only one of these is ever created.
45        // - This is the entry point of this program, so this will be dropped when
46        // no more FB C++ code is running.
47        let _destroy_guard = unsafe { fbinit::DestroyGuard::new() };
48        bootstrap_or_die().await;
49    })
50}

52#[pyfunction]
53pub fn run_worker_loop_forever(_py: Python<'_>, address: &str) -> PyResult<PyPythonTask> {
54    let addr = ChannelAddr::from_zmq_url(address)?;
55
56    // Check if we're running in a PAR/XAR build by looking for FB_XAR_INVOKED_NAME environment variable
57    let invoked_name = std::env::var("FB_XAR_INVOKED_NAME");
58
59    let mut env: std::collections::HashMap<String, String> = std::env::vars().collect();
60
61    let command = Some(if let Ok(invoked_name) = invoked_name {
62        // For PAR/XAR builds: use argv[0] from Python's sys.argv as the current executable
63        let current_exe = std::path::PathBuf::from(&invoked_name);
64
65        // For PAR/XAR builds: set PAR_MAIN_OVERRIDE and no additional args
66        env.insert(
67            "PAR_MAIN_OVERRIDE".to_string(),
68            "monarch._src.actor.bootstrap_main".to_string(),
69        );
70        BootstrapCommand {
71            program: current_exe,
72            arg0: Some(invoked_name),
73            args: vec![],
74            env,
75        }
76    } else {
77        // For regular Python builds: use current_exe() and -m arguments
78        let current_exe = std::env::current_exe().map_err(|e| {
79            pyo3::PyErr::new::<pyo3::exceptions::PyRuntimeError, _>(format!(
80                "Failed to get current executable: {}",
81                e
82            ))
83        })?;
84        let current_exe_str = current_exe.to_string_lossy().to_string();
85        BootstrapCommand {
86            program: current_exe,
87            arg0: Some(current_exe_str),
88            args: vec![
89                "-m".to_string(),
90                "monarch._src.actor.bootstrap_main".to_string(),
91            ],
92            env,
93        }
94    });
95
96    let boot = Bootstrap::Host {
97        addr,
98        config: None,
99        command,
100    };
101
102    PyPythonTask::new(async {
103        let err = boot.bootstrap().await;
104        Err(err).map_pyerr()?;
105        Ok(())
106    })
107}

109#[pyfunction]
110pub fn attach_to_workers<'py>(
111    workers: Vec<Bound<'py, PyPythonTask>>,
112    name: Option<&str>,
113) -> PyResult<PyPythonTask> {
114    let tasks = workers
115        .into_iter()
116        .map(|x| x.borrow_mut().take_task())
117        .collect::<PyResult<Vec<_>>>()?;
118
119    let name =
120        Name::new(name.unwrap_or("hosts")).map_err(|err| PyException::new_err(err.to_string()))?;
121    PyPythonTask::new(async move {
122        let results = try_join_all(tasks).await?;
123
124        let addresses: Result<Vec<ChannelAddr>, anyhow::Error> = monarch_with_gil(|py| {
125            results
126                .into_iter()
127                .map(|result| {
128                    let url_str: String = result.bind(py).extract()?;
129                    ChannelAddr::from_zmq_url(&url_str)
130                })
131                .collect()
132        })
133        .await;
134        let addresses = addresses?;
135
136        let host_mesh = HostMesh::take(HostMeshRef::from_hosts(name, addresses));
137        Ok(PyHostMesh::new_owned(host_mesh))
138    })
139}

141pub fn register_python_bindings(hyperactor_mod: &Bound<'_, PyModule>) -> PyResult<()> {
142    let f = wrap_pyfunction!(bootstrap_main, hyperactor_mod)?;
143    f.setattr(
144        "__module__",
145        "monarch._rust_bindings.monarch_hyperactor.bootstrap",
146    )?;
147    hyperactor_mod.add_function(f)?;
148
149    let f = wrap_pyfunction!(run_worker_loop_forever, hyperactor_mod)?;
150    f.setattr(
151        "__module__",
152        "monarch._rust_bindings.monarch_hyperactor.bootstrap",
153    )?;
154    hyperactor_mod.add_function(f)?;
155
156    let f = wrap_pyfunction!(attach_to_workers, hyperactor_mod)?;
157    f.setattr(
158        "__module__",
159        "monarch._rust_bindings.monarch_hyperactor.bootstrap",
160    )?;
161    hyperactor_mod.add_function(f)?;
162
163    Ok(())
164}
