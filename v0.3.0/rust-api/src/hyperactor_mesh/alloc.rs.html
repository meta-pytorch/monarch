48/// Errors that occur during allocation operations.
49#[derive(Debug, thiserror::Error)]
50pub enum AllocatorError {
51    #[error("incomplete allocation; expected: {0}")]
52    Incomplete(Extent),
53
54    /// The requested shape is too large for the allocator.
55    #[error("not enough resources; requested: {requested}, available: {available}")]
56    NotEnoughResources { requested: Extent, available: usize },
57
58    /// An uncategorized error from an underlying system.
59    #[error(transparent)]
60    Other(#[from] anyhow::Error),
61}

71/// Specifies how to interpret the extent dimensions for allocation.
72#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq)]
73#[derive(Default)]
74pub enum ProcAllocationMode {
75    /// Proc-level allocation: splits extent to allocate multiple processes per host.
76    /// Requires at least 2 dimensions (e.g., [hosts: N, gpus: M]).
77    /// Splits by second-to-last dimension, creating N regions with M processes each.
78    /// Used by MastAllocator.
79    #[default]
80    ProcLevel,
81    /// Host-level allocation: each point in the extent is a host (no sub-host splitting).
82    /// For extent!(region = 2, host = 4), create 8 regions, each representing 1 host.
83    /// Used by MastHostAllocator.
84    HostLevel,
85}

130/// A proc's status. A proc can only monotonically move from
131/// `Created` to `Running` to `Stopped`.
132#[derive(
133    Clone,
134    Debug,
135    PartialEq,
136    EnumAsInner,
137    Serialize,
138    Deserialize,
139    AsRefStr,
140    Named
141)]
142pub enum ProcState {
143    /// A proc was added to the alloc.
144    Created {
145        /// A key to uniquely identify a created proc. The key is used again
146        /// to identify the created proc as Running.
147        create_key: ShortUuid,
148        /// Its assigned point (in the alloc's extent).
149        point: Point,
150        /// The system process ID of the created child process.
151        pid: u32,
152    },
153    /// A proc was started.
154    Running {
155        /// The key used to identify the created proc.
156        create_key: ShortUuid,
157        /// The proc's assigned ID.
158        proc_id: ProcId,
159        /// Reference to this proc's mesh agent. In the future, we'll reserve a
160        /// 'well known' PID (0) for this purpose.
161        mesh_agent: ActorRef<ProcMeshAgent>,
162        /// The address of this proc. The endpoint of this address is
163        /// the proc's mailbox, which accepts [`hyperactor::mailbox::MessageEnvelope`]s.
164        addr: ChannelAddr,
165    },
166    /// A proc was stopped.
167    Stopped {
168        create_key: ShortUuid,
169        reason: ProcStopReason,
170    },
171    /// Allocation process encountered an irrecoverable error. Depending on the
172    /// implementation, the allocation process may continue transiently and calls
173    /// to next() may return some events. But eventually the allocation will not
174    /// be complete. Callers can use the `description` to determine the reason for
175    /// the failure.
176    /// Allocation can then be cleaned up by calling `stop()`` on the `Alloc` and
177    /// drain the iterator for clean shutdown.
178    Failed {
179        /// The world ID of the failed alloc.
180        ///
181        /// TODO: this is not meaningful with direct addressing.
182        world_id: WorldId,
183        /// A description of the failure.
184        description: String,
185    },
186}
187
188impl fmt::Display for ProcState {
189    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
190        match self {
191            ProcState::Created {
192                create_key,
193                point,
194                pid,
195            } => {
196                write!(f, "{}: created at ({}) with PID {}", create_key, point, pid)
197            }
198            ProcState::Running { proc_id, addr, .. } => {
199                write!(f, "{}: running at {}", proc_id, addr)
200            }
201            ProcState::Stopped { create_key, reason } => {
202                write!(f, "{}: stopped: {}", create_key, reason)
203            }
204            ProcState::Failed {
205                description,
206                world_id,
207            } => {
208                write!(f, "{}: failed: {}", world_id, description)
209            }
210        }
211    }
212}

214/// The reason a proc stopped.
215#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, EnumAsInner)]
216pub enum ProcStopReason {
217    /// The proc stopped gracefully, e.g., with exit code 0.
218    Stopped,
219    /// The proc exited with the provided error code and stderr
220    Exited(i32, String),
221    /// The proc was killed. The signal number is indicated;
222    /// the flags determines whether there was a core dump.
223    Killed(i32, bool),
224    /// The proc failed to respond to a watchdog request within a timeout.
225    Watchdog,
226    /// The host running the proc failed to respond to a watchdog request
227    /// within a timeout.
228    HostWatchdog,
229    /// The proc failed for an unknown reason.
230    Unknown,
231}
232
233impl fmt::Display for ProcStopReason {
234    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
235        match self {
236            Self::Stopped => write!(f, "stopped"),
237            Self::Exited(code, stderr) => {
238                if stderr.is_empty() {
239                    write!(f, "exited with code {}", code)
240                } else {
241                    write!(f, "exited with code {}: {}", code, stderr)
242                }
243            }
244            Self::Killed(signal, dumped) => {
245                write!(f, "killed with signal {} (core dumped={})", signal, dumped)
246            }
247            Self::Watchdog => write!(f, "proc watchdog failure"),
248            Self::HostWatchdog => write!(f, "host watchdog failure"),
249            Self::Unknown => write!(f, "unknown"),
250        }
251    }
252}

254/// An alloc is a specific allocation, returned by an [`Allocator`].
255#[automock]
256#[async_trait]
257pub trait Alloc {
258    /// Return the next proc event. `None` indicates that there are
259    /// no more events, and that the alloc is stopped.
260    async fn next(&mut self) -> Option<ProcState>;
261
262    /// The spec against which this alloc is executing.
263    fn spec(&self) -> &AllocSpec;
264
265    /// The shape of the alloc.
266    fn extent(&self) -> &Extent;
267
268    /// The shape of the alloc. (Deprecated.)
269    fn shape(&self) -> Shape {
270        let slice = Slice::new_row_major(self.extent().sizes());
271        Shape::new(self.extent().labels().to_vec(), slice).unwrap()
272    }
273
274    /// The world id of this alloc, uniquely identifying the alloc.
275    /// Note: This will be removed in favor of a different naming scheme,
276    /// once we exise "worlds" from hyperactor core.
277    fn world_id(&self) -> &WorldId;
278
279    /// The channel transport used the procs in this alloc.
280    fn transport(&self) -> ChannelTransport {
281        self.spec().transport.clone()
282    }
283
284    /// Stop this alloc, shutting down all of its procs. A clean
285    /// shutdown should result in Stop events from all allocs,
286    /// followed by the end of the event stream.
287    async fn stop(&mut self) -> Result<(), AllocatorError>;
288
289    /// Stop this alloc and wait for all procs to stop. Call will
290    /// block until all ProcState events have been drained.
291    async fn stop_and_wait(&mut self) -> Result<(), AllocatorError> {
292        tracing::error!(
293            name = "AllocStatus",
294            alloc_name = %self.world_id(),
295            status = "StopAndWait",
296        );
297        self.stop().await?;
298        while let Some(event) = self.next().await {
299            tracing::debug!(
300                alloc_name = %self.world_id(),
301                "drained event: {event:?}"
302            );
303        }
304        tracing::error!(
305            name = "AllocStatus",
306            alloc_name = %self.world_id(),
307            status = "Stopped",
308        );
309        Ok(())
310    }
311
312    /// Returns whether the alloc is a local alloc: that is, its procs are
313    /// not independent processes, but just threads in the selfsame process.
314    fn is_local(&self) -> bool {
315        false
316    }
317
318    /// The address that should be used to serve the client's router.
319    fn client_router_addr(&self) -> ChannelAddr {
320        ChannelAddr::any(self.transport())
321    }
322}

342#[async_trait]
343pub(crate) trait AllocExt {
344    /// Perform initial allocation, consuming events until the alloc is fully
345    /// running. Returns the ranked procs.
346    async fn initialize(&mut self) -> Result<Vec<AllocatedProc>, AllocatorError>;
347}
348
349#[async_trait]
350impl<A: ?Sized + Send + Alloc> AllocExt for A {
351    async fn initialize(&mut self) -> Result<Vec<AllocatedProc>, AllocatorError> {
352        // We wait for the full allocation to be running before returning the mesh.
353        let shape = self.shape().clone();
354
355        let mut created = Ranks::new(shape.slice().len());
356        let mut running = Ranks::new(shape.slice().len());
357
358        while !running.is_full() {
359            let Some(state) = self.next().await else {
360                // Alloc finished before it was fully allocated.
361                return Err(AllocatorError::Incomplete(self.extent().clone()));
362            };
363
364            let name = tracing::Span::current()
365                .metadata()
366                .map(|m| m.name())
367                .unwrap_or("initialize");
368            let status = format!("ProcState:{}", state.arm().unwrap_or("unknown"));
369
370            match state {
371                ProcState::Created {
372                    create_key, point, ..
373                } => {
374                    let rank = point.rank();
375                    if let Some(old_create_key) = created.insert(rank, create_key.clone()) {
376                        tracing::warn!(
377                            name,
378                            status,
379                            rank,
380                            "rank {rank} reassigned from {old_create_key} to {create_key}"
381                        );
382                    }
383                    tracing::info!(
384                        name,
385                        status,
386                        rank,
387                        "proc with create key {}, rank {}: created",
388                        create_key,
389                        rank
390                    );
391                }
392                ProcState::Running {
393                    create_key,
394                    proc_id,
395                    mesh_agent,
396                    addr,
397                } => {
398                    let Some(rank) = created.rank(&create_key) else {
399                        tracing::warn!(
400                            name,
401                            %proc_id,
402                            status,
403                            "proc id {proc_id} with create key {create_key} \
404                            is running, but was not created"
405                        );
406                        continue;
407                    };
408
409                    let allocated_proc = AllocatedProc {
410                        create_key,
411                        proc_id: proc_id.clone(),
412                        addr: addr.clone(),
413                        mesh_agent: mesh_agent.clone(),
414                    };
415                    if let Some(old_allocated_proc) = running.insert(*rank, allocated_proc.clone())
416                    {
417                        tracing::warn!(
418                            name,
419                            %proc_id,
420                            status,
421                            rank,
422                            "duplicate running notifications for {rank}: \
423                            old:{old_allocated_proc}; \
424                            new:{allocated_proc}"
425                        )
426                    }
427                    tracing::info!(
428                        name,
429                        %proc_id,
430                        status,
431                        "proc {} rank {}: running at addr:{addr} mesh_agent:{mesh_agent}",
432                        proc_id,
433                        rank
434                    );
435                }
436                // TODO: We should push responsibility to the allocator, which
437                // can choose to either provide a new proc or emit a
438                // ProcState::Failed to fail the whole allocation.
439                ProcState::Stopped { create_key, reason } => {
440                    tracing::error!(
441                        name,
442                        status,
443                        "allocation failed for proc with create key {}: {}",
444                        create_key,
445                        reason
446                    );
447                    return Err(AllocatorError::Other(anyhow::Error::msg(reason)));
448                }
449                ProcState::Failed {
450                    world_id,
451                    description,
452                } => {
453                    tracing::error!(
454                        name,
455                        status,
456                        "allocation failed for world {}: {}",
457                        world_id,
458                        description
459                    );
460                    return Err(AllocatorError::Other(anyhow::Error::msg(description)));
461                }
462            }
463        }
464
465        // We collect all the ranks at this point of completion, so that we can
466        // avoid holding Rcs across awaits.
467        Ok(running.into_iter().map(Option::unwrap).collect())
468    }
469}

471/// If addr is Tcp or Metatls, use its IP address or hostname to create
472/// a new addr with port unspecified.
473///
474/// for other types of addr, return "any" address.
475pub(crate) fn with_unspecified_port_or_any(addr: &ChannelAddr) -> ChannelAddr {
476    match addr {
477        ChannelAddr::Tcp(socket) => {
478            let mut new_socket = socket.clone();
479            new_socket.set_port(0);
480            ChannelAddr::Tcp(new_socket)
481        }
482        ChannelAddr::MetaTls(MetaTlsAddr::Socket(socket)) => {
483            let mut new_socket = socket.clone();
484            new_socket.set_port(0);
485            ChannelAddr::MetaTls(MetaTlsAddr::Socket(new_socket))
486        }
487        ChannelAddr::MetaTls(MetaTlsAddr::Host { hostname, port: _ }) => {
488            ChannelAddr::MetaTls(MetaTlsAddr::Host {
489                hostname: hostname.clone(),
490                port: 0,
491            })
492        }
493        _ => addr.transport().any(),
494    }
495}

653    pub(crate) async fn test_allocator_basic(mut allocator: impl Allocator) {
654        let extent = extent!(replica = 4);
655        let mut alloc = allocator
656            .allocate(AllocSpec {
657                extent: extent.clone(),
658                constraints: Default::default(),
659                proc_name: None,
660                transport: default_transport(),
661                proc_allocation_mode: Default::default(),
662            })
663            .await
664            .unwrap();
665
666        // Get everything up into running state. We require that we get
667        // procs 0..4.
668        let mut procs = HashMap::new();
669        let mut created = HashMap::new();
670        let mut running = HashSet::new();
671        while running.len() != 4 {
672            match alloc.next().await.unwrap() {
673                ProcState::Created {
674                    create_key, point, ..
675                } => {
676                    created.insert(create_key, point);
677                }
678                ProcState::Running {
679                    create_key,
680                    proc_id,
681                    ..
682                } => {
683                    assert!(running.insert(create_key.clone()));
684                    procs.insert(proc_id, created.remove(&create_key).unwrap());
685                }
686                event => panic!("unexpected event: {:?}", event),
687            }
688        }
689
690        // We should have complete coverage of all points.
691        let points: HashSet<_> = procs.values().collect();
692        for x in 0..4 {
693            assert!(points.contains(&extent.point(vec![x]).unwrap()));
694        }
695
696        // Every proc should belong to the same "world" (alloc).
697        let worlds: HashSet<_> = procs.keys().map(|proc_id| proc_id.world_id()).collect();
698        assert_eq!(worlds.len(), 1);
699
700        // Now, stop the alloc and make sure it shuts down cleanly.
701
702        alloc.stop().await.unwrap();
703        let mut stopped = HashSet::new();
704        while let Some(ProcState::Stopped {
705            create_key, reason, ..
706        }) = alloc.next().await
707        {
708            assert_eq!(reason, ProcStopReason::Stopped);
709            stopped.insert(create_key);
710        }
711        assert!(alloc.next().await.is_none());
712        assert_eq!(stopped, running);
713    }

796    async fn test_allocator_stuck_task() {
797        // Override config.
798        // Use temporary config for this test
799        let config = hyperactor_config::global::lock();
800        let _guard = config.override_key(
801            hyperactor::config::PROCESS_EXIT_TIMEOUT,
802            Duration::from_secs(1),
803        );
804
805        let command = Command::new(crate::testresource::get(
806            "monarch/hyperactor_mesh/bootstrap",
807        ));
808        let mut allocator = ProcessAllocator::new(command);
809        let mut alloc = allocator
810            .allocate(AllocSpec {
811                extent: extent! { replica = 1 },
812                constraints: Default::default(),
813                proc_name: None,
814                transport: ChannelTransport::Unix,
815                proc_allocation_mode: Default::default(),
816            })
817            .await
818            .unwrap();
819
820        // Get everything up into running state. We require that we get
821        let mut procs = HashMap::new();
822        let mut running = HashSet::new();
823        let mut actor_ref = None;
824        let (router, client, client_proc, router_addr) = spawn_proc(alloc.transport()).await;
825        while running.is_empty() {
826            match alloc.next().await.unwrap() {
827                ProcState::Created {
828                    create_key, point, ..
829                } => {
830                    procs.insert(create_key, point);
831                }
832                ProcState::Running {
833                    create_key,
834                    proc_id,
835                    mesh_agent,
836                    addr,
837                } => {
838                    router.bind(Reference::Proc(proc_id.clone()), addr.clone());
839
840                    assert!(procs.contains_key(&create_key));
841                    assert!(!running.contains(&create_key));
842
843                    actor_ref = Some(
844                        spawn_test_actor(0, &client_proc, &client, router_addr, mesh_agent).await,
845                    );
846                    running.insert(create_key.clone());
847                    break;
848                }
849                event => panic!("unexpected event: {:?}", event),
850            }
851        }
852        assert!(actor_ref.unwrap().send(&client, Wait).is_ok());
853
854        // There is a stuck actor! We should get a watchdog failure.
855        alloc.stop().await.unwrap();
856        let mut stopped = HashSet::new();
857        while let Some(ProcState::Stopped {
858            create_key, reason, ..
859        }) = alloc.next().await
860        {
861            assert_eq!(reason, ProcStopReason::Watchdog);
862            stopped.insert(create_key);
863        }
864        assert!(alloc.next().await.is_none());
