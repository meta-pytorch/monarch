120    /// Pings the proc, returning whether it is alive. This will be replaced by a
121    /// finer-grained lifecycle status in the near future.
122    pub(crate) async fn status(&self, cx: &impl context::Actor) -> v1::Result<bool> {
123        let (port, mut rx) = cx.mailbox().open_port();
124        self.agent
125            .status(cx, port.bind())
126            .await
127            .map_err(|e| Error::CallError(self.agent.actor_id().clone(), e))?;
128        loop {
129            let (rank, status) = rx
130                .recv()
131                .await
132                .map_err(|e| Error::CallError(self.agent.actor_id().clone(), e.into()))?;
133            if rank == self.create_rank {
134                break Ok(status);
135            }
136        }
137    }

139    /// Get the supervision events for one actor with the given name.
140    #[allow(dead_code)]
141    async fn actor_state(
142        &self,
143        cx: &impl context::Actor,
144        name: Name,
145    ) -> v1::Result<resource::State<ActorState>> {
146        let (port, mut rx) = cx.mailbox().open_port::<resource::State<ActorState>>();
147        self.agent
148            .send(
149                cx,
150                resource::GetState::<ActorState> {
151                    name: name.clone(),
152                    reply: port.bind(),
153                },
154            )
155            .map_err(|e| Error::CallError(self.agent.actor_id().clone(), e.into()))?;
156        let state = rx
157            .recv()
158            .await
159            .map_err(|e| Error::CallError(self.agent.actor_id().clone(), e.into()))?;
160        if let Some(ref inner) = state.state {
161            let rank = inner.create_rank;
162            if rank == self.create_rank {
163                Ok(state)
164            } else {
165                Err(Error::CallError(
166                    self.agent.actor_id().clone(),
167                    anyhow::anyhow!(
168                        "Rank on mesh agent not matching for Actor {}: returned {}, expected {}",
169                        name,
170                        rank,
171                        self.create_rank
172                    ),
173                ))
174            }
175        } else {
176            Err(Error::CallError(
177                self.agent.actor_id().clone(),
178                anyhow::anyhow!("Actor {} does not exist", name),
179            ))
180        }
181    }

201    #[allow(dead_code)]

209impl ProcMesh {
210    async fn create<C: context::Actor>(
211        cx: &C,
212        name: Name,
213        allocation: ProcMeshAllocation,
214        spawn_comm_actor: bool,
215    ) -> v1::Result<Self>
216    where
217        C::A: Handler<MeshFailure>,
218    {
219        let comm_actor_name = if spawn_comm_actor {
220            Some(Name::new("comm").unwrap())
221        } else {
222            None
223        };
224
225        let region = allocation.extent().clone().into();
226        let ranks = allocation.ranks();
227        let root_comm_actor = comm_actor_name.as_ref().map(|name| {
228            ActorRef::attest(
229                ranks
230                    .first()
231                    .expect("root mesh cannot be empty")
232                    .actor_id(name),
233            )
234        });
235        let host_mesh = allocation.hosts();
236        let current_ref = ProcMeshRef::new(
237            name.clone(),
238            region,
239            ranks,
240            host_mesh.cloned(),
241            None, // this is the root mesh
242            None, // comm actor is not alive yet
243        )
244        .unwrap();
245
246        let mut proc_mesh = Self {
247            name,
248            allocation,
249            comm_actor_name: comm_actor_name.clone(),
250            current_ref,
251        };
252
253        if let Some(comm_actor_name) = comm_actor_name {
254            // CommActor satisfies `Actor + Referable`, so it can be
255            // spawned and safely referenced via ActorRef<CommActor>.
256            // It is a system actor that should not have a controller managing it.
257            let comm_actor_mesh: ActorMesh<CommActor> = proc_mesh
258                .spawn_with_name(cx, comm_actor_name, &Default::default(), None, true)
259                .await?;
260            let address_book: HashMap<_, _> = comm_actor_mesh
261                .iter()
262                .map(|(point, actor_ref)| (point.rank(), actor_ref))
263                .collect();
264            // Now that we have all of the spawned comm actors, kick them all into
265            // mesh mode.
266            for (rank, comm_actor) in &address_book {
267                comm_actor
268                    .send(cx, CommMeshConfig::new(*rank, address_book.clone()))
269                    .map_err(|e| Error::SendingError(comm_actor.actor_id().clone(), Box::new(e)))?
270            }
271
272            // The comm actor is now set up and ready to go.
273            proc_mesh.current_ref.root_comm_actor = root_comm_actor;
274        }
275
276        Ok(proc_mesh)
277    }

307    /// Allocate a new ProcMesh from the provided alloc.
308    /// Allocate does not require an owning actor because references are not owned.
309    #[track_caller]
310    pub async fn allocate<C: context::Actor>(
311        cx: &C,
312        alloc: Box<dyn Alloc + Send + Sync + 'static>,
313        name: &str,
314    ) -> v1::Result<Self>
315    where
316        C::A: Handler<MeshFailure>,
317    {
318        let caller = Location::caller();
319        Self::allocate_inner(cx, alloc, Name::new(name)?, caller).await
320    }

322    // Use allocate_inner to set field mesh_name in span
323    #[hyperactor::instrument(fields(proc_mesh=name.to_string()))]
324    async fn allocate_inner<C: context::Actor>(
325        cx: &C,
326        mut alloc: Box<dyn Alloc + Send + Sync + 'static>,
327        name: Name,
328        caller: &'static Location<'static>,
329    ) -> v1::Result<Self>
330    where
331        C::A: Handler<MeshFailure>,
332    {
333        let alloc_id = Self::alloc_counter().fetch_add(1, Ordering::Relaxed) + 1;
334        tracing::info!(
335            name = "ProcMeshStatus",
336            status = "Allocate::Attempt",
337            %caller,
338            alloc_id,
339            shape = ?alloc.shape(),
340            "allocating proc mesh"
341        );
342
343        let running = alloc
344            .initialize()
345            .instrument(tracing::info_span!(
346                "ProcMeshStatus::Allocate::Initialize",
347                alloc_id,
348                proc_mesh = %name
349            ))
350            .await?;
351
352        // Wire the newly created mesh into the proc, so that it is routable.
353        // We route all of the relevant prefixes into the proc's forwarder,
354        // and serve it on the alloc's transport.
355        //
356        // This will be removed with direct addressing.
357        let proc = cx.instance().proc();
358
359        // First make sure we can serve the proc:
360        let proc_channel_addr = {
361            let _guard =
362                tracing::info_span!("allocate_serve_proc", proc_id = %proc.proc_id()).entered();
363            let (addr, rx) = channel::serve(ChannelAddr::any(alloc.transport()))?;
364            proc.clone().serve(rx);
365            tracing::info!(
366                name = "ProcMeshStatus",
367                status = "Allocate::ChannelServe",
368                proc_mesh = %name,
369                %addr,
370                "proc started listening on addr: {addr}"
371            );
372            addr
373        };
374
375        let bind_allocated_procs = |router: &DialMailboxRouter| {
376            // Route all of the allocated procs:
377            for AllocatedProc { proc_id, addr, .. } in running.iter() {
378                if proc_id.is_direct() {
379                    continue;
380                }
381                router.bind(proc_id.clone().into(), addr.clone());
382            }
383        };
384
385        // Temporary for backward compatibility with ranked procs and v0 API.
386        // Proc meshes can be allocated either using the root client proc (which
387        // has a DialMailboxRouter forwarder) or a mesh agent proc (which has a
388        // ReconfigurableMailboxSender forwarder with an inner DialMailboxRouter).
389        if let Some(router) = proc.forwarder().downcast_ref() {
390            bind_allocated_procs(router);
391        } else if let Some(router) = proc
392            .forwarder()
393            .downcast_ref::<ReconfigurableMailboxSender>()
394        {
395            bind_allocated_procs(
396                router
397                    .as_inner()
398                    .map_err(|_| Error::UnroutableMesh())?
399                    .as_configured()
400                    .ok_or(Error::UnroutableMesh())?
401                    .downcast_ref()
402                    .ok_or(Error::UnroutableMesh())?,
403            );
404        } else {
405            return Err(Error::UnroutableMesh());
406        }
407
408        // Set up the mesh agents. Since references are not owned, we don't supervise it.
409        // Instead, we just let procs die when they have unhandled supervision events.
410        let address_book: HashMap<_, _> = running
411            .iter()
412            .map(
413                |AllocatedProc {
414                     addr, mesh_agent, ..
415                 }| { (mesh_agent.actor_id().proc_id().clone(), addr.clone()) },
416            )
417            .collect();
418
419        let (config_handle, mut config_receiver) = cx.mailbox().open_port();
420        for (rank, AllocatedProc { mesh_agent, .. }) in running.iter().enumerate() {
421            mesh_agent
422                .configure(
423                    cx,
424                    rank,
425                    proc_channel_addr.clone(),
426                    None, // no supervisor; we just crash
427                    address_book.clone(),
428                    config_handle.bind(),
429                    true,
430                )
431                .await
432                .map_err(Error::ConfigurationError)?;
433        }
434        let mut completed = Ranks::new(running.len());
435        while !completed.is_full() {
436            let rank = config_receiver
437                .recv()
438                .await
439                .map_err(|err| Error::ConfigurationError(err.into()))?;
440            if completed.insert(rank, rank).is_some() {
441                tracing::warn!("multiple completions received for rank {}", rank);
442            }
443        }
444
445        let ranks: Vec<_> = running
446            .into_iter()
447            .enumerate()
448            .map(|(create_rank, allocated)| ProcRef {
449                proc_id: allocated.proc_id,
450                create_rank,
451                agent: allocated.mesh_agent,
452            })
453            .collect();
454
455        let stop = Arc::new(Notify::new());
456        let extent = alloc.extent().clone();
457        let alloc_name = alloc.world_id().to_string();
458
459        let alloc_task = {
460            let stop = Arc::clone(&stop);
461
462            tokio::spawn(
463                async move {
464                    loop {
465                        tokio::select! {
466                            _ = stop.notified() => {
467                                // If we are explicitly stopped, the alloc is torn down.
468                                if let Err(error) = alloc.stop_and_wait().await {
469                                    tracing::error!(
470                                        name = "ProcMeshStatus",
471                                        alloc_name = %alloc.world_id(),
472                                        status = "FailedToStopAlloc",
473                                        %error,
474                                    );
475                                }
476                                break;
477                            }
478                            // We are mostly just using this to drive allocation events.
479                            proc_state = alloc.next() => {
480                                match proc_state {
481                                    // The alloc was stopped.
482                                    None => break,
483                                    Some(proc_state) => {
484                                        tracing::debug!(
485                                            alloc_name = %alloc.world_id(),
486                                            "unmonitored allocation event: {}", proc_state);
487                                    }
488                                }
489
490                            }
491                        }
492                    }
493                }
494                .instrument(tracing::info_span!("alloc_monitor")),
495            )
496        };
497
498        let mesh = Self::create(
499            cx,
500            name,
501            ProcMeshAllocation::Allocated {
502                alloc_name,
503                stop,
504                extent,
505                ranks: Arc::new(ranks),
506                alloc_task: Some(alloc_task),
507            },
508            true, // alloc-based meshes support comm actors
509        )
510        .await;
511        match &mesh {
512            Ok(_) => tracing::info!(name = "ProcMeshStatus", status = "Allocate::Created"),
513            Err(error) => {
514                tracing::info!(name = "ProcMeshStatus", status = "Allocate::Failed", %error)
515            }
516        }
517        mesh
518    }

527    /// Stop this mesh gracefully.
528    pub async fn stop(&mut self, cx: &impl context::Actor, reason: String) -> anyhow::Result<()> {
529        let region = self.region.clone();
530        match &mut self.allocation {
531            ProcMeshAllocation::Allocated {
532                stop,
533                alloc_task,
534                alloc_name,
535                ..
536            } => {
537                stop.notify_one();
538                // Wait for the alloc monitor task to complete, ensuring the
539                // alloc has fully stopped before we drop it.
540                if let Some(handle) = alloc_task.take() {
541                    if let Err(e) = handle.await {
542                        tracing::warn!(
543                            name = "ProcMeshStatus",
544                            proc_mesh = %self.name,
545                            alloc_name,
546                            %e,
547                            "alloc monitor task failed"
548                        );
549                    }
550                }
551                tracing::info!(
552                    name = "ProcMeshStatus",
553                    proc_mesh = %self.name,
554                    alloc_name,
555                    status = "StoppingAlloc",
556                    "alloc {alloc_name} has stopped",
557                );
558                Ok(())
559            }
560            ProcMeshAllocation::Owned { hosts, .. } => {
561                let procs = self.current_ref.proc_ids().collect::<Vec<ProcId>>();
562                // We use the proc mesh region rather than the host mesh region
563                // because the host agent stores one entry per proc, not per host.
564                hosts
565                    .stop_proc_mesh(cx, &self.name, procs, region, reason)
566                    .await
567            }
568        }
569    }

596/// Represents different ways ProcMeshes can be allocated.
597enum ProcMeshAllocation {
598    /// A mesh that has been allocated from an `Alloc`.
599    Allocated {
600        // The name of the alloc from which this mesh was allocated.
601        alloc_name: String,
602
603        // A cancellation token used to stop the task keeping the alloc alive.
604        stop: Arc<Notify>,
605
606        extent: Extent,
607
608        // The allocated ranks.
609        ranks: Arc<Vec<ProcRef>>,
610
611        // The task handle for the alloc monitor. Used to wait for clean shutdown.
612        alloc_task: Option<tokio::task::JoinHandle<()>>,
613    },
614
615    /// An owned allocation: this ProcMesh fully owns the set of ranks.
616    Owned {
617        /// The host mesh from which the proc mesh was spawned.
618        hosts: HostMeshRef,
619        // This is purely for storage: `hosts.extent()` returns a computed (by value)
620        // extent.
621        extent: Extent,
622        /// A proc reference for each rank in the mesh.
623        ranks: Arc<Vec<ProcRef>>,
624    },
625}
626
627impl ProcMeshAllocation {
628    fn extent(&self) -> &Extent {
629        match self {
630            ProcMeshAllocation::Allocated { extent, .. } => extent,
631            ProcMeshAllocation::Owned { extent, .. } => extent,
632        }
633    }
634
635    fn ranks(&self) -> Arc<Vec<ProcRef>> {
636        Arc::clone(match self {
637            ProcMeshAllocation::Allocated { ranks, .. } => ranks,
638            ProcMeshAllocation::Owned { ranks, .. } => ranks,
639        })
640    }
641
642    fn hosts(&self) -> Option<&HostMeshRef> {
643        match self {
644            ProcMeshAllocation::Allocated { .. } => None,
645            ProcMeshAllocation::Owned { hosts, .. } => Some(hosts),
646        }
647    }
648}

671/// A reference to a ProcMesh, consisting of a set of ranked [`ProcRef`]s,
672/// arranged into a region. ProcMeshes are named, uniquely identifying the
673/// ProcMesh from which the reference was derived.
674///
675/// ProcMeshes can be sliced to create new ProcMeshes with a subset of the
676/// original ranks.
677#[derive(Debug, Clone, PartialEq, Eq, Hash, Named, Serialize, Deserialize)]
678pub struct ProcMeshRef {
679    name: Name,
680    region: Region,
681    ranks: Arc<Vec<ProcRef>>,
682    // Some if this was spawned from a host mesh, else none.
683    host_mesh: Option<HostMeshRef>,
684    // Temporary: used to fit v1 ActorMesh with v0's casting implementation. This
685    // should be removed after we remove the v0 code.
686    // The root region of this mesh. None means this mesh itself is the root.
687    pub(crate) root_region: Option<Region>,
688    // Temporary: used to fit v1 ActorMesh with v0's casting implementation. This
689    // should be removed after we remove the v0 code.
690    // v0 casting requires root mesh rank 0 as the 1st hop, so we need to provide
691    // it here. For v1, this can be removed since v1 can use any rank.
692    pub(crate) root_comm_actor: Option<ActorRef<CommActor>>,
693}
694wirevalue::register_type!(ProcMeshRef);

696impl ProcMeshRef {
697    /// Create a new ProcMeshRef from the given name, region, ranks, and so on.
698    #[allow(clippy::result_large_err)]
699    fn new(
700        name: Name,
701        region: Region,
702        ranks: Arc<Vec<ProcRef>>,
703        host_mesh: Option<HostMeshRef>,
704        root_region: Option<Region>,
705        root_comm_actor: Option<ActorRef<CommActor>>,
706    ) -> v1::Result<Self> {
707        if region.num_ranks() != ranks.len() {
708            return Err(v1::Error::InvalidRankCardinality {
709                expected: region.num_ranks(),
710                actual: ranks.len(),
711            });
712        }
713        Ok(Self {
714            name,
715            region,

772    /// The supervision events of procs in this mesh.
773    pub async fn actor_states(
774        &self,
775        cx: &impl context::Actor,
776        name: Name,
777    ) -> v1::Result<ValueMesh<resource::State<ActorState>>> {
778        let agent_mesh = self.agent_mesh();
779        let (port, mut rx) = cx.mailbox().open_port::<resource::State<ActorState>>();
780        // TODO: Use accumulation to get back a single value (representing whether
781        // *any* of the actors failed) instead of a mesh.
782        agent_mesh.cast(
783            cx,
784            resource::GetState::<ActorState> {
785                name: name.clone(),
786                reply: port.bind(),
787            },
788        )?;
789        let expected = self.ranks.len();
790        let mut states = Vec::with_capacity(expected);
791        let timeout = hyperactor_config::global::get(GET_ACTOR_STATE_MAX_IDLE);
792        for _ in 0..expected {
793            // The agent runs on the same process as the running actor, so if some
794            // fatal event caused the process to crash (e.g. OOM, signal, process exit),
795            // the agent will be unresponsive.
796            // We handle this by setting a timeout on the recv, and if we don't get a
797            // message we assume the agent is dead and return a failed state.
798            let state = RealClock.timeout(timeout, rx.recv()).await;
799            if let Ok(state) = state {
800                // Handle non-timeout receiver error.
801                let state = state?;
802                match state.state {
803                    Some(ref inner) => {
804                        states.push((inner.create_rank, state));
805                    }
806                    None => {
807                        return Err(Error::NotExist(state.name));
808                    }
809                }
810            } else {
811                tracing::error!(
812                    "timeout waiting for a message after {:?} from proc mesh agent in mesh {}",
813                    timeout,
814                    agent_mesh
815                );
816                // Timeout error, stop reading from the receiver and send back what we have so far,
817                // padding with failed states.
818                let all_ranks = (0..self.ranks.len()).collect::<HashSet<_>>();
819                let completed_ranks = states.iter().map(|(rank, _)| *rank).collect::<HashSet<_>>();
820                let mut leftover_ranks = all_ranks.difference(&completed_ranks).collect::<Vec<_>>();
821                assert_eq!(leftover_ranks.len(), expected - states.len());
822                while states.len() < expected {
823                    let rank = *leftover_ranks
824                        .pop()
825                        .expect("leftover ranks should not be empty");
826                    let agent = agent_mesh.get(rank).expect("agent should exist");
827                    let agent_id = agent.actor_id().clone();
828                    states.push((
829                        // We populate with any ranks leftover at the time of the timeout.
830                        rank,
831                        resource::State {
832                            name: name.clone(),
833                            status: resource::Status::Timeout(timeout),
834                            // We don't know the ActorId that used to live on this rank.
835                            // But we do know the mesh agent id, so we'll use that.
836                            state: Some(ActorState {
837                                actor_id: agent_id.clone(),
838                                create_rank: rank,
839                                supervision_events: vec![ActorSupervisionEvent::new(
840                                    agent_id,
841                                    None,
842                                    ActorStatus::generic_failure(format!(
843                                        "timeout waiting for message from proc mesh agent while querying for \"{}\". The process likely crashed",
844                                        name,
845                                    )),
846                                    None,
847                                )],
848                            }),
849                        },
850                    ));
851                }
852                break;
853            }
854        }
855        // Ensure that all ranks have replied. Note that if the mesh is sliced,
856        // not all create_ranks may be in the mesh.
857        // Sort by rank, so that the resulting mesh is ordered.
858        states.sort_by_key(|(rank, _)| *rank);
859        let vm = states
860            .into_iter()
861            .map(|(_, state)| state)
862            .collect_mesh::<ValueMesh<_>>(self.region.clone())?;
863        Ok(vm)
864    }

947    ///   is delivered when the actors spawned in the mesh have a failure that
948    ///   isn't handled.
949    #[hyperactor::instrument(fields(

962    where

969        );

971        let result = self
972            .spawn_with_name_inner(cx, name, params, supervision_display_name, is_system_actor)
973            .await;
974        match &result {
975            Ok(_) => {
976                tracing::info!(
977                    name = "ProcMeshStatus",
978                    status = "ActorMesh::Spawn::Success",
979                );
980                tracing::info!(name = "ActorMeshStatus", status = "Spawn::Success");
981            }
982            Err(error) => {
983                tracing::error!(name = "ProcMeshStatus", status = "ActorMesh::Spawn::Failed", %error);
984                tracing::error!(name = "ActorMeshStatus", status = "Spawn::Failed", %error);
985            }
986        }
987        result

990    async fn spawn_with_name_inner<A: RemoteSpawn, C: context::Actor>(
991        &self,
992        cx: &C,
993        name: Name,
994        params: &A::Params,
995        supervision_display_name: Option<String>,
996        is_system_actor: bool,
997    ) -> v1::Result<ActorMesh<A>>
998    where
999        C::A: Handler<MeshFailure>,
1000    {
1001        let remote = Remote::collect();
1002        // `RemoteSpawn` + `remote!(A)` ensure that `A` has a
1003        // `SpawnableActor` entry in this registry, so
1004        // `name_of::<A>()` can resolve its global type name.
1005        let actor_type = remote
1006            .name_of::<A>()
1007            .ok_or(Error::ActorTypeNotRegistered(type_name::<A>().to_string()))?
1008            .to_string();
1009
1010        let serialized_params = bincode::serialize(params)?;
1011        let agent_mesh = self.agent_mesh();
1012
1013        agent_mesh.cast(
1014            cx,
1015            resource::CreateOrUpdate::<mesh_agent::ActorSpec> {
1016                name: name.clone(),
1017                rank: Default::default(),
1018                spec: mesh_agent::ActorSpec {
1019                    actor_type: actor_type.clone(),
1020                    params_data: serialized_params.clone(),
1021                },
1022            },
1023        )?;
1024
1025        let region = self.region().clone();
1026        // Open an accum port that *receives overlays* and *emits full
1027        // meshes*.
1028        //
1029        // NOTE: Mailbox initializes the accumulator state via
1030        // `Default`, which is an *empty* ValueMesh (0 ranks). Our
1031        // Accumulator<ValueMesh<T>> implementation detects this on
1032        // the first update and replaces it with the caller-supplied
1033        // template (the `self` passed into open_accum_port), which we
1034        // seed here as "full NotExist over the target region".
1035        let (port, rx) = cx.mailbox().open_accum_port_opts(
1036            // Initial state for the accumulator: full mesh seeded to
1037            // NotExist.
1038            crate::v1::StatusMesh::from_single(region.clone(), Status::NotExist),
1039            StreamingReducerOpts {
1040                max_update_interval: Some(Duration::from_millis(50)),
1041                initial_update_interval: None,
1042            },
1043        );
1044
1045        let mut reply = port.bind();
1046        // If this proc dies or some other issue renders the reply undeliverable,
1047        // the reply does not need to be returned to the sender.
1048        reply.return_undeliverable(false);
1049        // Send a message to all ranks. They reply with overlays to
1050        // `port`.
1051        agent_mesh.cast(
1052            cx,
1053            resource::GetRankStatus {
1054                name: name.clone(),
1055                reply,
1056            },
1057        )?;
1058
1059        let start_time = RealClock.now();
1060
1061        // Wait for all ranks to report a terminal or running status.
1062        // If any proc reports a failure (via supervision) or the mesh
1063        // times out, `wait()` returns Err with the final snapshot.
1064        //
1065        // `rx` is the accumulator output stream: each time reduced
1066        // overlays are applied, it emits a new StatusMesh snapshot.
1067        // `wait()` loops on it, deciding when the stream is
1068        // "complete" (no more NotExist) or times out.
1069        let (statuses, mut mesh) = match GetRankStatus::wait(
1070            rx,
1071            self.ranks.len(),
1072            hyperactor_config::global::get(ACTOR_SPAWN_MAX_IDLE),
1073            region.clone(), // fallback
1074        )
1075        .await
1076        {
1077            Ok(statuses) => {
1078                // Spawn succeeds only if no rank has reported a
1079                // supervision/terminal state. This preserves the old
1080                // `first_terminating().is_none()` semantics.
1081                let has_terminating = statuses.values().any(|s| s.is_terminating());
1082                if !has_terminating {
1083                    Ok((statuses, ActorMesh::new(self.clone(), name, None)))
1084                } else {
1085                    let legacy = mesh_to_rankedvalues_with_default(
1086                        &statuses,
1087                        Status::NotExist,
1088                        Status::is_not_exist,
1089                        self.ranks.len(),
1090                    );
1091                    Err(Error::ActorSpawnError { statuses: legacy })
1092                }
1093            }
1094            Err(complete) => {
1095                // Fill remaining ranks with a timeout status, now
1096                // handled via the legacy shim.
1097                let elapsed = start_time.elapsed();
1098                let legacy = mesh_to_rankedvalues_with_default(
1099                    &complete,
1100                    Status::Timeout(elapsed),
1101                    Status::is_not_exist,
1102                    self.ranks.len(),
1103                );
1104                Err(Error::ActorSpawnError { statuses: legacy })
1105            }
1106        }?;
1107        // We don't need controllers for a system actor like the CommActor.
1108        if !is_system_actor {
1109            // Spawn a unique mesh manager for each actor mesh, so the type of the
1110            // mesh can be preserved.
1111            let controller: ActorMeshController<A> = ActorMeshController::new(
1112                mesh.deref().clone(),
1113                supervision_display_name,
1114                Some(cx.instance().port().bind()),
1115                statuses,
1116            );
1117            let controller = controller
1118                .spawn(cx)
1119                .map_err(|e| Error::ControllerActorSpawnError(mesh.name().clone(), e))?;
1120            // Controller and ActorMesh both depend on references from each other, break
1121            // the cycle by setting the controller after the fact.
1122            mesh.set_controller(Some(controller.bind()));
1123        }
1124        Ok(mesh)
1125    }

1155    async fn stop_actor_by_name_inner(
1156        &self,
1157        cx: &impl context::Actor,
1158        mesh_name: Name,
1159        reason: String,
1160    ) -> v1::Result<ValueMesh<Status>> {
1161        let region = self.region().clone();
1162        let agent_mesh = self.agent_mesh();
1163        agent_mesh.cast(
1164            cx,
1165            resource::Stop {
1166                name: mesh_name.clone(),
1167                reason,
1168            },
1169        )?;
1170
1171        // Open an accum port that *receives overlays* and *emits full
1172        // meshes*.
1173        //
1174        // NOTE: Mailbox initializes the accumulator state via
1175        // `Default`, which is an *empty* ValueMesh (0 ranks). Our
1176        // Accumulator<ValueMesh<T>> implementation detects this on
1177        // the first update and replaces it with the caller-supplied
1178        // template (the `self` passed into open_accum_port), which we
1179        // seed here as "full NotExist over the target region".
1180        let (port, rx) = cx.mailbox().open_accum_port_opts(
1181            // Initial state for the accumulator: full mesh seeded to
1182            // NotExist.
1183            crate::v1::StatusMesh::from_single(region.clone(), Status::NotExist),
1184            StreamingReducerOpts {
1185                max_update_interval: Some(Duration::from_millis(50)),
1186                initial_update_interval: None,
1187            },
1188        );
1189        agent_mesh.cast(
1190            cx,
1191            resource::GetRankStatus {
1192                name: mesh_name,
1193                reply: port.bind(),
1194            },
1195        )?;
1196        let start_time = RealClock.now();
1197
1198        // Reuse actor spawn idle time.
1199        let max_idle_time = hyperactor_config::global::get(ACTOR_SPAWN_MAX_IDLE);
1200        match GetRankStatus::wait(
1201            rx,
1202            self.ranks.len(),
1203            max_idle_time,
1204            region.clone(), // fallback mesh if nothing arrives
1205        )
1206        .await
1207        {
1208            Ok(statuses) => {
1209                // Check that all actors are in some terminal state.
1210                // Failed is ok, because one of these actors may have failed earlier
1211                // and we're trying to stop the others.
1212                let all_stopped = statuses.values().all(|s| s.is_terminating());
1213                if all_stopped {
1214                    Ok(statuses)
1215                } else {
1216                    let legacy = mesh_to_rankedvalues_with_default(
1217                        &statuses,
1218                        Status::NotExist,
1219                        Status::is_not_exist,
1220                        self.ranks.len(),
1221                    );
1222                    Err(Error::ActorStopError { statuses: legacy })
1223                }
1224            }
1225            Err(complete) => {
1226                // Fill remaining ranks with a timeout status via the
1227                // legacy shim.
1228                let legacy = mesh_to_rankedvalues_with_default(
1229                    &complete,
1230                    Status::Timeout(start_time.elapsed()),
1231                    Status::is_not_exist,
1232                    self.ranks.len(),
1233                );
1234                Err(Error::ActorStopError { statuses: legacy })
1235            }
1236        }
1237    }
1238}
