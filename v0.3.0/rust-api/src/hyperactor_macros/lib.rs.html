66/// Represents a variant of an enum.
67#[allow(dead_code)]
68enum Variant {
69    /// A named variant (i.e., `MyVariant { .. }`).
70    Named {
71        enum_name: Ident,
72        name: Ident,
73        field_names: Vec<Ident>,
74        field_types: Vec<Type>,
75        field_flags: Vec<FieldFlag>,
76        is_struct: bool,
77        generics: syn::Generics,
78    },
79    /// An anonymous variant (i.e., `MyVariant(..)`).
80    Anon {
81        enum_name: Ident,
82        name: Ident,
83        field_types: Vec<Type>,
84        field_flags: Vec<FieldFlag>,
85        is_struct: bool,
86        generics: syn::Generics,
87    },
88}
89
90impl Variant {
91    /// The number of fields in the variant.
92    fn len(&self) -> usize {
93        self.field_types().len()
94    }
95
96    /// Returns whether this variant was defined as a struct.
97    fn is_struct(&self) -> bool {
98        match self {
99            Variant::Named { is_struct, .. } => *is_struct,
100            Variant::Anon { is_struct, .. } => *is_struct,
101        }
102    }
103
104    /// The name of the enum containing the variant.
105    fn enum_name(&self) -> &Ident {
106        match self {
107            Variant::Named { enum_name, .. } => enum_name,
108            Variant::Anon { enum_name, .. } => enum_name,
109        }
110    }
111
112    /// The name of the variant itself.
113    fn name(&self) -> &Ident {
114        match self {
115            Variant::Named { name, .. } => name,
116            Variant::Anon { name, .. } => name,
117        }
118    }
119
120    /// The generics of the variant itself.
121    #[allow(dead_code)]
122    fn generics(&self) -> &syn::Generics {
123        match self {
124            Variant::Named { generics, .. } => generics,
125            Variant::Anon { generics, .. } => generics,
126        }
127    }
128
129    /// The snake_name of the variant itself.
130    fn snake_name(&self) -> Ident {
131        Ident::new(
132            &self.name().to_string().to_case(Case::Snake),
133            self.name().span(),
134        )
135    }
136
137    /// The variant's qualified name.
138    fn qualified_name(&self) -> proc_macro2::TokenStream {
139        let enum_name = self.enum_name();
140        let name = self.name();
141
142        if self.is_struct() {
143            quote! { #enum_name }
144        } else {
145            quote! { #enum_name::#name }
146        }
147    }
148
149    /// Names of the fields in the variant. Anonymous variants are named
150    /// according to their position in the argument list.
151    fn field_names(&self) -> Vec<Ident> {
152        match self {
153            Variant::Named { field_names, .. } => field_names.clone(),
154            Variant::Anon { field_types, .. } => (0usize..field_types.len())
155                .map(|idx| format_ident!("arg{}", idx))
156                .collect(),
157        }
158    }
159
160    /// The types of the fields int the variant.
161    fn field_types(&self) -> &Vec<Type> {
162        match self {
163            Variant::Named { field_types, .. } => field_types,
164            Variant::Anon { field_types, .. } => field_types,
165        }
166    }
167
168    /// Return the field flags for this variant.
169    fn field_flags(&self) -> &Vec<FieldFlag> {
170        match self {
171            Variant::Named { field_flags, .. } => field_flags,
172            Variant::Anon { field_flags, .. } => field_flags,
173        }
174    }
175
176    /// The constructor for the variant, using the field names directly.
177    fn constructor(&self) -> proc_macro2::TokenStream {
178        let qualified_name = self.qualified_name();
179        let field_names = self.field_names();
180        match self {
181            Variant::Named { .. } => quote! { #qualified_name { #(#field_names),* } },
182            Variant::Anon { .. } => quote! { #qualified_name(#(#field_names),*) },
183        }
184    }
185}

187struct ReplyPort {
188    is_handle: bool,
189    is_once: bool,
190}
191
192impl ReplyPort {
193    fn from_last_segment(last_segment: &proc_macro2::Ident) -> ReplyPort {
194        ReplyPort {
195            is_handle: last_segment == "PortHandle" || last_segment == "OncePortHandle",
196            is_once: last_segment == "OncePortHandle" || last_segment == "OncePortRef",
197        }
198    }
199
200    fn open_op(&self) -> proc_macro2::TokenStream {
201        if self.is_once {
202            quote! { hyperactor::mailbox::open_once_port }
203        } else {
204            quote! { hyperactor::mailbox::open_port }
205        }
206    }
207
208    fn rx_modifier(&self) -> proc_macro2::TokenStream {
209        if self.is_once {
210            quote! {}
211        } else {
212            quote! { mut }
213        }
214    }
215}

217/// Represents a message that can be sent to a handler, each message is associated with
218/// a variant.
219#[allow(clippy::large_enum_variant)]
220enum Message {
221    /// A call message is a request-response message, the last argument is
222    /// a [`hyperactor::OncePortRef`] or [`hyperactor::OncePortHandle`].
223    Call {
224        variant: Variant,
225        /// Tells whether the reply argument is a handle.
226        reply_port: ReplyPort,
227        /// The underlying return type (i.e., the type of the reply port).
228        return_type: Type,
229        /// the log level for generated instrumentation for handlers of this message.
230        log_level: Option<Ident>,
231    },
232    OneWay {
233        variant: Variant,
234        /// the log level for generated instrumentation for handlers of this message.
235        log_level: Option<Ident>,
236    },
237}
238
239impl Message {
240    fn new(span: Span, variant: Variant, log_level: Option<Ident>) -> Result<Self, syn::Error> {
241        match &variant
242            .field_flags()
243            .iter()
244            .zip(variant.field_types())
245            .filter_map(|(flag, ty)| match flag {
246                FieldFlag::Reply => Some(ty),
247                FieldFlag::None => None,
248            })
249            .collect::<Vec<&Type>>()[..]
250        {
251            [] => Ok(Self::OneWay { variant, log_level }),
252            [reply_port_ty] => {
253                let syn::Type::Path(type_path) = reply_port_ty else {
254                    return Err(syn::Error::new(span, REPLY_VARIANT_ERROR));
255                };
256                let Some(last_segment) = type_path.path.segments.last() else {
257                    return Err(syn::Error::new(span, REPLY_VARIANT_ERROR));
258                };
259                if last_segment.ident != "OncePortRef"
260                    && last_segment.ident != "OncePortHandle"
261                    && last_segment.ident != "PortRef"
262                    && last_segment.ident != "PortHandle"
263                {
264                    return Err(syn::Error::new_spanned(last_segment, REPLY_VARIANT_ERROR));
265                }
266                let syn::PathArguments::AngleBracketed(args) = &last_segment.arguments else {
267                    return Err(syn::Error::new_spanned(last_segment, REPLY_VARIANT_ERROR));
268                };
269                let Some(syn::GenericArgument::Type(return_ty)) = args.args.first() else {
270                    return Err(syn::Error::new_spanned(&args.args, REPLY_VARIANT_ERROR));
271                };
272                let reply_port = ReplyPort::from_last_segment(&last_segment.ident);
273                let return_type = return_ty.clone();
274                Ok(Self::Call {
275                    variant,
276                    reply_port,
277                    return_type,
278                    log_level,
279                })
280            }
281            _ => Err(syn::Error::new(span, REPLY_USAGE_ERROR)),
282        }
283    }
284
285    /// The arguments of this message.
286    fn args(&self) -> Vec<(Ident, Type)> {
287        match self {
288            Message::Call { variant, .. } => variant
289                .field_names()
290                .into_iter()
291                .zip(variant.field_types().clone())
292                .take(variant.len() - 1)
293                .collect(),
294            Message::OneWay { variant, .. } => variant
295                .field_names()
296                .into_iter()
297                .zip(variant.field_types().clone())
298                .collect(),
299        }
300    }
301
302    fn variant(&self) -> &Variant {
303        match self {
304            Message::Call { variant, .. } => variant,
305            Message::OneWay { variant, .. } => variant,
306        }
307    }
308
309    fn reply_port_position(&self) -> Option<usize> {
310        self.variant()
311            .field_flags()
312            .iter()
313            .position(|flag| matches!(flag, FieldFlag::Reply))
314    }
315
316    /// The reply port argument of this message.
317    fn reply_port_arg(&self) -> Option<(Ident, Type)> {
318        match self {
319            Message::Call { variant, .. } => {
320                let pos = self.reply_port_position()?;
321                Some((
322                    variant.field_names()[pos].clone(),
323                    variant.field_types()[pos].clone(),
324                ))
325            }
326            Message::OneWay { .. } => None,
327        }
328    }
329}

331fn parse_log_level(attrs: &[Attribute]) -> Result<Option<Ident>, syn::Error> {
332    let level: Option<String> = match attrs.iter().find(|attr| attr.path().is_ident("log_level")) {
333        Some(attr) => {
334            let Ok(meta) = attr.meta.require_list() else {
335                return Err(syn::Error::new(
336                    Span::call_site(),
337                    indoc! {"
338                            `log_level` attribute must specify level. Supported levels = error, warn, info, debug, trace
339
340                            = help use `#[log_level(info)]` or `#[log_level(error)]`
341                        "},
342                ));
343            };
344            let parsed = meta.parse_args_with(Punctuated::<Ident, Token![,]>::parse_terminated)?;
345            if parsed.len() != 1 {
346                return Err(syn::Error::new(
347                    Span::call_site(),
348                    indoc! {"
349                            `log_level` attribute must specify exactly one level
350
351                            = help use `#[log_level(warn)]` or `#[log_level(info)]`
352                        "},
353                ));
354            };
355            Some(parsed.first().unwrap().to_string())
356        }
357        None => None,
358    };
359
360    if level.is_none() {
361        return Ok(None);
362    }
363    let level = level.unwrap();
364
365    match level.as_str() {
366        "error" | "warn" | "info" | "debug" | "trace" => {}
367        _ => {
368            return Err(syn::Error::new(
369                Span::call_site(),
370                indoc! {"
371                            `log_level` attribute must be one of 'error, warn, info, debug, trace'
372
373                            = help use `#[log_level(warn)]` or `#[log_level(info)]`
374                        "},
375            ));
376        }
377    }
378
379    Ok(Some(Ident::new(
380        level.to_ascii_uppercase().as_str(),
381        Span::call_site(),
382    )))
383}

385fn parse_field_flag(field: &Field) -> FieldFlag {
386    for attr in field.attrs.iter() {
387        match &attr.meta {
388            syn::Meta::Path(path) if path.is_ident("reply") => return FieldFlag::Reply,
389            _ => {}
390        }
391    }
392    FieldFlag::None
393}

395/// Parse a message enum or struct into its constituent messages.
396fn parse_messages(input: DeriveInput) -> Result<Vec<Message>, syn::Error> {
397    match &input.data {
398        Data::Enum(data_enum) => {
399            let mut messages = Vec::new();
400
401            for variant in &data_enum.variants {
402                let name = variant.ident.clone();
403                let attrs = &variant.attrs;
404
405                let message_variant = match &variant.fields {
406                    syn::Fields::Unnamed(fields_) => Variant::Anon {
407                        enum_name: input.ident.clone(),
408                        name,
409                        field_types: fields_
410                            .unnamed
411                            .iter()
412                            .map(|field| field.ty.clone())
413                            .collect(),
414                        field_flags: fields_.unnamed.iter().map(parse_field_flag).collect(),
415                        is_struct: false,
416                        generics: input.generics.clone(),
417                    },
418                    syn::Fields::Named(fields_) => Variant::Named {
419                        enum_name: input.ident.clone(),
420                        name,
421                        field_names: fields_
422                            .named
423                            .iter()
424                            .map(|field| field.ident.clone().unwrap())
425                            .collect(),
426                        field_types: fields_.named.iter().map(|field| field.ty.clone()).collect(),
427                        field_flags: fields_.named.iter().map(parse_field_flag).collect(),
428                        is_struct: false,
429                        generics: input.generics.clone(),
430                    },
431                    _ => {
432                        return Err(syn::Error::new_spanned(
433                            variant,
434                            indoc! {r#"
435                                `Handler` currently only supports named or tuple struct variants
436
437                                = help use `MyCall(Arg1Type, Arg2Type, ..)`,
438                                = help use `MyCall { arg1: Arg1Type, arg2: Arg2Type, .. }`,
439                                = help use `MyCall(Arg1Type, Arg2Type, .., #[reply] OncePortRef<ReplyType>)`
440                                = help use `MyCall { arg1: Arg1Type, arg2: Arg2Type, .., reply: #[reply] OncePortRef<ReplyType>}`
441                                = help use `MyCall(Arg1Type, Arg2Type, .., #[reply] OncePortHandle<ReplyType>)`
442                                = help use `MyCall { arg1: Arg1Type, arg2: Arg2Type, .., reply: #[reply] OncePortHandle<ReplyType>}`
443                              "#},
444                        ));
445                    }
446                };
447                let log_level = parse_log_level(attrs)?;
448
449                messages.push(Message::new(
450                    variant.fields.span(),
451                    message_variant,
452                    log_level,
453                )?);
454            }
455
456            Ok(messages)
457        }
458        Data::Struct(data_struct) => {
459            let struct_name = input.ident.clone();
460            let attrs = &input.attrs;
461
462            let message_variant = match &data_struct.fields {
463                syn::Fields::Unnamed(fields_) => Variant::Anon {
464                    enum_name: struct_name.clone(),
465                    name: struct_name,
466                    field_types: fields_
467                        .unnamed
468                        .iter()
469                        .map(|field| field.ty.clone())
470                        .collect(),
471                    field_flags: fields_.unnamed.iter().map(parse_field_flag).collect(),
472                    is_struct: true,
473                    generics: input.generics.clone(),
474                },
475                syn::Fields::Named(fields_) => Variant::Named {
476                    enum_name: struct_name.clone(),
477                    name: struct_name,
478                    field_names: fields_
479                        .named
480                        .iter()
481                        .map(|field| field.ident.clone().unwrap())
482                        .collect(),
483                    field_types: fields_.named.iter().map(|field| field.ty.clone()).collect(),
484                    field_flags: fields_.named.iter().map(parse_field_flag).collect(),
485                    is_struct: true,
486                    generics: input.generics.clone(),
487                },
488                syn::Fields::Unit => Variant::Anon {
489                    enum_name: struct_name.clone(),
490                    name: struct_name,
491                    field_types: Vec::new(),
492                    field_flags: Vec::new(),
493                    is_struct: true,
494                    generics: input.generics.clone(),
495                },
496            };
497
498            let log_level = parse_log_level(attrs)?;
499            let message = Message::new(data_struct.fields.span(), message_variant, log_level)?;
500
501            Ok(vec![message])
502        }
503        _ => Err(syn::Error::new_spanned(
504            input,
505            "handlers can only be derived for enums and structs",
506        )),
507    }
508}

510/// Derive a custom handler trait for given an enum containing tuple
511/// structs.  The handler trait defines a method corresponding
512/// to each of the enum's variants, and a `handle` function
513/// that dispatches messages to the correct method.  The macro
514/// supports two messaging patterns: "call" and "oneway". A call is a
515/// request-response message; a [`hyperactor::mailbox::OncePortRef`] or
516/// [`hyperactor::mailbox::OncePortHandle`] in the last position is used
517/// to send the return value.
518///
519/// The macro also derives a client trait that can be automatically implemented
520/// by specifying [`HandleClient`] for `ActorHandle<Actor>` and [`RefClient`]
521/// for `ActorRef<Actor>` accordingly. We require two implementations because
522/// not `ActorRef`s require that its message type is serializable.
523///
524/// The associated [`hyperactor_macros::handler`] macro can be used to add
525/// a dispatching handler directly to an [`hyperactor::Actor`].
526///
527/// # Example
528///
529/// The following example creates a "shopping list" actor responsible for
530/// maintaining a shopping list.
531///
532/// ```
533/// use std::collections::HashSet;
534/// use std::time::Duration;
535///
536/// use async_trait::async_trait;
537/// use hyperactor::Actor;
538/// use hyperactor::HandleClient;
539/// use hyperactor::Handler;
540/// use hyperactor::Instance;
541/// use hyperactor::OncePortRef;
542/// use hyperactor::RefClient;
543/// use hyperactor::proc::Proc;
544/// use serde::Deserialize;
545/// use serde::Serialize;
546/// use typeuri::Named;
547///
548/// #[derive(Handler, HandleClient, RefClient, Debug, Serialize, Deserialize, Named)]
549/// enum ShoppingList {
550///     // Oneway messages dispatch messages asynchronously, with no reply.
551///     Add(String),
552///     Remove(String),
553///
554///     // Call messages dispatch a request, expecting a reply to the
555///     // provided port, which must be in the last position.
556///     Exists(String, #[reply] OncePortRef<bool>),
557///
558///     List(#[reply] OncePortRef<Vec<String>>),
559/// }
560///
561/// // Define an actor.
562/// #[derive(Debug)]
563/// #[hyperactor::export(
564///     spawn = true,
565///     handlers = [
566///         ShoppingList,
567///     ],
568/// )]
569/// struct ShoppingListActor(HashSet<String>);
570///
571/// #[async_trait]
572/// impl Actor for ShoppingListActor {
573///     type Params = ();
574///
575///     async fn new(_params: ()) -> Result<Self, anyhow::Error> {
576///         Ok(Self(HashSet::new()))
577///     }
578/// }
579///
580/// // ShoppingListHandler is the trait generated by derive(Handler) above.
581/// // We implement the trait here for the actor, defining a handler for
582/// // each ShoppingList message.
583/// //
584/// // The `forward` attribute installs a handler that forwards messages
585/// // to the `ShoppingListHandler` implementation directly. This can also
586/// // be done manually:
587/// //
588/// // ```ignore
589/// //<ShoppingListActor as ShoppingListHandler>
590/// //     ::handle(self, comm, message).await
591/// // ```
592/// #[async_trait]
593/// #[hyperactor::forward(ShoppingList)]
594/// impl ShoppingListHandler for ShoppingListActor {
595///     async fn add(&mut self, _cx: &Context<Self>, item: String) -> Result<(), anyhow::Error> {
596///         eprintln!("insert {}", item);
597///         self.0.insert(item);
598///         Ok(())
599///     }
600///
601///     async fn remove(&mut self, _cx: &Context<Self>, item: String) -> Result<(), anyhow::Error> {
602///         eprintln!("remove {}", item);
603///         self.0.remove(&item);
604///         Ok(())
605///     }
606///
607///     async fn exists(
608///         &mut self,
609///         _cx: &Context<Self>,
610///         item: String,
611///     ) -> Result<bool, anyhow::Error> {
612///         Ok(self.0.contains(&item))
613///     }
614///
615///     async fn list(&mut self, _cx: &Context<Self>) -> Result<Vec<String>, anyhow::Error> {
616///         Ok(self.0.iter().cloned().collect())
617///     }
618/// }
619///
620/// #[tokio::main]
621/// async fn main() -> Result<(), anyhow::Error> {
622///     let mut proc = Proc::local();
623///
624///     // Spawn our actor, and get a handle for rank 0.
625///     let shopping_list_actor: hyperactor::ActorHandle<ShoppingListActor> =
626///         proc.spawn("shopping", ()).await?;
627///
628///     // We join the system, so that we can send messages to actors.
629///     let client = proc.attach("client").unwrap();
630///
631///     // todo: consider making this a macro to remove the magic names
632///
633///     // Derive(Handler) generates client methods, which call the
634///     // remote handler provided an actor instance,
635///     // the destination actor, and the method arguments.
636///
637///     shopping_list_actor.add(&client, "milk".into()).await?;
638///     shopping_list_actor.add(&client, "eggs".into()).await?;
639///
640///     println!(
641///         "got milk? {}",
642///         shopping_list_actor.exists(&client, "milk".into()).await?
643///     );
644///     println!(
645///         "got yoghurt? {}",
646///         shopping_list_actor
647///             .exists(&client, "yoghurt".into())
648///             .await?
649///     );
650///
651///     shopping_list_actor.remove(&client, "milk".into()).await?;
652///     println!(
653///         "got milk now? {}",
654///         shopping_list_actor.exists(&client, "milk".into()).await?
655///     );
656///
657///     println!(
658///         "shopping list: {:?}",
659///         shopping_list_actor.list(&client).await?
660///     );
661///
662///     let _ = proc
663///         .destroy_and_wait::<()>(Duration::from_secs(1), None)
664///         .await?;
665///     Ok(())
666/// }
667/// ```
668#[proc_macro_derive(Handler, attributes(reply))]
669pub fn derive_handler(input: TokenStream) -> TokenStream {
670    let input = parse_macro_input!(input as DeriveInput);
671    let name: Ident = input.ident.clone();
672    let (_, ty_generics, _) = input.generics.split_for_impl();
673
674    let messages = match parse_messages(input.clone()) {
675        Ok(messages) => messages,
676        Err(err) => return TokenStream::from(err.to_compile_error()),
677    };
678
679    // Trait definition methods for the handler.
680    let mut handler_trait_methods = Vec::new();
681
682    // The arms of the match used in the message dispatcher.
683    let mut match_arms = Vec::new();
684
685    // Trait implemented by clients.
686    let mut client_trait_methods = Vec::new();
687
688    let global_log_level = parse_log_level(&input.attrs).ok().unwrap_or(None);
689
690    for message in &messages {
691        match message {
692            Message::Call {
693                variant,
694                reply_port,
695                return_type,
696                log_level,
697            } => {
698                let (arg_names, arg_types): (Vec<_>, Vec<_>) = message.args().into_iter().unzip();
699                let variant_name_snake = variant.snake_name();
700                let variant_name_snake_deprecated =

1231    ty: Type,
1232    cast: bool,
1233}
1234
1235impl Parse for HandlerSpec {
1236    fn parse(input: ParseStream) -> syn::Result<Self> {
1237        let ty: Type = input.parse()?;
1238
1239        if input.peek(syn::token::Brace) {
1240            let content;
1241            syn::braced!(content in input);
1242            let key: Ident = content.parse()?;
1243            content.parse::<Token![=]>()?;
1244            let expr: Expr = content.parse()?;
1245
1246            let cast = if key == "cast" {
1247                if let Expr::Lit(ExprLit {
1248                    lit: Lit::Bool(b), ..
1249                }) = expr
1250                {
1251                    b.value
1252                } else {
1253                    return Err(syn::Error::new_spanned(expr, "expected boolean for `cast`"));
1254                }
1255            } else {
1256                return Err(syn::Error::new_spanned(
1257                    key,
1258                    "unsupported field (expected `cast`)",
1259                ));
1260            };
1261
1262            Ok(HandlerSpec { ty, cast })
1263        } else if input.is_empty() || input.peek(Token![,]) {
1264            Ok(HandlerSpec { ty, cast: false })
1265        } else {
1266            // Something unexpected follows the type
1267            let unexpected: proc_macro2::TokenTree = input.parse()?;
1268            Err(syn::Error::new_spanned(
1269                unexpected,
1270                "unexpected token after type — expected `{ ... }` or nothing",
1271            ))
1272        }
1273    }
1274}

1297impl Parse for ExportAttr {
1298    fn parse(input: ParseStream) -> syn::Result<Self> {
1299        let mut spawn = false;
1300        let mut handlers: Vec<HandlerSpec> = vec![];
1301
1302        while !input.is_empty() {
1303            let key: Ident = input.parse()?;
1304            input.parse::<Token![=]>()?;
1305
1306            if key == "spawn" {
1307                let expr: Expr = input.parse()?;
1308                if let Expr::Lit(ExprLit {
1309                    lit: Lit::Bool(b), ..
1310                }) = expr
1311                {
1312                    spawn = b.value;
1313                } else {
1314                    return Err(syn::Error::new_spanned(
1315                        expr,
1316                        "expected boolean for `spawn`",
1317                    ));
1318                }
1319            } else if key == "handlers" {
1320                let content;
1321                bracketed!(content in input);
1322                let raw_handlers = content.parse_terminated(HandlerSpec::parse, Token![,])?;
1323                handlers = raw_handlers.into_iter().collect();
1324            } else {
1325                return Err(syn::Error::new_spanned(
1326                    key,
1327                    "unexpected key in `#[export(...)]`. Only supports `spawn` and `handlers`",
1328                ));
1329            }
1330
1331            // optional trailing comma
1332            let _ = input.parse::<Token![,]>();
1333        }
1334
1335        Ok(ExportAttr { spawn, handlers })
1336    }
1337}

1339/// Exports handlers for this actor. The set of exported handlers
1340/// determine the messages that may be sent to remote references of
1341/// the actor ([`hyperaxtor::ActorRef`]). Only messages that implement
1342/// [`hyperactor::RemoteMessage`] may be exported.
1343///
1344/// Additionally, an exported actor may be remotely spawned,
1345/// indicated by `spawn = true`. Such actors must also ensure that
1346/// their parameter type implements [`hyperactor::RemoteMessage`].
1347///
1348/// # Example
1349///
1350/// In the following example, `MyActor` can be spawned remotely. It also has
1351/// exports handlers for two message types, `MyMessage` and `MyOtherMessage`.
1352/// Consequently, `ActorRef`s of the actor's type may dispatch messages of these
1353/// types.
1354///
1355/// ```ignore
1356/// #[export(
1357///     spawn = true,
1358///     handlers = [
1359///         MyMessage,
1360///         MyOtherMessage,
1361///     ],
1362/// )]
1363/// struct MyActor {}
1364/// ```
1365#[proc_macro_attribute]
1366pub fn export(attr: TokenStream, item: TokenStream) -> TokenStream {
1367    let input: DeriveInput = parse_macro_input!(item as DeriveInput);
1368    let data_type_name = &input.ident;
1369    let (impl_generics, ty_generics, where_clause) = input.generics.split_for_impl();
1370
1371    let ExportAttr { spawn, handlers } = parse_macro_input!(attr as ExportAttr);
1372    let tys = HandlerSpec::add_indexed(handlers);
1373
1374    let mut handles = Vec::new();
1375    let mut bindings = Vec::new();
1376    let mut type_registrations = Vec::new();
1377
1378    for ty in &tys {
1379        handles.push(quote! {
1380            impl #impl_generics hyperactor::actor::RemoteHandles<#ty> for #data_type_name #ty_generics #where_clause {}
1381        });
1382        bindings.push(quote! {
1383            ports.bind::<#ty>();
1384        });
1385        type_registrations.push(quote! {
1386            wirevalue::register_type!(#ty);
1387        });
1388    }
1389
1390    let mut expanded = quote! {
1391        #input
1392
1393        impl #impl_generics hyperactor::actor::Referable for #data_type_name #ty_generics #where_clause {}
1394
1395        #(#handles)*
1396
1397        #(#type_registrations)*
1398
1399        // Always export the `Signal` type.
1400        impl #impl_generics hyperactor::actor::RemoteHandles<hyperactor::actor::Signal> for #data_type_name #ty_generics #where_clause {}
1401
1402        impl #impl_generics hyperactor::actor::Binds<#data_type_name #ty_generics> for #data_type_name #ty_generics #where_clause {
1403            fn bind(ports: &hyperactor::proc::Ports<Self>) {
1404                #(#bindings)*
1405            }
1406        }
1407
1408        // TODO: just use Named derive directly here.
1409        impl #impl_generics typeuri::Named for #data_type_name #ty_generics #where_clause {
1410            fn typename() -> &'static str { concat!(std::module_path!(), "::", stringify!(#data_type_name #ty_generics)) }
1411        }
1412    };
1413
1414    if spawn {
1415        expanded.extend(quote! {
1416            hyperactor::remote!(#data_type_name);
1417        });
1418    }
1419
1420    TokenStream::from(expanded)
1421}

1445/// Create a [`Referable`] definition, handling a specific set of message types.
1446/// Behaviors are used to create an [`ActorRef`] without having to depend on the
1447/// actor's implementation. If the message type need to be cast, add `castable`
1448/// flag to those types. e.g. the following example creates a behavior with 5
1449/// message types, and 4 of which need to be cast.
1450///
1451/// ```
1452/// hyperactor::behavior!(
1453///     TestActorBehavior,
1454///     TestMessage { castable = true },
1455///     () {castable = true },
1456///     MyGeneric<()> {castable = true },
1457///     u64,
1458/// );
1459/// ```
1460///
1461/// This macro also supports generic behaviors:
1462/// ```
1463/// hyperactor::behavior!(
1464///     TestBehavior<T>,
1465///     Message<T> { castable = true },
1466///     u64,
1467/// );
1468/// ```
1469#[proc_macro]
1470pub fn behavior(input: TokenStream) -> TokenStream {
1471    let BehaviorInput {
1472        behavior,
1473        generics,
1474        handlers,
1475    } = parse_macro_input!(input as BehaviorInput);
1476    let tys = HandlerSpec::add_indexed(handlers);
1477
1478    // Add bounds to generics for Named, Serialize, Deserialize
1479    let mut bounded_generics = generics.clone();
1480    for param in bounded_generics.type_params_mut() {
1481        param.bounds.push(syn::parse_quote!(typeuri::Named));
1482        param.bounds.push(syn::parse_quote!(serde::Serialize));
1483        param.bounds.push(syn::parse_quote!(std::marker::Send));
1484        param.bounds.push(syn::parse_quote!(std::marker::Sync));
1485        param.bounds.push(syn::parse_quote!(std::fmt::Debug));
1486        // Note: lifetime parameters are not *actually* hygienic.
1487        // https://github.com/rust-lang/rust/issues/54727
1488        let lifetime =
1489            syn::Lifetime::new("'hyperactor_behavior_de", proc_macro2::Span::mixed_site());
1490        param
1491            .bounds
1492            .push(syn::parse_quote!(for<#lifetime> serde::Deserialize<#lifetime>));
1493    }
1494
1495    // Split the generics for use in different contexts
1496    let (impl_generics, ty_generics, where_clause) = bounded_generics.split_for_impl();
1497
1498    // Create a combined generics for the Binds impl that includes both A and the behavior's generics
1499    let mut binds_generics = bounded_generics.clone();
1500    binds_generics.params.insert(
1501        0,
1502        syn::GenericParam::Type(syn::TypeParam {
1503            attrs: vec![],
1504            ident: Ident::new("A", proc_macro2::Span::call_site()),
1505            colon_token: None,
1506            bounds: Punctuated::new(),
1507            eq_token: None,
1508            default: None,
1509        }),
1510    );
1511    let (binds_impl_generics, _, _) = binds_generics.split_for_impl();
1512
1513    // Determine typename and typehash implementation based on whether we have generics
1514    let type_params: Vec<_> = bounded_generics.type_params().collect();
1515    let has_generics = !type_params.is_empty();
1516
1517    let (typename_impl, typehash_impl) = if has_generics {
1518        // Create format string with placeholders for each generic parameter
1519        let placeholders = vec!["{}"; type_params.len()].join(", ");
1520        let placeholders_format_string = format!("<{}>", placeholders);
1521        let format_string = quote! { concat!(std::module_path!(), "::", stringify!(#behavior), #placeholders_format_string) };
1522
1523        let type_param_idents: Vec<_> = type_params.iter().map(|p| &p.ident).collect();
1524        (
1525            quote! {
1526                typeuri::intern_typename!(Self, #format_string, #(#type_param_idents),*)
1527            },
1528            quote! {
1529                typeuri::cityhasher::hash(Self::typename())
1530            },
1531        )
1532    } else {
1533        (
1534            quote! {
1535                concat!(std::module_path!(), "::", stringify!(#behavior))
1536            },
1537            quote! {
1538                static TYPEHASH: std::sync::LazyLock<u64> = std::sync::LazyLock::new(|| {
1539                    typeuri::cityhasher::hash(<#behavior as typeuri::Named>::typename())
1540                });
1541                *TYPEHASH
1542            },
1543        )
1544    };
1545
1546    let type_param_idents = generics.type_params().map(|p| &p.ident).collect::<Vec<_>>();
1547
1548    let expanded = quote! {
1549        #[doc = "The generated behavior struct."]
1550        #[derive(Debug, serde::Serialize, serde::Deserialize)]
1551        pub struct #behavior #impl_generics #where_clause {
1552            _phantom: std::marker::PhantomData<(#(#type_param_idents),*)>
1553        }
1554
1555        impl #impl_generics typeuri::Named for #behavior #ty_generics #where_clause {
1556            fn typename() -> &'static str {
1557                #typename_impl
1558            }
1559
1560            fn typehash() -> u64 {
1561                #typehash_impl
1562            }
1563        }
1564
1565        impl #impl_generics hyperactor::actor::Referable for #behavior #ty_generics #where_clause {}
1566
1567        impl #binds_impl_generics hyperactor::actor::Binds<A> for #behavior #ty_generics
1568        where
1569            A: hyperactor::Actor #(+ hyperactor::Handler<#tys>)*,
1570            #where_clause
1571        {
1572            fn bind(ports: &hyperactor::proc::Ports<A>) {
1573                #(
1574                    ports.bind::<#tys>();
1575                )*
1576            }
1577        }
1578
1579        #(
1580            impl #impl_generics hyperactor::actor::RemoteHandles<#tys> for #behavior #ty_generics #where_clause {}
1581        )*
1582    };
1583
1584    TokenStream::from(expanded)
1585}

1587fn include_in_bind_unbind(field: &Field) -> syn::Result<bool> {
1588    let mut is_included = false;
1589    for attr in &field.attrs {
1590        if attr.path().is_ident("binding") {
1591            // parse #[binding(include)] and look for exactly "include"
1592            attr.parse_nested_meta(|meta| {
1593                if meta.path.is_ident("include") {
1594                    is_included = true;
1595                    Ok(())
1596                } else {
1597                    let path = meta.path.to_token_stream().to_string().replace(' ', "");
1598                    Err(meta.error(format_args!("unknown binding variant attribute `{}`", path)))
1599                }
1600            })?
1601        }
1602    }
1603    Ok(is_included)
1604}

1606/// The field accessor in struct or enum variant.
1607/// e.g.:
1608///   struct NamedStruct { foo: u32 } => FieldAccessor::Named(Ident::new("foo", Span::call_site()))
1609///   struct UnnamedStruct(u32) => FieldAccessor::Unnamed(Index::from(0))
1610enum FieldAccessor {
1611    Named(Ident),
1612    Unnamed(Index),
1613}
1614
1615/// Result of parsing a field in a struct, or a enum variant.
1616struct ParsedField {
1617    accessor: FieldAccessor,
1618    ty: Type,
1619    included: bool,
1620}
1621
1622impl From<&ParsedField> for (Ident, Type) {
1623    fn from(field: &ParsedField) -> Self {
1624        let field_ident = match &field.accessor {
1625            FieldAccessor::Named(ident) => ident.clone(),
1626            FieldAccessor::Unnamed(i) => {
1627                Ident::new(&format!("f{}", i.index), proc_macro2::Span::call_site())
1628            }
1629        };
1630        (field_ident, field.ty.clone())
1631    }
1632}
1633
1634fn collect_all_fields(fields: &Fields) -> syn::Result<Vec<ParsedField>> {
1635    match fields {
1636        Fields::Named(named) => named
1637            .named
1638            .iter()
1639            .map(|f| {
1640                let accessor = FieldAccessor::Named(f.ident.clone().unwrap());
1641                Ok(ParsedField {
1642                    accessor,
1643                    ty: f.ty.clone(),
1644                    included: include_in_bind_unbind(f)?,
1645                })
1646            })
1647            .collect(),
1648        Fields::Unnamed(unnamed) => unnamed
1649            .unnamed
1650            .iter()
1651            .enumerate()
1652            .map(|(i, f)| {
1653                let accessor = FieldAccessor::Unnamed(Index::from(i));
1654                Ok(ParsedField {
1655                    accessor,
1656                    ty: f.ty.clone(),
1657                    included: include_in_bind_unbind(f)?,
1658                })
1659            })
1660            .collect(),
1661        Fields::Unit => Ok(Vec::new()),
1662    }
1663}

1665fn gen_struct_items<F>(
1666    fields: &Fields,
1667    make_item: F,
1668    is_mutable: bool,
1669) -> syn::Result<Vec<proc_macro2::TokenStream>>
1670where
1671    F: Fn(proc_macro2::TokenStream, Type) -> proc_macro2::TokenStream,
1672{
1673    let borrow = if is_mutable {
1674        quote! { &mut }
1675    } else {
1676        quote! { & }
1677    };
1678    let items: Vec<_> = collect_all_fields(fields)?
1679        .into_iter()
1680        .filter(|f| f.included)
1681        .map(
1682            |ParsedField {
1683                 accessor,
1684                 ty,
1685                 included,
1686             }| {
1687                assert!(included);
1688                let field_accessor = match accessor {
1689                    FieldAccessor::Named(ident) => quote! { #borrow self.#ident },
1690                    FieldAccessor::Unnamed(index) => quote! { #borrow self.#index },
1691                };
1692                make_item(field_accessor, ty)
1693            },
1694        )
1695        .collect();
1696    Ok(items)
1697}

1699/// Generate the field accessor for a enum variant in pattern matching. e.g.
1700/// the <GENERATED> parts in the following example:
1701///
1702///   match my_enum {
1703///     // e.g. MyEnum::Tuple(_, f1, f2, _)
1704///     MyEnum::Tuple(<GENERATED>) => { ... }
1705///     // e.g. MyEnum::Struct { field0: _, field1 }
1706///     MyEnum::Struct(<GENERATED>) => { ... }
1707///   }
1708fn gen_enum_field_accessors(all_fields: &[ParsedField]) -> Vec<proc_macro2::TokenStream> {
1709    all_fields
1710        .iter()
1711        .map(
1712            |ParsedField {
1713                 accessor,
1714                 ty: _,
1715                 included,
1716             }| {
1717                match accessor {
1718                    FieldAccessor::Named(ident) => {
1719                        if *included {
1720                            quote! { #ident }
1721                        } else {
1722                            quote! { #ident: _ }
1723                        }
1724                    }
1725                    FieldAccessor::Unnamed(i) => {
1726                        if *included {
1727                            let ident = Ident::new(
1728                                &format!("f{}", i.index),
1729                                proc_macro2::Span::call_site(),
1730                            );
1731                            quote! { #ident }
1732                        } else {
1733                            quote! { _ }
1734                        }
1735                    }
1736                }
1737            },
1738        )
1739        .collect()
1740}

1742/// Generate all the parts for enum variants. e.g. the <GENERATED> part in the
1743/// following example:
1744///
1745///   match my_enum {
1746///      <GENERATED>
1747///   }
1748fn gen_enum_arms<F>(data: &DataEnum, make_item: F) -> syn::Result<Vec<proc_macro2::TokenStream>>
1749where
1750    F: Fn(proc_macro2::TokenStream, Type) -> proc_macro2::TokenStream,
1751{
1752    data.variants
1753        .iter()
1754        .map(|variant| {
1755            let name = &variant.ident;
1756            let all_fields = collect_all_fields(&variant.fields)?;
1757            let field_accessors = gen_enum_field_accessors(&all_fields);
1758            let included_fields = all_fields.iter().filter(|f| f.included).collect::<Vec<_>>();
1759            let items = included_fields
1760                .iter()
1761                .map(|f| {
1762                    let (accessor, ty) = <(Ident, Type)>::from(*f);
1763                    make_item(quote! { #accessor }, ty)
1764                })
1765                .collect::<Vec<_>>();
1766
1767            Ok(match &variant.fields {
1768                Fields::Named(_) => {
1769                    quote! { Self::#name { #(#field_accessors),* } => { #(#items)* } }
1770                }
1771                Fields::Unnamed(_) => {
1772                    quote! { Self::#name( #(#field_accessors),* ) => { #(#items)* } }
1773                }
1774                Fields::Unit => quote! { Self::#name => { #(#items)* } },
1775            })
1776        })
1777        .collect()
1778}

1780/// Derive a custom implementation of [`hyperactor::message::Bind`] trait for
1781/// a struct or enum. This macro is normally used in tandem with [`fn derive_unbind`]
1782/// to make the applied struct or enum castable.
1783///
1784/// Specifically, the derived implementation iterates through fields annotated
1785/// with `#[binding(include)]` based on their order of declaration in the struct
1786/// or enum. These fields' types must implement `Bind` trait as well. During the
1787/// iteration, parameters from `bindings` are bound to these fields.
1788///
1789/// # Example
1790///
1791/// This macro supports named and unamed structs and enums. Below are examples
1792/// of the supported types:
1793///
1794/// ```
1795/// #[derive(Bind, Unbind)]
1796/// struct MyNamedStruct {
1797///     field0: u64,
1798///     field1: MyReply,
1799///     #[binding(include)]
1800/// nnnn     field2: PortRef<MyReply>,
1801///     field3: bool,
1802///     #[binding(include)]
1803///     field4: hyperactor::PortRef<u64>,
1804/// }
1805///
1806/// #[derive(Bind, Unbind)]
1807/// struct MyUnamedStruct(
1808///     u64,
1809///     MyReply,
1810///     #[binding(include)] hyperactor::PortRef<MyReply>,
1811///     bool,
1812///     #[binding(include)] PortRef<u64>,
1813/// );
1814///
1815/// #[derive(Bind, Unbind)]
1816/// enum MyEnum {
1817///     Unit,
1818///     NoopTuple(u64, bool),
1819///     NoopStruct {
1820///         field0: u64,
1821///         field1: bool,
1822///     },
1823///     Tuple(
1824///         u64,
1825///         MyReply,
1826///         #[binding(include)] PortRef<MyReply>,
1827///         bool,
1828///         #[binding(include)] hyperactor::PortRef<u64>,
1829///     ),
1830///     Struct {
1831///         field0: u64,
1832///         field1: MyReply,
1833///         #[binding(include)]
1834///         field2: PortRef<MyReply>,
1835///         field3: bool,
1836///         #[binding(include)]
1837///         field4: hyperactor::PortRef<u64>,
1838///     },
1839/// }
1840/// ```
1841///
1842/// The following shows what derived `Bind`` and `Unbind`` implementations for
1843/// `MyNamedStruct` will look like. The implementations of other types are
1844/// similar, and thus are not shown here.
1845/// ```ignore
1846/// impl Bind for MyNamedStruct {
1847/// fn bind(&mut self, bindings: &mut Bindings) -> anyhow::Result<()> {
1848///     Bind::bind(&mut self.field2, bindings)?;
1849///     Bind::bind(&mut self.field4, bindings)?;
1850///     Ok(())
1851/// }
1852///
1853/// impl Unbind for MyNamedStruct {
1854///     fn unbind(&self, bindings: &mut Bindings) -> anyhow::Result<()> {
1855///         Unbind::unbind(&self.field2, bindings)?;
1856///         Unbind::unbind(&self.field4, bindings)?;
1857///         Ok(())
1858///     }
1859/// }
1860/// ```
1861#[proc_macro_derive(Bind, attributes(binding))]
1862pub fn derive_bind(input: TokenStream) -> TokenStream {
1863    fn make_item(field_accessor: proc_macro2::TokenStream, _ty: Type) -> proc_macro2::TokenStream {
1864        quote! {
1865            hyperactor::message::Bind::bind(#field_accessor, bindings)?;
1866        }
1867    }
1868
1869    let input = parse_macro_input!(input as DeriveInput);
1870    let name = &input.ident;
1871    let inner = match &input.data {
1872        Data::Struct(DataStruct { fields, .. }) => {
1873            match gen_struct_items(fields, make_item, true) {
1874                Ok(collects) => {
1875                    quote! { #(#collects)* }
1876                }
1877                Err(e) => {
1878                    return TokenStream::from(e.to_compile_error());
1879                }
1880            }
1881        }
1882        Data::Enum(data_enum) => match gen_enum_arms(data_enum, make_item) {
1883            Ok(arms) => {
1884                quote! { match self { #(#arms),* } }
1885            }
1886            Err(e) => {
1887                return TokenStream::from(e.to_compile_error());
1888            }
1889        },
1890        _ => panic!("Bind can only be derived for structs and enums"),
1891    };
1892    let (impl_generics, ty_generics, where_clause) = input.generics.split_for_impl();
1893    let expand = quote! {
1894        #[automatically_derived]
1895        impl #impl_generics hyperactor::message::Bind for #name #ty_generics #where_clause {
1896            fn bind(&mut self, bindings: &mut hyperactor::message::Bindings) -> anyhow::Result<()> {
1897                #inner
1898                Ok(())
1899            }
1900        }
1901    };
1902    TokenStream::from(expand)
1903}

1905/// Derive a custom implementation of [`hyperactor::message::Unbind`] trait for
1906/// a struct or enum. This macro is normally used in tandem with [`fn derive_bind`]
1907/// to make the applied struct or enum castable.
1908///
1909/// Specifically, the derived implementation iterates through fields annoated
1910/// with `#[binding(include)]` based on their order of declaration in the struct
1911/// or enum. These fields' types must implement `Unbind` trait as well. During
1912/// the iteration, parameters from these fields are extracted and stored in
1913/// `bindings`.
1914///
1915/// # Example
1916///
1917/// See [`fn derive_bind`]'s documentation for examples.
1918#[proc_macro_derive(Unbind, attributes(binding))]
1919pub fn derive_unbind(input: TokenStream) -> TokenStream {
1920    fn make_item(field_accessor: proc_macro2::TokenStream, _ty: Type) -> proc_macro2::TokenStream {
1921        quote! {
1922            hyperactor::message::Unbind::unbind(#field_accessor, bindings)?;
1923        }
1924    }
1925
1926    let input = parse_macro_input!(input as DeriveInput);
1927    let name = &input.ident;
1928    let inner = match &input.data {
1929        Data::Struct(DataStruct { fields, .. }) => match gen_struct_items(fields, make_item, false)
1930        {
1931            Ok(collects) => {
1932                quote! { #(#collects)* }
1933            }
1934            Err(e) => {
1935                return TokenStream::from(e.to_compile_error());
1936            }
1937        },
1938        Data::Enum(data_enum) => match gen_enum_arms(data_enum, make_item) {
1939            Ok(arms) => {
1940                quote! { match self { #(#arms),* } }
1941            }
1942            Err(e) => {
1943                return TokenStream::from(e.to_compile_error());
1944            }
1945        },
1946        _ => panic!("Unbind can only be derived for structs and enums"),
1947    };
1948    let (impl_generics, ty_generics, where_clause) = input.generics.split_for_impl();
1949    let expand = quote! {
1950        #[automatically_derived]
1951        impl #impl_generics hyperactor::message::Unbind for #name #ty_generics #where_clause {
1952            fn unbind(&self, bindings: &mut hyperactor::message::Bindings) -> anyhow::Result<()> {
1953                #inner
1954                Ok(())
1955            }
1956        }
1957    };
1958    TokenStream::from(expand)
1959}

1961// Helper function for common parsing and validation
1962fn parse_observe_function(
1963    attr: TokenStream,
1964    item: TokenStream,
1965) -> syn::Result<(ItemFn, String, String)> {
1966    let input = syn::parse::<ItemFn>(item)?;
1967
1968    if input.sig.asyncness.is_none() {
1969        return Err(syn::Error::new(
1970            input.sig.span(),
1971            "observe macros can only be applied to async functions",
1972        ));
1973    }
1974
1975    let fn_name_str = input.sig.ident.to_string();
1976    let module_name_str = syn::parse::<syn::LitStr>(attr)?.value();
1977
1978    Ok((input, fn_name_str, module_name_str))
1979}

1981// Helper function for creating telemetry identifiers and setup code
1982fn create_telemetry_setup(
1983    module_name_str: &str,
1984    fn_name_str: &str,
1985    include_error: bool,
1986) -> (Ident, Ident, Option<Ident>, proc_macro2::TokenStream) {
1987    let module_and_fn = format!("{}_{}", module_name_str, fn_name_str);
1988    let latency_ident = Ident::new("latency", Span::from(proc_macro::Span::def_site()));
1989
1990    let success_ident = Ident::new("success", Span::from(proc_macro::Span::def_site()));
1991
1992    let error_ident = if include_error {
1993        Some(Ident::new(
1994            "error",
1995            Span::from(proc_macro::Span::def_site()),
1996        ))
1997    } else {
1998        None
1999    };
2000
2001    let error_declaration = if let Some(ref error_ident) = error_ident {
2002        quote! {
2003            hyperactor_telemetry::declare_static_counter!(#error_ident, concat!(#module_and_fn, ".error"));
2004        }
2005    } else {
2006        quote! {}
2007    };
2008
2009    let setup_code = quote! {
2010        use hyperactor_telemetry;
2011        hyperactor_telemetry::declare_static_timer!(#latency_ident, concat!(#module_and_fn, ".latency"), hyperactor_telemetry::TimeUnit::Micros);
2012        hyperactor_telemetry::declare_static_counter!(#success_ident, concat!(#module_and_fn, ".success"));
2013        #error_declaration
2014    };
2015
2016    (latency_ident, success_ident, error_ident, setup_code)
2017}

2019/// A procedural macro that automatically injects telemetry code into async functions
2020/// that return a Result type.
2021///
2022/// This macro wraps async functions and adds instrumentation to measure:
2023/// 1. Latency - how long the function takes to execute
2024/// 2. Error counter - function error count
2025/// 3. Success counter - function completion count
2026///
2027/// # Example
2028///
2029/// ```rust
2030/// use hyperactor_actor::observe_result;
2031///
2032/// #[observe_result("my_module")]
2033/// async fn process_request(user_id: &str) -> Result<String, Error> {
2034///     // Function implementation
2035///     // Telemetry will be automatically collected
2036/// }
2037/// ```
2038#[proc_macro_attribute]
2039pub fn observe_result(attr: TokenStream, item: TokenStream) -> TokenStream {
2040    let (input, fn_name_str, module_name_str) = match parse_observe_function(attr, item) {
2041        Ok(parsed) => parsed,
2042        Err(err) => return err.to_compile_error().into(),
2043    };
2044
2045    let fn_name = &input.sig.ident;
2046    let vis = &input.vis;
2047    let args = &input.sig.inputs;
2048    let return_type = &input.sig.output;
2049    let body = &input.block;
2050    let attrs = &input.attrs;
2051    let generics = &input.sig.generics;
2052
2053    let (latency_ident, success_ident, error_ident, telemetry_setup) =
2054        create_telemetry_setup(&module_name_str, &fn_name_str, true);
2055    let error_ident = error_ident.unwrap();
2056
2057    let result_ident = Ident::new("result", Span::from(proc_macro::Span::def_site()));
2058
2059    // Generate the instrumented function
2060    let expanded = quote! {
2061        #(#attrs)*
2062        #vis async fn #fn_name #generics(#args) #return_type {
2063            #telemetry_setup
2064
2065            let kv_pairs = hyperactor_telemetry::kv_pairs!("function" => #fn_name_str.clone());
2066            let _timer = #latency_ident.start(kv_pairs);
2067
2068            let #result_ident = async #body.await;
2069
2070            match &#result_ident {
2071                Ok(_) => {
2072                    #success_ident.add(
2073                        1,
2074                        hyperactor_telemetry::kv_pairs!("function" => #fn_name_str.clone())
2075                    );
2076                }
2077                Err(_) => {
2078                    #error_ident.add(
2079                        1,
2080                        hyperactor_telemetry::kv_pairs!("function" => #fn_name_str.clone())
2081                    );
2082                }
2083            }
2084
2085            #result_ident
2086        }
2087    };
2088
2089    expanded.into()
2090}
