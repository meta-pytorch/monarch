50pub fn main_impl(
51    serve_address: ChannelAddr,
52    program: Command,
53    timeout: Option<Duration>,
54) -> tokio::task::JoinHandle<Result<(), anyhow::Error>> {
55    #[cfg(unix)]
56    fn ignore_sigpipe() {
57        // SAFETY: ignore SIGPIPE as hyperactor tx/rx channels have the following issue:
58        // When tx tries to send a message, it can either do try_post/post or send.
59        // try_post/post is async and send is sync. However, both methods do not have a way to tell the receiver's state.
60        // For try_post/post, it will send messages in the background. However, if the receiver process gets killed,
61        // try_post/post will continue sending messages leading to SIGPIPE and crash the parent process.
62        // For send, it is a sync call. It waits on the receiver to ack. However, if the receiver is dead or not started yet,
63        // the send will be blocked.
64        //
65        // TODO: We need to fix the above issues but before that, ignore SIGPIPE as a mitigation.
66        unsafe {
67            assert!(libc::signal(libc::SIGPIPE, libc::SIG_IGN) != libc::SIG_ERR);
68        }
69    }
70    ignore_sigpipe();
71
72    tracing::info!("bind address is: {}", serve_address);
73    tracing::info!("program to spawn on allocation request: [{:?}]", &program);
74
75    tokio::spawn(async move {
76        RemoteProcessAllocator::new()
77            .start(program, serve_address, timeout)
78            .await
79    })
80}

126    async fn test_main_impl() -> Result<(), anyhow::Error> {
127        hyperactor::initialize_with_current_runtime();
128
129        let serve_address = ChannelAddr::any(ChannelTransport::Unix);
130        let program = Command::new("/bin/date"); // date is usually a unix built-in command
131        let server_handle = main_impl(serve_address.clone(), program, None);
132
133        let spec = alloc::AllocSpec {
134            // NOTE: x cannot be more than 1 since we created a single process-allocator server instance!
135            extent: extent! { x=1, y=4 },
136            constraints: Default::default(),
137            proc_name: None,
138            transport: ChannelTransport::Unix,
139            proc_allocation_mode: Default::default(),
140        };
141
142        let mut initializer = remoteprocess::MockRemoteProcessAllocInitializer::new();
143        initializer.expect_initialize_alloc().return_once(move || {
144            Ok(vec![remoteprocess::RemoteProcessAllocHost {
145                hostname: serve_address.to_string(),
146                id: serve_address.to_string(),
147            }])
148        });
149
150        let world_id = WorldId("__unused__".to_string());
151
152        let mut alloc =
153            remoteprocess::RemoteProcessAlloc::new(spec.clone(), world_id, 0, initializer)
154                .await
155                .unwrap();
156
157        // make sure we accounted for `world_size` number of Created and Stopped proc states
158        let world_size = spec.extent.num_ranks();
159        let mut created = HashSet::new();
160        let mut stopped = HashSet::new();
161
162        while stopped.len() < world_size {
163            let proc_state = alloc.next().await.unwrap();
164            match proc_state {
165                alloc::ProcState::Created { create_key, .. } => {
166                    // alloc.next() will keep creating procs and incrementing rank id
167                    // so we mod the rank by world_size to map it to its logical rank
168                    eprintln!("created: {}", create_key);
169                    created.insert(create_key);
170                }
171                alloc::ProcState::Stopped { create_key, .. } => {
172                    eprintln!("stopped: {}", create_key);
173                    assert!(created.remove(&create_key));
174                    stopped.insert(create_key);
175                }
176                _ => {}
177            }
178        }
179
180        server_handle.abort();
181        Ok(())
182    }

187    async fn test_timeout() -> Result<(), anyhow::Error> {
188        hyperactor::initialize_with_current_runtime();
189
190        let serve_address = ChannelAddr::any(ChannelTransport::Unix);
191        let program = Command::new("/bin/date"); // date is usually a unix built-in command
192        // 1 second quick timeout to check that it fails.
193        let timeout = Duration::from_millis(500);
194        let server_handle = main_impl(serve_address.clone(), program, Some(timeout));
195
196        let spec = alloc::AllocSpec {
197            // NOTE: x cannot be more than 1 since we created a single process-allocator server instance!
198            extent: extent! { x=1, y=4 },
199            constraints: Default::default(),
200            proc_name: None,
201            transport: ChannelTransport::Unix,
202            proc_allocation_mode: Default::default(),
203        };
204
205        let mut initializer = remoteprocess::MockRemoteProcessAllocInitializer::new();
206        initializer.expect_initialize_alloc().return_once(move || {
207            Ok(vec![remoteprocess::RemoteProcessAllocHost {
208                hostname: serve_address.to_string(),
209                id: serve_address.to_string(),
210            }])
211        });
212
213        let world_id = WorldId("__unused__".to_string());
214
215        // Wait at least as long as the timeout before sending any messages.
216        RealClock.sleep(timeout * 2).await;
217
218        // Attempt to allocate, it should fail because a timeout happens before
219        let mut alloc =
220            remoteprocess::RemoteProcessAlloc::new(spec.clone(), world_id.clone(), 0, initializer)
221                .await
222                .unwrap();
223        let res = alloc.next().await.unwrap();
224        // Should fail because the allocator timed out.
225        if let alloc::ProcState::Failed {
226            world_id: msg_world_id,
227            description,
228        } = res
229        {
230            assert_eq!(msg_world_id, world_id);
231            assert!(description.contains("no process has ever been allocated"));
232        } else {
233            panic!("Unexpected ProcState: {:?}", res);
234        }
235
236        server_handle.abort();
237        Ok(())
238    }

312    /// Tests that an allocator with a timeout, that has a process running and
313    /// receives no messages, will keep running as long as the processes do.
314    #[tokio::test]
315    async fn test_timeout_not_during_execution() -> Result<(), anyhow::Error> {
316        hyperactor::initialize_with_current_runtime();
317
318        let serve_address = ChannelAddr::any(ChannelTransport::Unix);
319        let mut program = Command::new("/usr/bin/sleep"); // use a command that waits for a while
320        program.arg("3");
321        let timeout = Duration::from_millis(500);
322        let server_handle = main_impl(serve_address.clone(), program, Some(timeout));
323
324        let spec = alloc::AllocSpec {
325            // NOTE: x cannot be more than 1 since we created a single process-allocator server instance!
326            extent: extent! { x=1, y=4 },
327            constraints: Default::default(),
328            proc_name: None,
329            transport: ChannelTransport::Unix,
330            proc_allocation_mode: Default::default(),
331        };
332
333        let mut initializer = remoteprocess::MockRemoteProcessAllocInitializer::new();
334        let alloc_host = remoteprocess::RemoteProcessAllocHost {
335            hostname: serve_address.to_string(),
336            id: serve_address.to_string(),
337        };
338        initializer
339            .expect_initialize_alloc()
340            .return_once(move || Ok(vec![alloc_host]));
341
342        let world_id = WorldId("__unused__".to_string());
343
344        // Attempt to allocate, it should succeed because a timeout happens before
345        let mut alloc =
346            remoteprocess::RemoteProcessAlloc::new(spec.clone(), world_id.clone(), 0, initializer)
347                .await
348                .unwrap();
349        // Ensure the process starts. Since the command is "sleep", it should
350        // start without stopping.
351        // make sure we accounted for `world_size` number of Created and Stopped proc states
352        let world_size = spec.extent.num_ranks();
353        let mut created = HashSet::new();
354
355        while created.len() < world_size {
356            let proc_state = alloc.next().await.unwrap();
357            match proc_state {
358                alloc::ProcState::Created { create_key, .. } => {
359                    created.insert(create_key);
360                }
361                _ => {
362                    panic!("Unexpected message: {:?}", proc_state)
363                }
364            }
365        }
366        // Now that all procs have started, wait at least as long as the timeout
367        // before sending any messages. This way we ensure the remote allocator
368        // stays alive as long as the child processes stay alive.
369        RealClock.sleep(timeout * 2).await;
370        // Now wait for more events and ensure they are ProcState::Stopped
371        while !created.is_empty() {
372            let proc_state = alloc.next().await.unwrap();
373            match proc_state {
374                alloc::ProcState::Created { .. } => {
375                    // created.insert(create_key);
376                }
377                alloc::ProcState::Stopped { create_key, .. } => {
378                    created.remove(&create_key);
379                }
380                _ => {
381                    panic!("Unexpected message: {:?}", proc_state)
382                }
383            }
384        }
385        assert!(created.is_empty());
386        server_handle.abort();
387        Ok(())
388    }
