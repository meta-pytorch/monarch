9//! BFS capture of a mesh topology into [`SnapshotData`].
10//!
11//! The entry point is [`capture_snapshot`], which walks the mesh from
12//! root, resolving each node exactly once, and folds the results into
13//! a relational row set ready for ingestion.
14//!
15//! # Capture invariants (CS-*)
16//!
17//! - **CS-1 (snapshot-row-once):** Each successful `capture_snapshot`
18//!   emits exactly one [`SnapshotRow`].
19//! - **CS-2 (resolve-once-per-node):** Each distinct `NodeRef` is
20//!   resolved at most once per capture.
21//! - **CS-3 (edge-per-parent-payload):** Every [`ChildRow`] emitted
22//!   comes from a successfully resolved parent payload's `children`,
23//!   in parent enumeration order.
24//! - **CS-4 (duplicate-node-single-row):** If the same `NodeRef` is
25//!   reachable from multiple parents, [`SnapshotData`] contains one
26//!   node projection but one [`ChildRow`] per observed parent→child
27//!   edge.
28//! - **CS-5 (fold-homomorphism):** Folding a
29//!   [`ConvertedNode`](crate::convert::ConvertedNode) via
30//!   [`SnapshotData::push_converted`] appends exactly one
31//!   [`NodeRow`], exactly one subtype-table row matching `kind_row`,
32//!   optionally one [`ActorFailureRow`], optionally one
33//!   [`ActorInboundOrderingRow`] (CV-8), all of its
34//!   [`OrderingSessionRow`]s (CV-9), optionally one
35//!   [`ActorExecutionRow`] (CV-10), all of its [`ActiveHandlerRow`]s
36//!   (CV-11), and all of its [`ChildRow`]s.
37//! - **CS-6 (resolution-error-boundary):** Resolver transport/query
38//!   failure aborts capture with `Err`. Only successfully resolved
39//!   payloads with `NodeProperties::Error` populate
40//!   `resolution_errors`.
41//! - **CS-7 (typed-visited-key):** Traversal dedup uses typed
42//!   `NodeRef` identity, not stringified IDs.
43//! - **CS-8 (snapshot-ts-capture-start):** `snapshot.snapshot_ts` is
44//!   captured once at traversal start and is independent of per-node
45//!   `as_of`.
46

74/// All row vectors produced by a single snapshot capture.
75#[derive(Debug)]
76pub struct SnapshotData {
77    /// Capture metadata (CS-1: exactly one per successful capture).
78    pub snapshot: SnapshotRow,
79    /// One row per resolved node (CS-2: each `NodeRef` resolved
80    /// once).
81    pub nodes: Vec<NodeRow>,
82    /// One row per parent→child edge (CS-3, CS-4: edges emitted
83    /// per-parent, so a multiply-reachable node has multiple edges).
84    pub children: Vec<ChildRow>,
85    /// Singleton on success — CS-2 applied to the single root entry
86    /// point means root is resolved exactly once.
87    pub root_nodes: Vec<RootNodeRow>,
88    /// One row per resolved host node.
89    pub host_nodes: Vec<HostNodeRow>,
90    /// One row per resolved proc node.
91    pub proc_nodes: Vec<ProcNodeRow>,
92    /// One row per resolved actor node.
93    pub actor_nodes: Vec<ActorNodeRow>,
94    /// One row per actor with `failure_info: Some(…)` (CV-3).
95    pub actor_failures: Vec<ActorFailureRow>,
96    /// One row per actor with `inbound_ordering: Some(…)` (CV-8).
97    pub actor_inbound_orderings: Vec<ActorInboundOrderingRow>,
98    /// One row per RETURNED session in any actor's inbound-ordering
99    /// snapshot (CV-9). Skipped sessions are NOT enumerated; they
100    /// appear only in the parent rollup's `skipped_session_count`.
101    pub ordering_sessions: Vec<OrderingSessionRow>,
102    /// One row per actor with `execution: Some(…)` (CV-10).
103    pub actor_executions: Vec<ActorExecutionRow>,
104    /// One row per in-flight handler in any actor's execution snapshot
105    /// (CV-11). A prefix of the N oldest when truncated; empty when the
106    /// rollup's `complete == false`.
107    pub active_handlers: Vec<ActiveHandlerRow>,
108    /// One row per successfully resolved `NodeProperties::Error`
109    /// (CS-6: distinct from resolver transport failures).
110    pub resolution_errors: Vec<ResolutionErrorRow>,
111}
112
113impl SnapshotData {
114    /// Fold a single converted node into the accumulator (CS-5).
115    ///
116    /// This is the only place that branches on [`NodeKindRow`]. BFS
117    /// calls this and nothing else.
118    pub fn push_converted(&mut self, converted: ConvertedNode) {
119        self.nodes.push(converted.node);
120        self.children.extend(converted.children);
121        if let Some(f) = converted.actor_failure {
122            self.actor_failures.push(f);
123        }
124        if let Some(io) = converted.actor_inbound_ordering {
125            self.actor_inbound_orderings.push(io);
126        }
127        self.ordering_sessions.extend(converted.ordering_sessions);
128        if let Some(e) = converted.actor_execution {
129            self.actor_executions.push(e);
130        }
131        self.active_handlers.extend(converted.active_handlers);
132        match converted.kind_row {
133            NodeKindRow::Root(r) => self.root_nodes.push(r),
134            NodeKindRow::Host(h) => self.host_nodes.push(h),
135            NodeKindRow::Proc(p) => self.proc_nodes.push(p),
136            NodeKindRow::Actor(a) => self.actor_nodes.push(a),
137            NodeKindRow::ResolutionError(e) => self.resolution_errors.push(e),
138        }
139    }
140}

142/// Capture a full mesh snapshot by BFS from root.
143///
144/// The `resolve` closure is called once per distinct `NodeRef`
145/// (CS-2). Resolver transport failures abort capture immediately
146/// (CS-6). Successfully resolved `NodeProperties::Error` payloads are
147/// valid rows, not capture failures.
148pub async fn capture_snapshot<F, Fut>(snapshot_id: &str, resolve: F) -> anyhow::Result<SnapshotData>
149where
150    F: Fn(&NodeRef) -> Fut,
151    Fut: Future<Output = anyhow::Result<NodePayload>>,
152{
153    // CS-1, CS-8: one snapshot row, timestamp at capture start.
154    let snapshot = SnapshotRow {
155        snapshot_id: snapshot_id.to_owned(),
156        snapshot_ts: to_micros(SystemTime::now())
157            .context("failed to compute snapshot timestamp")?,
158    };
159
160    let mut data = SnapshotData {
161        snapshot,
162        nodes: Vec::new(),
163        children: Vec::new(),
164        root_nodes: Vec::new(),
165        host_nodes: Vec::new(),
166        proc_nodes: Vec::new(),
167        actor_nodes: Vec::new(),
168        actor_failures: Vec::new(),
169        actor_inbound_orderings: Vec::new(),
170        ordering_sessions: Vec::new(),
171        actor_executions: Vec::new(),
172        active_handlers: Vec::new(),
173        resolution_errors: Vec::new(),
174    };
175
176    // CS-7: typed visited key.
177    let mut visited: HashSet<NodeRef> = HashSet::new();
178    let mut queue: VecDeque<NodeRef> = VecDeque::new();
179    queue.push_back(NodeRef::Root);
180
181    while let Some(node_ref) = queue.pop_front() {
182        // Dedup on dequeue — queue may contain already-visited refs.
183        if !visited.insert(node_ref.clone()) {
184            continue;
185        }
186
187        // CS-6: resolver failure is an immediate capture error.
188        let payload = resolve(&node_ref)
189            .await
190            .with_context(|| format!("failed to resolve {}", node_ref))?;
191
192        // Enqueue all children unconditionally; dedup happens on
193        // dequeue.
194        for child_ref in &payload.children {
195            queue.push_back(child_ref.clone());
196        }
197
198        // Project and fold.
199        let converted = convert_node(snapshot_id, &payload)
200            .with_context(|| format!("failed to convert {}", node_ref))?;
201        data.push_converted(converted);
202    }
203
204    Ok(data)
205}

253    /// Stub resolver backed by a map. Returns the payload for known
254    /// refs, error for unknown.
255    fn stub_resolver(
256        map: HashMap<NodeRef, NodePayload>,
257    ) -> impl Fn(&NodeRef) -> std::future::Ready<anyhow::Result<NodePayload>> {
258        move |node_ref: &NodeRef| {
259            let result = map
260                .get(node_ref)
261                .cloned()
262                .ok_or_else(|| anyhow::anyhow!("unknown ref: {}", node_ref));
263            std::future::ready(result)
264        }
265    }
266
267    /// Stub resolver that also records every NodeRef requested.
268    fn recording_stub_resolver(
269        map: HashMap<NodeRef, NodePayload>,
270        log: Arc<Mutex<Vec<NodeRef>>>,
271    ) -> impl Fn(&NodeRef) -> std::future::Ready<anyhow::Result<NodePayload>> {
272        move |node_ref: &NodeRef| {
273            log.lock().unwrap().push(node_ref.clone());
274            let result = map
275                .get(node_ref)
276                .cloned()
277                .ok_or_else(|| anyhow::anyhow!("unknown ref: {}", node_ref));
278            std::future::ready(result)
279        }
280    }
