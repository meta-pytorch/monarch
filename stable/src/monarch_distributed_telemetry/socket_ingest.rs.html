9//! Unix-socket ingestion for telemetry Arrow IPC frames.
10//!
11//! This module is the collector-side transport for telemetry producers that
12//! cannot write directly into `DatabaseScanner`. It owns the Unix socket bind,
13//! accepts producer connections, validates the frame envelope, decodes Arrow
14//! IPC payloads, and appends the decoded `RecordBatch` into a pre-registered
15//! `TableStore`.
16//!
17//! The wire format is intentionally narrow: each frame names one destination
18//! table and carries one Arrow IPC stream payload for that table. Table
19//! registration and schema ownership stay in `DatabaseScanner`/`TableStore`,
20//! so this layer only does transport validation and hands decoded batches to
21//! the storage contract.
22//!
23//! Frame layout:
24//!
25//! - `u16`: big-endian table-name byte length
26//! - `[u8]`: UTF-8 table name
27//! - `u32`: big-endian Arrow IPC payload byte length
28//! - `[u8]`: Arrow IPC stream containing exactly one non-empty `RecordBatch`
29

51/// Buffered read capacity for producer socket connections.
52const READER_BUFFER_CAPACITY: usize = 64 * 1024;

54/// Handle for the background socket ingest task.
55pub struct IngestServerHandle {
56    task: JoinHandle<()>,
57}

59impl Drop for IngestServerHandle {
60    fn drop(&mut self) {
61        self.task.abort();
62    }

65/// Bind the telemetry ingest socket for the active collector.
66///
67/// A stale socket file from a crashed collector is removed and replaced. A live
68/// collector is an activation error: callers should avoid starting duplicate
69/// collectors for the same host-local socket namespace.
70pub(crate) fn bind_ingest_socket(path: &Path) -> anyhow::Result<StdUnixListener> {
71    match bind_listener(path) {
72        Ok(listener) => Ok(listener),
73        Err(error) if error.kind() == ErrorKind::AddrInUse => {
74            // `AddrInUse` can mean either a live collector or a stale socket
75            // file. Probe by connecting first so we only remove stale files.
76            if StdUnixStream::connect(path).is_ok() {
77                return Err(anyhow::anyhow!(
78                    "telemetry socket already has a live collector: {}",
79                    path.display()
80                ));
81            }
82
83            std::fs::remove_file(path)
84                .with_context(|| format!("remove stale socket {}", path.display()))?;
85            bind_listener(path).with_context(|| format!("bind socket {}", path.display()))
86        }
87        Err(error) => Err(error).with_context(|| format!("bind socket {}", path.display())),
88    }
89}

91/// Run the ingest accept loop on the active Tokio runtime.
92pub fn run_ingest_server(
93    listener: StdUnixListener,
94    store: TableStore,
95) -> anyhow::Result<IngestServerHandle> {
96    // `bind_ingest_socket` returns a std listener because binding is a
97    // synchronous ownership decision, not part of the async accept loop. Once
98    // ownership is established, convert the fd into Tokio's listener for serving
99    // connections. Tokio requires the fd to be nonblocking before `from_std`.
100    listener.set_nonblocking(true)?;
101    // Called from a PyO3 thread with no ambient Tokio runtime. Use monarch's
102    // shared runtime.
103    let runtime = get_tokio_runtime();
104    let _guard = runtime.enter();
105    let listener = UnixListener::from_std(listener)?;
106    let task = runtime.spawn(async move {
107        loop {
108            match listener.accept().await {
109                Ok((stream, _addr)) => {
110                    let store = store.clone();
111                    tokio::spawn(async move {
112                        // Treat malformed frames as connection-level errors.
113                        // Once the envelope is invalid, stream alignment is no
114                        // longer trustworthy; producers can reconnect, and the
115                        // listener remains available for other connections.
116                        if let Err(error) = read_connection(stream, store).await {
117                            tracing::warn!("telemetry socket connection closed: {error}");
118                        }
119                    });
120                }
121                Err(error) => {
122                    // Most accept errors here are transient (FD exhaustion,
123                    // per-connection aborts). The listener fd stays valid, so
124                    // back off briefly and keep accepting rather than tearing
125                    // down ingest for the scanner lifetime.
126                    tracing::warn!("telemetry socket accept failed: {error}");
127                    tokio::time::sleep(Duration::from_millis(100)).await;
128                }
129            }
130        }
131    });
132
133    Ok(IngestServerHandle { task })
134}

136fn bind_listener(path: &Path) -> std::io::Result<StdUnixListener> {
137    let listener = StdUnixListener::bind(path)?;
138    listener.set_nonblocking(true)?;
139    Ok(listener)
140}

142async fn read_connection(stream: tokio::net::UnixStream, store: TableStore) -> anyhow::Result<()> {
143    let mut reader = BufReader::with_capacity(READER_BUFFER_CAPACITY, stream);
144
145    // A producer may keep one socket connection open and stream many frames.
146    // Each frame declares the destination table, then carries one Arrow IPC
147    // stream payload for that table.
148    loop {
149        let Some(table_name) = read_table_name(&mut reader).await? else {
150            return Ok(());
151        };
152        let payload = read_payload(&mut reader).await?;
153        let batch = decode_one_batch(&table_name, &payload)?;
154        store.push_to_registered(&table_name, batch).await?;
155    }
156}

158async fn read_table_name<R>(reader: &mut R) -> anyhow::Result<Option<String>>
159where
160    R: AsyncRead + Unpin,
161{
162    let mut name_len_bytes = [0; 2];
163    // EOF before the next table-name length is a clean connection close.
164    // Once any header bytes are present, a truncated frame is an error from
165    // the stricter reads below.
166    if !read_exact_or_eof(reader, &mut name_len_bytes).await? {
167        return Ok(None);
168    }
169
170    let name_len = u16::from_be_bytes(name_len_bytes) as usize;
171    if name_len == 0 || name_len > MAX_TABLE_NAME_LEN {
172        anyhow::bail!("invalid table name length {name_len}");
173    }
174
175    let mut name_bytes = vec![0; name_len];
176    reader.read_exact(&mut name_bytes).await?;
177    Ok(Some(String::from_utf8(name_bytes)?))
178}
179
180async fn read_payload<R>(reader: &mut R) -> anyhow::Result<Vec<u8>>
181where
182    R: AsyncRead + Unpin,
183{
184    let mut frame_len_bytes = [0; 4];
185    reader.read_exact(&mut frame_len_bytes).await?;
186
187    let frame_len = u32::from_be_bytes(frame_len_bytes) as usize;
188    if frame_len == 0 || frame_len > MAX_FRAME_LEN {
189        anyhow::bail!("invalid frame length {frame_len}");
190    }
191
192    let mut payload = vec![0; frame_len];
193    reader.read_exact(&mut payload).await?;
194    Ok(payload)
195}
196
197async fn read_exact_or_eof<R>(reader: &mut R, buf: &mut [u8]) -> anyhow::Result<bool>
198where
199    R: AsyncRead + Unpin,
200{
201    match reader.read_exact(buf).await {
202        Ok(_) => Ok(true),
203        Err(error) if error.kind() == ErrorKind::UnexpectedEof => Ok(false),
204        Err(error) => Err(error.into()),
205    }
206}

208fn decode_one_batch(table_name: &str, payload: &[u8]) -> anyhow::Result<RecordBatch> {
209    let mut reader = StreamReader::try_new(Cursor::new(payload), None)?;
210    // Keep the wire contract one table batch per frame. This lets append-time
211    // schema validation report one table context and avoids partial success
212    // when a later batch in the same frame is invalid.
213    let Some(batch) = reader.next().transpose()? else {
214        anyhow::bail!("frame for table {table_name} contained no record batch");
215    };
216
217    if batch.num_rows() == 0 {
218        anyhow::bail!("frame for table {table_name} contained an empty record batch");
219    }
220
221    if reader.next().transpose()?.is_some() {
222        anyhow::bail!("frame for table {table_name} contained multiple record batches");
223    }
224
225    Ok(batch)
226}

276    async fn write_frame(path: &Path, table_name: &str, batch: &RecordBatch) {
277        let payload = serialize_batch(batch).unwrap();
278        let mut frame = Vec::new();
279        append_frame_header(&mut frame, table_name, payload.len());
280        frame.extend_from_slice(&payload);
281
282        let mut stream = tokio::net::UnixStream::connect(path).await.unwrap();
283        stream.write_all(&frame).await.unwrap();
284        stream.shutdown().await.unwrap();
285    }
286
287    fn append_frame_header(buf: &mut Vec<u8>, table_name: &str, payload_len: usize) {
288        let name_len = u16::try_from(table_name.len()).unwrap();
289        let frame_len = u32::try_from(payload_len).unwrap();
290
291        buf.extend_from_slice(&name_len.to_be_bytes());
292        buf.extend_from_slice(table_name.as_bytes());
293        buf.extend_from_slice(&frame_len.to_be_bytes());
294    }

296    async fn count_rows(store: &TableStore, table_name: &str) -> usize {
297        let provider = store.table_provider(table_name).unwrap().unwrap();
298        let ctx = SessionContext::new();
299        ctx.register_table(table_name, provider).unwrap();
300        let batches = ctx
301            .sql(&format!("SELECT COUNT(*) AS cnt FROM {table_name}"))
302            .await
303            .unwrap()
304            .collect()
305            .await
306            .unwrap();
307        let counts = batches[0]
308            .column_by_name("cnt")
309            .unwrap()
310            .as_any()
311            .downcast_ref::<Int64Array>()
312            .unwrap();
313        counts.value(0) as usize
314    }

316    async fn wait_for_rows(store: &TableStore, table_name: &str, expected: usize) {
317        for _ in 0..50 {
318            if count_rows(store, table_name).await == expected {
319                return;
320            }
321            tokio::time::sleep(Duration::from_millis(20)).await;
322        }
323        assert_eq!(count_rows(store, table_name).await, expected);
324    }

431    #[tokio::test]
432    async fn ingest_server_keeps_accepting_after_malformed_frame() {
433        let path = socket_path("survive.sock");
434        let listener = bind_ingest_socket(&path).unwrap();
435        let store = TableStore::new_empty();
436        store.register_table("t", make_batch(&[]).schema()).unwrap();
437        let _handle = run_ingest_server(listener, store.clone()).unwrap();
438
439        // First connection sends a header that declares a name length of zero,
440        // which `read_table_name` rejects. The per-connection task should die
441        // while the accept loop stays alive.
442        let mut bad = tokio::net::UnixStream::connect(&path).await.unwrap();
443        bad.write_all(&0u16.to_be_bytes()).await.unwrap();
444        bad.shutdown().await.unwrap();
445        drop(bad);
446
447        // A subsequent valid connection on the same server should still be
448        // accepted and its batch pushed.
449        write_frame(&path, "t", &make_batch(&[1, 2, 3])).await;
450
451        wait_for_rows(&store, "t", 3).await;
452    }
