9//! RecordBatchSink - Collects tracing telemetry data as Arrow RecordBatches
10//!
11//! Implements TraceEventSink for tracing events (spans, log events).
12//!
13//! Produces three tables:
14//! - `spans`: Information about span creation (NewSpan events)
15//! - `span_events`: Enter/exit/close events for spans
16//! - `events`: Tracing events (e.g., tracing::info!())
17//!
18//! For entity lifecycle events (actors, meshes), see EntityBatchSink.
19

37/// Helper to convert FieldValue slice to JSON string.
38fn fields_to_json(fields: &[(&str, FieldValue)]) -> String {
39    monarch_telemetry_schema::fields_to_json(fields.iter().map(|(key, value)| {
40        let json_value = match value {
41            FieldValue::Bool(b) => serde_json::Value::Bool(*b),
42            FieldValue::I64(i) => serde_json::Value::Number((*i).into()),
43            FieldValue::U64(u) => serde_json::Value::Number((*u).into()),
44            FieldValue::F64(f) => serde_json::Number::from_f64(*f)
45                .map(serde_json::Value::Number)
46                .unwrap_or(serde_json::Value::Null),
47            FieldValue::Str(s) => serde_json::Value::String(s.clone()),
48            FieldValue::Debug(d) => serde_json::Value::String(d.clone()),
49        };
50        (*key, json_value)
51    }))
52}

54use std::sync::Arc;
55use std::sync::Mutex;
56
57/// Callback function type for flushing RecordBatches.
58/// Takes ownership of the RecordBatch. The callback should handle empty batches
59/// by creating the table with the schema but not appending the empty data.
60pub type FlushCallback = Box<dyn Fn(&str, RecordBatch) + Send>;
61
62/// Inner state of RecordBatchSink.
63struct RecordBatchSinkInner {
64    spans_buffer: SpanBuffer,
65    span_events_buffer: SpanEventBuffer,
66    events_buffer: EventBuffer,
67    batch_size: usize,
68    flush_callback: FlushCallback,
69}
70
71impl RecordBatchSinkInner {
72    fn flush_buffer<B: RecordBatchBuffer>(
73        buffer: &mut B,
74        table_name: &str,
75        callback: &FlushCallback,
76    ) -> anyhow::Result<()> {
77        // Always produce a batch (even if empty) - the callback handles empty batches
78        // by creating the table with the schema but not appending empty data
79        let batch = buffer.drain_to_record_batch()?;
80        callback(table_name, batch);
81        Ok(())
82    }
83
84    fn flush(&mut self) -> anyhow::Result<()> {
85        Self::flush_buffer(&mut self.spans_buffer, SPANS, &self.flush_callback)?;
86        Self::flush_buffer(
87            &mut self.span_events_buffer,
88            SPAN_EVENTS,
89            &self.flush_callback,
90        )?;
91        Self::flush_buffer(&mut self.events_buffer, EVENTS, &self.flush_callback)?;
92        Ok(())
93    }
94
95    fn flush_spans_if_full(&mut self) -> anyhow::Result<()> {
96        if self.spans_buffer.len() >= self.batch_size {
97            Self::flush_buffer(&mut self.spans_buffer, SPANS, &self.flush_callback)?;
98        }
99        Ok(())
100    }
101
102    fn flush_span_events_if_full(&mut self) -> anyhow::Result<()> {
103        if self.span_events_buffer.len() >= self.batch_size {
104            Self::flush_buffer(
105                &mut self.span_events_buffer,
106                SPAN_EVENTS,
107                &self.flush_callback,
108            )?;
109        }
110        Ok(())
111    }
112
113    fn flush_events_if_full(&mut self) -> anyhow::Result<()> {
114        if self.events_buffer.len() >= self.batch_size {
115            Self::flush_buffer(&mut self.events_buffer, EVENTS, &self.flush_callback)?;
116        }
117        Ok(())
118    }
119}

121/// Buffers tracing events and produces Arrow RecordBatches.
122///
123/// Implements TraceEventSink for tracing events (spans, log events).
124///
125/// This type can be cloned to get a handle for flushing from outside the
126/// telemetry system. Clone it before registering with telemetry if you need
127/// to call flush() later.
128///
129/// Produces three tables:
130/// - `spans`: Information about span creation (NewSpan events)
131/// - `span_events`: Enter/exit/close events for spans
132/// - `events`: Tracing events (e.g., tracing::info!())
133///
134/// For entity lifecycle events (actors, meshes), see EntityBatchSink.
135#[derive(Clone)]
136pub struct RecordBatchSink {
137    inner: Arc<Mutex<RecordBatchSinkInner>>,
138}
139
140impl RecordBatchSink {
141    /// Create a new RecordBatchSink with the specified batch size and flush callback.
142    ///
143    /// The callback receives (table_name, record_batch) when a batch is ready.
144    /// The callback should handle empty batches by creating the table with the
145    /// schema but not appending the empty data.
146    ///
147    /// # Arguments
148    /// * `batch_size` - Number of rows to buffer before flushing each table
149    /// * `flush_callback` - Called with (table_name, record_batch) when a batch is ready
150    pub fn new(batch_size: usize, flush_callback: FlushCallback) -> Self {
151        let inner = Arc::new(Mutex::new(RecordBatchSinkInner {
152            spans_buffer: SpanBuffer::default(),
153            span_events_buffer: SpanEventBuffer::default(),
154            events_buffer: EventBuffer::default(),
155            batch_size,
156            flush_callback,
157        }));
158        Self { inner }
159    }
160
161    /// Flush all buffers, emitting batches for all tables.
162    ///
163    /// This always emits batches for all three tables (spans, span_events, events),
164    /// even if they are empty. The callback is expected to handle empty batches
165    /// by creating the table with the correct schema but not appending empty data.
166    pub fn flush(&self) -> anyhow::Result<()> {
167        let mut inner = self
168            .inner
169            .lock()
170            .map_err(|_| anyhow::anyhow!("lock poisoned"))?;
171        inner.flush()
172    }
173}
174
175impl TraceEventSink for RecordBatchSink {
176    fn consume(&mut self, event: &TraceEvent) -> Result<(), anyhow::Error> {
177        let mut inner = self
178            .inner
179            .lock()
180            .map_err(|_| anyhow::anyhow!("lock poisoned"))?;
181
182        match event {
183            TraceEvent::NewSpan {
184                id,
185                name,
186                target,
187                level,
188                fields,
189                timestamp,
190                parent_id,
191                thread_name,
192                file,
193                line,
194            } => {
195                inner.spans_buffer.insert(Span {
196                    id: *id,
197                    name: name.to_string(),
198                    target: target.to_string(),
199                    level: level.to_string(),
200                    fields_json: fields_to_json(fields),
201                    timestamp_us: timestamp_to_micros(timestamp),
202                    parent_id: *parent_id,
203                    thread_name: thread_name.to_string(),
204                    file: file.map(|s| s.to_string()),
205                    line: *line,
206                });
207                inner.flush_spans_if_full()?;
208            }
209            TraceEvent::SpanEnter { id, timestamp, .. } => {
210                inner.span_events_buffer.insert(SpanEvent {
211                    id: *id,
212                    timestamp_us: timestamp_to_micros(timestamp),
213                    event_type: "enter".to_string(),
214                });
215                inner.flush_span_events_if_full()?;
216            }
217            TraceEvent::SpanExit { id, timestamp, .. } => {
218                inner.span_events_buffer.insert(SpanEvent {
219                    id: *id,
220                    timestamp_us: timestamp_to_micros(timestamp),
221                    event_type: "exit".to_string(),
222                });
223                inner.flush_span_events_if_full()?;
224            }
225            TraceEvent::SpanClose { id, timestamp } => {
226                inner.span_events_buffer.insert(SpanEvent {
227                    id: *id,
228                    timestamp_us: timestamp_to_micros(timestamp),
229                    event_type: "close".to_string(),
230                });
231                inner.flush_span_events_if_full()?;
232            }
233            TraceEvent::Event {
234                name,
235                target,
236                level,
237                fields,
238                timestamp,
239                parent_span,
240                thread_id,
241                thread_name,
242                module_path,
243                file,
244                line,
245            } => {
246                inner.events_buffer.insert(Event {
247                    name: name.to_string(),
248                    target: target.to_string(),
249                    level: level.to_string(),
250                    fields_json: fields_to_json(fields),
251                    timestamp_us: timestamp_to_micros(timestamp),
252                    parent_span: *parent_span,
253                    thread_id: thread_id.to_string(),
254                    thread_name: thread_name.to_string(),
255                    module_path: module_path.map(|s| s.to_string()),
256                    file: file.map(|s| s.to_string()),
257                    line: *line,
258                });
259                inner.flush_events_if_full()?;
260            }
261            TraceEvent::Entity(_) => {}
262        }
263        Ok(())
264    }
265
266    fn flush(&mut self) -> Result<(), anyhow::Error> {
267        // No-op: we don't flush on the periodic timer from the telemetry worker.
268        // Instead, flush happens:
269        // 1. Automatically in consume() when buffers reach batch_size
270        // 2. Explicitly via RecordBatchSink::flush() before queries
271        Ok(())
272    }
273
274    fn name(&self) -> &str {
275        "RecordBatchSink"
276    }
