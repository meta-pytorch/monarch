64/// Response message for streaming query results.
65/// Sent directly over Rust PortRef for efficiency.
66/// Completion is signaled by the scan endpoint returning, not via a message.
67#[derive(Debug, Clone, Serialize, Deserialize, Named)]
68pub struct QueryResponse {
69    /// A batch of data in Arrow IPC format.
70    /// Uses Part for zero-copy transfer across the actor system.
71    pub data: Part,
72}

78/// Helper to convert SystemTime to microseconds since Unix epoch.
79pub(crate) fn timestamp_to_micros(timestamp: &std::time::SystemTime) -> i64 {
80    timestamp
81        .duration_since(std::time::UNIX_EPOCH)
82        .unwrap_or_default()
83        .as_micros() as i64
84}

86pub(crate) fn serialize_schema(schema: &SchemaRef) -> anyhow::Result<Vec<u8>> {
87    let batch = RecordBatch::new_empty(schema.clone());
88    let mut buf = Vec::new();
89    let mut writer = StreamWriter::try_new(&mut buf, schema)?;
90    writer.write(&batch)?;
91    writer.finish()?;
92    Ok(buf)
93}

99/// Start Unix-socket ingest for a database scanner.
100#[pyfunction]
101fn _start_socket_ingest(scanner: PyRef<'_, DatabaseScanner>, socket_path: &str) -> PyResult<()> {
102    scanner
103        .start_socket_ingest(std::path::Path::new(socket_path))
104        .map_err(|error| pyo3::exceptions::PyRuntimeError::new_err(error.to_string()))
105}
106
107/// Register trace and entity schemas with a database scanner.
108#[pyfunction]
109fn _register_trace_entity_schemas(scanner: PyRef<'_, DatabaseScanner>) -> PyResult<()> {
110    register_table_schema::<SpanBuffer>(&scanner, SPANS)?;
111    register_table_schema::<SpanEventBuffer>(&scanner, SPAN_EVENTS)?;
112    register_table_schema::<EventBuffer>(&scanner, EVENTS)?;
113    register_table_schema::<ActorBuffer>(&scanner, ACTORS)?;
114    register_table_schema::<MeshBuffer>(&scanner, MESHES)?;
115    register_table_schema::<ActorStatusEventBuffer>(&scanner, ACTOR_STATUS_EVENTS)?;
116    register_table_schema::<SentMessageBuffer>(&scanner, SENT_MESSAGES)?;
117    register_table_schema::<MessageBuffer>(&scanner, MESSAGES)?;
118    register_table_schema::<MessageStatusEventBuffer>(&scanner, MESSAGE_STATUS_EVENTS)?;
119    Ok(())
120}
121
122fn register_table_schema<B>(scanner: &DatabaseScanner, table_name: &str) -> PyResult<()>
123where
124    B: RecordBatchBuffer + Default,
125{
126    scanner
127        .register_table(
128            table_name,
129            B::default()
130                .drain_to_record_batch()
131                .map_err(|error| pyo3::exceptions::PyRuntimeError::new_err(error.to_string()))?
132                .schema(),
133        )
134        .map_err(|error| pyo3::exceptions::PyRuntimeError::new_err(error.to_string()))
135}

137/// Activate the process-global Unix socket telemetry sink.
138#[pyfunction]
139fn _set_unix_socket_sink_path(socket_path: &str) -> PyResult<()> {
140    hyperactor_telemetry::set_unix_socket_sink_path(socket_path)
141        .map_err(|error| pyo3::exceptions::PyRuntimeError::new_err(error.to_string()))
142}
143
144pub fn register_python_bindings(module: &Bound<'_, PyModule>) -> PyResult<()> {
145    module.add_function(wrap_pyfunction!(_start_socket_ingest, module)?)?;
146    module.add_function(wrap_pyfunction!(_register_trace_entity_schemas, module)?)?;
147    module.add_function(wrap_pyfunction!(_set_unix_socket_sink_path, module)?)?;
148    Ok(())
149}
