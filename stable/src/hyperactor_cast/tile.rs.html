9//! Pure tiling geometry for cast routing.
10//!
11//! A [`Tile`] is an affine footprint that can be recursively decomposed by a
12//! [`Tiling`] implementation. The root tile is the affine slice of the view
13//! being routed; child tiles are affine subspaces of their parents.
14//!
15//! A [`Tiling`] first produces structural [`TileNode`]s: children labeled by
16//! their relationship to the parent split. Anchor children preserve the parent
17//! representative and sibling children introduce a distinct representative.
18//! Communication children are then derived by contracting anchor edges and
19//! keeping sibling edges.
20//!
21//! The anchor/sibling vocabulary is shared by all tilers in this module, but
22//! each tiler can use it differently. [`BlockPartitioning`] recursively splits
23//! the first varying dimension; later dimensions are discovered by following
24//! the anchor branch. [`BoundedFanout`] computes a bounded local frontier in
25//! one step, emitting sibling frontier tiles plus a terminal root anchor.
26//!
27//! This [`BlockPartitioning`] example shows anchor contraction in the simple
28//! recursive case:
29//!
30//! ```text
31//! structural decomposition, rendered by tile-root rank:
32//! T0 [ A B
33//!      C D ] Root
34//! |-- T1 [ A B ] Anchor  { dim=0, index=0 }
35//! |   |-- T3 [ A ] Anchor  { dim=1, index=0 }
36//! |   `-- T4 [ B ] Sibling { dim=1, index=1 }
37//! `-- T2 [ C D ] Sibling { dim=0, index=1 }
38//!     |-- T5 [ C ] Anchor  { dim=1, index=0 }
39//!     `-- T6 [ D ] Sibling { dim=1, index=1 }
40//!
41//! communication tree after contracting anchors:
42//! A
43//! |-- B
44//! `-- C
45//!     `-- D
46//! ```
47//!
48//! [`BoundedFanout`] uses the same structural vocabulary with a different
49//! local rule. It partitions the non-root ranks into one frontier slab per
50//! active dimension, splits those slabs into a bounded number of groups, emits
51//! each group as a sibling, and emits the root point as a terminal anchor.
52//!
53//! The fanout parameter controls how finely each tile's frontier is split.
54//! Small fanout produces fewer, larger child tiles and pushes work deeper into
55//! the recursive send tree. Large fanout produces more, smaller child tiles. At
56//! the high end, the immediate frontier becomes BlockPartitioning-like: the
57//! current root sends to the fine-grained frontier pieces that anchor
58//! contraction would expose. At the low end, the tree becomes more
59//! bisection-like: each hop delegates larger subtiles to child roots, keeping
60//! local branching small. These are related behaviors, not identical tiling
61//! strategies. In practice, the goal is the useful middle ground: geometric,
62//! roughly logarithmic-depth routing with explicit control over local
63//! fan-out.

80/// Decomposable affine tile.
81///
82/// A [`Tile`] wraps the affine [`Slice`] carried by one node of a tiling tree.
83/// The tile root is the slice offset, the natural representative for the
84/// tile's geometry.
85///
86/// Root tiles should be constructed with [`Tile::from_view`]. Descendant tiles
87/// should be produced by a [`Tiling`] implementation by selecting affine
88/// subspaces of a parent tile.
89///
90/// ```text
91/// root tile ranks:
92/// 0 1 2 3
93/// 4 5 6 7
94///
95/// bottom-row child tile:
96/// 4 5 6 7
97///
98/// child.root_rank() = 4
99/// child.ranks() = 4 5 6 7
100/// ```
101#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, typeuri::Named)]
102pub(crate) struct Tile(Slice);
103wirevalue::register_type!(Tile);
104
105impl Tile {
106    /// Construct the root tile for a view.
107    ///
108    /// The root tile is exactly the view's affine slice.
109    pub(crate) fn from_view(view: &Region) -> Self {
110        Self(view.slice().clone())
111    }
112
113    fn from_space(space: Slice) -> Self {
114        Self(space)
115    }
116
117    /// Affine footprint covered by this tile.
118    pub(crate) fn space(&self) -> &Slice {
119        &self.0
120    }
121
122    /// Ranks covered by this tile in affine iteration order.
123    pub(crate) fn ranks(&self) -> impl Iterator<Item = usize> + '_ {
124        self.space().iter()
125    }
126
127    /// Number of ranks covered by this tile.
128    pub(crate) fn rank_count(&self) -> usize {
129        self.space().len()
130    }
131
132    /// Natural representative rank for this tile's geometry.
133    pub(crate) fn root_rank(&self) -> usize {
134        self.space().offset()
135    }
136}

138/// A pure [`Tile`] zipped with one concrete item for each tile cell.
139///
140/// [`Tile`] remains the geometry primitive. `MaterializedTile<T>` adds the
141/// items that occupy the tile's ranks, keyed by rank so subtiles can share the
142/// same backing map while narrowing only the tile footprint.
143///
144/// ```text
145/// tile.ranks():
146/// 4 5 6 7
147///
148/// items_by_rank:
149/// 4 -> A4
150/// 5 -> A5
151/// 6 -> A6
152/// 7 -> A7
153///
154/// root_item() = item_at(4) = A4
155/// ```
156#[derive(Debug, Clone, PartialEq)]
157pub(crate) struct MaterializedTile<T> {
158    tile: Tile,
159    items: Arc<ValueMesh<T>>,
160}
161
162impl<T: Serialize + 'static> Serialize for MaterializedTile<T> {
163    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
164    where
165        S: Serializer,
166    {
167        // In memory, child materialized tiles created by `subtile()` share a
168        // root-level ValueMesh so that further subtiles are O(1). On the wire,
169        // send only the active tile's items in `tile.ranks()` order. The
170        // serialized tile carries the ranks, so the receiver can rebuild a
171        // compact ValueMesh without index shifts.
172        let mut state = serializer.serialize_struct("MaterializedTile", 2)?;
173        state.serialize_field("tile", &self.tile)?;
174        state.serialize_field("items", &self.items().collect::<Vec<_>>())?;
175        state.end()
176    }
177}
178
179impl<'de, T: Deserialize<'de> + 'static> Deserialize<'de> for MaterializedTile<T> {
180    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
181    where
182        D: Deserializer<'de>,
183    {
184        #[derive(Deserialize)]
185        struct MaterializedTileWire<T> {
186            tile: Tile,
187            items: Vec<T>,
188        }
189
190        // Wire carries only this tile's items, positional in `tile.ranks()` order
191        // (see `Serialize`); rebuild a `ValueMesh` scoped to the tile, not the
192        // sender's (shared, possibly larger) backing.
193        let wire = MaterializedTileWire::deserialize(deserializer)?;
194
195        let items = Arc::new(
196            ValueMesh::new(
197                Region::new(
198                    (0..wire.tile.space().sizes().len())
199                        .map(|dim| format!("dim{dim}"))
200                        .collect(),
201                    wire.tile.space().clone(),
202                ),
203                wire.items,
204            )
205            .map_err(|err| D::Error::custom(format!("invalid materialized tile items: {err}")))?,
206        );
207        Ok(Self::from_value_mesh_with_tile(wire.tile, items))
208    }
209}

211impl<T: 'static> MaterializedTile<T> {
212    /// Construct a materialized tile from a complete item mesh and explicit
213    /// tile geometry.
214    ///
215    /// `items` may cover a larger region than `tile`, but it must contain every
216    /// base rank covered by `tile`.
217    pub(crate) fn from_value_mesh_with_tile(tile: Tile, items: Arc<ValueMesh<T>>) -> Self {
218        debug_assert!(
219            tile.ranks()
220                .all(|rank| items.get_by_base_rank(rank).is_some()),
221            "tile ranks must be covered by the items mesh",
222        );
223        Self { tile, items }
224    }
225
226    /// Geometry carried by this materialized tile.
227    pub(crate) fn tile(&self) -> &Tile {
228        &self.tile
229    }
230
231    /// Natural representative rank for this tile's geometry.
232    pub(crate) fn root_rank(&self) -> usize {
233        self.tile.root_rank()
234    }
235
236    /// Item stored at `rank`, if `rank` is covered by this tile.
237    pub(crate) fn item_at(&self, rank: usize) -> Option<&T> {
238        if !self.tile.space().contains(rank) {
239            return None;
240        }
241        self.items.get_by_base_rank(rank)
242    }
243
244    /// Item at this tile's natural representative rank.
245    pub(crate) fn root_item(&self) -> Option<&T> {
246        self.item_at(self.root_rank())
247    }
248
249    /// Items for this tile, in `self.tile.ranks()` order.
250    pub(crate) fn items(&self) -> impl Iterator<Item = &T> {
251        self.tile.ranks().map(|rank| {
252            self.item_at(rank)
253                .expect("materialized tile must contain every rank in its tile")
254        })
255    }
256
257    /// Number of ranks covered by this materialized tile.
258    pub(crate) fn rank_count(&self) -> usize {
259        self.tile.rank_count()
260    }
261
262    /// Materialize `tile` as a child of this materialized tile.
263    ///
264    /// `tile` must be covered by this tile. The returned materialized tile uses
265    /// `tile` as its geometry and shares the same rank-addressed item map.
266    ///
267    /// ```text
268    /// parent MaterializedTile:
269    /// 0 -> A0  1 -> A1  2 -> A2  3 -> A3
270    /// 4 -> A4  5 -> A5  6 -> A6  7 -> A7
271    ///
272    /// child Tile input space:
273    /// Slice { offset: 4, sizes: [1, 4], strides: [4, 1] }
274    ///
275    /// subtile(child):
276    /// tile.ranks() = 4 5 6 7
277    /// root_item() = item_at(4) = A4
278    /// ```
279    pub(crate) fn subtile(&self, tile: Tile) -> Self {
280        debug_assert!(
281            tile.ranks()
282                .all(|rank| self.items.get_by_base_rank(rank).is_some())
283        );
284        Self {
285            tile,
286            items: Arc::clone(&self.items),
287        }
288    }
289}

291/// Interface for tile decomposition.
292pub(crate) trait Tiling {
293    /// Return the immediate structural child nodes of `tile`.
294    ///
295    /// Structural children describe geometry. [`TileRelation::Sibling`]
296    /// children introduce distinct roots and become communication children.
297    /// [`TileRelation::Anchor`] children preserve the parent root. They are not
298    /// returned by [`Tiling::children`]; instead, `children` is called on the
299    /// anchor tile and those results are returned in its place.
300    fn child_nodes(&self, tile: &Tile) -> Vec<TileNode>;
301
302    /// Return the immediate communication child tiles of `tile`.
303    ///
304    /// Sibling child tiles are returned directly. Anchor child tiles are not
305    /// returned; instead, `children` is called on the anchor tile and those
306    /// results are returned in its place.
307    ///
308    /// For example, [`BlockPartitioning`] over a `2 x 2` tile produces:
309    ///
310    /// ```text
311    /// structural decomposition, rendered by tile-root rank:
312    /// T0 [ A B
313    ///      C D ] Root
314    /// |-- T1 [ A B ] Anchor  { dim=0, index=0 }
315    /// |   |-- T3 [ A ] Anchor  { dim=1, index=0 }
316    /// |   `-- T4 [ B ] Sibling { dim=1, index=1 }
317    /// `-- T2 [ C D ] Sibling { dim=0, index=1 }
318    ///     |-- T5 [ C ] Anchor  { dim=1, index=0 }
319    ///     `-- T6 [ D ] Sibling { dim=1, index=1 }
320    ///
321    /// children(tile), rendered by tile-root rank:
322    /// |-- T4 [ B ]
323    /// `-- T2 [ C D ]
324    /// ```
325    fn children(&self, tile: &Tile) -> Vec<Tile> {
326        self.child_nodes(tile)
327            .into_iter()
328            .flat_map(|node| match node.relation {
329                TileRelation::Sibling(_) => vec![node.tile],
330                TileRelation::Anchor(_) => self.children(&node.tile),
331            })
332            .collect()
333    }
334}

336/// Dimensions of `space`, as (dim, extent) pairs in dimension order.
337fn dimension_extents(space: &Slice) -> impl Iterator<Item = (usize, usize)> + '_ {
338    space.sizes().iter().copied().enumerate()
339}

341/// Block-partitioning tiler.
342///
343/// `child_nodes` finds the first dimension with extent greater than one and
344/// fixes that dimension to each index. The index `0` child keeps the parent
345/// tile root and is an anchor; every other index shifts the tile root and is a
346/// sibling.
347///
348/// Through [`Tiling::children`], the anchor child is replaced by its own
349/// communication children. That is how later dimensions become communication
350/// children of the original tile.
351#[derive(Debug, Clone, Copy, PartialEq, Eq)]
352pub(crate) struct BlockPartitioning;
353
354/// First dimension whose extent is greater than one.
355///
356/// [`BlockPartitioning`] splits this dimension next; dimensions with extent 1
357/// are already fixed.
358fn first_non_singleton_dim(space: &Slice) -> Option<usize> {
359    dimension_extents(space).find_map(|(dim, extent)| (extent > 1).then_some(dim))
360}
361
362impl Tiling for BlockPartitioning {
363    fn child_nodes(&self, tile: &Tile) -> Vec<TileNode> {
364        let space = tile.space();
365        let Some(dim) = first_non_singleton_dim(space) else {
366            return vec![];
367        };
368
369        let size = space.sizes()[dim];
370
371        let siblings = (1..size).map(|index| TileNode {
372            tile: Tile::from_space(
373                space
374                    .select(dim, index, index + 1, 1)
375                    .expect("fixing a valid dimension should produce a valid child space"),
376            ),
377            relation: TileRelation::Sibling(Split { dim, index }),
378        });
379
380        let anchor = TileNode {
381            tile: Tile::from_space(
382                space
383                    .select(dim, 0, 1, 1)
384                    .expect("fixing the anchor dimension should produce a valid child space"),
385            ),
386            relation: TileRelation::Anchor(Split { dim, index: 0 }),
387        };
388
389        siblings.chain(std::iter::once(anchor)).collect()
390    }
391}

393/// Rectangular tiler with bounded communication fan-out.
394///
395/// Peels one rectangular frontier slab per active dimension, left-to-right.
396/// For a 2 x 4 tile, the row dim peels [E F G H]; the column dim is then
397/// constrained to row 0 and peels [B C D]; the corner [A] is the terminal
398/// anchor. [`Tiling::children`] does not return that anchor because calling
399/// `children` on the singleton root tile yields no communication children.
400/// The requested cap is honored when geometrically feasible; when below the
401/// active-dimension count, [`effective_fanout`] raises it to the geometric
402/// minimum.
403///
404/// The vocabulary used by this tiler is:
405///
406/// - an *active dimension* is a dimension whose extent is greater than 1.
407///   Only active dimensions can contribute frontier slabs.
408/// - a *slab* is the rectangular frontier an active dimension contributes
409///   (one slab per active dim).
410/// - [`bounded_intervals`] partitions a slab into *groups* (left-heavy
411///   chunks).
412/// - each group becomes one structural sibling tile via [`frontier_tile`];
413///   [`Tiling::children`] returns sibling tiles as communication children.
414#[derive(Debug, Clone, Copy, PartialEq, Eq)]
415pub(crate) struct BoundedFanout {
416    /// Requested cap on the number of communication child tiles per tile.
417    ///
418    /// [`BoundedFanout`] raises this to [`minimum_fanout`] when a tile has
419    /// more active dimensions than the requested cap.
420    pub fanout: NonZeroUsize,
421}
422
423impl Tiling for BoundedFanout {
424    fn child_nodes(&self, tile: &Tile) -> Vec<TileNode> {
425        // Active dimensions are the dimensions that can contribute frontier
426        // slabs. Each entry is (dim, extent), in dimension order. If none
427        // remain, the tile is a singleton and has no children.
428        let active_dims: Vec<(usize, usize)> = active_dimensions(tile.space());
429        if active_dims.is_empty() {
430            return vec![];
431        }
432
433        // One active dimension means one frontier slab, and each slab starts
434        // with one group. If the effective fanout is larger than the number of
435        // slabs, the surplus groups are assigned left-to-right.
436        //
437        // `groups_per_slab` is parallel to `active_dims`: groups_per_slab[i]
438        // is the number of groups to cut from the slab for active_dims[i].
439        //
440        // For a 3 x 4 tile with active dims [row, col]:
441        //   fanout 2 -> row: 1 group, col: 1 group
442        //   fanout 3 -> row: 2 groups, col: 1 group
443        //   fanout 5 -> row: 2 groups, col: 3 groups
444        let groups_per_slab =
445            allocate_groups_per_slab(effective_fanout(tile, self.fanout.get()), &active_dims);
446
447        let mut structural_children: Vec<TileNode> = Vec::new();
448        for (&(slab_dim, dim_extent), &group_count) in active_dims.iter().zip(&groups_per_slab) {
449            // The outer loop walks slabs. `slab_dim` is the active dimension
450            // whose slab we are cutting: in a 3 x 4 tile, slab_dim 0 is the
451            // row slab and slab_dim 1 is the column slab.
452            for (group_begin, group_end) in bounded_intervals(group_count, dim_extent) {
453                // The inner loop walks this slab's bounded intervals. Each
454                // interval is one away-from-root group, and each group becomes
455                // one structural sibling tile.
456                structural_children.push(TileNode {
457                    tile: frontier_tile(tile, &active_dims, slab_dim, group_begin, group_end),
458                    relation: TileRelation::Sibling(Split {
459                        dim: slab_dim,
460                        index: group_begin,
461                    }),
462                });
463            }
464        }
465
466        // The sibling tiles above cover every rank away from the root. Add the
467        // root point as a terminal anchor so `child_nodes` is a structural
468        // cover of the whole tile. `Tiling::children` will not return this
469        // anchor because the root point has no communication children.
470        //
471        // `TileRelation::Anchor` carries a `Split`, but this root point is the
472        // intersection of every active dimension's index-0 slice, not a group
473        // from one slab. Use the last active dimension as a stable structural
474        // label.
475        let (anchor_label_dim, _) = *active_dims.last().expect("active_dims is non-empty");
476        structural_children.push(TileNode {
477            tile: root_point_tile(tile, &active_dims),
478            relation: TileRelation::Anchor(Split {
479                dim: anchor_label_dim,
480                index: 0,
481            }),
482        });
483        structural_children
484    }
485}

487/// Active dimensions of `space`, as (dim, extent) pairs in dimension order.
488///
489/// Dimensions with extent 1 cannot contribute a frontier slab.
490fn active_dimensions(space: &Slice) -> Vec<(usize, usize)> {
491    dimension_extents(space)
492        .filter_map(|(dim, extent)| (extent > 1).then_some((dim, extent)))
493        .collect()
494}
495
496/// Minimum fan-out needed to give each active dimension one group.
497pub(crate) fn minimum_fanout(tile: &Tile) -> usize {
498    active_dimensions(tile.space()).len()
499}
500
501/// BoundedFanout policy: raise a requested cap to the tile's minimum fan-out.
502fn effective_fanout(tile: &Tile, requested: usize) -> usize {
503    requested.max(minimum_fanout(tile))
504}

506/// Allocate frontier groups to slabs in dimension order.
507///
508/// `available_groups` is the effective fan-out for this tile: the requested
509/// fan-out raised to at least one group per active dimension.
510///
511/// `active_dims[i]` identifies one frontier slab, and the returned
512/// `groups_per_slab[i]` says how many groups to cut from that slab. Each slab
513/// starts with one group because every active dimension must contribute at
514/// least one child. Any remaining budget is assigned left-to-right, capped by
515/// the number of away-from-root positions in that dimension.
516///
517/// Allocation is left-heavy across dimensions: earlier slabs receive surplus
518/// groups up to capacity before later slabs receive any.
519fn allocate_groups_per_slab(available_groups: usize, active_dims: &[(usize, usize)]) -> Vec<usize> {
520    // Base allocation: one group per slab.
521    let mut groups_per_slab: Vec<usize> = vec![1; active_dims.len()];
522    // Surplus groups left after the one-group-per-slab baseline.
523    let mut remaining_groups = available_groups.saturating_sub(active_dims.len());
524
525    for (slab_index, &(_dim, dim_extent)) in active_dims.iter().enumerate() {
526        // A dimension of extent n has n - 1 away-from-root positions, so its
527        // slab cannot be split into more than n - 1 groups.
528        let slab_capacity = dim_extent - 1;
529        // Give this slab as many surplus groups as possible, but no more than
530        // the gap between its capacity and how many groups it already has.
531        let additional_groups = remaining_groups.min(slab_capacity - groups_per_slab[slab_index]);
532
533        // Fill this slab by the amount just allocated.
534        groups_per_slab[slab_index] += additional_groups;
535        // Consume the same amount from the shared surplus budget.
536        remaining_groups -= additional_groups;
537    }
538
539    groups_per_slab
540}

542/// Split one slab into left-heavy index intervals.
543///
544/// `extent` is the size of the active dimension for this slab. The slab covers
545/// indices `1..extent` in that dimension; index 0 stays with the
546/// root/anchor path. Each returned `(begin, end)` is one group interval that
547/// the caller turns into a structural sibling tile with [`frontier_tile`].
548///
549/// For example, with `extent = 8`, the away-from-root indices are `1..8`.
550/// Splitting into 3 groups yields `[1, 4)`, `[4, 6)`, and `[6, 8)`.
551///
552/// If the intervals do not divide evenly, earlier groups receive one extra
553/// index before later groups. Returns `[]` when no groups are requested or
554/// the dimension has no away-from-root coordinates.
555fn bounded_intervals(group_count: usize, extent: usize) -> Vec<(usize, usize)> {
556    if group_count == 0 || extent <= 1 {
557        return vec![];
558    }
559    let remaining = extent - 1;
560    let groups = group_count.min(remaining);
561    let base = remaining / groups;
562    let extra = remaining % groups;
563    let mut out = Vec::with_capacity(groups);
564    let mut begin = 1;
565    for i in 0..groups {
566        let size = base + if i < extra { 1 } else { 0 };
567        let end = begin + size;
568        out.push((begin, end));
569        begin = end;
570    }
571    out
572}

574/// Prepare `base` for cutting the slab of `dim`.
575///
576/// Every earlier active dimension is fixed to index 0. For a 2 x 4 tile, this
577/// returns the whole tile for the row slab, and the top row [A B C D] for the
578/// column slab. [`frontier_tile`] then selects the away-from-root interval in
579/// `dim`, such as columns 1..4 to produce [B C D].
580fn anchor_prefix(base: &Tile, active_dims: &[(usize, usize)], dim: usize) -> Tile {
581    let mut space = base.space().clone();
582    for &(earlier_dim, _) in active_dims.iter().take_while(|&&(d, _)| d != dim) {
583        // Stay in the index-0 slice of each earlier slab dimension so this
584        // slab does not overlap ranks already covered by earlier slabs.
585        // Index 0 is tile-local: it is this tile's anchor slice for
586        // `earlier_dim`, not a global row or column. The select call keeps
587        // only that index-0 slice.
588        space = space
589            .select(earlier_dim, 0, 1, 1)
590            .expect("anchor_prefix: active dim must admit index 0");
591    }
592    Tile::from_space(space)
593}
594
595/// Build the child tile for one group in `slab_dim`'s slab.
596///
597/// [`anchor_prefix`] first fixes earlier active dimensions to their tile-local
598/// index-0 slices. Then this function selects the group's index interval in
599/// `slab_dim`. For a 3 x 4 tile, the column slab starts from row 0 and then
600/// selects columns 1..4, yielding [B C D].
601fn frontier_tile(
602    base: &Tile,
603    active_dims: &[(usize, usize)],
604    slab_dim: usize,
605    group_begin: usize,
606    group_end: usize,
607) -> Tile {
608    let anchored = anchor_prefix(base, active_dims, slab_dim);
609    // Keep this group's half-open interval in `slab_dim`, with unit stride.
610    let space = anchored
611        .space()
612        .select(slab_dim, group_begin, group_end, 1)
613        .expect("frontier_tile: bounded_intervals must produce valid intervals");
614    Tile::from_space(space)
615}

617/// Build the root-point tile left after all frontier slabs are removed.
618///
619/// This fixes every active dimension to this tile's index 0, leaving the
620/// singleton tile at the natural representative rank. `BoundedFanout` emits it
621/// as a terminal anchor so `child_nodes` covers the whole parent tile; because
622/// it is a singleton, [`Tiling::children`] returns no communication children
623/// for it.
624fn root_point_tile(base: &Tile, active_dims: &[(usize, usize)]) -> Tile {
625    let mut space = base.space().clone();
626    for &(active_dim, _) in active_dims {
627        // Keep only this tile's index-0 slice in each active dimension.
628        space = space
629            .select(active_dim, 0, 1, 1)
630            .expect("root_point_tile: active dim must admit index 0");
631    }
632    Tile::from_space(space)
633}

635/// Reference bisection tiler.
636///
637/// Splits the first non-singleton dimension into a lower anchor half
638/// (`0..mid`) and an upper sibling half (`mid..n`), where `mid = n / 2`.
639/// The structural split is binary, but communication children are computed
640/// after anchor contraction: sibling descendants inside the lower half are
641/// promoted to the current root. This yields a deeper send tree with less root
642/// fan-out than block partitioning on wide dimensions.
643///
644/// This tiler is included for model completeness, tests, and comparison with
645/// the reference implementation. [`BoundedFanout`] is the production-oriented
646/// fan-out control policy.
647#[derive(Debug, Clone, Copy, PartialEq, Eq)]
648pub(crate) struct Bisection;
649
650impl Tiling for Bisection {
651    fn child_nodes(&self, tile: &Tile) -> Vec<TileNode> {
652        let space = tile.space();
653        let Some(dim) = first_non_singleton_dim(space) else {
654            return vec![];
655        };
656        let n = space.sizes()[dim];
657        let lower = n / 2;
658
659        let sibling = TileNode {
660            tile: Tile::from_space(
661                space
662                    .select(dim, lower, n, 1)
663                    .expect("Bisection: upper half must be a valid slice"),
664            ),
665            relation: TileRelation::Sibling(Split { dim, index: lower }),
666        };
667        let anchor = TileNode {
668            tile: Tile::from_space(
669                space
670                    .select(dim, 0, lower, 1)
671                    .expect("Bisection: lower half must be a valid slice"),
672            ),
673            relation: TileRelation::Anchor(Split { dim, index: 0 }),
674        };
675        vec![sibling, anchor]
676    }
677}

679/// Serializable selector for a concrete tiling algorithm.
680///
681/// This keeps the tiling family open internally via [`Tiling`], while giving
682/// cross-process setup messages a small data value that can be serialized and
683/// reconstructed by remote actors.
684#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
685pub enum TilingPolicy {
686    /// Recursively split the first varying dimension and contract anchor edges.
687    BlockPartitioning,
688    /// Cap communication fan-out per node. When `fanout` is below the
689    /// active-dimension minimum, the minimum is used.
690    BoundedFanout { fanout: NonZeroUsize },
691    /// Recursively bisect the first non-singleton dimension at each step.
692    Bisection,
693}
694
695impl TilingPolicy {
696    pub(crate) fn children(&self, tile: &Tile) -> Vec<Tile> {
697        match self {
698            Self::BlockPartitioning => {
699                let tiling = BlockPartitioning;
700                tiling.children(tile)
701            }
702            Self::BoundedFanout { fanout } => {
703                let tiling = BoundedFanout { fanout: *fanout };
704                tiling.children(tile)
705            }
706            Self::Bisection => {
707                let tiling = Bisection;
708                tiling.children(tile)
709            }
710        }
711    }
712}

714/// Coordinates one structural decomposition step.
715#[derive(Debug, Clone, Copy, PartialEq, Eq)]
716pub(crate) struct Split {
717    dim: usize,
718    index: usize,
719}
720
721/// Relationship between a structural child tile and its parent.
722#[derive(Debug, Clone, Copy, PartialEq, Eq)]
723pub(crate) enum TileRelation {
724    /// Child whose natural representative rank is the same as the parent's.
725    ///
726    /// Anchors are structural only. [`Tiling::children`] does not return them;
727    /// it calls `children` on the anchor tile and returns those results
728    /// instead.
729    Anchor(Split),
730    /// Child whose natural representative rank differs from the parent's.
731    ///
732    /// Siblings become direct communication children of the parent.
733    Sibling(Split),
734}
735
736/// Structural child emitted by a [`Tiling`].
737///
738/// The `tile` is the child geometry. The `relation` records whether that child
739/// is a communication sibling or an anchor that [`Tiling::children`] should
740/// replace with the anchor tile's own communication children.
741#[derive(Debug, Clone, PartialEq)]
742pub(crate) struct TileNode {
743    tile: Tile,
744    relation: TileRelation,
745}
