38/// Parts produced by [`serve`]: the bound address, the per-kind halves,
39/// the accept-loop join handle, and the shared cancellation token.
40pub(crate) struct MuxServeParts<M, In, Out>
41where
42    M: RemoteMessage,
43    In: RemoteMessage,
44    Out: RemoteMessage,
45{
46    pub addr: ChannelAddr,
47    pub simplex: ChannelRx<M>,
48    pub duplex: DuplexServer<In, Out>,
49    pub join_handle: JoinHandle<Result<(), ServerError>>,
50    pub cancel: CancellationToken,
51}

53/// Start a muxed server on `addr`. The accept-loop's [`JoinHandle`] is
54/// returned alongside the per-kind halves and the shared cancellation
55/// token so callers can build a true awaitable shutdown handle on top
56/// — see [`MuxServer`](super::super::MuxServer) for the public
57/// wrapper.
58pub(crate) fn serve<M, In, Out>(
59    addr: ChannelAddr,
60    prebound_listener: Option<std::net::TcpListener>,
61) -> Result<MuxServeParts<M, In, Out>, ServerError>
62where
63    M: RemoteMessage,
64    In: RemoteMessage,
65    Out: RemoteMessage,
66{
67    let (listener, channel_addr) = super::listen_with_prebound(addr, prebound_listener)?;
68    // None: the muxed listener accepts both kinds and dispatches by
69    // `link_init.kind` after the header is read.
70    let prepare = super::preparer_for(channel_addr.clone(), None);
71    serve_with_prepare(listener, channel_addr, prepare)
72}

74/// Shared implementation for the muxed listener. Wires up the
75/// per-kind state, the dispatch closure, the accept loop, and the
76/// drain-aware coordinator that exposes `drained` to the per-half
77/// handles. Production [`serve`] passes a [`super::preparer_for`]
78/// closure; tests pass an inline prepare closure that boxes
79/// in-memory streams (see [`serve_with_listener`]).
80fn serve_with_prepare<M, In, Out, L, F, Fut>(
81    mut listener: L,
82    channel_addr: ChannelAddr,
83    prepare: F,
84) -> Result<MuxServeParts<M, In, Out>, ServerError>
85where
86    M: RemoteMessage,
87    In: RemoteMessage,
88    Out: RemoteMessage,
89    L: super::Listener + 'static,
90    F: Fn(L::Stream, ChannelAddr) -> Fut + Clone + Send + 'static,
91    Fut: std::future::Future<Output = Result<(super::LinkInit, Box<dyn Stream>), anyhow::Error>>
92        + Send
93        + 'static,
94{
95    let (simplex_tx, simplex_rx) = mpsc::channel::<M>(1024);
96    let (duplex_accept_tx, duplex_accept_rx) = mpsc::channel(16);
97
98    let simplex_sessions: Arc<DashMap<SessionId, mpsc::UnboundedSender<Box<dyn Stream>>>> =
99        Arc::new(DashMap::new());
100    let simplex_stream_state: Arc<
101        std::sync::Mutex<HashMap<SessionId, Arc<StreamState<Box<dyn Stream>>>>>,
102    > = Arc::new(std::sync::Mutex::new(HashMap::new()));
103    let duplex_sessions: Arc<DashMap<SessionId, mpsc::UnboundedSender<Box<dyn Stream>>>> =
104        Arc::new(DashMap::new());
105
106    // Accept-loop cancel. Cloned into both halves' `ServerHandle`s
107    // and returned to the caller so dropping any of the parts (or the
108    // returned token) tears down the shared listener.
109    let cancel_token = CancellationToken::new();
110    let child_token = cancel_token.child_token();
111
112    // Per-session cancellation propagated into dispatched sessions.
113    // These are children of `cancel_token` so cancelling the parent
114    // immediately propagates to in-flight sessions; without that, the
115    // accept loop's `connections.join_next()` drain would block on
116    // session tasks whose cancel tokens never fire, and shutdown would
117    // hang. Mirrors the simplex (`server::serve`) and duplex
118    // (`duplex::serve`) shapes, which also derive the dispatch cancel
119    // from the listener's parent token.
120    let simplex_child_cancel = cancel_token.child_token();
121    let duplex_child_cancel = cancel_token.child_token();
122
123    let dispatch = {
124        let simplex_tx = simplex_tx.clone();
125        let simplex_sessions = Arc::clone(&simplex_sessions);
126        let simplex_stream_state = Arc::clone(&simplex_stream_state);
127        let duplex_sessions = Arc::clone(&duplex_sessions);
128        let duplex_accept_tx = duplex_accept_tx.clone();
129        let simplex_cancel = simplex_child_cancel.clone();
130        let duplex_cancel = duplex_child_cancel.clone();
131        let dest = channel_addr.clone();
132        move |link_init: super::LinkInit, stream: Box<dyn Stream>| {
133            let simplex_tx = simplex_tx.clone();
134            let simplex_sessions = Arc::clone(&simplex_sessions);
135            let simplex_stream_state = Arc::clone(&simplex_stream_state);
136            let duplex_sessions = Arc::clone(&duplex_sessions);
137            let duplex_accept_tx = duplex_accept_tx.clone();
138            let s_cancel = simplex_cancel.child_token();
139            let d_cancel = duplex_cancel.child_token();
140            let dest = dest.clone();
141            async move {
142                match link_init.kind {
143                    ProtocolKind::Simplex => {
144                        let streams = resolve_stream(&simplex_stream_state, &link_init);
145                        dispatch_stream(
146                            link_init.session_id,
147                            streams,
148                            stream,
149                            &simplex_sessions,
150                            dest,
151                            simplex_tx,
152                            s_cancel,
153                        )
154                        .await;
155                    }
156                    ProtocolKind::Duplex => {
157                        dispatch_duplex_stream::<In, Out>(
158                            link_init.session_id,
159                            stream,
160                            duplex_sessions,
161                            dest,
162                            &duplex_accept_tx,
163                            d_cancel,
164                        )
165                        .await;
166                    }
167                }
168            }
169        }
170    };
171
172    let accept_ca = channel_addr.clone();
173    let cancel_after = cancel_token.clone();
174    // Real accept-loop task. Wrapped in a coordinator below that
175    // surfaces the listener's drain as a watch signal so the per-half
176    // `ServerHandle`s can honor `Rx::join`'s "all pending acks
177    // flushed" contract.
178    let inner_accept = tokio::spawn(async move {
179        let result = accept_loop(&mut listener, &accept_ca, &child_token, prepare, dispatch).await;
180        // If `accept_loop` returned for a reason other than parent
181        // cancellation (e.g., listener error), cancelling here cascades
182        // to the per-session children and tears any survivors down.
183        cancel_after.cancel();
184        result
185    });
186
187    // `drained_tx` flips to `true` once the accept loop has joined
188    // every per-session task — i.e., every dispatch's terminal cleanup
189    // (final ack flush + `Closed` emit) has finished. Per-half
190    // `ServerHandle::join` waits on this signal so `ChannelRx::join` and
191    // `DuplexServer::join` actually observe the drain rather than
192    // returning on bare cancellation.
193    let (drained_tx, drained_rx) = watch::channel(false);
194    let coord_addr = channel_addr.clone();
195    let join_handle = tokio::spawn(async move {
196        let result = match inner_accept.await {
197            Ok(r) => r,
198            Err(e) => Err(ServerError::Internal(coord_addr, anyhow::anyhow!(e))),
199        };
200        let _ = drained_tx.send(true);
201        result
202    });
203
204    // Each half gets a drain-aware `ServerHandle` whose `stop` cancels
205    // the shared listener and whose `await` resolves only after the
206    // accept loop has fully drained. Drop on either half also cancels
207    // (via [`ChannelRx::Drop`] / [`DuplexServer::Drop`]).
208    let simplex_handle = drain_aware_handle(
209        cancel_token.clone(),
210        drained_rx.clone(),
211        channel_addr.clone(),
212    );
213    let duplex_handle = drain_aware_handle(cancel_token.clone(), drained_rx, channel_addr.clone());
214
215    let net_rx = ChannelRx {
216        receiver: simplex_rx,
217        dest: channel_addr.clone(),
218        server: simplex_handle,
219    };
220    let duplex_server =
221        DuplexServer::from_parts(duplex_accept_rx, duplex_handle, channel_addr.clone());
222
223    Ok(MuxServeParts {
224        addr: channel_addr,
225        simplex: net_rx,
226        duplex: duplex_server,
227        join_handle,
228        cancel: cancel_token,
229    })
230}

232/// Build a [`ServerHandle`] for a muxed half. `stop` cancels the
233/// shared listener; `await` resolves only after both the cancel fires
234/// and the accept-loop coordinator has signaled `drained` — i.e.,
235/// every dispatch's terminal cleanup (final ack flush + `Closed` emit)
236/// has finished. This honors the [`Rx::join`](super::Rx::join)
237/// contract for [`ChannelRx`] and the equivalent contract for
238/// [`DuplexServer::join`](DuplexServer::join), neither of which would
239/// be satisfied by a bare cancel-only handle.
240fn drain_aware_handle(
241    cancel: CancellationToken,
242    drained: watch::Receiver<bool>,
243    addr: ChannelAddr,
244) -> ServerHandle {
245    let cancel_wait = cancel.clone();
246    let mut drained = drained;
247    let join = tokio::spawn(async move {
248        cancel_wait.cancelled().await;
249        let _ = drained.wait_for(|d| *d).await;
250        Ok::<(), ServerError>(())
251    });
252    ServerHandle::new(join, cancel, addr)
253}

255/// Test-only variant of [`serve`] that accepts an arbitrary `Listener`.
256/// Mirrors [`server::serve_with_listener`](super::server::serve_with_listener)
257/// and [`duplex::serve_with_listener`](super::duplex::serve_with_listener):
258/// used by mock-link tests that drive the mux through in-memory
259/// `DuplexStream`s instead of going through `net::listen_with_prebound`.
260///
261/// The returned [`MuxServeParts`] expose the same shape as production
262/// `serve`, including drain-aware per-half handles and the coordinator's
263/// drained-watch signal — so wire-level tests can exercise the
264/// `Rx::join` ack-flush contract and the shutdown ordering.
265#[cfg(test)]
266pub(super) fn serve_with_listener<M, In, Out, L>(
267    listener: L,
268    channel_addr: ChannelAddr,
269) -> Result<MuxServeParts<M, In, Out>, ServerError>
270where
271    M: RemoteMessage,
272    In: RemoteMessage,
273    Out: RemoteMessage,
274    L: super::Listener + 'static,
275    L::Stream: Unpin + std::fmt::Debug + 'static,
276{
277    // Box the test stream and read the LinkInit; mirrors production
278    // `preparer_for(_, None)` but without the TLS branches that
279    // `preparer_for` carries for real transports.
280    let prepare = |stream: L::Stream, source: ChannelAddr| async move {
281        let mut boxed: Box<dyn Stream> = Box::new(stream);
282        let link_init = super::read_link_init(&mut boxed)
283            .await
284            .map_err(|e| anyhow::anyhow!("LinkInit read failed from {}: {}", source, e))?;
285        Ok((link_init, boxed))
286    };
287    serve_with_prepare(listener, channel_addr, prepare)
288}

448    /// blocked in `tokio::select!` on the session cancel), calling

450    /// per-session children so the accept loop's `connections`
451    /// drain completes and the listener task exits promptly.

455    /// are themselves waiting on tokens that never fire — an unbounded

461                .unwrap();

465        // through each so the dispatch tasks are spawned and parked on
466        // their session cancels.
467        let simplex_tx = channel::dial::<u64>(addr.clone()).unwrap();
468        simplex_tx.post(1);

470        let duplex_tx = duplex_client.tx();

472        duplex_tx.post(2);

477            .expect("simplex recv failed");

580
581        async fn wait_for_stop(mut stopped_rx: tokio::sync::watch::Receiver<bool>) {
582            let ok = stopped_rx.wait_for(|stopped| *stopped).await.is_ok();
583            if !ok {
584                std::future::pending::<()>().await;
585            }
586        }
587
588        fn passthrough_mailbox_handle(mut rx: channel::ChannelRx<u64>) -> MailboxServerHandle {
589            let (stopped_tx, mut stopped_rx) = tokio::sync::watch::channel(false);
590            let join_handle = tokio::spawn(async move {
591                loop {
592                    tokio::select! {
593                        result = rx.recv() => {
594                            if result.is_err() { break; }
595                        }
596                        _ = stopped_rx.changed() => {
597                            if *stopped_rx.borrow() { break; }
598                        }
599                    }
600                }
601                Ok::<(), MailboxServerError>(())
602            });
603            MailboxServerHandle::from_parts(join_handle, stopped_tx)
604        }
605
606        let frontend =
607            channel::serve_mux::<u64, u64, u64>(ChannelAddr::any(ChannelTransport::Unix), None)
608                .unwrap();
609        let addr = frontend.addr().clone();
610
611        let mut client = channel::duplex::dial::<u64, u64>(addr).unwrap();
612        let client_tx = client.tx();
613        let mut client_rx = client.take_rx().unwrap();
614
615        let (accepted_tx, accepted_rx) = tokio::sync::oneshot::channel();
616        let (delivered_tx, delivered_rx) = tokio::sync::oneshot::channel();
617
618        let mut accepted_tx = Some(accepted_tx);
619        let mut delivered_tx = Some(delivered_tx);
620        let handle = frontend.serve(
621            passthrough_mailbox_handle,
622            move |mut duplex_server, stop_rx| async move {
623                let (mut server_rx, _server_tx) = duplex_server.accept().await.unwrap();
624                accepted_tx.take().unwrap().send(()).unwrap();
625
626                // Drain inbound messages until stop fires; this proves
627                // the recv-side path was active up to the shutdown
628                // point so any Closed frame is the result of the
629                // shutdown drain, not a starved session.
630                tokio::select! {
631                    _ = wait_for_stop(stop_rx) => {}
632                    msg = server_rx.recv() => {
633                        let _ = msg;
634                        delivered_tx.take().unwrap().send(()).unwrap();
635                    }
636                }
637
638                duplex_server.stop("test mux clean close");
639                duplex_server.join().await;
640            },
641        );
642
643        accepted_rx.await.unwrap();
644
645        // Drive at least one message through so the session is live.
646        client_tx.post(7);
647        tokio::time::timeout(Duration::from_secs(5), delivered_rx)
648            .await
649            .expect("server did not see inbound message before stop")
650            .unwrap();
651
652        handle.stop("test mux clean close");
653        tokio::time::timeout(Duration::from_secs(5), handle)
654            .await
655            .expect("mux handle did not stop")
656            .expect("mux task panicked")
657            .expect("mux task failed");
658
659        // After the mux fully stops, the duplex peer's recv side must
660        // observe a terminal close (`ChannelError::Closed`) rather
661        // than hang. This is the application-visible consequence of
662        // the dispatch's terminal `Closed` frame being delivered
663        // during shutdown drain.
664        let result = tokio::time::timeout(Duration::from_secs(5), client_rx.recv()).await;
665        assert!(
666            matches!(result, Ok(Err(_))),
667            "client_rx should observe a terminal close after mux shutdown, got {:?}",
668            result,
669        );
670    }
