9//! Remote supervision actors.
10//!
11//! [`Supervisor`] is the parent-side proxy. It supervises a [`Worker`], starts
12//! the configured liveness actor, forwards parent stop requests to the worker, and
13//! re-raises worker-reported supervision events through Hyperactor's ordinary
14//! `UnhandledSupervisionEvent` path.
15//!
16//! [`Worker`] is the worker-side container. It starts empty, accepts one child
17//! actor, accepts one active remote supervision session, starts the worker-side
18//! liveness actor, and reports child or liveness lifecycle events to the supervisor
19//! session.

80enum SupervisorBootstrap {
81    ExistingWorker {
82        worker: ActorRef<WorkerLike>,
83    },
84    Callback {
85        fallback_actor: ActorAddr,
86        callback: Option<BootstrapCallback>,
87    },
88}

90impl std::fmt::Debug for SupervisorBootstrap {
91    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
92        match self {
93            Self::ExistingWorker { worker } => f
94                .debug_struct("ExistingWorker")
95                .field("worker", worker)
96                .finish(),
97            Self::Callback { fallback_actor, .. } => f
98                .debug_struct("Callback")
99                .field("fallback_actor", fallback_actor)
100                .finish_non_exhaustive(),
101        }
102    }
103}
104
105impl SupervisorBootstrap {
106    fn fallback_actor(&self) -> ActorAddr {
107        match self {
108            Self::ExistingWorker { worker } => worker.actor_addr().clone(),
109            Self::Callback { fallback_actor, .. } => fallback_actor.clone(),
110        }
111    }
112
113    fn worker_endpoint(&self) -> Option<&ActorRef<WorkerLike>> {
114        match self {
115            Self::ExistingWorker { worker } => Some(worker),
116            Self::Callback { .. } => None,
117        }
118    }
119
120    fn start(&mut self, cx: &Instance<Supervisor>, supervise: Supervise) -> anyhow::Result<()> {
121        match self {
122            Self::ExistingWorker { worker } => {
123                (&*worker).post(cx, supervise);
124            }
125            Self::Callback { callback, .. } => {
126                callback
127                    .take()
128                    .expect("supervisor bootstrap callback started more than once")(
129                    cx, supervise
130                )?;
131            }
132        }
133        Ok(())
134    }
135}

137/// Parent-side proxy for one remote supervision relationship.
138#[derive(Debug)]
139pub struct Supervisor {
140    bootstrap: SupervisorBootstrap,
141    liveness: Option<KeepaliveLink>,
142    options: SupervisionOptions,
143    session_id: Uid,
144    liveness_handle: Option<ActorHandle<KeepaliveSupervisor>>,
145    session: Option<SupervisorSession>,
146    pending_stop: Option<PendingStop>,
147    /// If set, a bare readiness signal is posted here once the worker reports
148    /// [`SupervisorEvent::Linked`] — i.e., once the remote child has become
149    /// reachable. The child's address is returned synchronously by the spawn
150    /// call, so this carries no payload. Fires at most once, hence a
151    /// [`OncePortHandle`]; the supervisor and the caller that opened the port
152    /// are always on the same proc.
153    ready: Option<OncePortHandle<()>>,
154}
155
156impl Supervisor {
157    /// Create a supervisor proxy for `worker`.
158    pub fn new(
159        worker: ActorRef<WorkerLike>,
160        liveness: KeepaliveLink,
161        options: SupervisionOptions,
162    ) -> Self {
163        Self::new_uid(worker, liveness, options, hyperactor::Uid::anonymous())
164    }
165
166    /// Create a supervisor proxy with an explicit session id.
167    pub fn new_uid(
168        worker: ActorRef<WorkerLike>,
169        liveness: KeepaliveLink,
170        options: SupervisionOptions,
171        session_id: hyperactor::Uid,
172    ) -> Self {
173        Self::new_with_bootstrap(
174            SupervisorBootstrap::ExistingWorker { worker },
175            liveness,
176            options,
177            session_id,
178            None,
179        )
180    }
181
182    /// Create a supervisor proxy whose supervise request is delivered by `bootstrap`.
183    ///
184    /// During supervisor initialization, this starts the supervisor-side
185    /// liveness actor, stores its handle, builds the worker-side [`Supervise`]
186    /// request, and calls `bootstrap` exactly once. The callback is responsible for
187    /// delivering that `Supervise` to the worker that should be supervised, usually
188    /// by embedding it in another spawn or routing message. If the callback
189    /// returns an error, supervisor initialization fails.
190    ///
191    /// `bootstrap` must not retain the `Supervise` for later use. Until the worker
192    /// accepts the supervise request and reports [`SupervisorEvent::Linked`], this
193    /// supervisor has no worker control endpoint; stop requests are therefore
194    /// held pending. `fallback_actor` is used only to synthesize a supervision
195    /// event if liveness fails before the worker reports the supervised child.
196    /// When `ready` is set, a bare readiness signal is posted to it once the
197    /// worker reports `Linked` — the earliest point at which messages to the
198    /// child are guaranteed to route — so callers can wait for reachability
199    /// instead of guessing with a delay.
200    pub(crate) fn bootstrap_uid<F>(
201        liveness: KeepaliveLink,
202        options: SupervisionOptions,
203        session_id: Uid,
204        fallback_actor: ActorAddr,
205        ready: Option<OncePortHandle<()>>,
206        bootstrap: F,
207    ) -> Self
208    where
209        F: FnOnce(&Instance<Supervisor>, Supervise) -> anyhow::Result<()> + Send + 'static,
210    {
211        Self::new_with_bootstrap(
212            SupervisorBootstrap::Callback {
213                fallback_actor,
214                callback: Some(Box::new(bootstrap)),
215            },
216            liveness,
217            options,
218            session_id,
219            ready,
220        )
221    }
222
223    fn new_with_bootstrap(
224        bootstrap: SupervisorBootstrap,
225        liveness: KeepaliveLink,
226        options: SupervisionOptions,
227        session_id: Uid,
228        ready: Option<OncePortHandle<()>>,
229    ) -> Self {
230        Self {
231            bootstrap,
232            liveness: Some(liveness),
233            options,
234            session_id,
235            liveness_handle: None,
236            session: None,
237            pending_stop: None,
238            ready,
239        }
240    }
241}

243#[async_trait]
244impl Actor for Supervisor {
245    async fn init(&mut self, this: &Instance<Self>) -> anyhow::Result<()> {
246        let (liveness_handle, liveness) = self
247            .liveness
248            .take()
249            .expect("supervisor initialized more than once")
250            .spawn_supervisor(this)?;
251        self.liveness_handle = Some(liveness_handle);
252        self.bootstrap.start(
253            this,
254            Supervise {
255                session_id: self.session_id.clone(),
256                supervisor: this.port::<SupervisorEvent>().bind(),
257                parent: this.self_addr().clone(),
258                liveness,
259                options: self.options.clone(),
260            },
261        )?;
262        Ok(())
263    }
264
265    async fn handle_stop(
266        &mut self,
267        this: &Instance<Self>,
268        mode: StopMode,
269        reason: &str,
270    ) -> anyhow::Result<()> {
271        self.pending_stop = Some(PendingStop {
272            mode,
273            reason: reason.to_string(),
274        });
275        self.send_pending_worker_stop(this)
276    }
277
278    async fn handle_supervision_event(
279        &mut self,
280        _this: &Instance<Self>,
281        event: &ActorSupervisionEvent,
282    ) -> anyhow::Result<bool> {
283        if self
284            .liveness_handle
285            .as_ref()
286            .is_some_and(|handle| handle.actor_addr() == &event.actor_id)
287        {
288            let event = self.synthesize_unreachable_event("supervision liveness failed");
289            return self.propagate_event(event);
290        }
291        Ok(!event.is_error())
292    }
293}

295#[async_trait]
296impl Handler<SupervisorEvent> for Supervisor {
297    async fn handle(&mut self, cx: &Context<Self>, message: SupervisorEvent) -> anyhow::Result<()> {
298        match message {
299            SupervisorEvent::Linked {
300                session_id,
301                worker,
302                child,
303                display_name,
304            } => {
305                self.ensure_session(&session_id)?;
306                if let Some(ready) = self.ready.take() {
307                    ready.post(cx, ());
308                }
309                self.session = Some(SupervisorSession {
310                    worker,
311                    child,
312                    display_name,
313                });
314                self.send_pending_worker_stop(cx)?;
315                Ok(())
316            }
317            SupervisorEvent::SuperviseRejected { session_id, reason } => {
318                self.ensure_session(&session_id)?;
319                anyhow::bail!("remote supervision request rejected: {}", reason)
320            }
321            SupervisorEvent::SupervisionEvent {
322                session_id,
323                event,
324                disposition: _,
325            } => {
326                self.ensure_session(&session_id)?;
327                self.propagate_event(event)
328            }
329            SupervisorEvent::Unlinked { session_id, reason } => {
330                self.ensure_session(&session_id)?;
331                cx.exit(&reason)?;
332                Ok(())
333            }
334        }
335    }
336}

338impl Supervisor {
339    fn ensure_session(&self, session_id: &hyperactor::Uid) -> anyhow::Result<()> {
340        if session_id != &self.session_id {
341            anyhow::bail!("remote supervision session id mismatch");
342        }
343        Ok(())
344    }
345
346    fn send_pending_worker_stop(&mut self, cx: &Instance<Self>) -> anyhow::Result<()> {
347        let Some(stop) = self.pending_stop.take() else {
348            return Ok(());
349        };
350        let Some(session) = &self.session else {
351            let Some(worker) = self.bootstrap.worker_endpoint() else {
352                self.pending_stop = Some(stop);
353                return Ok(());
354            };
355            worker.post(
356                cx,
357                WorkerCommand::Stop {
358                    session_id: self.session_id.clone(),
359                    mode: stop.mode,
360                    reason: stop.reason,
361                },
362            );
363            return Ok(());
364        };
365        (&session.worker).post(
366            cx,
367            WorkerCommand::Stop {
368                session_id: self.session_id.clone(),
369                mode: stop.mode,
370                reason: stop.reason,
371            },
372        );
373        Ok(())
374    }
375
376    fn synthesize_unreachable_event(&self, reason: &str) -> ActorSupervisionEvent {
377        if let Some(session) = &self.session {
378            ActorSupervisionEvent::new(
379                session.child.clone(),
380                session.display_name.clone(),
381                ActorStatus::generic_failure(reason),
382                None,
383            )
384        } else {
385            ActorSupervisionEvent::new(
386                self.bootstrap.fallback_actor(),
387                None,
388                ActorStatus::generic_failure(reason),
389                None,
390            )
391        }
392    }
393
394    fn propagate_event<T>(&self, event: ActorSupervisionEvent) -> anyhow::Result<T> {
395        Err(anyhow::Error::new(
396            ActorErrorKind::UnhandledSupervisionEvent(Box::new(event)),
397        ))
398    }
399}

547        }

554    child_handle: Option<AnyActorHandle>,
555    child_display_name: Option<String>,
556    session: Option<WorkerSession>,

559impl SupervisedChild {
560    pub(crate) fn set_child(
561        &mut self,
562        child_handle: AnyActorHandle,
563        child_display_name: Option<String>,
564    ) -> anyhow::Result<()> {
565        if self.child_handle.is_some() {
566            anyhow::bail!("worker already has a supervised child");
567        }
568        self.child_handle = Some(child_handle);
569        self.child_display_name = child_display_name;
570        Ok(())
571    }
572
573    pub(crate) async fn supervise<C>(&mut self, cx: &C, message: Supervise) -> anyhow::Result<()>
574    where
575        C: context::Actor,
576        C::A: Handler<WorkerCommand>,
577    {
578        if self.session.is_some() {
579            let _ = message.supervisor.post(
580                cx,
581                SupervisorEvent::SuperviseRejected {
582                    session_id: message.session_id,
583                    reason: "worker already supervised".to_string(),
584                },
585            );
586            return Ok(());
587        }
588
589        if self.child_handle.is_none() {
590            let _ = message.supervisor.post(
591                cx,
592                SupervisorEvent::SuperviseRejected {
593                    session_id: message.session_id,
594                    reason: "worker has no supervised child".to_string(),
595                },
596            );
597            return Ok(());
598        }
599
600        let liveness_handle = message.liveness.spawn_worker(cx).await?;
601        let child_addr = self
602            .child_handle
603            .as_ref()
604            .expect("worker child must be spawned before linking")
605            .actor_id()
606            .clone();
607        let supervisor = message.supervisor.clone();
608        let session_id = SessionId(message.session_id.clone());
609        self.session = Some(WorkerSession {
610            session_id,
611            supervisor: message.supervisor,
612            options: message.options,
613            liveness_handle,
614        });
615        supervisor.post(
616            cx,
617            SupervisorEvent::Linked {
618                session_id: message.session_id.clone(),
619                worker: cx.instance().port::<WorkerCommand>().bind(),
620                child: child_addr,
621                display_name: self.child_display_name.clone(),
622            },
623        );
624        Ok(())
625    }

627    pub(crate) fn handle_supervision_event<A: Actor>(
628        &mut self,
629        this: &Instance<A>,
630        event: &ActorSupervisionEvent,
631    ) -> anyhow::Result<bool> {
632        if self.is_liveness_event(event) {
633            self.handle_liveness_event(this, event)?;
634            return Ok(true);
635        }
636        if self.is_child_event(event) {
637            if let Some(session) = &self.session {
638                let supervisor = session.supervisor.clone();
639                let session_id = session.session_id.clone();
640                supervisor.post(
641                    this,
642                    SupervisorEvent::SupervisionEvent {
643                        session_id: session_id.into_uid(),
644                        event: event.clone(),
645                        disposition: RemoteActorDisposition::Terminal,
646                    },
647                );
648            }
649            self.child_handle = None;
650            this.exit("supervised child stopped")?;
651            return Ok(true);
652        }
653
654        debug_assert!(
655            self.session.is_none(),
656            "worker with active session received supervision event for non-liveness non-child actor: {:?}",
657            event.actor_id
658        );
659        if self.session.is_none() {
660            Ok(true)
661        } else {
662            Ok(!event.is_error())
663        }
664    }

666    pub(crate) fn handle_supervised_worker<A: Actor>(
667        &mut self,
668        cx: &Context<A>,
669        message: WorkerCommand,
670    ) -> anyhow::Result<()> {
671        match message {
672            WorkerCommand::Stop {
673                session_id,
674                mode,
675                reason,
676            } => {
677                self.accept_session_id(session_id)?;
678                self.stop_child(mode, &reason)?;
679            }
680            WorkerCommand::Unlink { session_id, reason } => {
681                let session_id = self.accept_session_id(session_id)?;
682                if let Some(session) = self.session.take() {
683                    let _ = session.liveness_handle.stop(&reason);
684                    match session.options.orphan_policy {
685                        OrphanPolicy::Stop => self.stop_child(StopMode::Stop, &reason)?,
686                        OrphanPolicy::Detach => (),
687                    }
688
689                    let _ = session.supervisor.post(
690                        cx,
691                        SupervisorEvent::Unlinked {
692                            session_id: session_id.into_uid(),
693                            reason,
694                        },
695                    );
696                }
697            }
698        }
699        Ok(())
700    }

702    fn accept_session_id(&self, raw: hyperactor::Uid) -> anyhow::Result<SessionId> {
703        let Some(session) = &self.session else {
704            anyhow::bail!("worker is not linked");
705        };
706        let session_id = SessionId(raw);
707        if session_id != session.session_id {
708            anyhow::bail!("remote supervision session id mismatch");
709        }
710        Ok(session_id)
711    }
712
713    fn stop_child(&self, mode: StopMode, reason: &str) -> anyhow::Result<()> {
714        let Some(child) = &self.child_handle else {
715            return Ok(());
716        };
717        match mode {
718            StopMode::Stop => child.stop(reason)?,
719            StopMode::DrainAndStop => child.drain_and_stop(reason)?,
720        }
721        Ok(())
722    }

724    fn is_child_event(&self, event: &ActorSupervisionEvent) -> bool {
725        self.child_handle
726            .as_ref()
727            .is_some_and(|child| child.actor_id() == &event.actor_id)
728    }
729
730    fn is_liveness_event(&self, event: &ActorSupervisionEvent) -> bool {
731        self.session
732            .as_ref()
733            .is_some_and(|session| session.liveness_handle.actor_id() == &event.actor_id)
734    }

736    pub(crate) fn handle_orphaned_supervisor(&mut self, reason: &str) -> anyhow::Result<()> {
737        match self.unlink_orphaned_supervisor(reason) {
738            Some(OrphanPolicy::Stop) => self.stop_child(StopMode::Stop, reason),
739            Some(OrphanPolicy::Detach) | None => Ok(()),
740        }
741    }
742
743    fn unlink_orphaned_supervisor(&mut self, reason: &str) -> Option<OrphanPolicy> {
744        let session = self.session.take()?;
745        let _ = session.liveness_handle.stop(reason);
746        Some(session.options.orphan_policy)
747    }
748
749    fn handle_liveness_event<A: Actor>(
750        &mut self,
751        cx: &Instance<A>,
752        event: &ActorSupervisionEvent,
753    ) -> anyhow::Result<()> {

761            supervisor.post(
762                cx,
763                SupervisorEvent::SupervisionEvent {
764                    session_id: session_id.into_uid(),
765                    event: ActorSupervisionEvent::new(
766                        child.actor_id().clone(),
767                        self.child_display_name.clone(),
768                        ActorStatus::generic_failure(format!(
769                            "supervision liveness failed: {}",
770                            event
771                        )),
772                        None,
773                    ),
774                    disposition: RemoteActorDisposition::Unreachable,
775                },
776            );
777        }
778        let session = self
779            .session
780            .take()
781            .expect("liveness event must have an active session");
782        let _ = session.liveness_handle.stop("supervision liveness failed");
783        if orphan_policy == OrphanPolicy::Stop {
784            self.stop_child(StopMode::Stop, "supervision liveness failed")?;
785        }
786        Ok(())
