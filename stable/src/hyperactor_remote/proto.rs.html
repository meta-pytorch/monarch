9//! Message types for remote supervision sessions.
10//!
11//! A remote supervision relationship is represented by one worker actor
12//! supervising one child actor on behalf of one supervisor. The supervisor
13//! initiates the relationship by sending [`Supervise`] to the worker. The
14//! `supervisor` port in [`Supervise`] identifies the session endpoint that the
15//! worker uses for all replies and supervision events. The `liveness` spec in
16//! [`Supervise`] identifies the implementation actor that the worker spawns as
17//! the session liveness mechanism. Later supervisor commands are sent to the
18//! worker as [`WorkerCommand`] messages and carry the same `session_id`; the
19//! worker accepts only the active session.

37/// Global actor spawn (gspawn) specification.
38#[derive(Clone, Debug, Serialize, Deserialize, Named, PartialEq, Eq)]
39pub struct Gspawn {
40    actor_type: String,
41    uid: Uid,
42    params: Data,
43}
44wirevalue::register_type!(Gspawn);
45
46impl Gspawn {
47    /// Create a spawn specification for the registered actor type with a fresh uid.
48    pub fn for_actor<A>(params: A::Params) -> anyhow::Result<Self>
49    where
50        A: RemoteSpawn + Named,
51    {
52        Self::for_actor_uid::<A>(Uid::anonymous(), params)
53    }
54
55    /// Create a spawn specification for the registered actor type with an explicit uid.
56    pub fn for_actor_uid<A>(uid: Uid, params: A::Params) -> anyhow::Result<Self>
57    where
58        A: RemoteSpawn + Named,
59    {
60        Ok(Self::with_uid(
61            A::typename(),
62            uid,
63            bincode::serde::encode_to_vec(params, bincode::config::legacy())?,
64        ))
65    }
66
67    pub(crate) fn with_uid(actor_type: impl Into<String>, uid: Uid, params: Data) -> Self {
68        Self {
69            actor_type: actor_type.into(),
70            uid,
71            params,
72        }
73    }
74
75    /// Registered name of the actor to spawn.
76    pub fn actor_type(&self) -> &str {
77        &self.actor_type
78    }
79
80    /// The uid that will identify the spawned actor.
81    pub fn uid(&self) -> &Uid {
82        &self.uid
83    }
84
85    /// The serialized parameters passed to the actor.
86    pub fn params(&self) -> &[u8] {
87        &self.params
88    }
89
90    /// Spawn the actor as a supervised child of `parent`.
91    pub async fn spawn_child<C: hyperactor::context::Actor>(
92        self,
93        parent: &C,
94    ) -> anyhow::Result<hyperactor::AnyActorHandle> {
95        parent
96            .instance()
97            .gspawn_uid(&self.actor_type, self.uid, self.params)
98            .await
99    }
100}

197/// Commands sent by the supervisor proxy to the supervised worker actor.

205    Eq,

210pub enum WorkerCommand {
211    /// Stop the supervised child.
212    Stop {
213        /// Unique identifier for this supervision session.
214        session_id: Uid,

218        reason: String,
219    },
220    /// Unlink the supervisor session.

222        /// Unique identifier for this supervision session.

226    },
227}

232/// Messages sent by the worker actor to the supervisor session port.
233#[derive(Clone, Debug, Serialize, Deserialize, Named, PartialEq, Eq)]
234pub enum SupervisorEvent {
235    /// The worker accepted the session and identified its supervised child.
236    Linked {
237        /// Unique identifier for this supervision session.
238        session_id: Uid,
239        /// Worker actor that owns the supervised child.
240        worker: PortRef<WorkerCommand>,
241        /// Actor address for the supervised child.
242        child: ActorAddr,
243        /// Friendly display name for the supervised child, if available.
244        display_name: Option<String>,
245    },
246    /// The worker rejected the supervise request.
247    SuperviseRejected {
248        /// Unique identifier for this supervision session.
249        session_id: Uid,
250        /// Reason the supervise request failed.
251        reason: String,
252    },
253    /// A supervision event observed by, or synthesized for, the worker.
254    SupervisionEvent {
255        /// Unique identifier for this supervision session.
256        session_id: Uid,
257        /// Event to deliver to the supervisor side.
258        event: ActorSupervisionEvent,
259        /// What the worker knows about the remote actor state.
260        disposition: RemoteActorDisposition,
261    },
262    /// The worker unlinked the session without reporting a supervision event.
263    Unlinked {
264        /// Unique identifier for this supervision session.
265        session_id: Uid,
266        /// Reason for unlinking the session.
267        reason: String,
268    },
269}
270wirevalue::register_type!(SupervisorEvent);
