25/// Render the top status/header bar.
26///
27/// Displays a colorful, information-dense header with topology stats,
28/// selection context, and system state. Uses semantic colors from the
29/// scheme for visual hierarchy and readability.
30pub(crate) fn render_header(
31    frame: &mut ratatui::Frame<'_>,
32    area: Rect,
33    app: &App,
34    detail_clipped: bool,
35) {
36    let l = &app.theme.labels;
37
38    // Error state overrides normal display
39    if let Some(err) = &app.error {
40        let header = Paragraph::new(vec![
41            Line::from(Span::styled(l.app_name, app.theme.scheme.app_name)),
42            Line::from(Span::styled(
43                format!("ERROR: {}", err),
44                app.theme.scheme.error,
45            )),
46        ])
47        .block(
48            Block::default()
49                .borders(Borders::BOTTOM)
50                .border_style(app.theme.scheme.border),
51        );
52        frame.render_widget(header, area);
53        return;
54    }
55
56    // Gather stats
57    let selection = app.current_selection();
58    let sys_state = if app.show_system { l.sys_on } else { l.sys_off };
59    let stopped_state = if app.show_stopped {
60        l.stopped_on
61    } else {
62        l.stopped_off
63    };
64
65    // Extract uptime and username from root node
66    let (uptime_str, username) = if let Some(root_payload) =
67        app.get_cached_payload(&hyperactor_mesh::introspect::NodeRef::Root)
68    {
69        if let NodeProperties::Root {
70            started_at,
71            started_by,
72            ..
73        } = &root_payload.properties
74        {
75            (
76                Some(crate::format::format_system_time_uptime(started_at)),
77                Some(started_by.clone()),
78            )
79        } else {
80            (None, None)
81        }
82    } else {
83        (None, None)
84    };
85
86    // Line 1: App name • URL • up: 2h 34m • @user • sys:off • ⟳ 1s
87    let mut line1_spans = vec![
88        Span::styled(l.app_name, app.theme.scheme.app_name),
89        Span::styled(l.separator, app.theme.scheme.stat_label),
90        Span::styled(&app.base_url, app.theme.scheme.stat_url),
91    ];
92
93    // Add uptime if available
94    if let Some(uptime) = uptime_str {
95        line1_spans.extend(vec![
96            Span::styled(l.separator, app.theme.scheme.stat_label),
97            Span::styled(l.uptime, app.theme.scheme.stat_label),
98            Span::styled(uptime, app.theme.scheme.stat_timing),
99        ]);
100    }
101
102    // Add username if available
103    if let Some(user) = username {
104        line1_spans.extend(vec![
105            Span::styled(l.separator, app.theme.scheme.stat_label),
106            Span::styled(format!("@{}", user), app.theme.scheme.stat_system),
107        ]);
108    }
109
110    line1_spans.extend(vec![
111        Span::styled(l.separator, app.theme.scheme.stat_label),
112        Span::styled(l.system, app.theme.scheme.stat_label),
113        Span::styled(sys_state, app.theme.scheme.stat_system),
114    ]);
115
116    line1_spans.extend(vec![
117        Span::styled(l.separator, app.theme.scheme.stat_label),
118        Span::styled(l.stopped, app.theme.scheme.stat_label),
119        Span::styled(stopped_state, app.theme.scheme.stat_system),
120    ]);
121
122    // Show active theme and lang (skip defaults to reduce noise)
123    if !matches!(app.theme_name, ThemeName::Nord) {
124        line1_spans.extend(vec![
125            Span::styled(l.separator, app.theme.scheme.stat_label),
126            Span::styled(
127                format!("theme:{}", app.theme_name),
128                app.theme.scheme.stat_system,
129            ),
130        ]);
131    }
132    if !matches!(app.lang_name, LangName::En) {
133        line1_spans.extend(vec![
134            Span::styled(l.separator, app.theme.scheme.stat_label),
135            Span::styled(
136                format!("lang:{}", app.lang_name),
137                app.theme.scheme.stat_system,
138            ),
139        ]);
140    }
141
142    if !app.refresh_interval_label.is_empty() {
143        line1_spans.extend(vec![
144            Span::styled(l.separator, app.theme.scheme.stat_label),
145            Span::styled(l.refresh_icon, app.theme.scheme.stat_timing),
146            Span::styled(&app.refresh_interval_label, app.theme.scheme.stat_timing),
147        ]);
148    }
149
150    // Terminal too short to show all detail panes. Prepend the warning -- line 1
151    // truncates from the right, so a segment at the end is the first thing a
152    // narrow terminal loses -- and make it loud (reversed + blink) so a size
153    // problem is not mistaken for missing data.
154    if detail_clipped {
155        let mut alert = vec![
156            Span::styled(
157                l.header_content_clipped,
158                app.theme
159                    .scheme
160                    .error
161                    .add_modifier(Modifier::REVERSED | Modifier::BOLD | Modifier::SLOW_BLINK),
162            ),
163            Span::styled(l.separator, app.theme.scheme.stat_label),
164        ];
165        alert.append(&mut line1_spans);
166        line1_spans = alert;
167    }
168
169    // Line 2: Selection context
170    let mut line2_spans = vec![];
171
172    if let Some(node) = selection {
173        let type_str = node.node_type.label();
174        let type_style = app.theme.scheme.node_style(node.node_type);
175
176        line2_spans.extend(vec![
177            Span::styled(l.selection_caret, app.theme.scheme.stat_selection),
178            Span::styled(type_str, type_style),
179            Span::styled(" ", Style::default()),
180            Span::styled(&node.label, app.theme.scheme.stat_selection),
181        ]);
182
183        // Classification tag only for actors (failed/stopped/system/user)
184        if matches!(node.node_type, NodeType::Actor) {
185            let (class_label, class_style) = if node.failed {
186                ("failed", app.theme.scheme.node_failed)
187            } else if node.stopped {
188                ("stopped", app.theme.scheme.detail_stopped)
189            } else if node.is_system {
190                ("system", app.theme.scheme.node_system_actor)
191            } else {
192                ("user", app.theme.scheme.node_user_actor)
193            };
194            line2_spans.extend(vec![
195                Span::styled(" [", app.theme.scheme.header_class_bracket),
196                Span::styled(class_label, class_style),
197                Span::styled("]", app.theme.scheme.header_class_bracket),
198            ]);
199        }
200    } else {
201        line2_spans.push(Span::styled(l.no_selection, app.theme.scheme.info));
202    }
203
204    let header = Paragraph::new(vec![Line::from(line1_spans), Line::from(line2_spans)]).block(
205        Block::default()
206            .borders(Borders::BOTTOM)
207            .border_style(app.theme.scheme.border),
208    );
209
210    frame.render_widget(header, area);
211}

213/// Render the bottom help bar showing the keyboard shortcuts.
214///
215/// Shows mode-specific hints: topology navigation when the tree is
216/// active, diagnostics navigation when the diagnostics pane is
217/// active.
218pub(crate) fn render_footer(frame: &mut ratatui::Frame<'_>, area: Rect, app: &App) {
219    let text = ActiveJob::footer_text(&app.active_job, &app.theme.labels);
220    let footer = Paragraph::new(text)
221        .style(app.theme.scheme.footer_help)
222        .block(Block::default().borders(Borders::TOP));
223    frame.render_widget(footer, area);
224}

226#[cfg(test)]
227mod tests {
228    use super::*;
229    use crate::theme::Labels;
230    use crate::theme::LangName;
231    use crate::theme::Theme;
232    use crate::theme::ThemeName;
233
234    fn en_labels() -> Labels {
235        Theme::new(ThemeName::Nord, LangName::En).labels
236    }
237
238    // TUI-21: running diagnostics selects the diag-running help text.
239    #[test]
240    fn footer_diag_running() {
241        let labels = en_labels();
242        let job = Some(ActiveJob::Diagnostics {
243            results: vec![],
244            running: true,
245            rx: None,
246            completed_at: None,
247        });
248        assert_eq!(
249            ActiveJob::footer_text(&job, &labels),
250            labels.footer_diag_running_help_text
251        );
252    }
253
254    // TUI-21: completed diagnostics selects the diag-completed help text.
255    #[test]
256    fn footer_diag_completed() {
257        let labels = en_labels();
258        let job = Some(ActiveJob::Diagnostics {
259            results: vec![],
260            running: false,
261            rx: None,
262            completed_at: Some("12:00:00".to_string()),
263        });
264        assert_eq!(
265            ActiveJob::footer_text(&job, &labels),
266            labels.footer_diag_completed_help_text
267        );
268    }
269
270    // TUI-21: active py-spy overlay selects the py-spy help text.
271    #[test]
272    fn footer_pyspy_active() {
273        let labels = en_labels();
274        let job = Some(ActiveJob::PySpy {
275            rx: None,
276            short: "my_proc".to_string(),
277            lines: vec![],
278            completed_at: None,
279        });
280        assert_eq!(
281            ActiveJob::footer_text(&job, &labels),
282            labels.footer_pyspy_help_text
283        );
284    }
285
286    // TUI-21: no active job selects the default help text.
287    #[test]
288    fn footer_idle() {
289        let labels = en_labels();
290        let job: Option<ActiveJob> = None;
291        assert_eq!(
292            ActiveJob::footer_text(&job, &labels),
293            labels.footer_help_text
294        );
295    }
