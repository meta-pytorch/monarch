9//! Read-only FUSE filesystem backed by packed metadata and byte chunks.
10//!
11//! Replaces the Python fusepy-based ChunkedFS with a Rust implementation
12//! using the `fuse3` crate (libfuse3, async/tokio). Exposed to Python
13//! via PyO3.
14//!
15//! Block bytes live in a positional table `blocks: Vec<Option<Bytes>>` indexed by
16//! block id (`global_offset / block`); a `None` entry is a not-yet-delivered block.
17//! Each block has a paired `tokio::sync::Notify` in `block_notifys`. A FUSE read
18//! that touches a `None` block fires `fault_callback` (briefly acquiring the GIL) to
19//! signal the client and then parks on that block's `Notify`; the client materialises
20//! the block into a mount-owned buffer (`block_ptr`, wrapped as a memoryview and
21//! written via RDMA), then calls `receive_block`, which stores `Some(bytes)` at the
22//! slot (a zero-copy freeze of that buffer) and wakes the block's waiters. The parked
23//! read then re-reads. There is no on-disk cache.
24//!
25//! Staleness is per FILE, not per block: a file whose source diverged under the
26//! freshness fence cannot be reproduced, so the client garbage-fills that file's
27//! bytes in the delivered block and flips the file's `stale` bit, shipped alongside
28//! the bytes in `receive_block`'s stale list. A read of that file returns EIO, and
29//! co-located valid files sharing the same 64
30//! MiB block still serve. The bit is cleared for free on `refresh` (the metadata is
31//! rebuilt).
32//!
33//! `PyMountHandle::refresh` atomically swaps the metadata behind the `RwLock` and
34//! grows the (append-only) block table for new tail blocks. All FUSE methods take a
35//! read lock, so reads run concurrently and are only briefly blocked during the
36//! write-lock swap; a read's bytes are gathered (one copy) from the touched blocks
37//! under that read lock.

69enum FsEntry {
70    Dir {
71        attr: FileAttr,
72        children: Vec<String>,
73    },
74    File {
75        attr: FileAttr,
76        global_offset: usize,
77        file_len: usize,
78        /// Set from ``receive_block``'s stale list when the client could not reproduce this file's
79        /// bytes (its source diverged under the fence), so the delivered block holds
80        /// garbage there: a read of this file returns EIO. Rebuilt false on every
81        /// ``refresh`` (the metadata table is recreated from scratch).
82        stale: bool,
83    },
84    Symlink {
85        attr: FileAttr,
86        link_target: OsString,
87    },
88}

90impl FsEntry {
91    fn attr(&self) -> &FileAttr {
92        match self {
93            FsEntry::Dir { attr, .. } => attr,
94            FsEntry::File { attr, .. } => attr,
95            FsEntry::Symlink { attr, .. } => attr,
96        }
97    }
98}

100/// All filesystem state — metadata and chunk data — protected by a single RwLock.
101///
102/// Keeping them together ensures every FUSE read sees a fully consistent
103/// snapshot: no reader can observe new chunk data paired with old file
104/// offsets/sizes (or vice versa), because `refresh` applies both under
105/// one write-lock acquisition.
106struct FsState {
107    metadata: HashMap<OsString, FsEntry>,
108    /// Positional block table indexed by block id (= global_offset / block size);
109    /// ``None`` is not-yet-delivered, ``Some(bytes)`` is delivered. Sized to the block
110    /// count at mount and grown (append-only) on refresh. Staleness lives per file
111    /// (``FsEntry::File::stale``), not per block, so one diverged file does not poison
112    /// the co-located files that share its 64 MiB block.
113    blocks: Vec<Option<Bytes>>,
114    /// One ``Notify`` per block, parallel to ``blocks`` (same length). A read that
115    /// finds ``blocks[id] == None`` parks on ``block_notifys[id]``; ``receive_block``
116    /// wakes it after storing the bytes. In an ``Arc`` so a read can clone it out
117    /// under the read lock and await it after releasing the lock.
118    block_notifys: Vec<Arc<Notify>>,
119    /// Total concatenated layout size.
120    total_size: usize,
121    /// Mount-owned writable staging buffers, parallel to ``blocks`` and keyed by the same
122    /// block id: ``reserved[id]`` is the buffer block ``id`` is delivered into, or ``None``
123    /// when no delivery is outstanding for it. ``block_ptr`` provisions ``Some(zeroed
124    /// BLOCK_SIZE)`` on demand and hands the client its address (wrapped as a memoryview
125    /// and written via RDMA / staging); ``receive_block`` ``take``s it and freezes it
126    /// (zero-copy) into ``blocks[id]``, leaving the slot ``None`` again. Grown with
127    /// ``None`` (no eager allocation), so only outstanding blocks hold a 64 MiB buffer.
128    reserved: Vec<Option<BytesMut>>,
129}
130
131impl FsState {
132    fn lookup_entry(&self, path: &OsStr) -> Option<&FsEntry> {
133        self.metadata.get(path)
134    }
135
136    /// Grow the block table and its parallel per-block notifys to `num_blocks`,
137    /// appending empty (`None`) slots. Append-only -- never shrinks -- and keeps
138    /// the invariant `blocks.len() == block_notifys.len()`.
139    fn grow_to(&mut self, num_blocks: usize) {
140        if num_blocks > self.blocks.len() {
141            self.blocks.resize(num_blocks, None);
142        }
143        while self.block_notifys.len() < self.blocks.len() {
144            self.block_notifys.push(Arc::new(Notify::new()));
145        }
146    }
147
148    /// Store a delivered block's ``bytes`` at its slot, set the ``stale`` bit on any
149    /// files whose source diverged under the fence, and return the block's ``Notify``
150    /// (so the caller can wake parked reads after releasing the lock).
151    fn install_block(&mut self, block_id: usize, bytes: Bytes, stale: Vec<String>) -> Arc<Notify> {
152        // Defensive: a delivery that races ahead of the refresh that grew the table
153        // still lands (the block id is bounded by the layout size).
154        self.grow_to(block_id + 1);
155        self.blocks[block_id] = Some(bytes);
156        for path in stale {
157            if let Some(FsEntry::File { stale, .. }) = self.metadata.get_mut(OsStr::new(&path)) {
158                *stale = true;
159            }
160        }
161        self.block_notifys[block_id].clone()
162    }
163
164    /// The staging-buffer slot for block ``block_id``, growing the array with ``None`` to
165    /// cover it if needed (no eager allocation -- only outstanding blocks hold a buffer).
166    /// ``block_ptr`` calls this then ``get_or_insert_with`` to provision the buffer;
167    /// ``receive_block`` calls it then ``take`` to consume it, leaving the slot ``None``
168    /// so a later re-delivery (e.g. a refresh) re-provisions cleanly.
169    fn reserved_slot(&mut self, block_id: usize) -> &mut Option<BytesMut> {
170        if block_id >= self.reserved.len() {
171            self.reserved.resize_with(block_id + 1, || None);
172        }
173        &mut self.reserved[block_id]
174    }
175
176    /// Commit block ``block_id``: freeze its staged buffer into ``blocks`` (zero-copy) and
177    /// return its ``Notify`` so the caller can wake parked reads. With no staged buffer, an
178    /// already-served block is a duplicate/stray commit -> no-op (``None``), never
179    /// clobbering the served bytes with zeros; a never-delivered block lands a zeroed
180    /// buffer rather than panicking.
181    fn commit_block(&mut self, block_id: usize, stale: Vec<String>) -> Option<Arc<Notify>> {
182        match self.reserved_slot(block_id).take() {
183            Some(buf) => Some(self.install_block(block_id, buf.freeze(), stale)),
184            None if self.blocks.get(block_id).is_some_and(|b| b.is_some()) => None,
185            None => Some(self.install_block(
186                block_id,
187                BytesMut::zeroed(AVAILABILITY_BLOCK_SIZE).freeze(),
188                stale,
189            )),
190        }
191    }
192}

194/// Gather a FUSE read from the positional block table, under the state read lock.
195/// The read spans at most two 64 MiB blocks; copy each touched block's bytes into
196/// one buffer, zero-padding a short tail block or (defensively) a block the read
197/// handler's presence check should have excluded.
198fn read_blocks(
199    blocks: &[Option<Bytes>],
200    global_offset: usize,
201    file_len: usize,
202    offset: u64,
203    size: u32,
204) -> Bytes {
205    let offset = offset as usize;
206    if offset >= file_len {
207        return Bytes::new();
208    }
209    let len = std::cmp::min(size as usize, file_len - offset);
210    if len == 0 {
211        return Bytes::new();
212    }
213    let start = global_offset + offset;
214    let end = start + len;
215
216    let mut out: Vec<u8> = Vec::with_capacity(len);
217    let mut pos = start;
218    while pos < end {
219        let block_id = pos / AVAILABILITY_BLOCK_SIZE;
220        let block_base = block_id * AVAILABILITY_BLOCK_SIZE;
221        let chunk_end = std::cmp::min(block_base + AVAILABILITY_BLOCK_SIZE, end);
222        let want = chunk_end - pos;
223        let within = pos - block_base;
224        match blocks.get(block_id) {
225            Some(Some(b)) => {
226                // Copy what the block actually holds; a short (tail) block leaves
227                // the remainder zero-padded.
228                let take = want.min(b.len().saturating_sub(within));
229                out.extend_from_slice(&b[within..within + take]);
230                if take < want {
231                    out.resize(out.len() + (want - take), 0);
232                }
233            }
234            // Not-yet-delivered (the read handler gathers only once every touched
235            // block is present, so this is defensive) or out-of-range -> zero-pad.
236            _ => out.resize(out.len() + want, 0),
237        }
238        pos = chunk_end;
239    }
240    Bytes::from(out)
241}

243/// The first not-yet-delivered (``None``) block in the touched range, or ``None`` if
244/// every touched block is present and the read can gather. Factored out so it is
245/// unit-testable. enumerate() before take/skip so the index stays the absolute block
246/// id, not a range-relative one.
247fn first_missing_block(
248    blocks: &[Option<Bytes>],
249    first_block: usize,
250    last_block: usize,
251) -> Option<usize> {
252    blocks
253        .iter()
254        .enumerate()
255        .take(last_block + 1)
256        .skip(first_block)
257        .find(|(_, slot)| slot.is_none())
258        .map(|(b, _)| b)
259}

261fn join_path(parent: &OsStr, name: &OsStr) -> OsString {
262    let parent_s = parent.to_string_lossy();
263    let name_s = name.to_string_lossy();
264    if parent_s == "/" {
265        OsString::from(format!("/{name_s}"))
266    } else {
267        OsString::from(format!("{parent_s}/{name_s}"))
268    }
269}
270
271fn parent_path(path: &OsStr) -> OsString {
272    let s = path.to_string_lossy();
273    match s.rfind('/') {
274        Some(0) | None => OsString::from("/"),
275        Some(i) => OsString::from(&s[..i]),
276    }
277}

279fn f64_to_system_time(ts: f64) -> SystemTime {
280    if ts >= 0.0 {
281        UNIX_EPOCH + Duration::from_secs_f64(ts)
282    } else {
283        UNIX_EPOCH
284    }

296fn extract_attr(dict: &Bound<'_, PyDict>, kind: FileType) -> PyResult<FileAttr> {
297    let st_size: u64 = required_key(dict, "st_size")?;
298    Ok(FileAttr {
299        size: st_size,
300        blocks: st_size.div_ceil(512),
301        atime: f64_to_system_time(required_key(dict, "st_atime")?),
302        mtime: f64_to_system_time(required_key(dict, "st_mtime")?),
303        ctime: f64_to_system_time(required_key(dict, "st_ctime")?),
304        kind,
305        perm: (required_key::<u32>(dict, "st_mode")? & 0o7777) as u16,
306        nlink: required_key(dict, "st_nlink")?,
307        uid: required_key(dict, "st_uid")?,
308        gid: required_key(dict, "st_gid")?,
309        rdev: 0,
310        blksize: 4096,
311        #[cfg(target_os = "macos")]
312        crtime: std::time::SystemTime::UNIX_EPOCH,
313        #[cfg(target_os = "macos")]
314        flags: 0,
315    })
316}

318fn extract_metadata(dict: &Bound<'_, PyDict>) -> PyResult<HashMap<OsString, FsEntry>> {
319    let mut entries = HashMap::with_capacity(dict.len());
320    for (key, value) in dict.iter() {
321        let path: String = key.extract()?;
322        let entry_dict: &Bound<'_, PyDict> = value.downcast()?;
323        let attr_obj = entry_dict.get_item("attr")?.ok_or_else(|| {
324            PyRuntimeError::new_err(format!("missing 'attr' key for path: {path}"))
325        })?;
326        let attr_dict: &Bound<'_, PyDict> = attr_obj.downcast()?;
327
328        let fs_entry = if let Some(link_target) = entry_dict.get_item("link_target")? {
329            let target: String = link_target.extract()?;
330            FsEntry::Symlink {
331                attr: extract_attr(attr_dict, FileType::Symlink)?,
332                link_target: OsString::from(target),
333            }
334        } else if let Some(children_obj) = entry_dict.get_item("children")? {
335            let children: Vec<String> = children_obj.extract()?;
336            FsEntry::Dir {
337                attr: extract_attr(attr_dict, FileType::Directory)?,
338                children,
339            }
340        } else {
341            let global_offset: usize = entry_dict
342                .get_item("global_offset")?
343                .map(|v| v.extract())
344                .transpose()?
345                .unwrap_or(0);
346            let file_len: usize = entry_dict
347                .get_item("file_len")?
348                .map(|v| v.extract())
349                .transpose()?
350                .unwrap_or(0);
351            FsEntry::File {
352                attr: extract_attr(attr_dict, FileType::RegularFile)?,
353                global_offset,
354                file_len,
355                stale: false,
356            }
357        };
358        entries.insert(OsString::from(path), fs_entry);
359    }
360    Ok(entries)
361}

365struct ChunkedFuseFs {
366    state: Arc<RwLock<FsState>>,
367    ttl: Duration,
368    /// Called (briefly acquiring the GIL) from the read handler when a read faults
369    /// a missing block, so the client is signalled to materialise it.
370    fault_callback: Py<PyAny>,
371}
372
373impl PathFilesystem for ChunkedFuseFs {
374    type DirEntryStream<'a> = stream::Iter<std::vec::IntoIter<FuseResult<DirectoryEntry>>>;
375    type DirEntryPlusStream<'a> = stream::Iter<std::vec::IntoIter<FuseResult<DirectoryEntryPlus>>>;
376
377    async fn init(&self, _req: Request) -> FuseResult<ReplyInit> {
378        Ok(ReplyInit {
379            max_write: NonZeroU32::new(16 * 1024).expect("16KB is non-zero"),
380        })
381    }

383    async fn destroy(&self, _req: Request) {}
384
385    async fn lookup(&self, _req: Request, parent: &OsStr, name: &OsStr) -> FuseResult<ReplyEntry> {
386        let path = join_path(parent, name);
387        let state = self.state.read().unwrap();
388        let entry = state.lookup_entry(&path).ok_or_else(Errno::new_not_exist)?;
389        Ok(ReplyEntry {
390            ttl: self.ttl,
391            attr: *entry.attr(),
392        })
393    }
394
395    async fn getattr(
396        &self,
397        _req: Request,
398        path: Option<&OsStr>,
399        _fh: Option<u64>,
400        _flags: u32,
401    ) -> FuseResult<ReplyAttr> {
402        let path = path.ok_or_else(Errno::new_not_exist)?;
403        let state = self.state.read().unwrap();
404        let entry = state.lookup_entry(path).ok_or_else(Errno::new_not_exist)?;
405        Ok(ReplyAttr {
406            ttl: self.ttl,
407            attr: *entry.attr(),
408        })
409    }
410
411    async fn readlink(&self, _req: Request, path: &OsStr) -> FuseResult<ReplyData> {
412        let state = self.state.read().unwrap();
413        let entry = state.lookup_entry(path).ok_or_else(Errno::new_not_exist)?;
414        match entry {
415            FsEntry::Symlink { link_target, .. } => Ok(ReplyData {
416                data: Bytes::copy_from_slice(link_target.as_encoded_bytes()),
417            }),
418            _ => Err(libc::EINVAL.into()),
419        }
420    }
421
422    async fn open(&self, _req: Request, path: &OsStr, flags: u32) -> FuseResult<ReplyOpen> {
423        let state = self.state.read().unwrap();
424        let entry = state.lookup_entry(path).ok_or_else(Errno::new_not_exist)?;
425        if matches!(entry, FsEntry::Dir { .. }) {
426            return Err(Errno::new_is_dir());
427        }
428        Ok(ReplyOpen { fh: 0, flags })
429    }
430
431    async fn read(
432        &self,
433        _req: Request,
434        path: Option<&OsStr>,
435        _fh: u64,
436        offset: u64,
437        size: u32,
438    ) -> FuseResult<ReplyData> {
439        let path = path.ok_or_else(Errno::new_not_exist)?;
440        loop {
441            // Resolve the file and find the first touched block that is not yet
442            // present, under the read lock. If every touched block is present,
443            // gather and return without dropping the lock so no refresh can swap
444            // state mid-read; otherwise take that block's notify to wait on.
445            let (block, notify) = {
446                let state = self.state.read().unwrap();
447                let entry = state.lookup_entry(path).ok_or_else(Errno::new_not_exist)?;
448                let FsEntry::File {
449                    global_offset,
450                    file_len,
451                    stale,
452                    ..
453                } = entry
454                else {
455                    return Err(libc::EINVAL.into());
456                };
457                // This file's source diverged under the fence and could not be
458                // reproduced -- its bytes in the delivered block are garbage. EIO
459                // this file; co-located valid files (each checked via its own entry)
460                // still gather from the same block.
461                if *stale {
462                    return Err(libc::EIO.into());
463                }
464                let (global_offset, file_len) = (*global_offset, *file_len);
465                let offset_usize = offset as usize;
466                if offset_usize >= file_len {
467                    return Ok(ReplyData { data: Bytes::new() });
468                }
469                let len = (size as usize).min(file_len - offset_usize);
470                if len == 0 {
471                    return Ok(ReplyData { data: Bytes::new() });
472                }
473                let start = global_offset + offset_usize;
474                let first_block = start / AVAILABILITY_BLOCK_SIZE;
475                let last_block = (start + len - 1) / AVAILABILITY_BLOCK_SIZE;
476                match first_missing_block(&state.blocks, first_block, last_block) {
477                    None => {
478                        return Ok(ReplyData {
479                            data: read_blocks(&state.blocks, global_offset, file_len, offset, size),
480                        });
481                    }
482                    Some(b) => (b, state.block_notifys[b].clone()),
483                }
484            };
485
486            // The block is missing. Register interest before requesting it, so a
487            // delivery landing between the check above and the await below cannot be
488            // lost -- a lost wakeup would wedge the read in uninterruptible sleep. If
489            // it was delivered in that window, re-loop and re-decide (which also
490            // re-checks the file's stale bit); otherwise signal the client (briefly
491            // acquiring the GIL) to materialise it, then wait for `receive_block` to
492            // wake us and re-decide.
493            let notified = notify.notified();
494            tokio::pin!(notified);
495            notified.as_mut().enable();
496            if self.state.read().unwrap().blocks[block].is_some() {
497                continue;
498            }
499            monarch_with_gil_blocking(GilSite::EndpointDispatch, |py| {
500                if let Err(e) = self.fault_callback.bind(py).call1((block,)) {
501                    warn!("chunked_fuse fault callback failed: {e}");
502                }
503            });
504            notified.await;
505        }
506    }

508    async fn opendir(&self, _req: Request, path: &OsStr, flags: u32) -> FuseResult<ReplyOpen> {
509        let state = self.state.read().unwrap();
510        let entry = state.lookup_entry(path).ok_or_else(Errno::new_not_exist)?;
511        if !matches!(entry, FsEntry::Dir { .. }) {
512            return Err(Errno::new_is_not_dir());
513        }
514        Ok(ReplyOpen { fh: 0, flags })
515    }

517    async fn readdir<'a>(
518        &'a self,
519        _req: Request,
520        parent: &'a OsStr,
521        _fh: u64,
522        offset: i64,
523    ) -> FuseResult<ReplyDirectory<Self::DirEntryStream<'a>>> {
524        let state = self.state.read().unwrap();
525        let entry = state
526            .lookup_entry(parent)
527            .ok_or_else(Errno::new_not_exist)?;
528        let children = match entry {
529            FsEntry::Dir { children, .. } => children,
530            _ => return Err(Errno::new_is_not_dir()),
531        };
532
533        let offset = offset as u64;
534        let mut entries: Vec<FuseResult<DirectoryEntry>> = Vec::new();
535        let mut idx: u64 = 1;
536
537        if offset < idx {
538            entries.push(Ok(DirectoryEntry {
539                kind: FileType::Directory,
540                name: OsString::from("."),
541                offset: idx as i64,
542            }));
543        }
544        idx += 1;
545
546        if offset < idx {
547            entries.push(Ok(DirectoryEntry {
548                kind: FileType::Directory,
549                name: OsString::from(".."),
550                offset: idx as i64,
551            }));
552        }
553        idx += 1;
554
555        for child_name in children {
556            if offset < idx {
557                let child_path = join_path(parent, OsStr::new(child_name));
558                if let Some(child_entry) = state.lookup_entry(&child_path) {
559                    entries.push(Ok(DirectoryEntry {
560                        kind: child_entry.attr().kind,
561                        name: OsString::from(child_name),
562                        offset: idx as i64,
563                    }));
564                }
565            }
566            idx += 1;
567        }
568
569        Ok(ReplyDirectory {
570            entries: stream::iter(entries),
571        })
572    }

574    async fn readdirplus<'a>(
575        &'a self,
576        _req: Request,
577        parent: &'a OsStr,
578        _fh: u64,
579        offset: u64,
580        _lock_owner: u64,
581    ) -> FuseResult<ReplyDirectoryPlus<Self::DirEntryPlusStream<'a>>> {
582        let state = self.state.read().unwrap();
583        let entry = state
584            .lookup_entry(parent)
585            .ok_or_else(Errno::new_not_exist)?;
586        let children = match entry {
587            FsEntry::Dir { children, .. } => children,
588            _ => return Err(Errno::new_is_not_dir()),
589        };
590
591        let ttl = self.ttl;
592        let mut entries: Vec<FuseResult<DirectoryEntryPlus>> = Vec::new();
593        let mut idx: u64 = 1;
594
595        if offset < idx {
596            entries.push(Ok(DirectoryEntryPlus {
597                kind: FileType::Directory,
598                name: OsString::from("."),
599                offset: idx as i64,
600                attr: *entry.attr(),
601                entry_ttl: ttl,
602                attr_ttl: ttl,
603            }));
604        }
605        idx += 1;
606
607        if offset < idx {
608            let parent_key = parent_path(parent);
609            let dotdot_attr = match state.lookup_entry(&parent_key) {
610                Some(e) => *e.attr(),
611                None => {
612                    warn!(
613                        "readdirplus: parent path {:?} not found in metadata, \
614                         falling back to current dir attrs",
615                        parent_key
616                    );
617                    *entry.attr()
618                }
619            };
620            entries.push(Ok(DirectoryEntryPlus {
621                kind: FileType::Directory,
622                name: OsString::from(".."),
623                offset: idx as i64,
624                attr: dotdot_attr,
625                entry_ttl: ttl,
626                attr_ttl: ttl,
627            }));
628        }
629        idx += 1;
630
631        for child_name in children {
632            if offset < idx {
633                let child_path = join_path(parent, OsStr::new(child_name));
634                if let Some(child_entry) = state.lookup_entry(&child_path) {
635                    entries.push(Ok(DirectoryEntryPlus {
636                        kind: child_entry.attr().kind,
637                        name: OsString::from(child_name),
638                        offset: idx as i64,
639                        attr: *child_entry.attr(),
640                        entry_ttl: ttl,
641                        attr_ttl: ttl,
642                    }));
643                }
644            }
645            idx += 1;
646        }
647
648        Ok(ReplyDirectoryPlus {
649            entries: stream::iter(entries),
650        })
651    }
652
653    async fn access(&self, _req: Request, path: &OsStr, _mask: u32) -> FuseResult<()> {
654        let state = self.state.read().unwrap();
655        if state.lookup_entry(path).is_some() {
656            Ok(())
657        } else {
658            Err(Errno::new_not_exist())
659        }
660    }

697/// Handle to a running FUSE mount. Call `unmount()` to stop it.
698#[pyclass(
699    name = "FuseMountHandle",
700    module = "monarch._rust_bindings.monarch_extension.chunked_fuse"
701)]
702struct PyMountHandle {
703    unmount_tx: Option<oneshot::Sender<()>>,
704    mount_point: String,
705    state: Arc<RwLock<FsState>>,
706}
707
708#[pymethods]
709impl PyMountHandle {
710    /// Unmount the FUSE filesystem.
711    ///
712    /// Cleanup of the background FUSE session task is asynchronous: this
713    /// method returns once `fusermount3 -u` completes, but the tokio task
714    /// may still be winding down.
715    fn unmount(&mut self, py: Python<'_>) -> PyResult<()> {
716        if self.unmount_tx.take().is_none() {
717            return Ok(());
718        }
719        let mount_point = self.mount_point.clone();
720        py.detach(|| {
721            match std::process::Command::new("fusermount3")
722                .arg("-u")
723                .arg(&mount_point)
724                .output()
725            {
726                Ok(o) if !o.status.success() => {
727                    let stderr = String::from_utf8_lossy(&o.stderr);
728                    Err(PyRuntimeError::new_err(format!(
729                        "fusermount3 -u {mount_point} failed: {stderr}"
730                    )))
731                }
732                Err(e) => Err(PyRuntimeError::new_err(format!(
733                    "failed to run fusermount3: {e}"
734                ))),
735                _ => Ok(()),
736            }
737        })
738    }
739
740    /// Atomically swap filesystem metadata + sizing.
741    ///
742    /// Acquires the write lock once, so no FUSE read can observe a mix of
743    /// old metadata with new chunk data (or vice versa).
744    ///
745    /// Append-only: unchanged block_ids keep their in-memory bytes, and any
746    /// appended blocks fault in on demand after the swap -- so refresh only swaps
747    /// the tree + sizing, never the block contents. (A defrag, which reassigns
748    /// block_ids, re-mounts rather than refreshing.)
749    ///
750    /// Sleeps 3× the FUSE TTL after the lock is released so kernel
751    /// attr/dentry caches expire before returning.
752    fn refresh(
753        &self,
754        py: Python<'_>,
755        metadata: &Bound<'_, PyDict>,
756        new_total_size: usize,
757    ) -> PyResult<()> {
758        let new_metadata = extract_metadata(metadata)?;
759        {
760            let mut state = self.state.write().unwrap();
761            state.metadata = new_metadata;
762            state.total_size = new_total_size;
763            // Append-only: grow the block table (and its notifys) for the new tail
764            // blocks; existing ids keep their delivered bytes. A shrink never happens
765            // here -- a defrag re-mounts instead of refreshing. New blocks fault in
766            // on demand, so no parked reader needs waking here.
767            let num_blocks = new_total_size.div_ceil(AVAILABILITY_BLOCK_SIZE);
768            state.grow_to(num_blocks);
769        }
770        // Wait for kernel FUSE attr/page caches (TTL=200ms) to expire.
771        let settle = Duration::from_millis(DEFAULT_TTL_MS * 3);
772        py.detach(|| {
773            #[allow(clippy::disallowed_methods)]
774            std::thread::sleep(settle);
775            Ok(())
776        })
777    }
778
779    /// The stable heap address of block ``block_id``'s staging buffer, provisioning a
780    /// zeroed ``BLOCK_SIZE`` buffer for it if none is outstanding. The client wraps this
781    /// address as a writable memoryview (e.g. via ``ctypes``), writes the block into it
782    /// (RDMA read / staged copy), then commits it with ``receive_block``. Valid for the
783    /// buffer's life: the allocation does not move, and ``receive_block``'s freeze is
784    /// zero-copy (the same address carries into the served ``Bytes``).
785    fn block_ptr(&self, block_id: usize) -> usize {
786        let mut state = self.state.write().unwrap();
787        state
788            .reserved_slot(block_id)
789            .get_or_insert_with(|| BytesMut::zeroed(AVAILABILITY_BLOCK_SIZE))
790            .as_ptr() as usize
791    }
792
793    /// Commit block ``block_id``'s staging buffer with NO copy: ``take`` the ``BytesMut``
794    /// (already written via its ``block_ptr`` memoryview) and freeze it into ``blocks``,
795    /// mark diverged files stale, and wake parked reads. If no buffer is outstanding: a
796    /// duplicate/stray commit of an already-served block is a no-op -- it must not clobber
797    /// the served bytes with zeros; a call for a never-delivered block (no preceding
798    /// ``block_ptr``, nothing served) lands a zeroed buffer rather than panicking. A
799    /// legitimate re-delivery (e.g. after a refresh) always provisions a fresh buffer via
800    /// ``block_ptr`` first, so it takes the normal path and overwrites cleanly. The freeze
801    /// hands the buffer's allocation to ``blocks`` and leaves the staging slot ``None``, so
802    /// the memoryview the client still holds aliases the served bytes -- the client must
803    /// not write it again after committing. ``stale`` carries the vpaths whose source
804    /// diverged under the fence (their bytes here are garbage) so their reads EIO while
805    /// co-located fresh files serve; the bit is sticky until a ``refresh`` rebuilds the
806    /// metadata.
807    fn receive_block(&self, block_id: usize, stale: Vec<String>) {
808        // Commit under the write lock, then wake parked reads after releasing it. A no-op
809        // commit (a duplicate/stray call) returns no notify, so there is nothing to wake.
810        let notify = self.state.write().unwrap().commit_block(block_id, stale);
811        if let Some(notify) = notify {
812            notify.notify_waiters();
813        }
814    }
815}

817/// Mount a read-only FUSE filesystem serving blocks from memory.
818///
819/// Args:
820///     metadata: dict mapping paths to entry dicts (as produced by
821///         the layout builder), including each file's `global_offset`.
822///     total_size: total concatenated layout size (for block-count bounds).
823///     mount_point: path to mount the filesystem.
824/// Returns a FuseMountHandle. Call handle.unmount() to unmount, handle.refresh() to
825/// swap metadata, or block_ptr + receive_block to deliver a block into memory (write
826/// it into block_ptr's buffer, then receive_block to freeze it in and serve it).
827#[pyfunction]
828fn mount_chunked_fuse(
829    py: Python<'_>,
830    metadata: &Bound<'_, PyDict>,
831    total_size: usize,
832    mount_point: String,
833    fault_callback: Py<PyAny>,
834) -> PyResult<PyMountHandle> {
835    let ttl_ms = DEFAULT_TTL_MS;
836    if !mount_point.starts_with('/') {
837        return Err(PyRuntimeError::new_err(
838            "mount_point must be an absolute path",
839        ));
840    }
841
842    let metadata = extract_metadata(metadata)?;
843
844    // The block table is sized to the layout's block count, all ``None``, with a
845    // paired ``Notify`` per block; blocks are delivered into their slots on demand
846    // via receive_block (open's code prefill + read faults). There is no warm-load.
847    let num_blocks = total_size.div_ceil(AVAILABILITY_BLOCK_SIZE);
848    let state = Arc::new(RwLock::new(FsState {
849        metadata,
850        blocks: vec![None; num_blocks],
851        block_notifys: (0..num_blocks).map(|_| Arc::new(Notify::new())).collect(),
852        total_size,
853        reserved: Vec::new(),
854    }));
855
856    let fs = ChunkedFuseFs {
857        state: state.clone(),
858        ttl: Duration::from_millis(ttl_ms),
859        fault_callback,
860    };
861
862    let mount_path = mount_point.clone();
863    let (unmount_tx, unmount_rx) = oneshot::channel::<()>();
864    let (mount_result_tx, mount_result_rx) = oneshot::channel::<Result<(), String>>();
865
866    // The FUSE session task runs on the shared tokio runtime. We do not
867    // retain the JoinHandle because the task's lifetime is governed by
868    // fusermount3 -u (called in unmount()). If the Python interpreter
869    // exits without calling unmount(), the runtime's atexit shutdown will
870    // drop the task. This is acceptable: the kernel automatically cleans
871    // up the FUSE mount when the process exits.
872    let runtime = monarch_hyperactor::runtime::get_tokio_runtime();
873    runtime.spawn(async move {
874        let mut opts = MountOptions::default();
875        opts.read_only(true).force_readdir_plus(true).nonempty(true);
876
877        let mount_result = fuse3::path::Session::new(opts)
878            .mount_with_unprivileged(fs, &mount_path)
879            .await;
880
881        match mount_result {
882            Ok(mount_handle) => {
883                let _ = mount_result_tx.send(Ok(()));
884                // Run the FUSE session until unmount() calls fusermount3 -u,
885                // which causes the session future to complete.
886                if let Err(e) = mount_handle.await {
887                    warn!("fuse session error: {e}");
888                }
889            }
890            Err(e) => {
891                warn!("fuse mount failed: {e}");
892                let _ = mount_result_tx.send(Err(format!("{e}")));
893            }
894        }
895        drop(unmount_rx);
896    });
897
898    // Poll until mount appears in /proc/mounts or the task signals
899    // failure. Releases the GIL while waiting. We use blocking
900    // std::thread::sleep because this is a synchronous PyO3 context.
901    //
902    // NOTE: Polling /proc/mounts has an inherent TOCTOU race: the mount
903    // could theoretically appear and disappear between our check and the
904    // caller's first filesystem access. In practice this doesn't happen
905    // because nothing else unmounts the path, but inotify on /proc/mounts
906    // would be a more robust alternative if needed.
907    #[allow(clippy::disallowed_methods)]
908    py.detach(|| {
909        let start = std::time::Instant::now();
910        let mut mount_result_rx = Some(mount_result_rx);
911        while start.elapsed() < Duration::from_secs(50) {
912            // Check if the mount task reported an early failure.
913            if let Some(rx) = mount_result_rx.as_mut() {
914                match rx.try_recv() {
915                    Ok(Err(e)) => {
916                        return Err(PyRuntimeError::new_err(format!("fuse mount failed: {e}")));
917                    }
918                    Ok(Ok(())) => {
919                        // Mount succeeded; stop checking the channel but
920                        // still wait for /proc/mounts visibility.
921                        mount_result_rx = None;
922                    }
923                    Err(oneshot::error::TryRecvError::Closed) => {
924                        return Err(PyRuntimeError::new_err(
925                            "fuse mount task exited without reporting status \
926                             (possible panic, tokio runtime shutdown, or \
927                             task cancellation)",
928                        ));
929                    }
930                    Err(oneshot::error::TryRecvError::Empty) => {}
931                }
932            }
933
934            // Check /proc/mounts directly (authoritative source).
935            if let Ok(mounts) = std::fs::read_to_string("/proc/mounts") {
936                for line in mounts.lines() {
937                    if let Some(mp) = line.split_whitespace().nth(1)
938                        && mp == mount_point
939                    {
940                        return Ok(());
941                    }
942                }
943            }
944
945            std::thread::sleep(Duration::from_millis(100));
946        }
947        Err(PyRuntimeError::new_err("timed out waiting for FUSE mount"))
948    })?;
949
950    Ok(PyMountHandle {
951        unmount_tx: Some(unmount_tx),
952        mount_point,
953        state,
954    })
955}
