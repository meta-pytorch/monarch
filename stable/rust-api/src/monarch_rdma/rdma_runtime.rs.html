9//! A shared, dedicated Tokio runtime for the RDMA manager actors.
10//!
11//! RDMA buffer registration scans `torch.cuda` segments under the GIL
12//! (`backend::ibverbs::mlx_domain` -> `pytorch_cuda_segments`), so the RDMA actor
13//! loops run here rather than on the shared control-plane runtime.
14//!
15//! # Invariants (RR-*)
16//!
17//! - **RR-1 (process-singleton):** one shared rdma runtime exists per process
18//!   (built once, lazily), serving all RDMA actors rather than one per actor.
19//!   Sharing is safe because RDMA keeps no per-thread state that would pin an
20//!   actor to a worker, and the `QueuePairActor`s are cooperative (each re-arms a
21//!   `Tick` and yields rather than busy-polling), so a small worker pool serves
22//!   many without starving.
23//! - **RR-2 (shared-runtime-routing):** the RDMA actors (`RdmaManagerActor`,
24//!   `IbvManagerActor`, `QueuePairActor`) route `spawn_server_task` through
25//!   [`spawn_on_rdma_runtime`], not ad hoc `tokio::spawn`.
26//! - **RR-3 (rdma-is-data-plane):** [`spawn_on_rdma_runtime`] runs its future on
27//!   a `DataPlane("rdma")` thread.
28//! - **RR-4 (registration-off-control-plane):** consequently, the GIL taken
29//!   during buffer registration lands on a `DataPlane("rdma")` thread, never the
30//!   shared control-plane runtime (where a blocked GIL-holder would stall actor
31//!   dispatch, supervision, and networking).
32//! - **RR-5 (cross-runtime-joinhandle):** the `JoinHandle` from
33//!   [`spawn_on_rdma_runtime`] is awaitable from the caller's runtime, so it is
34//!   returned directly rather than bridged through a relay task.
35
36use std::sync::OnceLock;
37
38use hyperactor::runtime_identity::build_data_plane_runtime;
39
40/// Worker-thread count for the shared rdma runtime. >1 so concurrent
41/// `QueuePairActor` ticks make progress in parallel.
42const RDMA_RUNTIME_WORKER_THREADS: usize = 4;
43
44/// The process-wide rdma data-plane runtime, built on first use and tagged
45/// `DataPlane("rdma")`.
46fn rdma_runtime() -> &'static tokio::runtime::Handle {
47    static RT: OnceLock<tokio::runtime::Handle> = OnceLock::new();
48    RT.get_or_init(|| build_data_plane_runtime("rdma", RDMA_RUNTIME_WORKER_THREADS))
49}
50
51/// Spawn an actor server loop onto the shared rdma runtime and return its
52/// `JoinHandle`. Used as `Actor::spawn_server_task` by the RDMA actors. The
53/// rdma-runtime handle is returned directly: a tokio `JoinHandle` can be awaited
54/// from any runtime, including the caller's.
55///
56/// Must not be called after `shutdown_data_plane_runtimes` has torn the rdma
57/// runtime down: the cached handle would then point at a dropped runtime and
58/// `spawn` would panic. Safe in practice because the RDMA actors are drained
59/// before that atexit teardown runs.
60pub(crate) fn spawn_on_rdma_runtime<F>(future: F) -> tokio::task::JoinHandle<F::Output>
61where
62    F: std::future::Future + Send + 'static,
63    F::Output: Send + 'static,
64{
65    rdma_runtime().spawn(future)
66}

68#[cfg(test)]
69mod tests {
70    use hyperactor::Actor;
71    use hyperactor::runtime_identity::RuntimeKind;
72    use hyperactor::runtime_identity::current_runtime_kind;
73
74    use super::*;
75    use crate::RdmaManagerActor;
76
77    // RR invariant coverage:
78    // RR-1 (process-singleton): structural (built once via OnceLock).
79    // RR-2 (shared-runtime-routing): rdma_manager_routes_spawn_server_task_to_rdma_runtime;
80    //   the generic actors share the identical override (structural).
81    // RR-3 (rdma-is-data-plane): spawn_on_rdma_runtime_runs_on_data_plane.
82    // RR-4 (registration-off-control-plane): structural (RR-2 + RR-3); the full
83    //   end-to-end assertion is the Diff-4 fitness test (GPU/IB-gated).
84    // RR-5 (cross-runtime-joinhandle): handle_awaitable_from_caller_runtime.
85
86    // RR-3: a future routed through spawn_on_rdma_runtime runs on a DataPlane("rdma") thread.
87    #[test]
88    fn spawn_on_rdma_runtime_runs_on_data_plane() {
89        let (tx, rx) = std::sync::mpsc::channel();
90        let _join = spawn_on_rdma_runtime(async move {
91            let _ = tx.send(current_runtime_kind());
92        });
93        assert_eq!(rx.recv().unwrap(), Some(RuntimeKind::DataPlane("rdma")));
94    }
95
96    // RR-2: RdmaManagerActor's spawn_server_task override routes onto the rdma
97    // runtime (the default tokio::spawn would have no runtime here and panic).
98    #[test]
99    fn rdma_manager_routes_spawn_server_task_to_rdma_runtime() {
100        let (tx, rx) = std::sync::mpsc::channel();
101        let _join = <RdmaManagerActor as Actor>::spawn_server_task(async move {
102            let _ = tx.send(current_runtime_kind());
103        });
104        assert_eq!(rx.recv().unwrap(), Some(RuntimeKind::DataPlane("rdma")));
105    }
106
107    // RR-5: the rdma-runtime JoinHandle is awaitable from a different runtime.
108    #[tokio::test]
109    async fn handle_awaitable_from_caller_runtime() {
110        let out = spawn_on_rdma_runtime(async { 7 }).await.unwrap();
111        assert_eq!(out, 7);
112    }
113}
