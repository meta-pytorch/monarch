9//! ibverbs RDMA domain.
10//!
11//! An [`IbvDomain`] manages the device context and protection domain (PD)
12//! required for RDMA operations. It provides the foundation for creating
13//! queue pairs and establishing connections between RDMA devices.

33/// Manages RDMA resources including context and protection domain.
34///
35/// # Fields
36///
37/// * `device_info`: Metadata for the device this PD is allocated on.
38/// * `domain_impl`: The backend [`IbvDomainImpl`] strategy for this PD. It may
39///   own FFI resources allocated against the PD, so it is declared before `pd`
40///   and thus dropped first — releasing those resources while the PD is still
41///   alive.
42/// * `pd`: The protection domain (and, through [`IbvPd`], the device context).
43///   Held in an `Arc` because the resources built against it — memory regions
44///   and queue pairs — each keep their own clone (via [`Self::pd`]) to hold the
45///   PD open for as long as they need it, independent of this domain.
46///
47/// `I` is the backend [`IbvDomainImpl`] strategy parameterizing per-PD
48/// behavior.
49///
50/// `IbvDomain` is not `Clone`: it owns the backend strategy's FFI resources.
51/// Hand out [`Self::pd`] clones to share the protection domain.
52pub struct IbvDomain<I: IbvDomainImpl> {
53    pub device_info: IbvDeviceInfo,
54    domain_impl: I,
55    pd: Arc<IbvPd>,
56}
57
58impl<I: IbvDomainImpl> std::fmt::Debug for IbvDomain<I> {
59    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
60        f.debug_struct("IbvDomain")
61            .field("context", &format!("{:p}", self.pd.context().as_ptr()))
62            .field("pd", &format!("{:p}", self.pd.as_ptr()))
63            .field("device_info", &self.device_info)
64            .field("domain_impl", &self.domain_impl)
65            .finish()
66    }
67}

69impl<I: IbvDomainImpl> IbvDomain<I> {
70    /// Creates an `IbvDomain` over an already-opened device `context`.
71    ///
72    /// Builds the backend [`IbvDomainImpl`] strategy `I` from `config`, then
73    /// allocates the protection domain against `context`. The PD is allocated
74    /// only *after* the strategy is built, so a panicking
75    /// [`IbvDomainImpl::new`] never leaks a PD.
76    ///
77    /// Note:
78    /// Our memory region (MR) registration uses implicit ODP for RDMA access, which maps large virtual
79    /// address ranges without explicit pinning. This is convenient, but it broadens the memory footprint
80    /// exposed to the NIC and introduces a security liability.
81    ///
82    /// We currently assume a trusted, single-environment and are not enforcing finer-grained memory isolation
83    /// at this layer. We plan to investigate mitigations - such as memory windows or tighter registration
84    /// boundaries in future follow-ups.
85    ///
86    /// # Safety
87    ///
88    /// `context` must wrap a valid, live `ibv_context`: the `Arc<IbvContext>`
89    /// keeps it open for this call and for the returned domain's lifetime (the
90    /// PD allocated here is freed against that context on `Drop`).
91    ///
92    /// # Panics
93    ///
94    /// Panics if `context` is null.
95    ///
96    /// # Errors
97    ///
98    /// Returns an error if protection-domain allocation fails.
99    pub unsafe fn new(
100        context: Arc<IbvContext>,
101        device_info: IbvDeviceInfo,
102        config: &IbvConfig,
103    ) -> Result<Self, anyhow::Error> {
104        assert!(
105            !context.as_ptr().is_null(),
106            "IbvDomain::new requires a non-null ibv_context"
107        );
108        // Build the strategy first; the PD below is allocated only if this
109        // returns, so a panicking `IbvDomainImpl::new` never leaks a PD.
110        // SAFETY: per this function's contract `context` wraps a valid, live
111        // `ibv_context`, which is what `I::new` requires to query the device.
112        let domain_impl = unsafe { I::new(&context, &device_info, config) };
113        // SAFETY: per this function's contract `context` wraps a valid, live
114        // `ibv_context`, which `IbvPd::create` allocates the PD against. The PD
115        // takes ownership of `context`; the domain reaches it via `pd.context()`.
116        let pd = Arc::new(unsafe { IbvPd::create(context) }?);
117        Ok(Self {
118            device_info,
119            domain_impl,
120            pd,
121        })
122    }
123
124    /// Test-only constructor assembling a domain from parts, so unit tests can
125    /// fabricate a domain (typically with a null `pd` whose `Drop` is a no-op).
126    ///
127    /// # Safety
128    ///
129    /// `pd` must satisfy [`IbvPd`]'s validity contract.
130    #[cfg(test)]
131    pub(super) unsafe fn for_test(
132        pd: Arc<IbvPd>,
133        device_info: IbvDeviceInfo,
134        domain_impl: I,
135    ) -> Self {
136        Self {
137            device_info,
138            domain_impl,
139            pd,
140        }
141    }

194/// Per-backend strategy for a protection domain: how memory regions are
195/// registered and how queue pairs are built against the PD.
196///
197/// One strategy is constructed per domain via [`Self::new`], which
198/// inspects the device behind the context to decide backend-specific behavior
199/// up front, and is then stored in the [`IbvDomain`] it drives. The per-op
200/// methods are associated functions taking `&IbvDomain<Self>` and reach the
201/// strategy itself through [`IbvDomain::domain_impl`].
202pub trait IbvDomainImpl: std::fmt::Debug + Send + Sync + 'static + Sized {
203    /// The concrete queue-pair type built against this domain's PD.
204    type QueuePair: IbvQueuePair;
205
206    /// Build the strategy for the device behind `context` (whose queried
207    /// metadata is `device_info`), using `config` for any setup it performs.
208    ///
209    /// # Safety
210    ///
211    /// If `context` is non-null it must wrap a valid, live `ibv_context`;
212    /// implementations may query the device behind it.
213    unsafe fn new(context: &IbvContext, device_info: &IbvDeviceInfo, config: &IbvConfig) -> Self;
214
215    /// Access flags used when registering memory regions and creating queue
216    /// pairs on this domain.
217    fn access_flags(&self) -> i32;
218
219    /// Register `mem` against `domain`'s PD and return a view of the
220    /// resulting memory region.
221    ///
222    /// The default implementation covers host memory (`ibv_reg_mr`) and
223    /// the device-memory dmabuf path (`ibv_reg_dmabuf_mr`); backends
224    /// override to add hardware-specific registration, falling back to
225    /// this default for the cases they do not special-case.
226    ///
227    /// # Safety
228    ///
229    /// `domain`'s PD (`domain.as_ptr()`) must be null or a live protection
230    /// domain; `mem`'s backing memory must stay valid for the returned MR's
231    /// lifetime.
232    unsafe fn register_mr(
233        domain: &IbvDomain<Self>,
234        mem: &KeepaliveLocalMemory,
235    ) -> anyhow::Result<IbvMemoryRegionView> {
236        // SAFETY: `domain.as_ptr()` is null or a live PD (per this method's
237        // contract; `register_host_or_dmabuf_mr` errors on null), and the caller
238        // keeps `mem`'s backing memory valid for the MR's lifetime.
239        unsafe { register_host_or_dmabuf_mr(domain, mem) }
240    }
241
242    /// Create a queue pair against `domain`. The default builds [`Self::QueuePair`]
243    /// directly; backends override to construct their own queue-pair type.
244    fn create_queue_pair(
245        domain: &IbvDomain<Self>,
246        config: &IbvConfig,
247    ) -> anyhow::Result<Self::QueuePair> {
248        // SAFETY: a fully-constructed `IbvDomain` holds a null-or-live PD per
249        // its construction contract, which is what `IbvQueuePair::new` requires.
250        unsafe { Self::QueuePair::new(domain, config.clone()) }
251    }
252}

254/// Register host memory as a standard MR via `ibv_reg_mr`.
255///
256/// # Safety
257///
258/// If `pd` is non-null it must be a live protection domain whose context
259/// outlives this call. `[addr, addr + size)` must name a host mapping that
260/// stays valid for the lifetime of the returned MR.
261pub(super) unsafe fn register_host_mr(
262    pd: &Arc<IbvPd>,
263    addr: usize,
264    size: usize,
265    access_flags: i32,
266) -> anyhow::Result<IbvMr> {
267    if pd.as_ptr().is_null() {
268        anyhow::bail!("register_host_mr called with a null protection domain");
269    }
270    // SAFETY: `pd.as_ptr()` is non-null (checked above) and, per this function's
271    // contract, a live protection domain; `[addr, addr + size)` is a
272    // caller-guaranteed valid mapping. `ibv_reg_mr` returns null on failure,
273    // which we check before wrapping the pointer.
274    let mr =
275        unsafe { rdmaxcel_sys::ibv_reg_mr(pd.as_ptr(), addr as *mut c_void, size, access_flags) };
276    if mr.is_null() {
277        anyhow::bail!("failed to register standard MR");
278    }
279    // SAFETY: `mr` is non-null (checked above) and freshly registered against
280    // `pd`.
281    Ok(unsafe { IbvMr::from_raw(mr, pd.clone()) })
282}

284/// Register exactly `[addr, addr + size)` of device memory as a dmabuf MR via
285/// `ibv_reg_dmabuf_mr`, mapped at iova 0.
286///
287/// `cuMemGetHandleForAddressRange` requires both `addr` and `size` to be
288/// host-page aligned, so this errors if either is not.
289///
290/// # Safety
291///
292/// If `pd` is non-null it must be a live protection domain whose context
293/// outlives this call. `[addr, addr + size)` must name device memory that
294/// stays valid for the lifetime of the returned MR.
295pub(super) unsafe fn register_dmabuf_range(
296    pd: &Arc<IbvPd>,
297    addr: usize,
298    size: usize,
299    access_flags: i32,
300) -> anyhow::Result<IbvMr> {
301    if pd.as_ptr().is_null() {
302        anyhow::bail!("register_dmabuf_range called with a null protection domain");
303    }
304    // SAFETY: `sysconf` reads a process-global parameter and takes no pointers.
305    let page_size = unsafe { libc::sysconf(libc::_SC_PAGESIZE) };
306    anyhow::ensure!(
307        page_size > 0,
308        "sysconf(_SC_PAGESIZE) failed: {}",
309        Error::last_os_error()
310    );
311    let host_page_size = page_size as usize;
312    anyhow::ensure!(
313        addr.is_multiple_of(host_page_size) && size.is_multiple_of(host_page_size),
314        "dmabuf range is not host-page aligned (addr: 0x{addr:x}, size: {size}, page size: {host_page_size})"
315    );
316
317    let mut fd: i32 = -1;
318    // SAFETY: `rdmaxcel_cuMemGetHandleForAddressRange` writes the dmabuf fd for
319    // `[addr, addr + size)` into `fd` and touches no Rust memory beyond the
320    // `&mut fd` out-param; it reports failure via its return code, checked next.
321    let cu_err = unsafe {
322        rdmaxcel_sys::rdmaxcel_cuMemGetHandleForAddressRange(
323            &mut fd,
324            addr as rdmaxcel_sys::CUdeviceptr,
325            size,
326            rdmaxcel_sys::CU_MEM_RANGE_HANDLE_TYPE_DMA_BUF_FD,
327            0,
328        )
329    };
330    if cu_err != rdmaxcel_sys::CUDA_SUCCESS || fd < 0 {
331        anyhow::bail!(
332            "failed to get dmabuf handle for CUDA memory (addr: 0x{:x}, size: {}, cu_err: {}, fd: {})",
333            addr,
334            size,
335            cu_err,
336            fd
337        );
338    }
339    // SAFETY: `fd >= 0` (checked above) is a fresh dmabuf descriptor we
340    // exclusively own; wrapping it in `OwnedFd` closes it on drop and keeps it
341    // open across the registration below.
342    let fd = unsafe { OwnedFd::from_raw_fd(fd) };
343    // SAFETY: `pd.as_ptr()` is a non-null protection domain (checked above)
344    // belonging to a live context; `fd` is a valid dmabuf descriptor kept open
345    // by the `OwnedFd` across this call; `size` matches the range queried above.
346    // `ibv_reg_dmabuf_mr` returns null on failure, which we check.
347    let mr = unsafe {
348        rdmaxcel_sys::ibv_reg_dmabuf_mr(pd.as_ptr(), 0, size, 0, fd.as_raw_fd(), access_flags)
349    };
350    if mr.is_null() {
351        anyhow::bail!("failed to register dmabuf MR");
352    }
353    // SAFETY: `mr` is non-null (checked above) and freshly registered against
354    // `pd`.
355    Ok(unsafe { IbvMr::from_raw(mr, pd.clone()) })
356}

358/// Register the CUDA allocation containing `addr` as a dmabuf MR.
359/// The MR covers the *entire* allocation; the returned `usize` is
360/// the offset of `addr` within it.
361///
362/// The whole-allocation base and size come from `cuMemGetAddressRange`;
363/// [`register_dmabuf_range`] then enforces that both are host-page aligned.
364///
365/// # Safety
366///
367/// If `pd` is non-null it must be a live protection domain whose context
368/// outlives this call. `addr` must belong to a CUDA device allocation
369/// that stays valid for the lifetime of the returned MR.
370pub(super) unsafe fn register_dmabuf_mr(
371    pd: &Arc<IbvPd>,
372    addr: usize,
373    access_flags: i32,
374) -> anyhow::Result<(IbvMr, usize)> {
375    // `cuMemGetAddressRange` resolves the pointer in the *current* CUDA context,
376    // so make the pointer's own device context current first. Without this, in a
377    // multi-GPU process it fails with `CUDA_ERROR_NOT_FOUND` whenever the active
378    // context belongs to a different device than `addr`.
379    // SAFETY: this path is only taken for device memory (`is_device_ptr(addr)`
380    // in `register_host_or_dmabuf_mr`), so `addr` is a valid CUDA device pointer
381    // as `set_ctx_for_ptr` requires. The guard restores the prior context on drop.
382    let _ctx_guard = unsafe { crate::local_memory::set_ctx_for_ptr(addr)? };
383
384    // Resolve the base and size of the allocation containing `addr`; the dmabuf
385    // handle and MR cover the whole allocation.
386    let mut base: rdmaxcel_sys::CUdeviceptr = 0;
387    let mut alloc_size: usize = 0;
388    // SAFETY: `rdmaxcel_cuMemGetAddressRange` writes the allocation's base and
389    // size into the out-params and touches no other Rust memory; it reports
390    // failure via its return code, checked next.
391    let cu_err = unsafe {
392        rdmaxcel_sys::rdmaxcel_cuMemGetAddressRange(
393            &mut base,
394            &mut alloc_size,
395            addr as rdmaxcel_sys::CUdeviceptr,
396        )
397    };
398    if cu_err != rdmaxcel_sys::CUDA_SUCCESS {
399        anyhow::bail!(
400            "failed to get address range for CUDA memory (addr: 0x{:x}, cu_err: {})",
401            addr,
402            cu_err
403        );
404    }
405    let base = base as usize;
406
407    // SAFETY: forwards this function's contract; `register_dmabuf_range` checks
408    // that `base`/`alloc_size` are host-page aligned.
409    let mr = unsafe { register_dmabuf_range(pd, base, alloc_size, access_flags)? };
410    Ok((mr, addr - base))
411}

413/// Default MR registration: host memory via [`register_host_mr`], device
414/// memory via [`register_dmabuf_mr`]. Shared by the [`IbvDomainImpl`]
415/// default `register_mr` and by backends as the fallback for memory they
416/// do not special-case.
417///
418/// # Safety
419///
420/// `domain.as_ptr()` must be null or a live protection domain whose context
421/// outlives this call (a null PD yields an error). `mem`'s
422/// `[addr, addr + size)` must stay valid for the lifetime of the returned
423/// view's MR — the MR keepalive maintains the `ibv_mr` but does not keep the
424/// backing memory mapped.
425pub(super) unsafe fn register_host_or_dmabuf_mr<I: IbvDomainImpl>(
426    domain: &IbvDomain<I>,
427    mem: &KeepaliveLocalMemory,
428) -> anyhow::Result<IbvMemoryRegionView> {
429    let addr = mem.addr();
430    let size = mem.size();
431    let access_flags = domain.access_flags();
432    // `mr_offset` is the offset of `addr` within the MR. For device memory the
433    // MR covers the whole allocation, so the requested range starts partway in;
434    // for host memory the MR is the requested range itself, so the offset is 0.
435    // SAFETY: per this function's contract `domain.as_ptr()` is null or a live
436    // PD (the helpers error on null), and `[addr, addr + size)` stays valid for
437    // the returned MR's lifetime.
438    let (mr, mr_offset) = unsafe {
439        if is_device_ptr(addr) {
440            register_dmabuf_mr(domain.pd(), addr, access_flags)?
441        } else {
442            (register_host_mr(domain.pd(), addr, size, access_flags)?, 0)
443        }
444    };
445
446    // SAFETY: `mr` wraps a non-null, freshly-registered `ibv_mr` —
447    // `register_dmabuf_mr`/`register_host_mr` only return `Ok` with one — so
448    // reading its `addr`/`lkey`/`rkey` here is sound.
449    let (mr_addr, lkey, rkey) = unsafe {
450        let p = mr.as_ptr();
451        ((*p).addr as usize, (*p).lkey, (*p).rkey)
452    };
453    // The view addresses the requested sub-range, which sits at `mr_offset`
454    // within the MR's zero-based address space.
455    let rdma_addr = mr_addr + mr_offset;
456    let device_name = domain.device_info().name().to_string();
457    // The `IbvMr` guard owns the MR (deregistered on its `Drop`) and anchors the
458    // PD past that deregistration; it coerces to `Arc<dyn IbvMemoryRegionKeepalive>`.
459    let guard = Arc::new(mr);
460    Ok(IbvMemoryRegionView::new(
461        addr,
462        rdma_addr,
463        size,
464        lkey,
465        rkey,
466        device_name,
467        guard,
468    ))
469}

486    /// Open an [`IbvDomain`] on the NIC mapped to CUDA `device`. These tests
487    /// require a GPU with a mapped RDMA NIC, so a missing device, missing NIC, or
488    /// open/creation failure panics. The returned domain owns its context, so it
489    /// (and its PD) stays valid after the local [`IbvDevice`] drops.
490    fn open_domain_for_cuda_device(device: i32) -> IbvDomain<MlxDomain> {
491        let nic = get_cuda_device_to_ibv_device::<MlxDevice>()
492            .get(device as usize)
493            .and_then(|nic| nic.as_ref())
494            .expect("CUDA device should map to RDMA NIC")
495            .name()
496            .clone();
497        let mut config = IbvConfig::default();
498        MlxDevice::apply_config_defaults(&mut config);
499        let dev =
500            IbvDevice::<MlxDevice>::open(&nic, config.clone()).expect("mapped NIC should open");
501        // SAFETY: `dev.context()` wraps the live `ibv_context` opened above; the
502        // returned `Arc<IbvContext>` keeps it open for the new domain's lifetime.
503        unsafe { IbvDomain::new(dev.context(), dev.device_info().clone(), &config) }
504            .expect("domain creation should succeed")
505    }

507    /// A 4 MiB allocation, fully committed (reserved == committed) so it is
508    /// entirely mapped and `cuMemGetAddressRange` reports the whole extent.
509    fn committed_allocation() -> crate::backend::cuda_test_utils::CudaAllocation {
510        CudaAllocator::get().allocate(0, 4 * 1024 * 1024, 4 * 1024 * 1024)
511    }
512
513    /// `addr` (iova) and `length` of an `ibv_mr`.
514    ///
515    /// # Safety
516    ///
517    /// `mr` must be a live MR (e.g. owned by a not-yet-dropped [`IbvMr`]).
518    unsafe fn mr_extent(mr: *mut rdmaxcel_sys::ibv_mr) -> (usize, usize) {
519        // SAFETY: per this function's contract `mr` is a live MR.
520        unsafe { ((*mr).addr as usize, (*mr).length) }
521    }
522
523    // `register_dmabuf_mr` registers the whole enclosing allocation and reports
524    // the requested address's offset within it — even when that address is not
525    // host-page aligned, which is the case the bug fix targets.
526    #[test]
527    fn register_dmabuf_mr_covers_whole_allocation() {
528        let domain = open_domain_for_cuda_device(0);
529        let pd = domain.pd();
530        let access = domain.access_flags();
531        let alloc = committed_allocation();
532        let alloc_size = alloc.size();
533
534        // At the allocation base: offset 0, MR spans the whole allocation at iova
535        // 0. `mr` deregisters on drop, after the assertions below.
536        // SAFETY: `pd` is a live PD; `alloc.ptr()` is a live CUDA allocation kept
537        // mapped by `alloc` for the MR's lifetime.
538        let (mr, offset) = unsafe { register_dmabuf_mr(pd, alloc.ptr(), access) }.unwrap();
539        // SAFETY: `mr` owns a live MR and has not been dropped.
540        let (iova, length) = unsafe { mr_extent(mr.as_ptr()) };
541        assert_eq!(offset, 0, "base address sits at offset 0");
542        assert_eq!(iova, 0, "dmabuf MR is mapped at iova 0");
543        assert_eq!(length, alloc_size, "MR covers the whole allocation");
544
545        // At an unaligned interior address: the MR still spans the whole
546        // allocation and the offset locates the requested address.
547        let unaligned: usize = 257;
548        assert!(!unaligned.is_multiple_of(host_page_size()));

633    // The full `register_host_or_dmabuf_mr` path over device memory covering the
634    // whole allocation: the view reports the allocation's address and size, and
635    // an `rdma_addr` of 0 (the request sits at the MR's base).
636    #[test]
637    fn register_host_or_dmabuf_mr_covers_whole_allocation() {
638        let domain = open_domain_for_cuda_device(0);
639        let alloc = committed_allocation();
640        let mem = alloc.keepalive_slice(0, alloc.size());
641
642        // SAFETY: `domain`'s PD is live; `mem` keeps the allocation mapped for
643        // the view's MR lifetime.
644        let view = unsafe { register_host_or_dmabuf_mr(&domain, &mem) }.unwrap();
645        // Tie the view's lifetime to the allocation's lifetime so that the safety contract
646        // above holds.
647        mem.mr_slot()
648            .set(view.clone())
649            .expect("mr_slot not already set");
650        assert_eq!(view.virtual_addr, alloc.ptr());
651        assert_eq!(view.size, alloc.size());
652        assert_eq!(view.rdma_addr, 0, "whole allocation starts at MR offset 0");
653    }
