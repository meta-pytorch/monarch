44/// Server-side link that receives pre-established streams from the
45/// dispatcher via an unbounded mpsc channel. Implements [`Link`] so
46/// it can be used with [`Session::new`]. Dropping the link drops
47/// the receiver, which causes any feeder's later `send` to fail
48/// with `SendError` (the conn is dropped). The channel is unbounded
49/// so the dispatch task can publish its sender to the session map
50/// before draining the first connection without risking a deadlock
51/// against a concurrent feeder for the same session_id.
52pub(super) struct AcceptorLink<S: Stream> {
53    pub(super) dest: ChannelAddr,
54    pub(super) session_id: SessionId,
55    pub(super) stream: mpsc::UnboundedReceiver<S>,
56    pub(super) cancel: CancellationToken,
57}
58
59impl<S: Stream> fmt::Debug for AcceptorLink<S> {
60    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
61        f.debug_struct("AcceptorLink")
62            .field("dest", &self.dest)
63            .field("session_id", &self.session_id)
64            .finish()
65    }
66}

68#[async_trait]
69impl<S: Stream> Link for AcceptorLink<S> {
70    type Stream = S;
71
72    fn dest(&self) -> ChannelAddr {
73        self.dest.clone()
74    }
75
76    fn link_id(&self) -> SessionId {
77        self.session_id
78    }
79
80    async fn next(&mut self) -> Result<S, ClientError> {
81        let dest = self.dest.clone();
82        let make_err = move || {
83            ClientError::Connect(
84                dest.clone(),
85                std::io::Error::other("acceptor closed"),
86                "acceptor channel closed".into(),
87            )
88        };
89        tokio::select! {
90            result = self.stream.recv() => result.ok_or_else(&make_err),
91            _ = self.cancel.cancelled() => Err(make_err()),
92        }
93    }
94}

96/// Shared state for multi-stream sessions. Each additional stream
97/// (stream_id > 0) gets its own connection-handoff channel. The
98/// [`AckWatermark`] is shared across all streams so cumulative acks
99/// reflect the global sequence space.
100pub(super) struct StreamState<S: Stream> {
101    /// Per-stream connection-handoff senders. Keyed by stream_id.
102    /// Accessed concurrently from multiple accept-loop dispatch tasks
103    /// (one per incoming connection for the same session), but only
104    /// for a brief `entry().or_insert_with()` — a plain std Mutex is
105    /// cheaper than DashMap here.
106    streams: std::sync::Mutex<HashMap<u8, mpsc::UnboundedSender<S>>>,
107    /// Shared ack tracker: each stream records its received seqs here.
108    ack_watermark: tokio::sync::Mutex<session::AckWatermark>,
109}
110
111impl<S: Stream> StreamState<S> {
112    pub(super) fn new() -> Self {
113        let ack_msg_interval: u64 =
114            hyperactor_config::global::get(config::MESSAGE_ACK_EVERY_N_MESSAGES);
115        let ack_time_interval = hyperactor_config::global::get(config::MESSAGE_ACK_TIME_INTERVAL);
116        Self {
117            streams: std::sync::Mutex::new(HashMap::new()),
118            ack_watermark: tokio::sync::Mutex::new(session::AckWatermark::new(
119                ack_msg_interval,
120                ack_time_interval,
121            )),
122        }
123    }
124}

126/// If `link_init` is for a multi-stream connection (`stream_id > 0`),
127/// resolve (or create) the shared per-session state and return
128/// `Some((stream_id, state))`. Otherwise return `None`.
129fn resolve_stream<S: Stream>(
130    stream_state: &std::sync::Mutex<HashMap<SessionId, Arc<StreamState<S>>>>,
131    link_init: &super::LinkInit,
132) -> Option<(u8, Arc<StreamState<S>>)> {
133    if link_init.stream_id == 0 {
134        return None;
135    }
136    let state = stream_state
137        .lock()
138        .unwrap()
139        .entry(link_init.session_id)
140        .or_insert_with(|| Arc::new(StreamState::new()))
141        .clone();
142    Some((link_init.stream_id, state))
143}

145/// Dispatch a newly accepted connection to the right session.
146///
147/// `streams = None` is a single-stream (strictly in-order) session;
148/// `Some((stream_id, state))` is the `stream_id`-th stream of a
149/// multi-stream session sharing `state`.
150///
151/// Structured concurrency: the first dispatch for a session runs the
152/// recv-loop inline and only returns after its terminal cleanup
153/// (flush any pending ack, emit `Closed` on cancellation).
154/// Reconnects hand the connection off via the per-session
155/// channel and return immediately. [`accept_loop`] joins every
156/// dispatch in its `connections` `JoinSet`, so it finishes only
157/// after every recv-loop has finished.
158pub(super) async fn dispatch_stream<M: RemoteMessage, S: Stream>(
159    session_id: SessionId,
160    streams: Option<(u8, Arc<StreamState<S>>)>,
161    conn: S,
162    sessions: &DashMap<SessionId, mpsc::UnboundedSender<S>>,
163    dest: ChannelAddr,
164    tx: mpsc::Sender<M>,
165    cancel: CancellationToken,
166) {
167    if let Some((stream_id, state)) = streams {
168        // Multi-stream: route to the per-stream dispatch handler.
169        dispatch_multi_stream::<M, S>(session_id, stream_id, conn, state, dest, tx, cancel).await;
170        return;
171    }
172
173    // Single-stream: insert into the session map and drop the
174    // DashMap shard guard before any await. Vacant inserts the
175    // sender side; occupied returns the existing sender so this
176    // dispatch acts as a feeder. Unbounded so this task can publish
177    // the sender before draining the first conn without deadlocking
178    // a concurrent feeder for the same session_id.
179    let entry_result = {
180        let entry = sessions.entry(session_id);
181        match entry {
182            dashmap::mapref::entry::Entry::Occupied(e) => Err(e.get().clone()),
183            dashmap::mapref::entry::Entry::Vacant(e) => {
184                let (sender, receiver) = mpsc::unbounded_channel::<S>();
185                e.insert(sender.clone());
186                Ok((sender, receiver))
187            }
188        }
189    };
190
191    let (sender, receiver) = match entry_result {
192        Err(sender) => {
193            // Feeder: forward the conn through the existing channel.
194            // Send returns Err if the processor task exited and
195            // dropped the receiver — drop the conn in that case.
196            let _ = sender.send(conn);
197            return;
198        }
199        Ok(pair) => pair,
200    };
201
202    let link = AcceptorLink {
203        dest: dest.clone(),
204        session_id,
205        stream: receiver,
206        cancel: cancel.clone(),
207    };
208    let deliver_tx = tx;
209    let ct = cancel;
210
211    // Hand the first connection off through the channel; the
212    // recv-loop below picks it up via `session.connect().await`.
213    let _ = sender.send(conn);
214    drop(sender);
215
216    let mut session = Session::new(link);
217    let mut next = Next { seq: 0, ack: 0 };
218
219    loop {
220        let connected = match session.connect().await {
221            Ok(s) => s,
222            Err(_) => break,
223        };
224
225        let result = {
226            let conn = connected.stream(super::INITIATOR_TO_ACCEPTOR);
227            tokio::select! {
228                r = session::recv_connected::<M, _, _>(
229                    &conn,
230                    &deliver_tx,
231                    &mut next,
232                ) => r,
233                _ = ct.cancelled() => Err(session::RecvLoopError::Cancelled),
234            }
235        };
236
237        // Flush a final ack if behind.
238        if next.ack < next.seq {
239            let ack = super::serialize_response(super::NetRxResponse::Ack(next.seq - 1)).unwrap();
240            let conn = connected.stream(super::INITIATOR_TO_ACCEPTOR);
241            let mut completion = conn.write(ack);
242            match completion.drive().await {
243                Ok(()) => {
244                    next.ack = next.seq;
245                }
246                Err(e) => {
247                    tracing::debug!(
248                        error = %e,
249                        "failed to flush acks during cleanup"
250                    );
251                }
252            }
253        }
254
255        // Send reject or closed response if appropriate.
256        let terminal_response = match &result {
257            Err(session::RecvLoopError::SequenceError(reason)) => {
258                Some(super::NetRxResponse::Reject(reason.clone()))
259            }
260            Err(session::RecvLoopError::Cancelled) => Some(super::NetRxResponse::Closed),
261            _ => None,
262        };
263        if let Some(rsp) = terminal_response {
264            let data = super::serialize_response(rsp).unwrap();
265            let conn = connected.stream(super::INITIATOR_TO_ACCEPTOR);
266            let mut completion = conn.write(data);
267            let _ = completion.drive().await;
268        }
269
270        let recoverable = matches!(&result, Ok(()) | Err(session::RecvLoopError::Io(_)));
271
272        match &result {
273            Ok(()) => {
274                tracing::info!(
275                    %dest,
276                    %session_id,
277                    "recv_connected returned EOF, awaiting reconnect"
278                );
279            }
280            Err(e) => {
281                tracing::info!(
282                    %dest,
283                    %session_id,
284                    error = %e,
285                    recoverable,
286                    "recv_connected returned error"
287                );
288            }
289        }
290
291        session = connected.release();
292
293        if recoverable {
294            continue;
295        }
296        break;
297    }
298
299    // Recv-loop is finished — no further conns will be drained.
300    // Drop the session entry so a later reconnect for the same
301    // session_id starts a fresh dispatch task instead of feeding a
302    // dead channel; any in-flight feeder's send fails after the
303    // receiver above is dropped.
304    sessions.remove(&session_id);
305}

307/// Handle a multi-stream connection (stream_id > 0). All streams of
308/// a session share an `AckWatermark` for cumulative out-of-order
309/// acking. Same structured-concurrency contract as
310/// [`dispatch_stream`]: first dispatch for a given (`session_id`,
311/// `stream_id`) runs the recv-loop inline; reconnects hand off via
312/// the per-stream channel and return.
313async fn dispatch_multi_stream<M: RemoteMessage, S: Stream>(
314    session_id: SessionId,
315    stream_id: u8,
316    conn: S,
317    state: Arc<StreamState<S>>,
318    dest: ChannelAddr,
319    tx: mpsc::Sender<M>,
320    cancel: CancellationToken,
321) {
322    tracing::info!(
323        %dest, %session_id, stream_id,
324        "dispatch_multi_stream: new connection"
325    );
326
327    // Insert into the per-session map and drop the std::sync::Mutex
328    // guard before any await — its guard is !Send. Vacant inserts
329    // the sender side; occupied returns the existing sender so this
330    // dispatch acts as a feeder. Unbounded so this task can publish
331    // the sender before draining the first conn without deadlocking
332    // a concurrent feeder for the same (session_id, stream_id).
333    let entry_result = {
334        use std::collections::hash_map::Entry;
335        let mut streams = state.streams.lock().unwrap();
336        match streams.entry(stream_id) {
337            Entry::Occupied(e) => Err(e.get().clone()),
338            Entry::Vacant(e) => {
339                let (sender, receiver) = mpsc::unbounded_channel::<S>();
340                e.insert(sender.clone());
341                Ok((sender, receiver))
342            }
343        }
344    };
345
346    let (sender, receiver) = match entry_result {
347        Err(sender) => {
348            // Feeder: forward the conn through the existing channel.
349            // Send returns Err if the processor task exited and
350            // dropped the receiver — drop the conn in that case.
351            let _ = sender.send(conn);
352            return;
353        }
354        Ok(pair) => pair,
355    };
356
357    let link = AcceptorLink {
358        dest: dest.clone(),
359        session_id,
360        stream: receiver,
361        cancel: cancel.clone(),
362    };
363    let deliver_tx = tx;
364    let ct = cancel;
365    let shared_state = state;
366
367    // Hand the first connection off through the channel; the
368    // recv-loop below picks it up via `session.connect().await`.
369    let _ = sender.send(conn);
370    drop(sender);
371
372    let mut session = Session::new(link);
373
374    loop {
375        let connected = match session.connect().await {
376            Ok(s) => s,
377            Err(_) => break,
378        };
379
380        let result = {
381            let conn = connected.stream(super::INITIATOR_TO_ACCEPTOR);
382            tokio::select! {
383                r = session::multi_stream_recv_connected::<M, _, _>(
384                    &conn,
385                    &shared_state.ack_watermark,
386                    &deliver_tx,
387                ) => r,
388                _ = ct.cancelled() => Err(session::RecvLoopError::Cancelled),
389            }
390        };
391
392        // Each stream emits the cumulative watermark on its own wire
393        // so the peer's per-wire NetTx sees an ack for messages it
394        // sent on this connection.
395        let pending_ack = shared_state
396            .ack_watermark
397            .lock()
398            .await
399            .highest_uncommitted();
400        if let Some(ack_val) = pending_ack {
401            let ack = super::serialize_response(super::NetRxResponse::Ack(ack_val)).unwrap();
402            let conn = connected.stream(super::INITIATOR_TO_ACCEPTOR);
403            let mut completion = conn.write(ack);
404            match completion.drive().await {
405                Ok(()) => {
406                    shared_state.ack_watermark.lock().await.commit(ack_val);
407                }
408                Err(e) => {
409                    tracing::debug!(
410                        error = %e,
411                        stream_id,
412                        "failed to flush multi-stream ack during cleanup"
413                    );
414                }
415            }
416        }
417
418        // Send reject or closed response if appropriate.
419        let terminal_response = match &result {
420            Err(session::RecvLoopError::SequenceError(reason)) => {
421                Some(super::NetRxResponse::Reject(reason.clone()))
422            }
423            Err(session::RecvLoopError::Cancelled) => Some(super::NetRxResponse::Closed),
424            _ => None,
425        };
426        if let Some(rsp) = terminal_response {
427            let data = super::serialize_response(rsp).unwrap();
428            let conn = connected.stream(super::INITIATOR_TO_ACCEPTOR);
429            let mut completion = conn.write(data);
430            let _ = completion.drive().await;
431        }
432
433        let recoverable = matches!(&result, Ok(()) | Err(session::RecvLoopError::Io(_)));
434
435        match &result {
436            Ok(()) => {
437                tracing::info!(
438                    %dest,
439                    %session_id,
440                    stream_id,
441                    "multi-stream recv EOF, awaiting reconnect"
442                );
443            }
444            Err(e) => {
445                tracing::info!(
446                    %dest,
447                    %session_id,
448                    stream_id,
449                    error = %e,
450                    recoverable,
451                    "multi-stream recv error"
452                );
453            }
454        }
455
456        session = connected.release();
457
458        if recoverable {
459            continue;
460        }
461        break;
462    }
463
464    // Recv-loop is finished — no further conns will be drained for
465    // this (session_id, stream_id). Drop the per-stream sender entry
466    // so a later reconnect starts a fresh dispatch task instead of
467    // feeding a dead channel.
468    shared_state.streams.lock().unwrap().remove(&stream_id);
469}

471/// Generic accept loop. Accepts connections from `listener`, transforms
472/// each via `prepare` (which may do TLS negotiation), then hands them
473/// to `dispatch`.
474pub(super) async fn accept_loop<S, L, F, Fut, D, DFut>(
475    listener: &mut L,
476    listener_addr: &ChannelAddr,
477    parent_cancel: &CancellationToken,
478    prepare: F,
479    dispatch: D,
480) -> Result<(), ServerError>
481where
482    S: Stream,
483    L: super::Listener,
484    F: Fn(L::Stream, ChannelAddr) -> Fut + Clone + Send + 'static,
485    Fut: Future<Output = Result<(super::LinkInit, S), anyhow::Error>> + Send + 'static,
486    D: Fn(super::LinkInit, S) -> DFut + Clone + Send + 'static,
487    DFut: Future<Output = ()> + Send + 'static,
488{
489    let mut connections: JoinSet<Result<(), anyhow::Error>> = JoinSet::new();
490
491    let heartbeat_interval = hyperactor_config::global::get(config::SERVER_HEARTBEAT_INTERVAL);
492    let mut heartbeat_timer: Interval = tokio::time::interval(heartbeat_interval);
493
494    let result: Result<(), ServerError> = loop {
495        tokio::select! {
496            result = listener.accept() => {
497                match result {
498                    Ok((stream, source)) => {
499                        tracing::debug!(
500                            source = %source,
501                            dest = %listener_addr,
502                            "new connection accepted"
503                        );
504                        metrics::CHANNEL_CONNECTIONS.add(
505                            1,
506                            hyperactor_telemetry::kv_pairs!(
507                                "transport" => listener_addr.transport().to_string(),
508                                "operation" => "accept"
509                            ),
510                        );
511
512                        let prepare = prepare.clone();
513                        let dispatch = dispatch.clone();
514                        connections.spawn(async move {
515                            let (link_init, stream) = prepare(stream, source).await?;
516                            dispatch(link_init, stream).await;
517                            Ok(())
518                        });
519                    }
520                    Err(err) => {
521                        metrics::CHANNEL_ERRORS.add(
522                            1,
523                            hyperactor_telemetry::kv_pairs!(
524                                "transport" => listener_addr.transport().to_string(),
525                                "operation" => "accept",
526                                "error" => err.to_string(),
527                                "error_type" => metrics::ChannelErrorType::ConnectionError.as_str(),
528                            ),
529                        );
530                        tracing::info!(
531                            dest = %listener_addr,
532                            error = %err,
533                            "accept error"
534                        );
535                    }
536                }
537            }
538
539            _ = heartbeat_timer.tick() => {
540                metrics::SERVER_HEARTBEAT.add(
541                    1,
542                    hyperactor_telemetry::kv_pairs!(
543                        "dest" => listener_addr.to_string()
544                    ),
545                );
546            }
547
548            _ = parent_cancel.cancelled() => {
549                tracing::info!(
550                    dest = %listener_addr,
551                    "received parent token cancellation"
552                );
553                break Ok(());
554            }
555
556            result = session::join_nonempty(&mut connections) => {
557                if let Err(err) = result {
558                    tracing::info!(
559                        dest = %listener_addr,
560                        error = %err,
561                        "connection task join error"
562                    );
563                }
564            }
565        }
566    };
567
568    while connections.join_next().await.is_some() {}
569    result
570}

572/// Handle to an underlying server that is actively listening.
573#[derive(Debug)]
574pub struct ServerHandle {
575    /// Join handle of the listening task.
576    join_handle: JoinHandle<Result<(), ServerError>>,
577
578    /// A cancellation token used to indicate that a stop has been
579    /// initiated.
580    cancel_token: CancellationToken,
581
582    /// Address of the channel that was used to create the listener. This can be used to dial the that listener.
583    channel_addr: ChannelAddr,
584}
585
586impl fmt::Display for ServerHandle {
587    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
588        write!(f, "{}", self.channel_addr)
589    }
590}
591
592impl ServerHandle {
593    /// Create a new server handle.
594    pub(super) fn new(
595        join_handle: JoinHandle<Result<(), ServerError>>,
596        cancel_token: CancellationToken,
597        channel_addr: ChannelAddr,
598    ) -> Self {
599        Self {
600            join_handle,
601            cancel_token,
602            channel_addr,
603        }
604    }
605
606    /// Signal the server to stop. This will stop accepting new
607    /// incoming connection requests, and drain pending operations
608    /// on active connections. After draining is completed, the
609    /// connections are closed.
610    pub(crate) fn stop(&self, reason: &str) {
611        tracing::info!(
612            name = "ChannelServerStatus",
613            dest = %self.channel_addr,
614            status = "Stop::Sent",
615            reason,
616            "sent Stop signal; check server logs for the stop progress"
617        );
618        self.cancel_token.cancel();
619    }
620}
621
622impl Future for ServerHandle {
623    type Output = <JoinHandle<Result<(), ServerError>> as Future>::Output;
624
625    fn poll(self: Pin<&mut Self>, cx: &mut std::task::Context<'_>) -> Poll<Self::Output> {
626        // SAFETY: This is safe to do because self is pinned.
627        let join_handle_pinned =
628            unsafe { self.map_unchecked_mut(|container| &mut container.join_handle) };
629        join_handle_pinned.poll(cx)
630    }
631}

633/// Serve new connections on the given address, optionally using a pre-opened TCP listener.
634/// When `prebound_listener` is `Some`, it is used instead of binding a new socket.
635/// This is only supported for TCP-based transports (Tcp, Tls, MetaTls).
636pub(in crate::channel) fn serve<M: RemoteMessage>(
637    addr: ChannelAddr,
638    prebound_listener: Option<std::net::TcpListener>,
639) -> Result<(ChannelAddr, NetRx<M>), ServerError> {
640    let (mut listener, channel_addr) = super::listen_with_prebound(addr, prebound_listener)?;
641
642    metrics::CHANNEL_CONNECTIONS.add(
643        1,
644        hyperactor_telemetry::kv_pairs!(
645            "transport" => channel_addr.transport().to_string(),
646            "operation" => "serve"
647        ),
648    );
649
650    let (tx, rx) = mpsc::channel::<M>(1024);
651    let cancel_token = CancellationToken::new();
652    let child_token = cancel_token.child_token();
653
654    let is_tls = matches!(
655        channel_addr.transport(),
656        ChannelTransport::Tls | ChannelTransport::MetaTls(_)
657    );
658    let dest = channel_addr.clone();
659    let prepare = move |stream: Box<dyn Stream>, source: ChannelAddr| {
660        let dest = dest.clone();
661        async move {
662            if is_tls {
663                let tls_acceptor = match dest.transport() {
664                    ChannelTransport::Tls => tls::tls_acceptor()?,
665                    _ => meta::tls_acceptor(true)?,
666                };
667                let mut tls_stream = tls_acceptor.accept(stream).await?;
668                let link_init = read_link_init(&mut tls_stream)
669                    .await
670                    .map_err(|e| anyhow::anyhow!("LinkInit read failed from {}: {}", source, e))?;
671                Ok((link_init, Box::new(tls_stream) as Box<dyn Stream>))
672            } else {
673                let mut stream = stream;
674                let link_init = read_link_init(&mut stream)
675                    .await
676                    .map_err(|e| anyhow::anyhow!("LinkInit read failed from {}: {}", source, e))?;
677                Ok((link_init, stream))
678            }
679        }
680    };
681
682    let sessions: Arc<DashMap<SessionId, mpsc::UnboundedSender<Box<dyn Stream>>>> =
683        Arc::new(DashMap::new());
684    let stream_state: Arc<std::sync::Mutex<HashMap<SessionId, Arc<StreamState<Box<dyn Stream>>>>>> =
685        Arc::new(std::sync::Mutex::new(HashMap::new()));
686    let dispatch_dest = channel_addr.clone();
687    let dispatch_cancel = child_token.clone();
688    let dispatch = {
689        let sessions = Arc::clone(&sessions);
690        let stream_state = Arc::clone(&stream_state);
691        let tx = tx.clone();
692        let dest = dispatch_dest;
693        move |link_init: super::LinkInit, stream: Box<dyn Stream>| {
694            let sessions = Arc::clone(&sessions);
695            let stream_state = Arc::clone(&stream_state);
696            let tx = tx.clone();
697            let cancel = dispatch_cancel.clone();
698            let dest = dest.clone();
699            async move {
700                let streams = resolve_stream(&stream_state, &link_init);
701                dispatch_stream(
702                    link_init.session_id,
703                    streams,
704                    stream,
705                    &sessions,
706                    dest,
707                    tx,
708                    cancel,
709                )
710                .await;
711            }
712        }
713    };
714
715    let ca: ChannelAddr = channel_addr.clone();
716    let join_handle = tokio::spawn(async move {
717        accept_loop(&mut listener, &ca, &child_token, prepare, dispatch).await
718    });
719

724    };
725

728        NetRx(rx, channel_addr, server_handle),
729    ))
730}

732/// Test-only variant that accepts an arbitrary `Listener`. Used by
733/// mock-link tests that cannot go through `net::listen()`.
734#[cfg(test)]
735pub(super) fn serve_with_listener<M, L>(
736    mut listener: L,
737    channel_addr: ChannelAddr,
738) -> Result<(ChannelAddr, NetRx<M>), ServerError>
739where
740    M: RemoteMessage,
741    L: super::Listener + 'static,
742    L::Stream: Unpin + std::fmt::Debug + 'static,
743{
744    let (tx, rx) = mpsc::channel::<M>(1024);
745    let cancel_token = CancellationToken::new();
746    let child_token = cancel_token.child_token();
747
748    let prepare = |mut stream: L::Stream, source: ChannelAddr| async move {
749        let link_init = read_link_init(&mut stream)
750            .await
751            .map_err(|e| anyhow::anyhow!("LinkInit read failed from {}: {}", source, e))?;
752        Ok((link_init, stream))
753    };
754
755    let sessions: Arc<DashMap<SessionId, mpsc::UnboundedSender<L::Stream>>> =
756        Arc::new(DashMap::new());
757    let stream_state: Arc<std::sync::Mutex<HashMap<SessionId, Arc<StreamState<L::Stream>>>>> =
758        Arc::new(std::sync::Mutex::new(HashMap::new()));
759    let dispatch_cancel = child_token.clone();
760    let dispatch = {
761        let sessions = Arc::clone(&sessions);
762        let stream_state = Arc::clone(&stream_state);
763        let tx = tx.clone();
764        let dest = channel_addr.clone();
765        move |link_init: super::LinkInit, stream: L::Stream| {
766            let sessions = Arc::clone(&sessions);
767            let stream_state = Arc::clone(&stream_state);
768            let tx = tx.clone();
769            let cancel = dispatch_cancel.clone();
770            let dest = dest.clone();
771            async move {
772                let streams = resolve_stream(&stream_state, &link_init);
773                dispatch_stream(
774                    link_init.session_id,
775                    streams,
776                    stream,
777                    &sessions,
778                    dest,
779                    tx,
780                    cancel,
781                )
782                .await;
783            }
784        }
785    };
786
787    let ca = channel_addr.clone();
788    let join_handle = tokio::spawn(async move {
789        accept_loop(&mut listener, &ca, &child_token, prepare, dispatch).await
790    });
791
792    let server_handle = ServerHandle {
793        join_handle,
794        cancel_token,
795        channel_addr: channel_addr.clone(),
796    };

798    Ok((
799        server_handle.channel_addr.clone(),
