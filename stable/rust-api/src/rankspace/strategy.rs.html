9//! Property-based generators for [`RankRect`] and related types.
10//!
11//! These strategies are test infrastructure for code that reasons about
12//! rank-space geometry. They generate dense and row-major-derived affine
13//! [`RankRect`]s, including selected sub-rectangles whose base ranks remain in
14//! the same coordinate frame as their parent.
15//!
16//! The public API is intentionally small. It covers the shapes downstream
17//! tiling/routing tests need today and leaves richer generators, such as
18//! column-major/permuted strides or sparse [`crate::RankSpace`] values, as
19//! follow-up work.
20//!
21//! # Example
22//!
23//! ```ignore
24//! use proptest::prelude::*;
25//! use rankspace::strategy::gen_rank_rect_strided;
26//!
27//! proptest! {
28//!     #[test]
29//!     fn my_tiler_respects_affine_frame(
30//!         rect in gen_rank_rect_strided(1..=4, 4, 3, 32),
31//!     ) {
32//!         // assertions on rect ...
33//!     }
34//! }
35//! ```

49/// Generates a random [`Extent`] with dimensionality in `dims` and
50/// per-dimension sizes in `1..=max_size`.
51///
52/// Dimension names are generated as `d0, d1, ...`.
53pub fn gen_extent(dims: RangeInclusive<usize>, max_size: usize) -> impl Strategy<Value = Extent> {
54    prop::collection::vec(1..=max_size, dims).prop_map(|sizes| {
55        Extent::new(
56            sizes
57                .into_iter()
58                .enumerate()
59                .map(|(index, size)| Dim::new(format!("d{index}"), size))
60                .collect(),
61        )
62        .expect("generated dimensions are valid")
63    })
64}

66/// Generates a random dense row-major [`RankRect`].
67///
68/// The returned rectangle has base offset zero and row-major strides. Use
69/// [`gen_rank_rect_strided`] for affine rectangles with non-trivial offsets
70/// or strides.
71pub fn gen_rank_rect(
72    dims: RangeInclusive<usize>,
73    max_size: usize,
74) -> impl Strategy<Value = RankRect> {
75    gen_extent(dims, max_size).prop_map(|extent| {
76        RankRect::new(extent).expect("dense rect built from a generated extent is valid")
77    })
78}

80/// Generates a row-major-derived affine [`RankRect`].
81///
82/// The generated rectangle starts from a row-major parent with dimensionality
83/// in `dims`, per-dimension sizes in `1..=max_size`, and parent base offset in
84/// `0..=max_base_offset`. It then applies one selection per dimension. Each
85/// selection may move the output offset and may multiply that dimension's
86/// stride by a step in `1..=max_step`.
87///
88/// This is the rankspace replacement for `ndslice::strategy::gen_region_strided`.
89/// It is the load-bearing generator for property tests that need to prove
90/// affine-frame preservation — for example, that tile decomposition
91/// children stay in the same base-rank frame as their parent.
92///
93/// TODO: column-major and permuted-stride coverage are not produced by this
94/// generator. Add a dedicated `gen_rank_rect_permuted` if downstream tests
95/// need those shapes.
96pub fn gen_rank_rect_strided(
97    dims: RangeInclusive<usize>,
98    max_size: usize,
99    max_step: usize,
100    max_base_offset: usize,
101) -> impl Strategy<Value = RankRect> {
102    gen_rank_rect_with_selections(dims, max_size, max_step, max_base_offset)
103        .prop_map(|(_parent, _selections, subrect)| subrect)
104}

106/// Generates a `(parent, subrect)` pair in that order.
107///
108/// `parent` is a row-major [`RankRect`] with possible non-zero base offset.
109/// `subrect` is produced from `parent` by applying one [`RankRect::select`]
110/// per dimension with random `(begin, step)` choices. The pair satisfies
111/// `parent.contains_rect(&subrect)`, and `subrect`'s ranks are still base ranks
112/// in `parent`'s coordinate frame.
113///
114/// Use this when the test needs both endpoints — for example, to assert
115/// containment of the selected subrect within the parent. For tests that
116/// only need the strided child, prefer [`gen_rank_rect_strided`].
117pub fn gen_rank_rect_and_strided_subrect(
118    dims: RangeInclusive<usize>,
119    max_size: usize,
120    max_step: usize,
121    max_base_offset: usize,
122) -> impl Strategy<Value = (RankRect, RankRect)> {
123    gen_rank_rect_with_selections(dims, max_size, max_step, max_base_offset)
124        .prop_map(|(parent, _selections, subrect)| (parent, subrect))
125}

127/// Internal generator that also returns the per-dim `(begin, step)` selections
128/// used to derive the subrect from the parent.
129///
130/// Strategy tests use the selections to project subrect coordinates back into
131/// parent coordinates and prove base-rank-frame preservation. The public API
132/// intentionally hides them: downstream tests need generated shapes, not the
133/// generator's construction trace.
134fn gen_rank_rect_with_selections(
135    dims: RangeInclusive<usize>,
136    max_size: usize,
137    max_step: usize,
138    max_base_offset: usize,
139) -> impl Strategy<Value = (RankRect, Vec<(usize, usize)>, RankRect)> {
140    let max_step = max_step.max(1);
141    gen_extent(dims, max_size).prop_flat_map(move |extent| {
142        let sizes: Vec<usize> = extent.sizes().collect();
143        let begins_strategy: BoxedStrategy<Vec<usize>> =
144            sizes
145                .iter()
146                .fold(Just(Vec::<usize>::new()).boxed(), |acc, &size| {
147                    let size = size.max(1);
148                    (acc, 0..size)
149                        .prop_map(|(mut begins, begin)| {
150                            begins.push(begin);
151                            begins
152                        })
153                        .boxed()
154                });
155        let steps_raw_strategy = prop::collection::vec(1..=max_step, sizes.len());
156        (
157            Just(extent),
158            0..=max_base_offset,
159            begins_strategy,
160            steps_raw_strategy,
161        )
162            .prop_map(move |(extent, offset, begins, steps_raw)| {
163                // Clamp steps so they satisfy steps[i] % steps[i+1] == 0.
164                // Without this, per-dim independent stride choices on a
165                // row-major source can yield NonrectangularStrides. Mirrors
166                // ndslice::strategy::gen_region_strided's clamp pass.
167                let steps = clamp_steps_to_divisibility_chain(steps_raw, max_step);
168                let selections: Vec<(usize, usize)> = begins.into_iter().zip(steps).collect();
169
170                let strides = row_major_strides(extent.sizes());
171                let parent = if offset == 0 {
172                    RankRect::new(extent.clone()).expect("dense rect is valid")
173                } else {
174                    RankRect::affine(extent.clone(), Rank(offset), strides)
175                        .expect("row-major rect with offset is valid")
176                };
177                let dim_names: Vec<String> = extent
178                    .dims()
179                    .iter()
180                    .map(|dim| dim.name().to_string())
181                    .collect();
182                let mut subrect = parent.clone();
183                for (dim_index, &(begin, step)) in selections.iter().enumerate() {
184                    let step_nonzero =
185                        NonZeroUsize::new(step.max(1)).expect("step max(1) is non-zero");
186                    subrect = subrect
187                        .select(
188                            &dim_names[dim_index],
189                            DimRange::new(begin, None, step_nonzero),
190                        )
191                        .expect("clamped steps preserve rectangularity; select must succeed");
192                }
193                (parent, selections, subrect)
194            })
195    })
196}

198/// Clamp per-dim selection steps into the divisibility chain required by
199/// [`RankRect::affine`].
200///
201/// For row-major-derived rectangles, selecting outer dimensions with arbitrary
202/// independent steps can create non-rectangular stride layouts. Ensuring
203/// `steps[i] % steps[i + 1] == 0` keeps the selected strides representable as a
204/// single [`RankRect`].
205fn clamp_steps_to_divisibility_chain(mut steps: Vec<usize>, max_step: usize) -> Vec<usize> {
206    let max_step = max_step.max(1);
207    if steps.is_empty() {
208        return steps;
209    }
210    let last = steps.len() - 1;
211    steps[last] = steps[last].max(1).min(max_step);
212    for i in (0..last).rev() {
213        let inner = steps[i + 1].max(1);
214        let max_mult = (max_step / inner).max(1);
215        let m = ((steps[i].max(1) - 1) % max_mult) + 1;
216        steps[i] = inner.saturating_mul(m);
217    }
218    steps
219}

221fn row_major_strides(sizes: impl IntoIterator<Item = usize>) -> Vec<usize> {
222    let sizes = sizes.into_iter().collect::<Vec<_>>();
223    let mut strides = vec![1; sizes.len()];
224    for index in (0..sizes.len().saturating_sub(1)).rev() {
225        strides[index] = strides[index + 1] * sizes[index + 1];
226    }
227    strides
228}

240    fn coords_for(extent: &Extent) -> Vec<Vec<usize>> {
241        (0..extent.cardinality())
242            .map(|index| coord_at(index, extent.sizes()).expect("index is in bounds"))
243            .collect()
244    }
245
246    fn coord_at(index: usize, sizes: impl IntoIterator<Item = usize>) -> Option<Vec<usize>> {
247        let sizes = sizes.into_iter().collect::<Vec<_>>();
248        let cardinality = sizes.iter().product::<usize>();
249        if index >= cardinality {
250            return None;
251        }
252        let mut rest = index;
253        let mut coord = vec![0; sizes.len()];
254        for (axis, size) in sizes.iter().copied().enumerate().rev() {
255            coord[axis] = rest % size;
256            rest /= size;
257        }
258        Some(coord)
259    }

261    proptest! {
262        // Tests that the dense rectangle generator emits valid row-major
263        // rectangles: every generated coordinate resolves to a rank, and that
264        // rank resolves back to the original coordinate.
265        #[test]
266        fn prop_gen_rank_rect_emits_valid_rectangles(rect in gen_rank_rect(1..=4, 8)) {
267            for coord in coords_for(rect.extent()) {
268                let rank = rect.rank_of(&coord).expect("generated coord is in range");
269                prop_assert_eq!(
270                    rect.coord_of(rank),
271                    Some(Coord::new(coord.clone())),
272                    "rank/coord roundtrip failed for {:?}", coord
273                );
274            }
275        }
276
277        // Tests that the strided rectangle generator still emits valid affine
278        // rectangles after base-offset and per-dimension selection. This guards
279        // the generator itself against producing shapes `RankRect` cannot
280        // round-trip through rank/coordinate conversion.
281        #[test]
282        fn prop_gen_rank_rect_strided_emits_valid_rectangles(
283            rect in gen_rank_rect_strided(1..=4, 4, 3, 32),
284        ) {
285            for coord in coords_for(rect.extent()) {
286                let rank = rect.rank_of(&coord).expect("generated coord is in range");
287                prop_assert_eq!(
288                    rect.coord_of(rank),
289                    Some(Coord::new(coord.clone())),
290                    "rank/coord roundtrip failed for {:?}", coord
291                );
292            }
293        }
294
295        // Tests the public pair-generator contract: the returned subrect is
296        // contained in the returned parent, rather than being an unrelated
297        // valid rectangle.
298        #[test]
299        fn prop_gen_rank_rect_and_strided_subrect_satisfies_containment(
300            (parent, subrect) in gen_rank_rect_and_strided_subrect(1..=4, 4, 3, 32),
301        ) {
302            prop_assert!(parent.contains_rect(&subrect));
303        }
304
305        // Tests the load-bearing affine-frame invariant for future tiler
306        // proptests: each subrect coordinate maps to the same base rank as the
307        // corresponding selected parent coordinate. This catches generators
308        // that accidentally rebase children into a fresh `0..n` rank frame.
309        #[test]
310        fn prop_gen_rank_rect_and_strided_subrect_preserves_base_rank_embedding(
311            (parent, selections, subrect) in gen_rank_rect_with_selections(1..=4, 4, 3, 32),
312        ) {
313            for coord in coords_for(subrect.extent()) {
314                let projected: Vec<usize> = coord
315                    .iter()
316                    .zip(&selections)
317                    .map(|(c, &(begin, step))| begin + *c * step.max(1))
318                    .collect();
319                prop_assert_eq!(
320                    subrect.rank_of(&coord),
321                    parent.rank_of(&projected),
322                    "subrect coord {:?} did not embed at projected parent coord {:?}",
323                    coord, projected
324                );
325            }
326        }
327    }

329    fn sample_n<T>(strategy: impl Strategy<Value = T>, n: usize) -> Vec<T> {
330        let mut runner = TestRunner::new(Config::default());
331        (0..n)
332            .map(|_| strategy.new_tree(&mut runner).unwrap().current())
333            .collect()
334    }

336    // Tests that `max_base_offset` is consumed by the generator. This inspects
337    // the parent via the internal helper because a subrect's final offset can
338    // also become non-zero from `select(begin > 0)`.
339    #[test]
340    fn gen_rank_rect_strided_can_emit_non_zero_offset() {
341        let samples = sample_n(gen_rank_rect_with_selections(1..=4, 4, 3, 32), 256);
342        assert!(
343            samples
344                .iter()
345                .any(|(parent, _, _)| parent.offset().get() != 0),
346            "expected at least one sample with a non-zero parent offset"
347        );
348    }

350    // Tests that the generator can apply `step > 1` selections. Checking the
351    // subrect's strides alone would be too weak because multi-dimensional
352    // row-major parents naturally have strides greater than one.
353    #[test]
354    fn gen_rank_rect_strided_can_emit_non_unit_stride() {
355        let samples = sample_n(gen_rank_rect_with_selections(1..=4, 4, 3, 32), 256);
356        assert!(
357            samples
358                .iter()
359                .any(|(_, selections, _)| selections.iter().any(|&(_, step)| step > 1)),
360            "expected at least one sample with a per-dim selection step > 1"
361        );
362    }
363
364    // Tests that the generator can emit different selection steps across
365    // dimensions, not just a uniform stride scale. With the divisibility-chain
366    // clamp, non-uniformity means at least one outer step is a proper multiple
367    // of an inner step. Requires at least two dimensions.
368    #[test]
369    fn gen_rank_rect_strided_can_emit_non_uniform_per_dim_stride() {
370        let samples = sample_n(gen_rank_rect_with_selections(2..=4, 4, 3, 32), 256);
371        let non_uniform_seen = samples.iter().any(|(_, selections, _)| {
372            if selections.len() < 2 {
373                return false;
374            }
375            let first_step = selections[0].1;
376            selections.iter().any(|&(_, step)| step != first_step)
377        });
378        assert!(
379            non_uniform_seen,
380            "expected at least one sample with non-uniform per-dim selection steps"
381        );
382    }
