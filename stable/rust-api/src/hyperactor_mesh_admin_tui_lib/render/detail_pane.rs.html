49/// Render the contextual details pane (right side).
50///
51/// Precedence: the help glossary (`app.show_help`, TUI-22) takes priority,
52/// then the active `app.overlay` (py-spy / config / diagnostics), then node
53/// detail. For node detail, if a `NodePayload` for the current selection is
54/// available in `app.detail`, dispatches to `render_node_detail` to show a
55/// type-specific view (root/host/proc/actor); otherwise shows either the
56/// last fetch error (`app.detail_error`) or a neutral "select a node"
57/// placeholder message.
58pub(crate) fn render_detail_pane(frame: &mut ratatui::Frame<'_>, area: Rect, app: &App) {
59    if app.show_help {
60        render_help_overlay(frame, area, app.detail.as_ref(), &app.theme.scheme);
61        return;
62    }
63    if let Some(overlay) = &app.overlay {
64        render_overlay(frame, area, overlay, &app.theme.scheme);
65        return;
66    }
67    match &app.detail {
68        Some(payload) => {
69            render_node_detail(frame, area, payload, &app.theme.scheme, &app.theme.labels)
70        }
71        None => {
72            let msg = app
73                .detail_error
74                .as_deref()
75                .unwrap_or("Select a node to view details");
76            let msg_style = if app.detail_error.is_some() {
77                app.theme.scheme.error
78            } else {
79                app.theme.scheme.info
80            };
81            let block = Block::default()
82                .title(app.theme.labels.pane_details)
83                .borders(Borders::ALL)
84                .border_style(app.theme.scheme.border);
85            let p = Paragraph::new(Span::styled(msg, msg_style)).block(block);
86            frame.render_widget(p, area);
87        }
88    }
89}

91/// A single field-help entry: a field name, its one-line meaning, and an
92/// optional subdued note line for caveats.
93pub(crate) struct HelpEntry {
94    pub(crate) field: &'static str,
95    pub(crate) meaning: &'static str,
96    pub(crate) note: Option<&'static str>,
97}
98
99const fn help(field: &'static str, meaning: &'static str) -> HelpEntry {
100    HelpEntry {
101        field,
102        meaning,
103        note: None,
104    }
105}

107const fn noted(field: &'static str, meaning: &'static str, note: &'static str) -> HelpEntry {
108    HelpEntry {
109        field,
110        meaning,
111        note: Some(note),
112    }
113}

115// First-pass glossary is intentionally English-only. Move into localized
116// help topics if/when the feature is promoted.
117//
118// This is a glossary of the non-obvious fields, not exhaustive row coverage:
119// one entry per confusing rendered row, at that row's granularity. Field
120// names are canonical English, independent of the localized `Labels`.
121pub(crate) fn help_content(props: &NodeProperties) -> (&'static str, &'static [HelpEntry]) {
122    // Entries live in `const` items so the returned slices are `'static`
123    // (a `&[..]` temporary built in a match arm is not promoted and would
124    // not outlive the function).
125    const ROOT: &[HelpEntry] = &[
126        help("hosts", "number of host processes in the mesh"),
127        help("started by", "identity that launched this root"),
128        help("uptime", "time since this root was started"),
129    ];
130    const HOST: &[HelpEntry] = &[
131        help("address", "network address of this host process"),
132        help("procs", "number of worker processes on this host"),
133        help("rss", "resident set size: physical RAM used by the process"),
134        help("vm size", "virtual address space reserved by the process"),
135    ];
136    const PROC: &[HelpEntry] = &[
137        help("actors", "live actors running in this process"),
138        help("rss", "resident set size: physical RAM used by the process"),
139        help("vm size", "virtual address space reserved by the process"),
140        noted(
141            "queue depth",
142            "accepted handler work not yet dequeued (total across actors; max = largest single actor)",
143            "max is a point-in-time snapshot, not a historical high-water mark",
144        ),
145        help(
146            "peak depth",
147            "maximum proc-wide queue total observed since startup",
148        ),
149        help(
150            "last busy",
151            "time since proc queue total was last observed non-zero",
152        ),
153        help(
154            "poisoned",
155            "proc has failed and should no longer be treated as healthy",
156        ),
157    ];
158    const ACTOR: &[HelpEntry] = &[
159        help(
160            "status",
161            "current actor lifecycle state: ok / degraded / failed",
162        ),
163        help("instance", "stable UUID for this actor instance"),
164        noted(
165            "queue depth",
166            "accepted handler work not yet dequeued by this actor",
167            "includes out-of-order work held by the reorder buffer",
168        ),
169        help("msgs processed", "messages handled since actor start"),
170        help(
171            "processing time",
172            "cumulative time spent in message handlers",
173        ),
174        help("last handler", "most recent handler method invoked"),
175        help("children", "actors spawned by this actor"),
176        help(
177            "ordering enabled",
178            "whether inbound sequence tracking is active",
179        ),
180        help(
181            "sessions stalled",
182            "sender sessions waiting on a missing next sequence",
183        ),
184        noted(
185            "buffered",
186            "out-of-order frames held by inbound ordering",
187            "independent diagnostic; do not compare arithmetically with queue depth",
188        ),
189    ];
190    const ERROR: &[HelpEntry] = &[
191        help("code", "error code returned when resolving this node"),
192        help("message", "error detail"),
193    ];
194    match props {
195        NodeProperties::Root { .. } => ("root", ROOT),
196        NodeProperties::Host { .. } => ("host", HOST),
197        NodeProperties::Proc { .. } => ("proc", PROC),
198        NodeProperties::Actor { .. } => ("actor", ACTOR),
199        NodeProperties::Error { .. } => ("error", ERROR),
200    }
201}

203/// Render the static help glossary overlay (TUI-22) into the detail pane.
204///
205/// Read-only: lists field meanings for the selected node kind, with the
206/// kind shown in the block title (e.g. " ? actor help "). Non-scrollable;
207/// content is kept short enough to fit common pane heights.
208fn render_help_overlay(
209    frame: &mut ratatui::Frame<'_>,
210    area: Rect,
211    payload: Option<&NodePayload>,
212    scheme: &ColorScheme,
213) {
214    // Title carries the node kind for context, e.g. " ? actor help ".
215    let (title, lines): (String, Vec<Line<'static>>) = match payload {
216        None => (
217            " ? help ".to_string(),
218            vec![
219                Line::from(Span::styled(
220                    "Select a node to view field help",
221                    scheme.info,
222                )),
223                Line::raw(""),
224                Line::from(Span::styled("press any key to dismiss", scheme.info)),
225            ],
226        ),
227        Some(p) => {
228            let (kind, entries) = help_content(&p.properties);
229            let mut lines = Vec::new();
230            for e in entries {
231                lines.push(Line::from(vec![
232                    Span::styled(format!("{:<22}", e.field), scheme.detail_label),
233                    Span::raw(e.meaning),
234                ]));
235                if let Some(n) = e.note {
236                    lines.push(Line::from(vec![
237                        Span::raw(format!("{:<22}", "")),
238                        Span::styled(n, scheme.info),
239                    ]));
240                }
241            }
242            lines.push(Line::raw(""));
243            lines.push(Line::from(Span::styled(
244                "press any key to dismiss",
245                scheme.info,
246            )));
247            (format!(" ? {kind} help "), lines)
248        }
249    };
250
251    let block = Block::default()
252        .title(title)
253        .borders(Borders::ALL)
254        .border_style(scheme.border);
255    let inner = block.inner(area);
256    frame.render_widget(block, area);
257    frame.render_widget(Paragraph::new(lines).wrap(Wrap { trim: false }), inner);
258}

260/// Render the details view for a resolved node.
261///
262/// This is the main dispatcher for the right-hand pane: it matches on
263/// `payload.properties` and forwards to the appropriate renderer
264/// (`render_root_detail`, `render_host_detail`, `render_proc_detail`,
265/// or `render_actor_detail`) with the relevant fields extracted from
266/// the payload.
267pub(crate) fn render_node_detail(
268    frame: &mut ratatui::Frame<'_>,
269    area: Rect,
270    payload: &NodePayload,
271    scheme: &ColorScheme,
272    labels: &Labels,
273) {
274    match &payload.properties {
275        NodeProperties::Root {
276            num_hosts,
277            started_at,
278            started_by,
279            ..
280        } => {
281            render_root_detail(
282                frame, area, payload, *num_hosts, started_at, started_by, scheme, labels,
283            );
284        }
285        NodeProperties::Host {
286            addr,
287            num_procs,
288            memory,
289            ..
290        } => {
291            render_host_detail(
292                frame, area, payload, addr, *num_procs, memory, scheme, labels,
293            );
294        }
295        NodeProperties::Proc {
296            proc_name,
297            num_actors,
298            is_poisoned,
299            failed_actor_count,
300            debug,
301            ..
302        } => {
303            render_proc_detail(
304                frame,
305                area,
306                payload,
307                proc_name,
308                *num_actors,
309                *is_poisoned,
310                *failed_actor_count,
311                debug,
312                scheme,
313                labels,
314            );
315        }
316        NodeProperties::Actor {
317            actor_status,
318            actor_type,
319            instance_id,
320            messages_processed,
321            queue_depth,
322            created_at,
323            last_message_handler,
324            total_processing_time_us,
325            flight_recorder,
326            inbound_ordering,
327            execution,
328            failure_info,
329            ..
330        } => {
331            render_actor_detail(
332                frame,
333                area,
334                payload,
335                actor_status,
336                actor_type,
337                instance_id,
338                *queue_depth,
339                *messages_processed,
340                created_at,
341                last_message_handler.as_deref(),
342                *total_processing_time_us,
343                flight_recorder.as_deref(),
344                inbound_ordering.as_deref(),
345                execution.as_deref(),
346                failure_info.as_ref(),
347                scheme,
348                labels,
349            );
350        }
351        NodeProperties::Error { code, message } => {
352            let text = format!("Error: {} — {}", code, message);
353            let paragraph = Paragraph::new(text)
354                .block(
355                    Block::default()
356                        .borders(Borders::ALL)
357                        .border_style(scheme.border)
358                        .title(labels.pane_error),
359                )
360                .style(scheme.error)
361                .wrap(Wrap { trim: true });
362            frame.render_widget(paragraph, area);
363        }
364    }
365}

379/// Render the right-pane detail view for the mesh root node.
380///
381/// Shows a simple summary (host count) and then lists the root's
382/// immediate children (host references) so the user can quickly see
383/// which hosts are currently registered under the mesh.
384fn render_root_detail(
385    frame: &mut ratatui::Frame<'_>,
386    area: Rect,
387    payload: &NodePayload,
388    num_hosts: usize,
389    started_at: &SystemTime,
390    started_by: &str,
391    scheme: &ColorScheme,
392    l: &Labels,
393) {
394    let block = Block::default()
395        .title(l.pane_root_details)
396        .borders(Borders::ALL)
397        .border_style(scheme.border);
398
399    let uptime_str = format_system_time_uptime(started_at);
400
401    let mut lines = vec![
402        detail_line(l.hosts, num_hosts.to_string(), scheme),
403        detail_line(l.started_by, started_by, scheme),
404        detail_line(l.uptime_detail, &uptime_str, scheme),
405        detail_line(l.started_at, format_system_time_local(started_at), scheme),
406        detail_line(
407            l.data_as_of,
408            format_system_time_relative(&payload.as_of),
409            scheme,
410        ),
411        Line::default(),
412    ];
413    for child in &payload.children {
414        let child_str = child.to_string();
415        lines.push(Line::from(vec![
416            Span::styled("  ", Style::default()),
417            Span::styled(child_str, scheme.node_host),
418        ]));
419    }
420
421    let p = Paragraph::new(lines).block(block);
422    frame.render_widget(p, area);
423}

481    frame.render_widget(p, area);

484/// Render the right-pane detail view for a proc node.
485///
486/// Shows the proc's full name and actor count, then lists up to the
487/// first 50 child actor references (with an elision line if there are
488/// more) to keep the UI responsive and the pane readable.
489fn render_proc_detail(
490    frame: &mut ratatui::Frame<'_>,
491    area: Rect,
492    payload: &NodePayload,
493    proc_name: &str,
494    num_actors: usize,
495    is_poisoned: bool,
496    failed_actor_count: usize,
497    debug: &hyperactor_mesh::introspect::ProcDebugStats,
498    scheme: &ColorScheme,
499    l: &Labels,
500) {
501    let block = Block::default()
502        .title(l.pane_proc_details)
503        .borders(Borders::ALL)
504        .border_style(scheme.border);
505
506    let mut lines = vec![
507        detail_line(l.name, proc_name, scheme),
508        detail_line(l.actors, num_actors.to_string(), scheme),
509    ];
510    // Hosting-process memory — always show so absence is explicit.
511    lines.push(detail_line(
512        l.rss,
513        debug
514            .memory
515            .process_rss_bytes
516            .map_or_else(|| "N/A".to_string(), format_bytes),
517        scheme,
518    ));
519    lines.push(detail_line(
520        l.vm_size,
521        debug
522            .memory
523            .process_vm_size_bytes
524            .map_or_else(|| "N/A".to_string(), format_bytes),
525        scheme,
526    ));
527    // Queue pressure — always show so absence of pressure is explicit.
528    lines.push(detail_line(
529        l.queue_depth,
530        format!(
531            "{} total, {} max",
532            debug.actor_work_queue_depth_total, debug.actor_work_queue_depth_max
533        ),
534        scheme,
535    ));
536    // Retained queue-pressure evidence (PD-6, PD-7).
537    if debug.actor_work_queue_depth_high_water_mark > 0 {
538        lines.push(detail_line(
539            l.peak_depth,
540            debug.actor_work_queue_depth_high_water_mark.to_string(),
541            scheme,
542        ));
543    }
544    lines.push(detail_line(
545        l.last_busy,
546        match debug.last_nonzero_queue_depth_age_ms {
547            None => "never".to_string(),
548            Some(ms) if ms < 1000 => format!("{}ms ago", ms),
549            Some(ms) => format!("{}s ago", ms / 1000),
550        },
551        scheme,
552    ));
553    lines.push(detail_line(
554        l.data_as_of,
555        format_system_time_relative(&payload.as_of),
556        scheme,
557    ));
558
559    if is_poisoned {
560        lines.push(Line::from(vec![
561            Span::styled(l.poisoned, scheme.node_failed),
562            Span::styled(l.yes, scheme.node_failed),
563        ]));
564        lines.push(Line::from(vec![
565            Span::styled(l.failed_actors, scheme.node_failed),
566            Span::styled(failed_actor_count.to_string(), scheme.node_failed),
567        ]));
568    }
569
570    lines.push(Line::default());
571    for (i, actor) in payload.children.iter().enumerate() {
572        if i >= 50 {
573            lines.push(Line::from(Span::styled(
574                format!("  ... and {} more", payload.children.len() - 50),
575                scheme.detail_stopped,
576            )));
577            break;
578        }
579        lines.push(Line::from(vec![
580            Span::styled("  ", Style::default()),
581            Span::raw(actor.to_string()),
582        ]));
583    }
584
585    let p = Paragraph::new(lines).block(block);
586    frame.render_widget(p, area);
587}

589/// Render the right-pane detail view for an actor node, including
590/// flight-recorder events.
591///
592/// The top section summarizes actor status/type, message/processing
593/// stats, creation time, last handler, and child count; the bottom
594/// section parses the optional flight-recorder JSON and displays a
595/// compact, timestamped list of recent events.
596const TOP_N: usize = 10;
597const MIN_FLIGHT_RECORDER_HEIGHT: u16 = 5;
598/// A bordered pane needs border(2) + 1 row to show any content; below this it
599/// draws only a border sliver, so treat it as hidden (0 height + clip alert).
600const MIN_VISIBLE_PANE_HEIGHT: u16 = 3;
601const RED_STALL_THRESHOLD: usize = 100; // placeholder; tune later.
602
603#[allow(clippy::too_many_arguments)]
604fn render_actor_detail(
605    frame: &mut ratatui::Frame<'_>,
606    area: Rect,
607    payload: &NodePayload,
608    actor_status: &str,
609    actor_type: &str,
610    instance_id: &str,
611    queue_depth: u64,
612    messages_processed: u64,
613    created_at: &Option<SystemTime>,
614    last_message_handler: Option<&str>,
615    total_processing_time_us: u64,
616    flight_recorder_json: Option<&str>,
617    inbound_ordering: Option<&InboundOrdering>,
618    execution: Option<&Execution>,
619    failure_info: Option<&FailureInfo>,
620    scheme: &ColorScheme,
621    l: &Labels,
622) {
623    let info_height = actor_info_height(failure_info);
624    let ordering_height = compute_ordering_height(area.height, info_height, inbound_ordering);
625    // Execution section is hidden (height 0) when the actor reports no
626    // execution (None = unsupported) or nothing is in flight; its budget
627    // comes after info + ordering, preserving the flight recorder's minimum.
628    let execution_height =
629        compute_execution_height(area.height, info_height, ordering_height, execution);
630    let chunks = Layout::default()
631        .direction(Direction::Vertical)
632        .constraints([
633            Constraint::Length(info_height),
634            Constraint::Length(ordering_height),
635            Constraint::Length(execution_height),
636            Constraint::Min(MIN_FLIGHT_RECORDER_HEIGHT),
637        ])
638        .split(area);
639
640    // Actor info
641    let info_block = Block::default()
642        .title(l.pane_actor_details)
643        .borders(Borders::ALL)
644        .border_style(scheme.border);
645
646    let status_style = if failure_info.is_some() {
647        scheme.detail_status_failed
648    } else if actor_status == "Running" {
649        scheme.detail_status_ok
650    } else {
651        scheme.detail_status_warn
652    };
653
654    let created_str = created_at
655        .as_ref()
656        .map(format_system_time_iso)
657        .unwrap_or_else(|| "-".to_string());
658
659    let mut lines = vec![
660        Line::from(vec![
661            Span::styled(l.status, scheme.detail_label),
662            Span::styled(actor_status, status_style),
663        ]),
664        detail_line(l.instance_id_label, instance_id, scheme),
665        detail_line(l.queue_depth, queue_depth.to_string(), scheme),
666        detail_line(
667            l.data_as_of,
668            format_system_time_relative(&payload.as_of),
669            scheme,
670        ),
671        detail_line(l.actor_type, actor_type, scheme),
672        detail_line(l.messages, messages_processed.to_string(), scheme),
673        detail_line(
674            l.processing_time,
675            humantime::format_duration(std::time::Duration::from_micros(total_processing_time_us))
676                .to_string(),
677            scheme,
678        ),
679        detail_line(l.created, &created_str, scheme),
680        detail_line(l.last_handler, last_message_handler.unwrap_or("-"), scheme),
681        detail_line(l.children, payload.children.len().to_string(), scheme),
682    ];
683
684    if let Some(fi) = failure_info {
685        lines.push(Line::default());
686        lines.push(Line::from(vec![
687            Span::styled(l.error_message, scheme.node_failed),
688            Span::styled(&fi.error_message, scheme.node_failed),
689        ]));
690        let root_cause_display = match &fi.root_cause_name {
691            Some(name) => format!("{} ({})", name, fi.root_cause_actor),
692            None => fi.root_cause_actor.to_string(),
693        };
694        lines.push(Line::from(vec![
695            Span::styled(l.root_cause, scheme.detail_label),
696            Span::raw(root_cause_display),
697        ]));
698        lines.push(detail_line(
699            l.failed_at,
700            format_system_time_iso(&fi.occurred_at),
701            scheme,
702        ));
703        lines.push(Line::from(vec![
704            Span::styled(l.propagated, scheme.detail_label),
705            Span::raw(if fi.is_propagated { l.yes } else { l.no }),
706        ]));
707    }
708
709    let info = Paragraph::new(lines)
710        .block(info_block)
711        .wrap(Wrap { trim: false });
712    frame.render_widget(info, chunks[0]);
713
714    // Inbound ordering
715    render_inbound_ordering(frame, chunks[1], inbound_ordering, scheme, l);
716
717    // Execution (hidden via a zero-height chunk when None / active_count 0).
718    render_execution(frame, chunks[2], execution, scheme, l);
719
720    // Flight recorder
721    let recorder_block = Block::default()
722        .title(l.pane_flight_recorder)
723        .borders(Borders::ALL)
724        .border_style(scheme.border);
725
726    let recorded_events: Vec<RecordedEvent> = flight_recorder_json
727        .and_then(|json| serde_json::from_str(json).ok())
728        .unwrap_or_default();
729
730    let events: Vec<Line> = if recorded_events.is_empty() {
731        vec![Line::from(Span::styled("No events", scheme.detail_label))]
732    } else {
733        recorded_events
734            .iter()
735            .map(|event| {
736                let level_style = match event.level.as_str() {
737                    "ERROR" => scheme.error,
738                    "WARN" => scheme.detail_status_warn,
739                    "INFO" => scheme.detail_status_ok,
740                    "DEBUG" => scheme.info,
741                    _ => scheme.detail_label,
742                };
743                // Event text is less-trusted mesh data; strip control bytes so a
744                // crafted event cannot emit terminal escape/OSC sequences into
745                // the operator's terminal.
746                let level_char = event
747                    .level
748                    .chars()
749                    .next()
750                    .filter(|c| !c.is_control())
751                    .unwrap_or('?');
752                Line::from(vec![
753                    Span::styled(format!("{level_char} "), level_style),
754                    Span::styled(
755                        format!(
756                            "{} ",
757                            sanitize_control(&format_local_time(&event.timestamp))
758                        ),
759                        scheme.detail_label,
760                    ),
761                    Span::raw(sanitize_control(&format_event_summary(
762                        &event.name,
763                        &event.fields,
764                    ))),
765                ])
766            })
767            .collect()
768    };
769
770    // TUI-FR-1 (flight-recorder-autoscroll): events render chronologically
771    // (oldest at top) and auto-scroll so the newest row sits on the bottom line.
772    // Read line_count before attaching the block; a Paragraph carrying the block
773    // subtracts the border from the width again, over-counting and scrolling past
774    // the newest line.
775    let inner_width = chunks[3].width.saturating_sub(2);
776    let inner_height = chunks[3].height.saturating_sub(2);
777    let recorder = Paragraph::new(events).wrap(Wrap { trim: true });
778    // Guard the degenerate zero-width case: nothing renders at zero inner width,
779    // and line_count(0) has no well-defined wrap.
780    let total_rows = if inner_width == 0 {
781        0
782    } else {
783        recorder.line_count(inner_width) as u16
784    };
785    let scroll = total_rows.saturating_sub(inner_height);
786    let recorder = recorder.block(recorder_block).scroll((scroll, 0));
787    frame.render_widget(recorder, chunks[3]);
788}

860/// Compute the desired height for the inbound-ordering section, or 0 to hide it
861/// (per `ordering_has_content`), capped so the flight recorder keeps
862/// `MIN_FLIGHT_RECORDER_HEIGHT`. The row budget mirrors `build_inbound_ordering_lines`
863/// exactly, the way `compute_execution_height` mirrors `build_execution_lines`.
864pub(crate) fn compute_ordering_height(
865    area_height: u16,
866    info_height: u16,
867    inbound_ordering: Option<&InboundOrdering>,
868) -> u16 {
869    let Some(io) = inbound_ordering else {
870        return 0;
871    };
872    if !ordering_has_content(io) {
873        return 0;
874    }
875    let stalled = io.sessions.iter().filter(|s| s.buffered_count > 0).count();
876    // border(2) + rollup line 1(1) + rollup line 2(0/1). A partial snapshot with
877    // no returned stalls stops there; otherwise add the table: spacer(1) +
878    // header(1) + rows + more(0/1) + footer spacer(1) + footer(1).
879    let rollup2 = u16::from(io.returned_buffered_message_count > 0);
880    let table = if stalled == 0 {
881        0
882    } else {
883        let rows = stalled.min(TOP_N) as u16;
884        let more_row = u16::from(stalled > TOP_N);
885        1 + 1 + rows + more_row + 1 + 1
886    };
887    let want = 2 + 1 + rollup2 + table;
888    let budget = area_height
889        .saturating_sub(info_height)
890        .saturating_sub(MIN_FLIGHT_RECORDER_HEIGHT);
891    let height = want.min(budget);
892    // Below one content row it is a border-only sliver -> hide it (0), so the
893    // header clip-alert fires instead of drawing a content-less box.
894    if height < MIN_VISIBLE_PANE_HEIGHT {
895        0
896    } else {
897        height
898    }
899}

901/// Compute the desired height for the Execution section.
902///
903/// Returns 0 when the actor reports no execution (`None` = unsupported) or
904/// nothing is in flight (`active_count == 0`), hiding the section entirely.
905/// Otherwise sizes to the rendered line count, capped against the remaining
906/// budget so the flight recorder keeps its minimum height. Must match the
907/// line count produced by `build_execution_lines`.
908pub(crate) fn compute_execution_height(
909    area_height: u16,
910    info_height: u16,
911    ordering_height: u16,
912    execution: Option<&Execution>,
913) -> u16 {
914    let Some(e) = execution else {
915        return 0;
916    };
917    if e.active_count == 0 {
918        return 0;
919    }
920    let budget = area_height
921        .saturating_sub(info_height)
922        .saturating_sub(ordering_height)
923        .saturating_sub(MIN_FLIGHT_RECORDER_HEIGHT);
924    // border(2) + count line(1) + oldest line(1 iff a row exists)
925    // + per-name rows + truncation marker(1 iff truncated).
926    let oldest_line = u16::from(!e.active_handlers.is_empty());
927    let rows = e.active_handlers.len() as u16;
928    let more_row = u16::from(e.truncated);
929    let want = 2 + 1 + oldest_line + rows + more_row;
930    let height = want.min(budget);
931    // Below one content row it is a border-only sliver -> hide it (0), so the
932    // header clip-alert fires instead of drawing a content-less box.
933    if height < MIN_VISIBLE_PANE_HEIGHT {
934        0
935    } else {
936        height
937    }
938}

980        e.active_count.to_string(),

984    // Oldest live invocation: the first row (aggregated by name and sorted
985    // oldest-first upstream, EX-4). Absent when per-handler detail was
986    // momentarily unavailable on this poll (EX-2: complete == false leaves

1008    }
1009
1010    // Truncation marker (EX-4): some of the oldest names were dropped. We
1011    // don't carry a precise hidden-name count, so the marker is unqualified.

1013        lines.push(Line::from(Span::styled(
1014            l.execution_more_names,
1015            scheme.detail_label,

1062fn build_inbound_ordering_lines<'a>(
1063    io: &'a InboundOrdering,
1064    scheme: &ColorScheme,
1065    l: &'a Labels,
1066) -> Vec<Line<'a>> {
1067    let mut lines: Vec<Line> = Vec::new();
1068
1069    // Filter to stalled-only, sort by severity then by oldest gap.
1070    let mut stalled: Vec<&hyperactor::ordering::OrderingSessionSnapshot> = io
1071        .sessions
1072        .iter()
1073        .filter(|s| s.buffered_count > 0)
1074        .collect();
1075    stalled.sort_by(|a, b| {
1076        b.buffered_count.cmp(&a.buffered_count).then_with(|| {
1077            match (a.oldest_buffered_seq, b.oldest_buffered_seq) {
1078                (Some(ax), Some(bx)) => ax.cmp(&bx),
1079                (Some(_), None) => std::cmp::Ordering::Less,
1080                (None, Some(_)) => std::cmp::Ordering::Greater,
1081                (None, None) => std::cmp::Ordering::Equal,
1082            }
1083        })
1084    });
1085
1086    // Rollup line 1: "enabled · 2 of 3 sessions stalled (partial: N skipped)"
1087    let partial_suffix = if !io.snapshot_complete {
1088        format!(
1089            " {}",
1090            l.ordering_sessions_partial
1091                .replace("{n}", &io.skipped_session_count.to_string())
1092        )
1093    } else {
1094        String::new()
1095    };
1096    lines.push(Line::from(vec![
1097        Span::styled(l.ordering_buffering_enabled, scheme.detail_label),
1098        Span::raw(format!(
1099            " · {} {} {} {} {}{}",
1100            io.returned_buffered_session_count,
1101            l.ordering_sessions_known,
1102            io.known_session_count,
1103            l.ordering_sessions_label,
1104            l.ordering_sessions_stalled,
1105            partial_suffix,
1106        )),
1107    ]));
1108
1109    // Rollup line 2 only when something is buffered; a partial snapshot with no
1110    // returned stalls shows just line 1 ("0 of N stalled (partial: M skipped)").
1111    if io.returned_buffered_message_count > 0 {
1112        lines.push(Line::from(Span::raw(format!(
1113            "{} {}, {} {}{}",
1114            io.returned_buffered_message_count,
1115            l.ordering_buffered_label,
1116            l.ordering_max_in_worst,
1117            io.returned_max_buffered_count,
1118            l.ordering_returned_total,
1119        ))));
1120    }
1121
1122    // Partial snapshot with nothing returned as stalled: the rollup (with its
1123    // "(partial: N skipped)" suffix) is why the pane is shown, so stop before the
1124    // per-session table.
1125    if stalled.is_empty() {
1126        return lines;
1127    }
1128
1129    // Spacer + column header.
1130    lines.push(Line::default());
1131    lines.push(Line::from(Span::styled(
1132        format!(
1133            "{:<owner$} {:>seq$}  {}",
1134            l.ordering_col_owner,
1135            l.ordering_col_missing_seq,
1136            l.ordering_col_buffered,
1137            owner = OWNER_COL_WIDTH,
1138            seq = NEED_SEQ_COL_WIDTH,
1139        ),
1140        scheme.detail_label,
1141    )));
1142
1143    // Per-session rows. Color is themed: stalled rows use
1144    // `scheme.detail_stalled`; severe stalls (above threshold) escalate
1145    // to `scheme.detail_stalled_severe` (adds bold).
1146    let visible = stalled.iter().take(TOP_N);
1147    for s in visible {
1148        let owner_full = match &s.sender {
1149            Some(addr) => trim_owner_for_display(&addr.to_string()),
1150            None => l
1151                .ordering_sender_fallback
1152                .replace("{id}", &short_session_id(&s.session_id.to_string())),
1153        };
1154        let owner = truncate_cell(&owner_full, OWNER_COL_WIDTH);
1155        let buffered_cell = match (s.oldest_buffered_seq, s.newest_buffered_seq) {
1156            (Some(oldest), Some(newest)) => {
1157                format!("{} ({}-{})", s.buffered_count, oldest, newest)
1158            }
1159            _ => s.buffered_count.to_string(),
1160        };
1161        let row_style = if s.buffered_count > RED_STALL_THRESHOLD {
1162            scheme.detail_stalled_severe
1163        } else {
1164            scheme.detail_stalled
1165        };
1166        let row_text = format!(
1167            "{:<owner$} {:>seq$}  {}",
1168            owner,
1169            s.expected_next_seq,
1170            buffered_cell,
1171            owner = OWNER_COL_WIDTH,
1172            seq = NEED_SEQ_COL_WIDTH,
1173        );
1174        lines.push(Line::from(Span::styled(row_text, row_style)));
1175    }
1176    if stalled.len() > TOP_N {
1177        let more = stalled.len() - TOP_N;
1178        lines.push(Line::from(Span::styled(
1179            l.ordering_more_row.replace("{n}", &more.to_string()),
1180            scheme.detail_label,
1181        )));
1182    }
1183
1184    // Footer (only with stalled sessions visible — this branch).
1185    lines.push(Line::default());
1186    lines.push(Line::from(Span::styled(
1187        l.ordering_footer,
1188        scheme.detail_label,
1189    )));
1190
1191    lines
1192}

1194/// Strip `<base58>` instance suffixes and the `@location` tail from an
1195/// `ActorAddr` string, leaving the human-meaningful
1196/// `{actor_label}.{proc_label}` form. Full address is still observable
1197/// via the API; this is the at-a-glance form for the TUI.
1198fn trim_owner_for_display(addr: &str) -> String {
1199    let pre_at = addr.split('@').next().unwrap_or(addr);
1200    let mut out = String::with_capacity(pre_at.len());
1201    let mut depth: i32 = 0;
1202    for c in pre_at.chars() {
1203        match c {
1204            '<' => depth += 1,
1205            '>' => depth -= 1,
1206            _ if depth == 0 => out.push(c),
1207            _ => {}
1208        }
1209    }
1210    out
1211}

1213/// Truncate `s` to `width` chars; if shorter, returns owned copy.
1214/// Truncation marker is `…` to make the cut visible to operators.
1215fn truncate_cell(s: &str, width: usize) -> String {
1216    if s.chars().count() <= width {
1217        return s.to_string();
1218    }
1219    let kept: String = s.chars().take(width.saturating_sub(1)).collect();
1220    format!("{kept}…")
1221}

1224    s.chars().take(8).collect()

1362/// Format one diagnostic probe result as a TUI row.
1363fn diag_result_line(r: &DiagResult, scheme: &ColorScheme, labels: &Labels) -> Line<'static> {
1364    let (icon, icon_style) = match &r.outcome {
1365        DiagOutcome::Pass { .. } => ("✓", scheme.detail_status_ok),
1366        DiagOutcome::Slow { .. } => ("⚠", scheme.detail_status_warn),
1367        DiagOutcome::Fail { .. } => ("✗", scheme.error),
1368    };
1369    let timing = match &r.outcome {
1370        DiagOutcome::Pass { elapsed_ms } | DiagOutcome::Slow { elapsed_ms } => {
1371            format!(" {}ms", elapsed_ms)
1372        }
1373        DiagOutcome::Fail { elapsed_ms, error } => format!(" {}ms — {}", elapsed_ms, error),
1374    };
1375    let mut spans = vec![
1376        Span::styled(format!(" {} ", icon), icon_style),
1377        Span::raw(r.label.clone()),
1378        Span::styled(timing, scheme.detail_label),
1379    ];
1380    if let Some(role) = r.note {
1381        let note = match role {
1382            DiagNodeRole::AdminServer => labels.diag_note_admin_server,
1383            DiagNodeRole::HostAgent => labels.diag_note_host_agent,
1384            DiagNodeRole::AdminServiceProc => labels.diag_note_admin_service_proc,
1385            DiagNodeRole::LocalClientProc => labels.diag_note_local_client_proc,
1386            DiagNodeRole::IntrospectionHandler => labels.diag_note_introspection_handler,
1387            DiagNodeRole::ActorLifecycleManager => labels.diag_note_actor_lifecycle_manager,
1388            DiagNodeRole::RootClientBridge => labels.diag_note_root_client_bridge,
1389            DiagNodeRole::CastActor => labels.diag_note_cast_actor,
1390            DiagNodeRole::ProcAgent => labels.diag_note_proc_agent,
1391            DiagNodeRole::UserProc => labels.diag_note_user_proc,
1392            DiagNodeRole::UserActor => labels.diag_note_user_actor,
1393        };
1394        spans.push(Span::styled(format!("  — {}", note), scheme.stat_url));
1395    }
1396    Line::from(spans)
1397}

1399/// Render a generic scrollable overlay in the detail pane area.
1400///
1401/// Supports an optional pinned status line (used by diagnostics)
1402/// above the scrollable content.
1403fn render_overlay(
1404    frame: &mut ratatui::Frame<'_>,
1405    area: Rect,
1406    overlay: &crate::overlay::Overlay,
1407    scheme: &ColorScheme,
1408) {
1409    let block = Block::default()
1410        .title(overlay.title.clone())
1411        .borders(Borders::ALL)
1412        .border_style(scheme.border);
1413    let inner = block.inner(area);
1414    frame.render_widget(block, area);
1415
1416    // If there's a pinned status line, split inner into status + scrollable.
1417    let content_area = if let Some(status) = &overlay.status_line {
1418        let chunks = Layout::default()
1419            .direction(Direction::Vertical)
1420            .constraints([Constraint::Length(1), Constraint::Min(1)])
1421            .split(inner);
1422        frame.render_widget(Paragraph::new(status.clone()), chunks[0]);
1423        chunks[1]
1424    } else {
1425        inner
1426    };
1427
1428    // Show "fetching…" only when loading with no lines AND no status line.
1429    // When a status_line is present it already communicates the loading state
1430    // (e.g. "Running… • fetching stack trace"), so the placeholder is redundant.
1431    let lines: Vec<Line<'static>> =
1432        if overlay.loading && overlay.lines.is_empty() && overlay.status_line.is_none() {
1433            vec![Line::from("fetching…")]
1434        } else {
1435            overlay.lines.clone()
1436        };
1437
1438    let max_scroll = (lines.len() as u16).saturating_sub(content_area.height);
1439    overlay.max_scroll.set(max_scroll);
1440    let scroll = overlay.scroll.get().min(max_scroll);
1441    overlay.scroll.set(scroll);
1442
1443    let paragraph = Paragraph::new(lines)
1444        .wrap(Wrap { trim: false })
1445        .scroll((scroll, 0));
1446    frame.render_widget(paragraph, content_area);
1447}
