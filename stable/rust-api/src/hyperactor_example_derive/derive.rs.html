9use std::collections::HashSet;
10use std::fmt::Debug;
11use std::time::Duration;
12
13use async_trait::async_trait;
14use hyperactor::Actor;
15use hyperactor::Context;
16use hyperactor::HandleClient;
17use hyperactor::Handler;
18use hyperactor::RefClient;
19use hyperactor::proc::Proc;
20use hyperactor::reference;
21use serde::Deserialize;
22use serde::Serialize;
23use typeuri::Named;
24
25#[derive(Handler, HandleClient, RefClient, Debug, Serialize, Deserialize, Named)]
26enum ShoppingList {
27    // Oneway messages dispatch messages asynchronously, with no reply.
28    Add(String),
29    Remove(String),
30
31    // Call messages dispatch a request, expecting a reply to the
32    // provided port, which must be in the last position.
33    Exists(String, #[reply] reference::OncePortRef<bool>),
34
35    List(#[reply] reference::OncePortRef<Vec<String>>),
36}
37
38// Example struct-based message types (demonstrating the new struct support)
39#[derive(Handler, HandleClient, RefClient, Debug, Serialize, Deserialize, Named)]
40struct ClearList {
41    reason: String,
42}
43
44#[derive(Handler, HandleClient, RefClient, Debug, Serialize, Deserialize, Named)]
45struct GetItemCount<C> {
46    category_filter: String,
47    #[reply]
48    reply: reference::OncePortRef<C>,
49}
50
51// Define an actor.
52#[derive(Debug, Default)]
53struct ShoppingListActor(HashSet<String>);
54
55impl Actor for ShoppingListActor {}
56
57// ShoppingListHandler is the trait generated by derive(Handler) above.
58// We implement the trait here for the actor, defining a handler for
59// each ShoppingList message.
60//
61// The `handle` attribute installs a handler that routes messages
62// to the `ShoppingListHandler` implementation directly. This can also
63// be done manually:
64//
65// ```ignore
66//<ShoppingListActor as ShoppingListHandler>
67//     ::handle(self, comm, message).await
68// ```
69#[async_trait]
70#[hyperactor::handle(ShoppingList)]
71impl ShoppingListHandler for ShoppingListActor {
72    async fn add(&mut self, _cx: &Context<Self>, item: String) -> Result<(), anyhow::Error> {
73        eprintln!("insert {}", item);
74        self.0.insert(item);
75        Ok(())
76    }
77
78    async fn remove(&mut self, _cx: &Context<Self>, item: String) -> Result<(), anyhow::Error> {
79        eprintln!("remove {}", item);
80        self.0.remove(&item);
81        Ok(())
82    }
83
84    async fn exists(&mut self, _cx: &Context<Self>, item: String) -> Result<bool, anyhow::Error> {
85        Ok(self.0.contains(&item))
86    }
87
88    async fn list(&mut self, _cx: &Context<Self>) -> Result<Vec<String>, anyhow::Error> {
89        Ok(self.0.iter().cloned().collect())
90    }
91}
92
93#[async_trait]
94#[hyperactor::handle(ClearList)]
95impl ClearListHandler for ShoppingListActor {
96    async fn clear_list(
97        &mut self,
98        _cx: &Context<Self>,
99        reason: String,
100    ) -> Result<(), anyhow::Error> {
101        eprintln!("clearing list: {}", reason);
102        self.0.clear();
103        Ok(())
104    }
105}
106
107#[async_trait]
108#[hyperactor::handle(GetItemCount<usize>)]
109impl GetItemCountHandler<usize> for ShoppingListActor {
110    async fn get_item_count(
111        &mut self,
112        _cx: &Context<Self>,
113        category_filter: String,
114    ) -> Result<usize, anyhow::Error> {
115        // Simple filter: count items containing the category_filter string
116        let count = self
117            .0
118            .iter()
119            .filter(|item| item.contains(&category_filter))
120            .count();
121        eprintln!("counting items with '{}': {}", category_filter, count);
122        Ok(count)
123    }
124}
125
126// Define a behavior `ShoppingApi`. Clients can use
127// `ActorRef<ShoppingApi>` instead of referencing the concrete
128// `ShoppingListActor` directly.
129hyperactor::behavior!(ShoppingApi, ShoppingList, ClearList, GetItemCount<usize>,);

131#[tokio::main]
132async fn main() -> Result<(), anyhow::Error> {
133    let mut proc = Proc::local();
134
135    // Spawn our actor, and get a handle for rank 0.
136    let shopping_list_actor: hyperactor::ActorHandle<ShoppingListActor> =
137        proc.spawn("shopping", ShoppingListActor::default())?;
138    let shopping_api: reference::ActorRef<ShoppingApi> = shopping_list_actor.bind();
139    // We join the system, so that we can send messages to actors.
140    let (client, _) = proc.instance("client").unwrap();
141
142    // todo: consider making this a macro to remove the magic names
143
144    // Derive(Handler) generates client methods, which call the
145    // remote handler provided an instance (send + open capability),
146    // the destination actor, and the method arguments.
147
148    shopping_list_actor.add(&client, "milk".into()).await?;
149    // We can perfrom operations through `shopping_api` too.
150    shopping_api.add(&client, "eggs".into()).await?;
151
152    println!(
153        "got milk? {}",
154        shopping_list_actor.exists(&client, "milk".into()).await?
155    );
156    println!(
157        "got yoghurt? {}",
158        shopping_list_actor
159            .exists(&client, "yoghurt".into())
160            .await?
161    );
162
163    shopping_list_actor.remove(&client, "milk".into()).await?;
164    println!(
165        "got milk now? {}",
166        shopping_list_actor.exists(&client, "milk".into()).await?
167    );
168
169    println!(
170        "shopping list: {:?}",
171        shopping_list_actor.list(&client).await?
172    );
173
174    // Add some more items to test filtering
175    shopping_list_actor
176        .add(&client, "dairy milk".into())
177        .await?;
178    shopping_list_actor
179        .add(&client, "whole grain bread".into())
180        .await?;
181
182    // Test the GetItemCount struct message (call message with reply)
183    let dairy_count = shopping_list_actor
184        .get_item_count(&client, "dairy".into())
185        .await?;
186    println!("items containing 'dairy': {}", dairy_count);
187
188    let grain_count = shopping_list_actor
189        .get_item_count(&client, "grain".into())
190        .await?;
191    println!("items containing 'grain': {}", grain_count);
192
193    // Test the ClearList struct message (oneway message)
194    shopping_list_actor
195        .clear_list(&client, "end of shopping session".into())
196        .await?;
197
198    println!(
199        "shopping list after clear: {:?}",
200        shopping_list_actor.list(&client).await?
201    );
202
203    let _ = proc
204        .destroy_and_wait::<()>(Duration::from_secs(1), None, "example cleanup")
205        .await?;
206    Ok(())
207}
