9//! Durable snapshot bundle export and import.
10//!
11//! A snapshot bundle is a directory containing 13 Arrow IPC files (one
12//! per table) and a JSON manifest. It represents exactly one snapshot
13//! and is the persistence mechanism for historical TUI debugging.
14//!
15//! # Bundle invariants (BN-*)
16//!
17//! - **BN-1 (one snapshot per bundle):** One bundle directory
18//!   represents exactly one `snapshot_id`. `snapshots.arrow` contains
19//!   exactly one row.
20//! - **BN-2 (one batch per table file):** Each `{table}.arrow` file
21//!   contains exactly one `RecordBatch`. `write_bundle` writes one
22//!   batch; `import_snapshot_bundle` reads one batch.
23//! - **BN-3 (portable manifest):** `manifest.json` contains no
24//!   path-dependent fields. The bundle remains valid if the directory
25//!   is moved or copied.
26//! - **BN-4 (canonical timestamp):** `snapshot_ts` is the only
27//!   persisted timestamp in the manifest. No `created_at`.
28//! - **BN-5 (manifest written last):** `manifest.json` is written
29//!   only after all table files are successfully written. An
30//!   incomplete bundle has no manifest and is rejected by import.
31//! - **BN-6 (import rejects invalid bundles):**
32//!   `import_snapshot_bundle` errors on: missing `manifest.json`,
33//!   unparseable manifest, unknown `version`, `tables` not matching
34//!   [`SNAPSHOT_TABLE_NAMES`] exactly (same entries, same order), or
35//!   missing `.arrow` file for any listed table.
36//! - **BN-7 (path derivation ownership):** The service generates
37//!   `snapshot_id` and derives
38//!   `{export_root}/snapshot-{snapshot_id}/`. The directory name and
39//!   the snapshot ID cannot diverge.
40//! - **BN-8 (round-trip fidelity):** For any `SnapshotData`,
41//!   `capture(export_root=some) → import → query` produces identical
42//!   SQL results to `capture(table_store=some) → query`.
43

66/// Portable metadata for a snapshot bundle.
67///
68/// Persisted as `manifest.json` inside the bundle directory. Contains
69/// no path-dependent fields (BN-3) and no `created_at` (BN-4).
70#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
71pub struct BundleManifest {
72    /// Format version. Currently always `1`.
73    pub version: u32,
74    /// Unique identifier for the snapshot (UUID v4).
75    pub snapshot_id: String,
76    /// Capture timestamp in microseconds since epoch (BN-4).
77    pub snapshot_ts: i64,
78    /// Summary counts of captured entities.
79    pub node_counts: NodeCounts,
80    /// Table names in canonical order, matching
81    /// [`SNAPSHOT_TABLE_NAMES`].
82    pub tables: Vec<String>,
83}

85/// Extract `snapshot_id` and `snapshot_ts` from the snapshots batch.
86///
87/// The snapshots batch must have exactly one row (BN-1) with columns
88/// `snapshot_id` (Utf8) and `snapshot_ts` (Int64).
89fn extract_snapshot_row(batch: &RecordBatch) -> anyhow::Result<(String, i64)> {
90    anyhow::ensure!(
91        batch.num_rows() == 1,
92        "snapshots batch must have exactly 1 row, found {}",
93        batch.num_rows(),
94    );
95    let id_col = batch
96        .column_by_name("snapshot_id")
97        .context("snapshots batch missing snapshot_id column")?;
98    let ts_col = batch
99        .column_by_name("snapshot_ts")
100        .context("snapshots batch missing snapshot_ts column")?;
101    let id = id_col.as_string::<i32>().value(0).to_owned();
102    let ts = ts_col.as_primitive::<Int64Type>().value(0);
103    Ok((id, ts))
104}

106/// Derive [`NodeCounts`] from batch row counts (write path).
107fn counts_from_batches(batches: &[NamedBatch]) -> NodeCounts {
108    let row_count = |name: &str| -> usize {
109        batches
110            .iter()
111            .find(|(n, _)| *n == name)
112            .map(|(_, b)| b.num_rows())
113            .unwrap_or(0)
114    };
115    NodeCounts {
116        nodes: row_count("nodes"),
117        children: row_count("children"),
118        root_nodes: row_count("root_nodes"),
119        host_nodes: row_count("host_nodes"),
120        proc_nodes: row_count("proc_nodes"),
121        actor_nodes: row_count("actor_nodes"),
122        actor_failures: row_count("actor_failures"),
123        actor_inbound_orderings: row_count("actor_inbound_orderings"),
124        ordering_sessions: row_count("ordering_sessions"),
125        actor_executions: row_count("actor_executions"),
126        active_handlers: row_count("active_handlers"),
127        resolution_errors: row_count("resolution_errors"),
128    }
129}

131/// Derive [`NodeCounts`] from loaded table name/row-count pairs
132/// (import path).
133fn counts_from_loaded(loaded: &[(&str, usize)]) -> NodeCounts {
134    let row_count = |name: &str| -> usize {
135        loaded
136            .iter()
137            .find(|(n, _)| *n == name)
138            .map(|(_, c)| *c)
139            .unwrap_or(0)
140    };
141    NodeCounts {
142        nodes: row_count("nodes"),
143        children: row_count("children"),
144        root_nodes: row_count("root_nodes"),
145        host_nodes: row_count("host_nodes"),
146        proc_nodes: row_count("proc_nodes"),
147        actor_nodes: row_count("actor_nodes"),
148        actor_failures: row_count("actor_failures"),
149        actor_inbound_orderings: row_count("actor_inbound_orderings"),

151        actor_executions: row_count("actor_executions"),
152        active_handlers: row_count("active_handlers"),
153        resolution_errors: row_count("resolution_errors"),
154    }
155}

157/// Write a snapshot bundle to `dir`.
158///
159/// All manifest metadata is derived from the batches themselves —
160/// `snapshot_id`, `snapshot_ts`, and `node_counts` are read from the
161/// data, not supplied separately. This makes it impossible to write
162/// a bundle whose manifest disagrees with its table files.
163///
164/// Creates `dir` (errors if it already exists), writes one `.arrow`
165/// file per table, then writes `manifest.json` last (BN-5).
166///
167/// On failure, the partially written directory is left in place — no
168/// cleanup attempt. Incomplete bundles lack `manifest.json` and are
169/// rejected by [`import_snapshot_bundle`].
170pub fn write_bundle(dir: &Path, batches: &[NamedBatch]) -> anyhow::Result<()> {
171    // Validate that batches matches SNAPSHOT_TABLE_NAMES exactly
172    // before writing anything.
173    let batch_names: Vec<&str> = batches.iter().map(|(n, _)| *n).collect();
174    let expected_names: Vec<&str> = SNAPSHOT_TABLE_NAMES.to_vec();
175    anyhow::ensure!(
176        batch_names == expected_names,
177        "batches {:?} do not match expected {:?}",
178        batch_names,
179        expected_names,
180    );
181
182    // Derive manifest metadata from the batches.
183    let snapshots_batch = &batches
184        .iter()
185        .find(|(n, _)| *n == "snapshots")
186        .context("missing snapshots batch")?
187        .1;
188    let (snapshot_id, snapshot_ts) = extract_snapshot_row(snapshots_batch)?;
189    let node_counts = counts_from_batches(batches);
190
191    // BN-5: create dir first; error if exists (UUID collision = bug).
192    fs::create_dir(dir).with_context(|| format!("failed to create bundle directory {:?}", dir))?;
193
194    // Write table files before manifest (BN-5).
195    for (name, batch) in batches {
196        let path = dir.join(format!("{}.arrow", name));
197        let file =
198            fs::File::create(&path).with_context(|| format!("failed to create {:?}", path))?;
199        let writer = BufWriter::new(file);
200        let mut ipc_writer = FileWriter::try_new(writer, &batch.schema())
201            .with_context(|| format!("failed to create IPC writer for {:?}", path))?;
202        ipc_writer
203            .write(batch)
204            .with_context(|| format!("failed to write batch to {:?}", path))?;
205        ipc_writer
206            .finish()
207            .with_context(|| format!("failed to finish IPC file {:?}", path))?;
208    }
209
210    // BN-5: manifest written last, derived from batch data.
211    let manifest = BundleManifest {
212        version: BUNDLE_VERSION,
213        snapshot_id,
214        snapshot_ts,
215        node_counts,
216        tables: SNAPSHOT_TABLE_NAMES.iter().map(|s| s.to_string()).collect(),
217    };
218    let manifest_path = dir.join("manifest.json");
219    let manifest_file = fs::File::create(&manifest_path)
220        .with_context(|| format!("failed to create {:?}", manifest_path))?;
221    serde_json::to_writer_pretty(BufWriter::new(manifest_file), &manifest)
222        .context("failed to write manifest.json")?;
223
224    Ok(())
225}

227/// Import a snapshot bundle from `dir`.
228///
229/// Validates the manifest (BN-6), reads one `RecordBatch` per table
230/// file (BN-2), and ingests into a fresh [`TableStore`].
231pub async fn import_snapshot_bundle(dir: &Path) -> anyhow::Result<(TableStore, BundleManifest)> {
232    // Read and validate manifest.
233    let manifest_path = dir.join("manifest.json");
234    let manifest_file = fs::File::open(&manifest_path)
235        .with_context(|| format!("failed to open {:?}", manifest_path))?;
236    let manifest: BundleManifest = serde_json::from_reader(BufReader::new(manifest_file))
237        .with_context(|| format!("failed to parse {:?}", manifest_path))?;
238
239    // BN-6: validate version.
240    anyhow::ensure!(
241        manifest.version == BUNDLE_VERSION,
242        "unsupported bundle version {} (expected {})",
243        manifest.version,
244        BUNDLE_VERSION,
245    );
246
247    // BN-6: validate tables match SNAPSHOT_TABLE_NAMES exactly.
248    let expected: Vec<String> = SNAPSHOT_TABLE_NAMES.iter().map(|s| s.to_string()).collect();
249    anyhow::ensure!(
250        manifest.tables == expected,
251        "manifest tables {:?} do not match expected {:?}",
252        manifest.tables,
253        expected,
254    );
255
256    // Load each table file and ingest into a fresh store.
257    // Track row counts per table for node_counts cross-check.
258    let store = TableStore::new_empty();
259    let mut loaded_batches: Vec<(&str, usize)> = Vec::new();
260    for table_name in &manifest.tables {
261        let path = dir.join(format!("{}.arrow", table_name));
262        let file = fs::File::open(&path).with_context(|| format!("failed to open {:?}", path))?;
263        let reader = FileReader::try_new(BufReader::new(file), None)
264            .with_context(|| format!("failed to create IPC reader for {:?}", path))?;
265
266        // BN-2: read exactly one batch.
267        let mut batch_count = 0;
268        let mut row_count = 0;
269        for batch_result in reader {
270            let batch =
271                batch_result.with_context(|| format!("failed to read batch from {:?}", path))?;
272
273            // BN-1: verify snapshots table has one row and matches
274            // the manifest.
275            if table_name == "snapshots" {
276                let (row_id, row_ts) = extract_snapshot_row(&batch)
277                    .context("snapshots.arrow failed BN-1 validation")?;
278                anyhow::ensure!(
279                    row_id == manifest.snapshot_id,
280                    "snapshots.arrow snapshot_id {:?} does not match \
281                     manifest {:?}",
282                    row_id,
283                    manifest.snapshot_id,
284                );
285                anyhow::ensure!(
286                    row_ts == manifest.snapshot_ts,
287                    "snapshots.arrow snapshot_ts {} does not match \
288                     manifest {}",
289                    row_ts,
290                    manifest.snapshot_ts,
291                );
292            }
293
294            row_count += batch.num_rows();
295            store.ingest_batch(table_name, batch).await?;
296            batch_count += 1;
297        }
298        anyhow::ensure!(
299            batch_count == 1,
300            "expected exactly 1 batch in {:?}, found {}",
301            path,
302            batch_count,
303        );
304        loaded_batches.push((table_name, row_count));
305    }
306
307    // Cross-check manifest.node_counts against actual file row
308    // counts. The manifest was derived from the batches on the write
309    // side, so a mismatch here indicates tampering or corruption.
310    let actual_counts = counts_from_loaded(&loaded_batches);
311    anyhow::ensure!(
312        actual_counts == manifest.node_counts,
313        "manifest node_counts {:?} do not match actual file row \
314         counts {:?}",
315        manifest.node_counts,
316        actual_counts,
317    );
318
319    Ok((store, manifest))
320}

744    // BN-8: query equivalence — import-path and push-path produce
745    // identical results for a representative query.
746    #[tokio::test]
747    async fn test_bundle_query_equivalence_with_push() {
748        // Push path.
749        let push_store = TableStore::new_empty();
750        let push_data = populated_snapshot("equiv");
751        let push_batches = drain_to_batches(push_data).unwrap();
752        for (name, batch) in &push_batches {
753            push_store.ingest_batch(name, batch.clone()).await.unwrap();
754        }
755
756        // Import path.
757        let dir = tempfile::tempdir().unwrap();
758        let bundle_dir = dir.path().join("snapshot-equiv");
759        let import_data = populated_snapshot("equiv");
760        let import_batches = drain_to_batches(import_data).unwrap();
761        write_bundle(&bundle_dir, &import_batches).unwrap();
762        let (import_store, _) = import_snapshot_bundle(&bundle_dir).await.unwrap();
763
764        // Query both paths.
765        let push_ctx = SessionContext::new();
766        register_all(&push_store, &push_ctx).await.unwrap();
767        let import_ctx = SessionContext::new();
768        register_all(&import_store, &import_ctx).await.unwrap();
769
770        // Compare row counts for every table.
771        for table in SNAPSHOT_TABLE_NAMES {
772            let push_count = query_row_count(&push_ctx, table).await.unwrap();
773            let import_count = query_row_count(&import_ctx, table).await.unwrap();
774            assert_eq!(
775                push_count, import_count,
776                "BN-8: row count mismatch for table {}",
777                table,
778            );
779        }
780
781        // Spot-check a representative value.
782        let push_df = push_ctx
783            .sql("SELECT proc_name FROM proc_nodes LIMIT 1")
784            .await
785            .unwrap();
786        let push_result = push_df.collect().await.unwrap();
787        let import_df = import_ctx
788            .sql("SELECT proc_name FROM proc_nodes LIMIT 1")
789            .await
790            .unwrap();
791        let import_result = import_df.collect().await.unwrap();
792        assert_eq!(push_result, import_result, "BN-8: value mismatch");
793    }
