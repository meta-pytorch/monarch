9//! Snapshot capture service.
10//!
11//! [`SnapshotService`] owns the capture pipeline as a single
12//! operation and publishes to the telemetry HTTP API, a durable bundle,
13//! or both.
14//!
15//! The service captures once via BFS, drains to `RecordBatch` pairs
16//! once via [`drain_to_batches`], and publishes the same batches to
17//! whichever destinations are active.
18//!
19//! # Usage
20//!
21//! [`SnapshotService::capture`] takes a *resolver* — a closure
22//! `Fn(&NodeRef) -> Future<Result<NodePayload>>` that resolves a
23//! single node reference via the mesh admin. In production this calls
24//! `MeshAdminAgent::resolve`; in tests it can be a stub backed by a
25//! `HashMap`.
26//!
27//! **One-shot capture** — capture a mesh snapshot on demand:
28//!
29//! ```ignore
30//! // Build a resolver that calls MeshAdminAgent::resolve for
31//! // each NodeRef.
32//! let resolve = |node_ref: &NodeRef| {
33//!     let admin_ref = admin_ref.clone();
34//!     let ref_string = node_ref.to_string();
35//!     async move {
36//!         let resp = admin_ref.resolve(instance, ref_string).await?;
37//!         resp.0.map_err(|e| anyhow::anyhow!("{}", e))
38//!     }
39//! };
40//!
41//! let service = SnapshotService::new(Some(HttpPublisher::new(base_url)));
42//! let result = service.capture(resolve, None).await?;
43//! println!("{} nodes captured", result.node_counts.nodes);
44//! ```
45//!
46//! [`capture`](SnapshotService::capture) is the full pipeline: BFS
47//! traversal, drain to `RecordBatch` pairs, publish to the configured
48//! live publisher and/or bundle export.
49//!
50//! **Periodic capture** — spawn a [`SnapshotCaptureActor`]:
51//!
52//! ```ignore
53//! let actor = SnapshotCaptureActor::new(
54//!     HttpPublisher::new(base_url),
55//!     admin_ref,
56//!     Duration::from_secs(30),
57//! );
58//! proc.spawn_with_uid(
59//!     hyperactor::Uid::singleton(hyperactor::Label::strip("snapshot_capture")),
60//!     actor,
61//! )?;
62//! // Actor is stopped by framework lifecycle on proc teardown.
63//! ```
64//!
65//! The actor reuses the same capture pipeline but is live-ingest only
66//! (`export_root` is always `None`). Overlapping ticks are skipped,
67//! not queued. Capture errors are logged and do not stop the timer.
68//!
69//! # Service invariants (SV-*)
70//!
71//! - **SV-1 (destination required):** `capture` returns `Err` when both
72//!   live telemetry and `export_root` are absent.
73//! - **SV-2 (single capture):** Each `capture` call performs exactly
74//!   one BFS traversal and one `drain_to_batches`.
75//! - **SV-3 (live publication):** When live telemetry is configured,
76//!   all 13 tables are ingested or published (delegates to PS-1..PS-7).
77//!   Publication is not atomic — a failure partway through may leave
78//!   some tables ingested/published and others not.
79//! - **SV-4 (counts before drain):** [`NodeCounts`] is computed from
80//!   [`SnapshotData`] before `drain_to_batches` consumes it.
81//! - **SV-5 (metadata correctness):** [`CaptureResult`] contains the
82//!   snapshot ID generated for this capture, `snapshot_ts` from the
83//!   snapshot row produced by `capture_snapshot`, accurate node
84//!   counts, non-negative wall-clock duration, and `bundle_path`
85//!   when a bundle was exported.
86//! - **SV-6 (bundle path derivation):** When `export_root` is `Some`,
87//!   the service derives the bundle directory as
88//!   `{export_root}/snapshot-{snapshot_id}/` after generating the
89//!   snapshot ID. The two cannot diverge (delegates to BN-7).
90//! - **SV-7 (cross-destination non-atomicity):** When live telemetry and
91//!   bundle export are both active, the operation is not atomic. If live ingest
92//!   succeeds and bundle writing fails (or vice versa), you have
93//!   partial success. The destinations are independent and the service
94//!   reports the error.
95//!
96//! # Periodic-trigger invariants (PT-*)
97//!
98//! - **PT-1 (positive interval):** Zero interval rejected before
99//!   spawn.
100//! - **PT-2 (live publisher by construction):** The periodic path takes a
101//!   concrete `HttpPublisher`, not an `Option`. A live publisher is guaranteed
102//!   by the API shape.
103//! - **PT-3 (immediate first fire):** First capture fires at spawn
104//!   time. Subsequent captures fire after each interval.
105//! - **PT-4 (single in-flight):** Overlapping ticks skipped via CAS.
106//!   `in_flight` is consulted only by the periodic loop; on-demand
107//!   `capture` calls do not check it.
108//! - **PT-5 (actor lifecycle):** The actor is stopped by framework
109//!   lifecycle (proc teardown via `DrainAndStop`). The framework
110//!   guarantees the current handler runs to completion before
111//!   stopping. At most one additional queued capture may execute
112//!   during drain. After stop, snapshot count stabilizes — no
113//!   unbounded reschedule tail. Tested by
114//!   `test_pt5_drain_halts_future_captures`.
115//! - **PT-6 (failure resilience):** Capture `Err` logged, loop
116//!   continues.
117//! - **PT-7 (live-ingest only):** Always `export_root = None`.

151/// Publishes snapshot batches to the telemetry sidecar HTTP ingest API.
152#[derive(Clone)]
153pub struct HttpPublisher {
154    base_url: String,
155    client: reqwest::Client,
156}
157
158impl HttpPublisher {
159    /// Build a publisher that writes to the telemetry sidecar HTTP API.
160    pub fn new(base_url: impl Into<String>) -> Self {
161        Self {
162            base_url: base_url.into().trim_end_matches('/').to_owned(),
163            client: reqwest::Client::new(),
164        }
165    }
166
167    async fn publish_batch(&self, table_name: &str, batch: &RecordBatch) -> anyhow::Result<()> {
168        let payload = serialize_batch(batch)?;
169        let url = format!("{}/api/ingest_snapshot/{}", self.base_url, table_name);
170        self.client
171            .post(url)
172            .header(CONTENT_TYPE, "application/vnd.apache.arrow.stream")
173            .body(payload)
174            .send()
175            .await?
176            .error_for_status()?;
177        Ok(())
178    }
179}

181/// Snapshot capture service.
182///
183/// Owns the capture-and-publish pipeline. Captures once per call and
184/// publishes to configured destinations.
185#[derive(Clone)]
186pub struct SnapshotService {
187    /// Live telemetry publisher. `None` means bundle export only.
188    publisher: Option<HttpPublisher>,
189    /// Overlap guard for the periodic trigger. CAS to acquire, reset
190    /// on completion.
191    in_flight: Arc<AtomicBool>,
192}
193
194impl SnapshotService {
195    /// Create a new snapshot service.
196    ///
197    /// When `publisher` is `Some`, captured snapshots are published into
198    /// live telemetry storage. When `None`, only bundle export is available.
199    pub fn new(publisher: Option<HttpPublisher>) -> Self {
200        Self {
201            publisher,
202            in_flight: Arc::new(AtomicBool::new(false)),
203        }
204    }
205
206    /// Capture a mesh snapshot and publish to configured destinations.
207    ///
208    /// Captures once via BFS, drains to `RecordBatch` pairs once,
209    /// then publishes to whichever destinations are active:
210    /// - If `self.publisher` is `Some`, publish to live telemetry storage.
211    /// - If `export_root` is `Some`, write a durable bundle to
212    ///   `{export_root}/snapshot-{snapshot_id}/`.
213    ///
214    /// At least one destination must be active (`publisher` or
215    /// `export_root`), otherwise the capture has no destination and
216    /// returns an error (SV-1).
217    pub async fn capture<F, Fut>(
218        &self,
219        resolve: F,
220        export_root: Option<&Path>,
221    ) -> anyhow::Result<CaptureResult>
222    where
223        F: Fn(&NodeRef) -> Fut,
224        Fut: Future<Output = anyhow::Result<NodePayload>>,
225    {
226        // SV-1: at least one destination must be active.
227        if self.publisher.is_none() && export_root.is_none() {
228            anyhow::bail!(
229                "snapshot capture requires at least one active destination \
230                 (live telemetry or export_root)"
231            );
232        }
233
234        let snapshot_id = Uuid::new_v4().to_string();
235        let t0 = Instant::now();
236        let data = capture_snapshot(&snapshot_id, &resolve).await?;
237        // SV-4: counts computed before drain consumes data.
238        let node_counts = NodeCounts::from_data(&data);
239        let snapshot_ts = data.snapshot.snapshot_ts;
240
241        // SV-2: drain once, publish to active destinations from the
242        // same batches.
243        let batches = drain_to_batches(data)?;
244
245        if let Some(ref publisher) = self.publisher {
246            for (name, batch) in &batches {
247                publisher.publish_batch(name, batch).await?;
248            }
249        }
250
251        // SV-6: service owns snapshot_id and derives directory name.
252        let bundle_path = match export_root {
253            Some(root) => {
254                let dir = root.join(format!("snapshot-{}", snapshot_id));
255                write_bundle(&dir, &batches)?;
256                Some(dir)
257            }
258            None => None,
259        };
260
261        Ok(CaptureResult {
262            snapshot_id,
263            snapshot_ts,
264            node_counts,
265            capture_duration_ms: t0.elapsed().as_secs_f64() * 1000.0,
266            bundle_path,
267        })
268    }
269}

271/// Private drop guard that resets `in_flight` to `false` on all exit
272/// paths (success, error, or early return).
273struct InFlightGuard<'a>(&'a AtomicBool);
274
275impl Drop for InFlightGuard<'_> {
276    fn drop(&mut self) {
277        self.0.store(false, Ordering::Release);
278    }
279}

281/// Execute one periodic capture tick.
282///
283/// Owns the CAS overlap check (PT-4), drop guard, capture call
284/// (PT-7: always `export_root = None`), and error logging (PT-6).
285/// Returns `true` if a capture was attempted (CAS succeeded),
286/// `false` if skipped due to overlap.
287///
288/// This is the unit of per-tick behavior, factored out of the
289/// spawned timer loop so it can be tested deterministically.
290async fn run_periodic_tick<F, Fut>(service: &SnapshotService, resolve: F) -> bool
291where
292    F: Fn(&NodeRef) -> Fut,
293    Fut: Future<Output = anyhow::Result<NodePayload>>,
294{
295    // PT-4: CAS to acquire overlap guard.
296    if service
297        .in_flight
298        .compare_exchange(false, true, Ordering::Acquire, Ordering::Relaxed)
299        .is_err()
300    {
301        tracing::warn!("periodic capture skipped: previous capture still in flight");
302        return false;
303    }
304    let _guard = InFlightGuard(&service.in_flight);
305
306    // PT-7: always None for export_root.
307    match service.capture(resolve, None).await {
308        Ok(result) => {
309            let c = &result.node_counts;
310            let short_id = &result.snapshot_id[..6];
311            if c.resolution_errors > 0 {
312                tracing::warn!(
313                    "capture partial: {}/{} nodes ({} resolution errors) in {:.0}ms [snap_id={}]",
314                    c.nodes - c.resolution_errors,
315                    c.nodes,
316                    c.resolution_errors,
317                    result.capture_duration_ms,
318                    short_id,
319                );
320            } else {
321                tracing::info!(
322                    "capture ok: {} nodes ({} hosts, {} procs, {} actors) in {:.0}ms [snap_id={}]",
323                    c.nodes,
324                    c.host_nodes,
325                    c.proc_nodes,
326                    c.actor_nodes,
327                    result.capture_duration_ms,
328                    short_id,
329                );
330            }
331        }
332        // PT-6: log and continue.
333        Err(e) => tracing::warn!("periodic capture failed: {:#}", e),
334    }
335    true
336}

344/// Periodic snapshot capture actor. Owns scheduling and lifecycle;
345/// delegates per-tick execution to [`run_periodic_tick`].
346///
347/// The spawn site sends the first `CaptureSnapshot` (PT-3). The
348/// handler reschedules after each tick via `post_after`.
349/// Stopped by framework lifecycle (`DrainAndStop` on proc teardown).
350#[hyperactor::export(handlers = [CaptureSnapshot])]
351pub struct SnapshotCaptureActor {
352    /// Shared snapshot capture pipeline and live-ingest publisher.
353    service: SnapshotService,
354    /// Typed admin actor reference used to resolve `NodeRef`s during
355    /// capture.
356    admin_ref: ActorRef<MeshAdminAgent>,
357    /// Delay between periodic capture ticks after the initial
358    /// immediate fire.
359    interval: Duration,
360}
361
362#[async_trait]
363impl Actor for SnapshotCaptureActor {
364    async fn init(&mut self, this: &Instance<Self>) -> Result<(), anyhow::Error> {
365        this.set_system();
366        Ok(())
367    }
368}
369
370#[async_trait]
371impl Handler<CaptureSnapshot> for SnapshotCaptureActor {
372    async fn handle(
373        &mut self,
374        cx: &Context<Self>,
375        _message: CaptureSnapshot,
376    ) -> Result<(), anyhow::Error> {
377        let admin_ref = self.admin_ref.clone();
378        let resolve = |node_ref: &NodeRef| {
379            let admin_ref = admin_ref.clone();
380            let ref_string = node_ref.to_string();
381            async move {
382                let resp = admin_ref.resolve(cx, ref_string).await?;
383                resp.0.map_err(|e| anyhow::anyhow!("{}", e))
384            }
385        };
386        run_periodic_tick(&self.service, resolve).await;
387
388        // Reschedule through the actor runtime instead of detached work.
389        cx.post_after(cx, CaptureSnapshot, self.interval);
390        Ok(())
391    }
392}

603    fn http_request_complete(buf: &[u8]) -> bool {
604        let Some(header_end) = buf.windows(4).position(|window| window == b"\r\n\r\n") else {
605            return false;
606        };
607        let headers = String::from_utf8_lossy(&buf[..header_end]);
608        let content_len = headers
609            .lines()
610            .find_map(|line| {
611                let (name, value) = line.split_once(':')?;
612                if name.eq_ignore_ascii_case("content-length") {
613                    value.trim().parse::<usize>().ok()
614                } else {
615                    None
616                }
617            })
618            .unwrap_or(0);
619        buf.len() >= header_end + 4 + content_len
620    }

622    fn spawn_snapshot_http_server(expected_requests: usize) -> (String, JoinHandle<Vec<Vec<u8>>>) {
623        let listener = TcpListener::bind("127.0.0.1:0").unwrap();
624        let addr = listener.local_addr().unwrap();
625        let handle = std::thread::spawn(move || {
626            let mut requests = Vec::new();
627            for _ in 0..expected_requests {
628                let (mut stream, _) = listener.accept().unwrap();
629                stream
630                    .set_read_timeout(Some(Duration::from_secs(5)))
631                    .unwrap();
632                let mut request = Vec::new();
633                loop {
634                    let mut chunk = [0_u8; 4096];
635                    let read = stream.read(&mut chunk).unwrap();
636                    if read == 0 {
637                        break;
638                    }
639                    request.extend_from_slice(&chunk[..read]);
640                    if http_request_complete(&request) {
641                        break;
642                    }
643                }
644                stream
645                    .write_all(
646                        b"HTTP/1.1 204 No Content\r\nContent-Length: 0\r\nConnection: close\r\n\r\n",
647                    )
648                    .unwrap();
649                requests.push(request);
650            }
651            requests
652        });
653        (format!("http://{}", addr), handle)
654    }
