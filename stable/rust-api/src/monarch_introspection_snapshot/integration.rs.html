9//! Integration helpers for wiring snapshot capture into the live
10//! telemetry system.
11//!
12//! - [`register_snapshot_schemas`] pre-registers empty snapshot
13//!   tables so the `QueryEngine` discovers them at setup time.
14//! - [`start_periodic_snapshots`] spawns a
15//!   [`SnapshotCaptureActor`](crate::service::SnapshotCaptureActor)
16//!   on the given proc.
17//!
18//! # Snapshot integration invariants (SI-*)
19//!
20//! - **SI-1 (snapshot tables discoverable):** After
21//!   `register_snapshot_schemas`, the 13 snapshot table names appear
22//!   in `DatabaseScanner.table_names()` and are discoverable by
23//!   `QueryEngine.setup_tables()`.
24//! - **SI-2 (snapshot tables queryable):** After periodic capture
25//!   fires, snapshot rows are queryable through the live telemetry
26//!   query path.
27//! - **SI-3 (shared storage):** The `TableStore` handle used by
28//!   `SnapshotService` shares the same underlying storage as the
29//!   `DatabaseScanner`. Snapshot ingestion is visible to telemetry
30//!   queries immediately.
31//! - **SI-4 (startup ordering):** Schema pre-registration happens
32//!   after `DatabaseScanner` creation but before `QueryEngine`
33//!   construction. Periodic snapshots start only after both telemetry
34//!   and admin are running.
35//! - **SI-5 (shutdown):** The snapshot capture actor is stopped by
36//!   framework lifecycle (proc teardown via `DrainAndStop`). The
37//!   framework guarantees the current handler runs to completion.
38//!   After stop, snapshot count stabilizes. No Python code calls stop
39//!   explicitly.
40//! - **SI-6 (unconditional schemas):** Schema pre-registration runs
41//!   whenever telemetry starts, regardless of whether periodic
42//!   capture is enabled. The query schema does not depend on config.
43//! - **SI-7 (resolver provenance):** The resolver's
44//!   `ActorRef<MeshAdminAgent>` comes directly from `spawn_admin`'s
45//!   typed return value. It crosses the Python boundary only as an
46//!   opaque capability token (`PyMeshAdminRef`). It is not
47//!   reconstructed from actor identity.

75/// Pre-register the 13 snapshot table schemas into `table_store`.
76///
77/// Each table is registered with a zero-row `RecordBatch` carrying
78/// the correct Arrow schema. This must be called before the
79/// `QueryEngine` constructs its `SessionContext`, because table
80/// discovery is static (one-shot at construction time).
81///
82/// Uses the same pattern as pyspy table pre-registration in
83/// `DatabaseScanner::new()` (`database_scanner.rs:251`).
84pub async fn register_snapshot_schemas(table_store: &TableStore) -> anyhow::Result<()> {
85    // Order matches SNAPSHOT_TABLE_NAMES (sorted).
86    let batches = [
87        (
88            "active_handlers",
89            ActiveHandlerRowBuffer::default().drain_to_record_batch()?,
90        ),
91        (
92            "actor_executions",
93            ActorExecutionRowBuffer::default().drain_to_record_batch()?,
94        ),
95        (
96            "actor_failures",
97            ActorFailureRowBuffer::default().drain_to_record_batch()?,
98        ),
99        (
100            "actor_inbound_orderings",
101            ActorInboundOrderingRowBuffer::default().drain_to_record_batch()?,
102        ),
103        (
104            "actor_nodes",
105            ActorNodeRowBuffer::default().drain_to_record_batch()?,
106        ),
107        (
108            "children",
109            ChildRowBuffer::default().drain_to_record_batch()?,
110        ),
111        (
112            "host_nodes",
113            HostNodeRowBuffer::default().drain_to_record_batch()?,
114        ),
115        ("nodes", NodeRowBuffer::default().drain_to_record_batch()?),
116        (
117            "ordering_sessions",
118            OrderingSessionRowBuffer::default().drain_to_record_batch()?,
119        ),
120        (
121            "proc_nodes",
122            ProcNodeRowBuffer::default().drain_to_record_batch()?,
123        ),
124        (
125            "resolution_errors",
126            ResolutionErrorRowBuffer::default().drain_to_record_batch()?,
127        ),
128        (
129            "root_nodes",
130            RootNodeRowBuffer::default().drain_to_record_batch()?,
131        ),
132        (
133            "snapshots",
134            SnapshotRowBuffer::default().drain_to_record_batch()?,
135        ),
136    ];
137
138    for (name, batch) in batches {
139        table_store.ingest_batch(name, batch).await?;
140    }
141
142    Ok(())
143}

145/// Spawn periodic snapshot capture as a `SnapshotCaptureActor`.
146///
147/// The actor is spawned on the given proc (same proc as the mesh
148/// admin). Lifecycle is framework-managed: proc teardown stops the
149/// actor via `DrainAndStop`. Fire-and-forget — returns `()`.
150///
151/// `cx` is any actor context for sending the initial
152/// `CaptureSnapshot` message to the spawned actor.
153pub fn start_periodic_snapshots(
154    cx: &impl hyperactor::context::Actor,
155    publisher: HttpPublisher,
156    admin_ref: hyperactor::ActorRef<MeshAdminAgent>,
157    interval: Duration,
158) -> anyhow::Result<()> {
159    let actor = SnapshotCaptureActor::new(publisher, admin_ref, interval);
160    start_periodic_snapshot_actor(cx, actor, interval)
161}
162
163fn start_periodic_snapshot_actor(
164    cx: &impl hyperactor::context::Actor,
165    actor: SnapshotCaptureActor,
166    interval: Duration,
167) -> anyhow::Result<()> {
168    anyhow::ensure!(
169        !interval.is_zero(),
170        "periodic capture interval must be non-zero"
171    );
172    let proc = cx.instance().proc();
173    let handle = proc.spawn_with_uid(Uid::singleton(Label::strip("snapshot_capture")), actor)?;
174    // PT-3: first capture fires at spawn time.
175    handle.post(cx, CaptureSnapshot);
176    Ok(())
177}

179#[cfg(test)]
180mod tests {
181    use super::*;
182    use crate::push::SNAPSHOT_TABLE_NAMES;
183
184    // SI-1: register_snapshot_schemas populates a TableStore with 13
185    // table names matching SNAPSHOT_TABLE_NAMES.
186    #[tokio::test]
187    async fn test_register_snapshot_schemas() {
188        let store = TableStore::new_empty();
189        register_snapshot_schemas(&store).await.unwrap();
190
191        let names = store.table_names().unwrap();
192        assert_eq!(names.len(), 13);
193
194        let expected: Vec<String> = SNAPSHOT_TABLE_NAMES.iter().map(|s| s.to_string()).collect();
195        assert_eq!(names, expected);
196    }
197
198    // SI-1: each pre-registered table has zero rows but a valid
199    // schema (non-zero columns).
200    #[tokio::test]
201    async fn test_register_snapshot_schemas_empty_but_valid() {
202        let store = TableStore::new_empty();
203        register_snapshot_schemas(&store).await.unwrap();
204
205        for name in SNAPSHOT_TABLE_NAMES {
206            let provider = store.table_provider(name).unwrap();
207            assert!(
208                provider.is_some(),
209                "table '{}' should have a provider",
210                name,
211            );
212        }
213    }
