9//! Relational row definitions for mesh introspection snapshots.
10//!
11//! Each struct maps to one Arrow table. `#[derive(RecordBatchRow)]`
12//! generates a `FooBuffer` type with `insert()`, `schema()`, `len()`,
13//! and `drain_to_record_batch()`.
14//!
15//! # Conversion principle
16//!
17//! Typed in Rust, canonical string IDs in SQL, numeric time in SQL.
18//! ID columns (`node_id`, `parent_id`, `child_id`,
19//! `failure_root_cause_actor`) are opaque strings derived from typed
20//! refs (`NodeRef`, `ActorAddr`, `ProcAddr`) at the conversion boundary.
21//! Timestamps are `i64` microseconds since epoch. Queries should
22//! treat ID columns as opaque join keys — do not parse them in SQL.
23//!
24//! # How to read this schema
25//!
26//! This schema stores one mesh-admin snapshot as a small relational
27//! graph.
28//!
29//! - [`SnapshotRow`] is the capture itself: one row per snapshot.
30//! - [`NodeRow`] is the base table of all nodes seen in that snapshot.
31//!   Every node has a stable opaque ID string (`node_id`) and a
32//!   `node_kind` telling you which subtype table holds its properties.
33//! - The subtype tables ([`RootNodeRow`], [`HostNodeRow`],
34//!   [`ProcNodeRow`], [`ActorNodeRow`], [`ResolutionErrorRow`]) store
35//!   the kind-specific columns for each node.
36//! - [`ChildRow`] stores the edges between nodes: each row says
37//!   "within this snapshot, parent P has child C at position K", plus
38//!   link-level classifications such as `is_system` and `is_stopped`.
39//! - [`ActorFailureRow`] stores optional failure detail for actor nodes
40//!   that were failed at snapshot time.
41//!
42//! The important modeling choice is that SQL IDs are opaque strings even
43//! though the in-memory model is typed. Conversion into these string IDs
44//! happens at the boundary when rows are produced.
45//!
46//! Another important choice is that [`ChildRow`] models the admin
47//! navigation graph, not a single ownership relation. The same node
48//! may legitimately appear under more than one parent when different
49//! navigation relations are projected into the same edge table.
50//!
51//! Example from the TUI:
52//!
53//! ```text
54//! │  ▼ unix:@EEgtjsbaWCOCPl3zsDeX4KNH  (2 procs)
55//! │    ├─ ▼ local  (447 actors: 5 system, 442 user)
56//! │      ├─ ▼ sieve-13dKVEf934hK[0]
57//! │        └─ ▼ sieve-13dKVEf934hK[1]
58//! │▸         └─ ▶ sieve-13dKVEf934hK[2]
59//! │      ├─ ▶ sieve-13dKVEf934hK[1]
60//! │      ├─ ▶ sieve-13dKVEf934hK[2]
61//! ```
62//!
63//! Here `sieve-...[2]` appears both as a direct child of proc `local`
64//! and as a child of actor `sieve-...[1]`. Within each relation the
65//! parent is unique, but the snapshot schema currently stores both
66//! relations in the same [`ChildRow`] table. Queries should therefore
67//! treat [`ChildRow`] as an edge table of parent→child links, not
68//! assume one globally unique parent for every non-root node.
69//!
70//! # Cardinality invariants
71//!
72//! - One snapshot has many nodes.
73//! - One node can have many outgoing child links.
74//! - Each node has exactly one kind-specific payload row, chosen by
75//!   `node_kind`.
76//! - An actor node has zero or one failure-detail row.
77//! - Root never appears as a child.
78//! - Non-root nodes may have more than one parent link, so the graph is
79//!   a rooted DAG rather than a strict tree.

83// Schema-row invariants (SR-*)
84//
85// These invariants are local to the Arrow schema defined in this
86// module. They describe guarantees enforced by the row definitions
87// and verified by the tests below.
88//
89// SR-1: Generated Arrow field names match the SQL-facing column names
90//       (the Rust field name IS the column name).
91// SR-2: ID and identity-string columns (`snapshot_id`, `node_id`,
92//       `parent_id`, `child_id`, `failure_root_cause_actor`,
93//       `instance_id`, `session_id`, `sender`, `name`) are Arrow `Utf8`.
94// SR-3: Time and count columns (`snapshot_ts`, `as_of`, `started_at`,
95//       `created_at`, `failure_occurred_at`, `num_hosts`,
96//       `host_num_procs`, `num_actors`, `stopped_retention_cap`,
97//       `failed_actor_count`, `messages_processed`,
98//       `total_processing_time_us`, `queue_depth`, `child_sort_key`,
99//       `skipped_session_count`, `known_session_count`,
100//       `returned_buffered_session_count`,
101//       `returned_buffered_message_count`,
102//       `returned_max_buffered_count`, `last_released_seq`,
103//       `expected_next_seq`, `buffered_count`, `oldest_buffered_seq`,
104//       `newest_buffered_seq`, `active_count`, `oldest_since`) are
105//       Arrow `Int64`.
106// SR-4: Flag columns (`is_system`, `is_stopped`, `is_poisoned`,
107//       `failure_is_propagated`, `enabled`, `snapshot_complete`,
108//       `complete`, `truncated`) are Arrow `Boolean`.
109// SR-5: Optional source fields map to nullable Arrow columns; required
110//       fields are non-nullable. The optional fields are:
111//       `ActorNodeRow.created_at`, `ActorNodeRow.last_message_handler`,
112//       `ActorFailureRow.failure_root_cause_name`,
113//       `OrderingSessionRow.sender`,
114//       `OrderingSessionRow.oldest_buffered_seq`,
115//       `OrderingSessionRow.newest_buffered_seq`.
116// SR-6: `drain_to_record_batch()` preserves row count and empties the
117//       buffer (zero rows remain after drain).

128/// Base table for all resolved nodes in a snapshot.
129///
130/// Tree structure is expressed solely through [`ChildRow`] — there
131/// is no parent column here.
132#[derive(Debug, Clone, PartialEq, RecordBatchRow)]
133pub struct NodeRow {
134    /// PK component. FK → [`SnapshotRow::snapshot_id`].
135    pub snapshot_id: String,
136    /// PK component. Canonical opaque node key from
137    /// `NodeRef::to_string()` — do not parse in SQL.
138    pub node_id: String,
139    /// Discriminator: `"root"` | `"host"` | `"proc"` | `"actor"` |
140    /// `"error"`. Determines which subtype table holds this node's
141    /// properties.
142    pub node_kind: String,
143    /// Payload freshness, microseconds since epoch. From
144    /// `NodePayload.as_of`.
145    pub as_of: i64,
146}

148/// Canonical parent→child relation. The model is a rooted DAG, not a
149/// strict tree: the same logical node may legitimately appear under
150/// more than one parent in the admin navigation model. This happens
151/// when one node is shown through multiple structural views rather
152/// than because of a cycle. The TUI already accounts for this by
153/// disambiguating appearances by `(reference, depth)` and rejecting
154/// only true ancestor cycles.
155#[derive(Debug, Clone, PartialEq, RecordBatchRow)]
156pub struct ChildRow {
157    /// PK component. FK → [`SnapshotRow::snapshot_id`].
158    pub snapshot_id: String,
159    /// PK component. FK → `Node(snapshot_id, node_id)`.
160    pub parent_id: String,
161    /// PK component. FK → `Node(snapshot_id, node_id)`.
162    pub child_id: String,
163    /// Sibling ordering within one parent.
164    pub child_sort_key: i64,
165    /// Link classification: child appears in parent's
166    /// `system_children` set (Root/Host/Proc).
167    pub is_system: bool,
168    /// Link classification: child appears in parent's
169    /// `stopped_children` set (Proc only).
170    pub is_stopped: bool,
171}

235    /// Actor type name.
236    pub actor_type: String,
237    /// Stable per-instance identifier (UUID canonical form). Sourced
238    /// from `ActorAttrsView::instance_id` (`Uuid::now_v7` assigned at
239    /// `Instance::new`).
240    pub instance_id: String,
241    /// Cumulative messages processed.
242    pub messages_processed: i64,
243    /// Actor creation time, microseconds since epoch. Nullable —
244    /// `Option<SystemTime>` in source model.

247    /// `Option<String>` in source model.
248    pub last_message_handler: Option<String>,
249    /// Cumulative message-processing wall time in microseconds.
250    pub total_processing_time_us: i64,
251    /// Per-actor work queue depth at snapshot time. Cardinality of the
252    /// actor's handler input queue; independent diagnostic from
253    /// [`ActorInboundOrderingRow::returned_buffered_message_count`]

255    pub queue_depth: i64,
256    /// Intrinsic system-actor classification from the actor itself.
257    pub is_system: bool,

260/// Snapshot-time inbound-ordering rollup for an actor. PK:
261/// `(snapshot_id, node_id)`, FK → `ActorNode(snapshot_id, node_id)`.
262///
263/// Conditional row: one row per actor that goes through the
264/// `OrderedSender` path (IO-1: `inbound_ordering: Some(...)`). Actors
265/// with `inbound_ordering: None` (structural absence — e.g., instance
266/// not built through `Instance::new`) have NO row, matching
267/// [`ActorFailureRow`] for `failure_info: None` actors.
268///
269/// `enabled == false` means buffering is off for this actor (messages
270/// flow via `direct_send`); the join to [`OrderingSessionRow`] on
271/// `(snapshot_id, node_id)` will be empty in that case.
272///
273/// `snapshot_complete == false` indicates IO-2 partial snapshot: rows
274/// in [`OrderingSessionRow`] for this key are a lower bound (skipped
275/// sessions are not enumerated). Use `skipped_session_count` to
276/// detect.
277#[derive(Debug, Clone, PartialEq, RecordBatchRow)]
278pub struct ActorInboundOrderingRow {
279    /// PK component. FK → `ActorNode(snapshot_id, node_id)`.
280    pub snapshot_id: String,
281    /// PK component. FK → `ActorNode(snapshot_id, node_id)`.
282    pub node_id: String,
283    /// `true` when reorder buffering is enabled for this actor.
284    pub enabled: bool,
285    /// IO-4: `true` iff `skipped_session_count == 0`.
286    pub snapshot_complete: bool,
287    /// Sessions held by a concurrent send at snapshot time
288    /// (`OrderedSender::snapshot` uses `try_lock`). Not enumerated in
289    /// [`OrderingSessionRow`].
290    pub skipped_session_count: i64,
291    /// IO-5: total live sessions = returned + skipped. The only
292    /// rollup that totals across both.
293    pub known_session_count: i64,
294    /// IO-6: sessions with `buffered_count > 0` AMONG RETURNED
295    /// sessions. Lower bound when `snapshot_complete = false`.
296    pub returned_buffered_session_count: i64,
297    /// IO-6: sum of `buffered_count` OVER RETURNED sessions. Lower
298    /// bound when `snapshot_complete = false`.
299    pub returned_buffered_message_count: i64,
300    /// IO-6: max of `buffered_count` OVER RETURNED sessions. Lower
301    /// bound when `snapshot_complete = false`.
302    pub returned_max_buffered_count: i64,
303}
304
305/// Per-returned-session detail of inbound-ordering state. PK:
306/// `(snapshot_id, node_id, session_id)`, FK →
307/// `ActorInboundOrdering(snapshot_id, node_id)`.
308///
309/// One row per RETURNED session at snapshot time. Skipped sessions
310/// (IO-2: held by a concurrent send during snapshotting) are not
311/// enumerated — only counted in
312/// [`ActorInboundOrderingRow::skipped_session_count`]. To detect
313/// "is this a complete view?", join to [`ActorInboundOrderingRow`]
314/// and check `snapshot_complete`.
315///
316/// `sender` is the session OWNER ActorAddr — the actor whose
317/// `Sequencer` assigned this session's SEQ_INFO. For direct sends and
318/// V1 cast that's the logical sender; for V0 legacy cast that's the
319/// forwarding `CommActor`. `None` is rare and occurs only when every
320/// message in this session bypassed the stamping path.
321#[derive(Debug, Clone, PartialEq, RecordBatchRow)]
322pub struct OrderingSessionRow {
323    /// PK component. FK → `ActorInboundOrdering(snapshot_id, node_id)`.
324    pub snapshot_id: String,
325    /// PK component. FK → `ActorInboundOrdering(snapshot_id, node_id)`.
326    pub node_id: String,
327    /// PK component. Session identifier (UUID canonical form).
328    pub session_id: String,
329    /// Session owner `ActorAddr` (canonical string form). `None` only
330    /// when every message in this session bypassed the normal
331    /// stamping path.
332    pub sender: Option<String>,
333    /// Highest seq released from the reorder buffer into the actor
334    /// work queue.
335    pub last_released_seq: i64,
336    /// `last_released_seq + 1` — the seq the next contiguous send
337    /// must carry for delivery without further buffering.
338    pub expected_next_seq: i64,
339    /// Messages held in the reorder buffer waiting for a seq gap to
340    /// be filled. Zero on healthy in-order sessions.
341    pub buffered_count: i64,
342    /// Lowest seq currently buffered. `None` when `buffered_count == 0`.
343    pub oldest_buffered_seq: Option<i64>,
344    /// Highest seq currently buffered. `None` when `buffered_count == 0`.
345    pub newest_buffered_seq: Option<i64>,
346}

348/// Snapshot-time execution rollup for an actor. PK:
349/// `(snapshot_id, node_id)`, FK → `ActorNode(snapshot_id, node_id)`.
350///
351/// Conditional row: one row per actor that reports execution
352/// (EX-1: `execution: Some(...)`). Actors with `execution: None`
353/// (unsupported — no snapshot callback installed) have NO row, matching
354/// [`ActorFailureRow`] / [`ActorInboundOrderingRow`].
355///
356/// `active_count` is the authoritative in-flight total (EX-3), never
357/// summed from [`ActiveHandlerRow`] rows (which may be a truncated
358/// prefix). `complete == false` (EX-2) means the per-handler detail was
359/// momentarily unavailable on that read — [`ActiveHandlerRow`] rows for
360/// this key are then empty while `active_count` stays authoritative.
361/// `truncated == true` (EX-4) means [`ActiveHandlerRow`] holds only the
362/// N oldest handler names.
363#[derive(Debug, Clone, PartialEq, RecordBatchRow)]
364pub struct ActorExecutionRow {
365    /// PK component. FK → `ActorNode(snapshot_id, node_id)`.
366    pub snapshot_id: String,
367    /// PK component. FK → `ActorNode(snapshot_id, node_id)`.
368    pub node_id: String,
369    /// EX-3: authoritative count of in-flight handler invocations.
370    pub active_count: i64,
371    /// EX-2: `true` iff the per-handler detail was captured on this read.
372    pub complete: bool,
373    /// EX-4: `true` iff [`ActiveHandlerRow`] is a prefix of the N oldest.
374    pub truncated: bool,
375}
376
377/// Per-handler detail of an actor's in-flight execution. PK:
378/// `(snapshot_id, node_id, name)`, FK →
379/// `ActorExecution(snapshot_id, node_id)`.
380///
381/// One row per in-flight handler name at snapshot time, aggregated by
382/// name and ordered oldest-first upstream. A prefix of the N oldest when
383/// the parent rollup's `truncated == true`; empty when its
384/// `complete == false` (EX-2). The parent's `active_count` stays the full
385/// total regardless.
386#[derive(Debug, Clone, PartialEq, RecordBatchRow)]
387pub struct ActiveHandlerRow {
388    /// PK component. FK → `ActorExecution(snapshot_id, node_id)`.
389    pub snapshot_id: String,
390    /// PK component. FK → `ActorExecution(snapshot_id, node_id)`.
391    pub node_id: String,
392    /// PK component. Handler name (e.g. a Python endpoint method name) —
393    /// the per-actor aggregation key.
394    pub name: String,
395    /// In-flight invocations of this handler.
396    pub active_count: i64,
397    /// Start time of the oldest in-flight invocation, microseconds since
398    /// epoch. Non-null (`SystemTime` in source, not `Option`).
399    pub oldest_since: i64,
400}

420    pub failure_root_cause_name: Option<String>,

425}
