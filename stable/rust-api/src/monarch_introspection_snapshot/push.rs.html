9//! Drain [`SnapshotData`] into [`TableStore`] tables.
10//!
11//! The shared infrastructure is [`drain_to_batches`], which converts
12//! a [`SnapshotData`] into 13 named `RecordBatch` pairs. The public
13//! entry point [`push_snapshot`] uses this to ingest all tables into
14//! a [`TableStore`].
15//!
16//! # Push-snapshot invariants (PS-*)
17//!
18//! - **PS-1 (full table coverage):** Every `push_snapshot` call
19//!   ingests all thirteen logical tables, including empty ones.
20//! - **PS-2 (canonical routing):** Each row family goes only to its
21//!   canonical table.
22//! - **PS-3 (append-preserving counts):** Each table's row count
23//!   increases by exactly the number of source rows for that family.
24//! - **PS-4 (empty-table queryability):** Empty row families still
25//!   become registered, queryable tables through
26//!   [`TableStore::table_provider()`].
27//! - **PS-5 (snapshot anchor):** The [`SnapshotRow`] is always
28//!   persisted to `snapshots`, including when all per-node vectors
29//!   are empty.
30//! - **PS-6 (query handoff):** After push, all registered tables can
31//!   be wired into a DataFusion `SessionContext` without extra
32//!   adaptation.
33//! - **PS-7 (error propagation):** Drain/ingest failure returns `Err`
34//!   and does not silently skip tables.
35

55/// Canonical table names in sorted order.
56///
57/// Every snapshot contains exactly these 13 tables. Both
58/// [`drain_to_batches`] and [`push_snapshot`] use this ordering.
59pub const SNAPSHOT_TABLE_NAMES: &[&str] = &[
60    "active_handlers",
61    "actor_executions",
62    "actor_failures",
63    "actor_inbound_orderings",
64    "actor_nodes",
65    "children",
66    "host_nodes",
67    "nodes",
68    "ordering_sessions",
69    "proc_nodes",
70    "resolution_errors",
71    "root_nodes",
72    "snapshots",
73];
74
75/// A table name paired with its drained `RecordBatch`.
76pub type NamedBatch = (&'static str, RecordBatch);
77
78/// Drain all row families in `data` to named `RecordBatch` pairs.
79///
80/// Returns exactly 13 pairs, one per table, in
81/// [`SNAPSHOT_TABLE_NAMES`] order. Empty families produce zero-row
82/// batches with correct schemas (PS-4).
83pub fn drain_to_batches(data: SnapshotData) -> anyhow::Result<Vec<NamedBatch>> {
84    // Each block: create buffer, insert rows, drain to RecordBatch.
85    // Order matches SNAPSHOT_TABLE_NAMES (sorted).
86
87    let mut failure_buf = ActorFailureRowBuffer::default();
88    for row in data.actor_failures {
89        failure_buf.insert(row);
90    }
91
92    let mut io_buf = ActorInboundOrderingRowBuffer::default();
93    for row in data.actor_inbound_orderings {
94        io_buf.insert(row);
95    }
96
97    let mut actor_buf = ActorNodeRowBuffer::default();
98    for row in data.actor_nodes {
99        actor_buf.insert(row);
100    }
101
102    let mut child_buf = ChildRowBuffer::default();
103    for row in data.children {
104        child_buf.insert(row);
105    }
106
107    let mut host_buf = HostNodeRowBuffer::default();
108    for row in data.host_nodes {
109        host_buf.insert(row);
110    }
111
112    let mut node_buf = NodeRowBuffer::default();
113    for row in data.nodes {
114        node_buf.insert(row);
115    }
116
117    let mut session_buf = OrderingSessionRowBuffer::default();
118    for row in data.ordering_sessions {
119        session_buf.insert(row);
120    }
121
122    let mut proc_buf = ProcNodeRowBuffer::default();
123    for row in data.proc_nodes {
124        proc_buf.insert(row);
125    }
126
127    let mut error_buf = ResolutionErrorRowBuffer::default();
128    for row in data.resolution_errors {
129        error_buf.insert(row);
130    }
131
132    // Logically singleton per snapshot (CS-2: root resolved exactly
133    // once), but modeled as a row family like the other subtype
134    // tables.
135    let mut root_buf = RootNodeRowBuffer::default();
136    for row in data.root_nodes {
137        root_buf.insert(row);
138    }
139
140    // PS-5: snapshot anchor is always persisted.
141    let mut snapshot_buf = SnapshotRowBuffer::default();
142    snapshot_buf.insert(data.snapshot);
143
144    let mut execution_buf = ActorExecutionRowBuffer::default();
145    for row in data.actor_executions {
146        execution_buf.insert(row);
147    }
148
149    let mut handler_buf = ActiveHandlerRowBuffer::default();
150    for row in data.active_handlers {
151        handler_buf.insert(row);
152    }
153
154    Ok(vec![
155        ("active_handlers", handler_buf.drain_to_record_batch()?),
156        ("actor_executions", execution_buf.drain_to_record_batch()?),
157        ("actor_failures", failure_buf.drain_to_record_batch()?),
158        ("actor_inbound_orderings", io_buf.drain_to_record_batch()?),
159        ("actor_nodes", actor_buf.drain_to_record_batch()?),
160        ("children", child_buf.drain_to_record_batch()?),
161        ("host_nodes", host_buf.drain_to_record_batch()?),
162        ("nodes", node_buf.drain_to_record_batch()?),
163        ("ordering_sessions", session_buf.drain_to_record_batch()?),
164        ("proc_nodes", proc_buf.drain_to_record_batch()?),
165        ("resolution_errors", error_buf.drain_to_record_batch()?),
166        ("root_nodes", root_buf.drain_to_record_batch()?),
167        ("snapshots", snapshot_buf.drain_to_record_batch()?),
168    ])
169}

171/// Drain a captured snapshot into table storage.
172///
173/// Calls [`drain_to_batches`] and ingests each batch into its
174/// canonical table via [`TableStore::ingest_batch`]. Empty families
175/// produce zero-row batches that still register the table schema
176/// (PS-4).
177pub async fn push_snapshot(table_store: &TableStore, data: SnapshotData) -> anyhow::Result<()> {
178    let batches = drain_to_batches(data)?;
179    for (name, batch) in batches {
180        table_store.ingest_batch(name, batch).await?;
181    }
182    Ok(())
183}

224    async fn register_all(store: &TableStore, ctx: &SessionContext) -> anyhow::Result<()> {
225        for name in store.table_names()? {
226            if let Some(provider) = store.table_provider(&name)? {
227                ctx.register_table(&name, provider)?;
228            }
229        }
230        Ok(())

244            .as_any()
245            .downcast_ref::<datafusion::arrow::array::Int64Array>()
246            .expect("cnt not Int64");

715    /// Canonical-UUID shape check: 36 chars with hyphens at positions
716    /// 8/13/18/23 and hex digits elsewhere. Used for PS-9 and the
717    /// PS-2 sweep on `instance_id`.
718    fn is_uuid_canonical(s: &str) -> bool {
719        s.len() == 36
720            && s.as_bytes()[8] == b'-'
721            && s.as_bytes()[13] == b'-'
722            && s.as_bytes()[18] == b'-'
723            && s.as_bytes()[23] == b'-'
724            && s.chars()
725                .enumerate()
726                .all(|(i, c)| matches!(i, 8 | 13 | 18 | 23) || c.is_ascii_hexdigit())
727    }
