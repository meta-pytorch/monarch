9//! Remote supervision actors.
10//!
11//! [`Supervisor`] is the parent-side proxy. It links to a [`Worker`], starts
12//! the configured link actor, forwards parent stop requests to the worker, and
13//! re-raises worker-reported supervision events through Hyperactor's ordinary
14//! `UnhandledSupervisionEvent` path.
15//!
16//! [`Worker`] is the worker-side container. It supervises one local child actor,
17//! accepts one active remote supervision session, starts the worker-side link
18//! actor, and reports child or link lifecycle events to the supervisor session.
19

70/// Parent-side proxy for one remote supervision relationship.
71#[derive(Debug)]
72pub struct Supervisor {
73    worker: ActorRef<WorkerLike>,
74    link: Option<KeepaliveLink>,
75    options: LinkOptions,
76    session_id: hyperactor::Uid,
77    link_handle: Option<ActorHandle<KeepaliveSupervisor>>,
78    session: Option<SupervisorSession>,
79    pending_stop: Option<PendingStop>,
80}

82impl Supervisor {
83    /// Create a supervisor proxy for `worker`.
84    pub fn new(worker: ActorRef<WorkerLike>, link: KeepaliveLink, options: LinkOptions) -> Self {
85        Self::new_uid(worker, link, options, hyperactor::Uid::anonymous())
86    }
87
88    /// Create a supervisor proxy with an explicit session id.
89    pub fn new_uid(
90        worker: ActorRef<WorkerLike>,
91        link: KeepaliveLink,
92        options: LinkOptions,
93        session_id: hyperactor::Uid,
94    ) -> Self {
95        Self {
96            worker,
97            link: Some(link),
98            options,
99            session_id,
100            link_handle: None,
101            session: None,
102            pending_stop: None,
103        }
104    }
105}

107#[async_trait]
108impl Actor for Supervisor {
109    async fn init(&mut self, this: &Instance<Self>) -> anyhow::Result<()> {
110        let (link_handle, link) = self
111            .link
112            .take()
113            .expect("supervisor initialized more than once")
114            .spawn_supervisor(this)?;
115        self.link_handle = Some(link_handle);
116        (&self.worker).post(
117            this,
118            Link {
119                session_id: self.session_id.clone(),
120                supervisor: this.port::<WorkerSupervisor>().bind(),
121                parent: this.self_addr().clone(),
122                link,
123                options: self.options.clone(),
124            },
125        );
126        Ok(())
127    }
128
129    async fn handle_stop(
130        &mut self,
131        this: &Instance<Self>,
132        mode: StopMode,
133        reason: &str,
134    ) -> anyhow::Result<()> {
135        self.pending_stop = Some(PendingStop {
136            mode,
137            reason: reason.to_string(),
138        });
139        self.send_pending_worker_stop(this)
140    }
141
142    async fn handle_supervision_event(
143        &mut self,
144        _this: &Instance<Self>,
145        event: &ActorSupervisionEvent,
146    ) -> anyhow::Result<bool> {
147        if self
148            .link_handle
149            .as_ref()
150            .is_some_and(|handle| handle.actor_addr() == &event.actor_id)
151        {
152            let event = self.synthesize_unreachable_event("supervision link failed");
153            return self.propagate_event(event);
154        }
155        Ok(!event.is_error())
156    }
157}

159#[async_trait]
160impl Handler<WorkerSupervisor> for Supervisor {
161    async fn handle(
162        &mut self,
163        cx: &Context<Self>,
164        message: WorkerSupervisor,
165    ) -> anyhow::Result<()> {
166        match message {
167            WorkerSupervisor::Linked {
168                session_id,
169                child,
170                display_name,
171            } => {
172                self.ensure_session(&session_id)?;
173                self.session = Some(SupervisorSession {
174                    child,
175                    display_name,
176                });
177                self.send_pending_worker_stop(cx)?;
178                Ok(())
179            }
180            WorkerSupervisor::LinkRejected { session_id, reason } => {
181                self.ensure_session(&session_id)?;
182                anyhow::bail!("remote supervision link rejected: {}", reason)
183            }
184            WorkerSupervisor::SupervisionEvent {
185                session_id,
186                event,
187                disposition: _,
188            } => {
189                self.ensure_session(&session_id)?;
190                self.propagate_event(event)
191            }
192            WorkerSupervisor::Unlinked { session_id, reason } => {
193                self.ensure_session(&session_id)?;
194                cx.exit(&reason)?;
195                Ok(())
196            }
197        }
198    }
199}
200
201impl Supervisor {
202    fn ensure_session(&self, session_id: &hyperactor::Uid) -> anyhow::Result<()> {
203        if session_id != &self.session_id {
204            anyhow::bail!("remote supervision session id mismatch");
205        }
206        Ok(())
207    }
208
209    fn send_pending_worker_stop(&mut self, cx: &Instance<Self>) -> anyhow::Result<()> {
210        let Some(stop) = self.pending_stop.take() else {
211            return Ok(());
212        };
213        (&self.worker).post(
214            cx,
215            SupervisedWorker::Stop {
216                session_id: self.session_id.clone(),
217                mode: stop.mode,
218                reason: stop.reason,
219            },
220        );
221        Ok(())
222    }
223
224    fn synthesize_unreachable_event(&self, reason: &str) -> ActorSupervisionEvent {
225        if let Some(session) = &self.session {
226            ActorSupervisionEvent::new(
227                session.child.clone(),
228                session.display_name.clone(),
229                ActorStatus::generic_failure(reason),
230                None,
231            )
232        } else {
233            ActorSupervisionEvent::new(
234                self.worker.actor_addr().clone(),
235                None,
236                ActorStatus::generic_failure(reason),
237                None,
238            )
239        }
240    }
241
242    fn propagate_event<T>(&self, event: ActorSupervisionEvent) -> anyhow::Result<T> {
243        Err(anyhow::Error::new(
244            ActorErrorKind::UnhandledSupervisionEvent(Box::new(event)),
245        ))
246    }
247}

279#[async_trait]
280impl<C> Actor for Worker<C>
281where
282    C: Actor + Send + 'static,
283{
284    async fn init(&mut self, this: &Instance<Self>) -> anyhow::Result<()> {
285        let child = self
286            .child
287            .take()
288            .expect("worker initialized more than once");
289        self.child_display_name = child.display_name();
290        self.child_handle = Some(this.spawn(child)?);
291        Ok(())
292    }
293
294    async fn handle_supervision_event(
295        &mut self,
296        this: &Instance<Self>,
297        event: &ActorSupervisionEvent,
298    ) -> anyhow::Result<bool> {
299        if self.is_link_event(event) {
300            self.handle_link_event(this, event)?;
301            return Ok(true);
302        }
303        if self.is_child_event(event) {
304            if let Some(session) = &self.session {
305                let supervisor = session.supervisor.clone();
306                let session_id = session.session_id.clone();
307                supervisor.post(
308                    this,
309                    WorkerSupervisor::SupervisionEvent {
310                        session_id: session_id.into_uid(),
311                        event: event.clone(),
312                        disposition: RemoteActorDisposition::Terminal,
313                    },
314                );
315            }
316            self.child_handle = None;
317            this.exit("supervised child stopped")?;
318            return Ok(true);
319        }
320
321        debug_assert!(
322            self.session.is_none(),
323            "worker with active session received supervision event for non-link non-child actor: {:?}",
324            event.actor_id
325        );
326        if self.session.is_none() {
327            Ok(true)
328        } else {
329            Ok(!event.is_error())
330        }
331    }
332
333    async fn handle_undeliverable_message(
334        &mut self,
335        _this: &Instance<Self>,
336        _envelope: Undeliverable<MessageEnvelope>,
337    ) -> anyhow::Result<()> {
338        self.handle_orphaned_supervisor("supervisor session undeliverable")
339    }
340}

342#[async_trait]
343impl<C> Handler<Link> for Worker<C>
344where
345    C: Actor + Send + 'static,
346{
347    async fn handle(&mut self, cx: &Context<Self>, message: Link) -> anyhow::Result<()> {
348        if self.session.is_some() {
349            let _ = message.supervisor.post(
350                cx,
351                WorkerSupervisor::LinkRejected {
352                    session_id: message.session_id,
353                    reason: "worker already linked".to_string(),
354                },
355            );
356            return Ok(());
357        }
358
359        let link_handle = message.link.spawn_worker(cx).await?;
360        let child_addr = self
361            .child_handle
362            .as_ref()
363            .expect("worker child must be spawned before linking")
364            .actor_addr()
365            .clone();
366        let supervisor = message.supervisor.clone();
367        let session_id = SessionId(message.session_id.clone());
368        self.session = Some(WorkerSession {
369            session_id,
370            supervisor: message.supervisor,
371            options: message.options,
372            link_handle,
373        });
374        supervisor.post(
375            cx,
376            WorkerSupervisor::Linked {
377                session_id: message.session_id.clone(),
378                child: child_addr,
379                display_name: self.child_display_name.clone(),
380            },
381        );
382        Ok(())
383    }
384}

386#[async_trait]
387impl<C> Handler<SupervisedWorker> for Worker<C>
388where
389    C: Actor + Send + 'static,
390{
391    async fn handle(
392        &mut self,
393        cx: &Context<Self>,
394        message: SupervisedWorker,
395    ) -> anyhow::Result<()> {
396        match message {
397            SupervisedWorker::Stop {
398                session_id,
399                mode,
400                reason,
401            } => {
402                self.accept_session_id(session_id)?;
403                self.stop_child(mode, &reason)?;
404            }
405            SupervisedWorker::Unlink { session_id, reason } => {
406                let session_id = self.accept_session_id(session_id)?;
407                if let Some(session) = self.session.take() {
408                    let _ = session.link_handle.stop(&reason);
409                    match session.options.orphan_policy {
410                        OrphanPolicy::Stop => self.stop_child(StopMode::Stop, &reason)?,
411                        OrphanPolicy::Detach => (),
412                    }
413
414                    let _ = session.supervisor.post(
415                        cx,
416                        WorkerSupervisor::Unlinked {
417                            session_id: session_id.into_uid(),
418                            reason,
419                        },
420                    );
421                }
422            }
423        }
424        Ok(())
425    }
426}

428impl<C: Actor> Worker<C> {
429    fn accept_session_id(&self, raw: hyperactor::Uid) -> anyhow::Result<SessionId> {
430        let Some(session) = &self.session else {
431            anyhow::bail!("worker is not linked");
432        };
433        let session_id = SessionId(raw);
434        if session_id != session.session_id {
435            anyhow::bail!("remote supervision session id mismatch");
436        }
437        Ok(session_id)
438    }
439
440    fn stop_child(&self, mode: StopMode, reason: &str) -> anyhow::Result<()> {
441        let Some(child) = &self.child_handle else {
442            return Ok(());
443        };
444        match mode {
445            StopMode::Stop => child.stop(reason)?,
446            StopMode::DrainAndStop => child.drain_and_stop(reason)?,
447        }
448        Ok(())
449    }
450
451    fn is_child_event(&self, event: &ActorSupervisionEvent) -> bool {
452        self.child_handle
453            .as_ref()
454            .is_some_and(|child| child.actor_addr() == &event.actor_id)
455    }
456
457    fn is_link_event(&self, event: &ActorSupervisionEvent) -> bool {
458        self.session
459            .as_ref()
460            .is_some_and(|session| session.link_handle.actor_id() == &event.actor_id)
461    }
462
463    fn handle_orphaned_supervisor(&mut self, reason: &str) -> anyhow::Result<()> {
464        match self.unlink_orphaned_supervisor(reason) {
465            Some(OrphanPolicy::Stop) => self.stop_child(StopMode::Stop, reason),
466            Some(OrphanPolicy::Detach) | None => Ok(()),
467        }
468    }
469
470    fn unlink_orphaned_supervisor(&mut self, reason: &str) -> Option<OrphanPolicy> {
471        let session = self.session.take()?;
472        let _ = session.link_handle.stop(reason);
473        Some(session.options.orphan_policy)
474    }
475
476    fn handle_link_event(
477        &mut self,
478        cx: &Instance<Self>,
479        event: &ActorSupervisionEvent,
480    ) -> anyhow::Result<()> {
481        let Some(session) = &self.session else {
482            return Ok(());
483        };
484        let supervisor = session.supervisor.clone();
485        let session_id = session.session_id.clone();
486        let orphan_policy = session.options.orphan_policy;
487        if let Some(child) = &self.child_handle {
488            supervisor.post(
489                cx,
490                WorkerSupervisor::SupervisionEvent {
491                    session_id: session_id.into_uid(),
492                    event: ActorSupervisionEvent::new(
493                        child.actor_addr().clone(),
494                        self.child_display_name.clone(),
495                        ActorStatus::generic_failure(format!("supervision link failed: {}", event)),
496                        None,
497                    ),
498                    disposition: RemoteActorDisposition::Unreachable,
499                },
500            );
501        }
502        let session = self
503            .session
504            .take()
505            .expect("link event must have an active session");
506        let _ = session.link_handle.stop("supervision link failed");
507        if orphan_policy == OrphanPolicy::Stop {
508            self.stop_child(StopMode::Stop, "supervision link failed")?;
509        }
510        Ok(())
511    }
512}

703    // Sets up the `Parent → Supervisor → Worker<TestChild>` harness
704    // (not Grandparent or synthetic-link). Spawns the worker, awaits
705    // the child's ready signal, then spawns the parent with
706    // `Supervisor::new_uid` using the supplied session id and
707    // options. Any post-spawn `Link`/`Linked` sleep is intentionally
708    // omitted - tests add it themselves next to the message send that
709    // depends on it.
710    async fn spawn_supervised_pair(
711        proc: &Proc,
712        inst: &Instance<()>,
713        session_id: hyperactor::Uid,
714        options: LinkOptions,
715        child_action: Option<TestChildAction>,
716    ) -> (
717        ActorAddr,                           // child address
718        ActorHandle<Worker<TestChild>>,      // worker handle
719        ActorHandle<Parent>,                 // parent handle
720        PortReceiver<String>,                // child stop notification
721        PortReceiver<ActorSupervisionEvent>, // parent's supervision events
722    ) {
723        // The ready receiver is consumed since no test needs more
724        // than one child address.
725        let (ready, mut ready_rx) = inst.open_port::<ActorAddr>();
726        let (stopped, stopped_rx) = inst.open_port::<String>();
727        let (events, events_rx) = inst.open_port::<ActorSupervisionEvent>();
728        let worker = proc
729            .spawn(
730                "worker",
731                Worker::new(test_child(ready, stopped, child_action)),
732            )
733            .unwrap();
734        let child_addr = ready_rx.recv().await.unwrap();
735        // Spawning Parent starts the link handshake: Parent::init
736        // spawns the Supervisor whose own init sends `Link` to the
737        // worker.
738        let parent = proc
739            .spawn(
740                "parent",
741                Parent {
742                    supervisor: Some(Supervisor::new_uid(
743                        worker.bind::<WorkerLike>(),
744                        KeepaliveLink::new(Duration::from_millis(5), Duration::from_secs(60)),
745                        options,
746                        session_id,
747                    )),
748                    events,
749                },
750            )
751            .unwrap();
752        (child_addr, worker, parent, stopped_rx, events_rx)
753    }
