18/// Derive a human-friendly label for a resolved node payload.
19///
20/// Kept as a free function (rather than an inherent `NodePayload`
21/// method) because `NodePayload` lives in
22/// `hyperactor_mesh::mesh_admin`; adding an extension trait here
23/// would be more ceremony than it's worth for a small formatting
24/// helper.
25///
26/// Uses `NodeProperties` to format a concise label for the tree view:
27/// roots show host counts, hosts show proc counts, procs show actor
28/// counts, and actors are rendered as `name[pid]` when the identity
29/// parses as an `ActorAddr`.
30pub(crate) fn derive_label(payload: &NodePayload) -> String {
31    match &payload.properties {
32        NodeProperties::Root { num_hosts, .. } => format!("Mesh Root ({} hosts)", num_hosts),
33        NodeProperties::Host {
34            addr, num_procs, ..
35        } => {
36            format!("{}  ({} procs)", addr, num_procs)
37        }
38        NodeProperties::Proc {
39            proc_name,
40            num_actors,
41            system_children,
42            stopped_children,
43            stopped_retention_cap,
44            is_poisoned,
45            failed_actor_count,
46            ..
47        } => {
48            let short = hyperactor::ProcAddr::from_str(proc_name)
49                .map(|pid| {
50                    pid.label()
51                        .map(|l| l.as_str().to_string())
52                        .unwrap_or_else(|| pid.id().to_string())
53                })
54                .unwrap_or_else(|_| proc_name.clone());
55            let num_system = system_children.len();
56            let num_stopped = stopped_children.len();
57            let num_user = num_actors.saturating_sub(num_system);
58            let total = num_actors + num_stopped;
59            let mut parts = Vec::new();
60            if num_system > 0 {
61                parts.push(format!("{} system", num_system));
62            }
63            if num_user > 0 {
64                parts.push(format!("{} user", num_user));
65            }
66            if num_stopped > 0 {
67                if num_stopped >= *stopped_retention_cap && *stopped_retention_cap > 0 {
68                    parts.push(format!("{} stopped (max retained)", num_stopped));
69                } else {
70                    parts.push(format!("{} stopped", num_stopped));
71                }
72            }
73            let base = if parts.is_empty() {
74                format!("{}  ({} actors)", short, total)
75            } else {
76                format!("{}  ({} actors: {})", short, total, parts.join(", "))
77            };
78            if *is_poisoned {
79                format!("{}  [POISONED: {} failed]", base, failed_actor_count)
80            } else {
81                base
82            }
83        }
84        NodeProperties::Actor { .. } => match &payload.identity {
85            NodeRef::Actor(actor_id) => format!(
86                "{}[{}]",
87                actor_id.log_name(),
88                actor_id
89                    .label()
90                    .map(|label| label.as_str().to_string())
91                    .unwrap_or_else(|| actor_id.uid().to_string())
92            ),
93            other => other.to_string(),
94        },
95        NodeProperties::Error { code, message } => {
96            format!("[error] {}: {}", code, message)
97        }
98    }
99}

101/// Derive a display label from a typed node reference without
102/// fetching.
103///
104/// For actor references, format as `name[uid]`; for all others, fall
105/// back to the `Display` representation.
106pub(crate) fn derive_label_from_ref(reference: &NodeRef) -> String {
107    match reference {
108        NodeRef::Actor(actor_id) => format!(
109            "{}[{}]",
110            actor_id.log_name(),
111            actor_id
112                .label()
113                .map(|label| label.as_str().to_string())
114                .unwrap_or_else(|| actor_id.uid().to_string())
115        ),
116        other => other.to_string(),
117    }
118}

120/// Produce a compact, human-readable summary string for a recorded
121/// event.
122///
123/// Prefers common message-like fields (`message`, then `msg`),
124/// otherwise renders a useful hint such as `handler: ...`. As a
125/// fallback, formats up to three key/value pairs from the event
126/// fields (using `format_value`) to keep the TUI line short; if
127/// nothing matches, falls back to the event `name`.
128pub(crate) fn format_event_summary(name: &str, fields: &Value) -> String {
129    if let Some(obj) = fields.as_object() {
130        if let Some(msg) = obj.get("message").and_then(|v| v.as_str()) {
131            return msg.to_string();
132        }
133        if let Some(msg) = obj.get("msg").and_then(|v| v.as_str()) {
134            return msg.to_string();
135        }
136        if let Some(handler) = obj.get("handler").and_then(|v| v.as_str()) {
137            return format!("handler: {}", handler);
138        }
139        if !obj.is_empty() {
140            let summary: Vec<String> = obj
141                .iter()
142                .take(3)
143                .map(|(k, v)| format!("{}={}", k, format_value(v)))
144                .collect();
145            if !summary.is_empty() {
146                return summary.join(" ");
147            }
148        }
149    }
150    name.to_string()
151}

153/// Format a JSON value into a short, single-line representation
154/// suitable for the TUI.
155///
156/// Strings/numbers/bools render as-is; `null` renders as `"null"`.
157/// Arrays and objects are summarized by their length/field count
158/// (e.g. `"[3]"`, `"{5}"`) to avoid dumping large payloads into the
159/// event list.
160pub(crate) fn format_value(v: &Value) -> String {
161    match v {
162        Value::String(s) => s.clone(),
163        Value::Number(n) => n.to_string(),
164        Value::Bool(b) => b.to_string(),
165        Value::Null => "null".to_string(),
166        Value::Array(arr) => format!("[{}]", arr.len()),
167        Value::Object(obj) => format!("{{{}}}", obj.len()),
168    }
169}

171/// Convert an ISO 8601 UTC timestamp (e.g.
172/// "2026-02-11T19:11:01.265Z") to a local-timezone HH:MM:SS string.
173/// Falls back to extracting the raw UTC time portion if parsing
174/// fails.
175pub(crate) fn format_local_time(timestamp: &str) -> String {
176    chrono::DateTime::parse_from_rfc3339(timestamp)
177        .map(|dt| {
178            dt.with_timezone(&chrono::Local)
179                .format("%H:%M:%S")
180                .to_string()
181        })
182        .unwrap_or_else(|_| timestamp.get(11..19).unwrap_or(timestamp).to_string())
183}

185/// Convert a `SystemTime` to a `chrono::DateTime<Utc>`.
186fn system_time_to_chrono(t: &SystemTime) -> chrono::DateTime<chrono::Utc> {
187    let dur = t.duration_since(std::time::UNIX_EPOCH).unwrap_or_default();
188    chrono::DateTime::from_timestamp(dur.as_secs() as i64, dur.subsec_nanos()).unwrap_or_default()
189}
190
191/// Format a `SystemTime` as a local-timezone HH:MM:SS string.
192pub(crate) fn format_system_time_local(t: &SystemTime) -> String {
193    system_time_to_chrono(t)
194        .with_timezone(&chrono::Local)
195        .format("%H:%M:%S")
196        .to_string()
197}

199/// Format a `SystemTime` as a human-readable relative time from now
200/// (e.g. "just now", "5s ago", "3m 12s ago").
201pub(crate) fn format_system_time_relative(t: &SystemTime) -> String {
202    match t.elapsed() {
203        Ok(d) if d.as_secs() < 2 => "just now".to_string(),
204        Ok(d) => format!("{} ago", humantime::format_duration(d)),
205        Err(_) => "just now".to_string(),
206    }
207}
208
209/// Format uptime from a `SystemTime` start point.
210///
211/// Rounds to nearest 30 seconds for cleaner display.
212pub(crate) fn format_system_time_uptime(started_at: &SystemTime) -> String {
213    match started_at.elapsed() {
214        Ok(d) => {
215            let total_secs = d.as_secs();
216            let rounded_secs = ((total_secs + 15) / 30) * 30;
217            humantime::format_duration(Duration::from_secs(rounded_secs)).to_string()
218        }
219        Err(_) => "unknown".to_string(),
220    }
221}
222
223/// Format a `SystemTime` as an ISO-8601 string for display.
224pub(crate) fn format_system_time_iso(t: &SystemTime) -> String {
225    system_time_to_chrono(t).to_rfc3339()
226}

228/// Format a byte count as a human-readable string.
229pub(crate) fn format_bytes(bytes: u64) -> String {
230    const KIB: u64 = 1024;
231    const MIB: u64 = 1024 * KIB;
232    const GIB: u64 = 1024 * MIB;
233    if bytes >= GIB {
234        format!("{:.1} GiB", bytes as f64 / GIB as f64)
235    } else if bytes >= MIB {
236        format!("{:.1} MiB", bytes as f64 / MIB as f64)
237    } else if bytes >= KIB {
238        format!("{:.1} KiB", bytes as f64 / KIB as f64)
239    } else {
240        format!("{bytes} B")
241    }
242}
