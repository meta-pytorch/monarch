9//! BFS capture of a mesh topology into [`SnapshotData`].
10//!
11//! The entry point is [`capture_snapshot`], which walks the mesh from
12//! root, resolving each node exactly once, and folds the results into
13//! a relational row set ready for ingestion.
14//!
15//! # Capture invariants (CS-*)
16//!
17//! - **CS-1 (snapshot-row-once):** Each successful `capture_snapshot`
18//!   emits exactly one [`SnapshotRow`].
19//! - **CS-2 (resolve-once-per-node):** Each distinct `NodeRef` is
20//!   resolved at most once per capture.
21//! - **CS-3 (edge-per-parent-payload):** Every [`ChildRow`] emitted
22//!   comes from a successfully resolved parent payload's `children`,
23//!   in parent enumeration order.
24//! - **CS-4 (duplicate-node-single-row):** If the same `NodeRef` is
25//!   reachable from multiple parents, [`SnapshotData`] contains one
26//!   node projection but one [`ChildRow`] per observed parent→child
27//!   edge.
28//! - **CS-5 (fold-homomorphism):** Folding a
29//!   [`ConvertedNode`](crate::convert::ConvertedNode) via
30//!   [`SnapshotData::push_converted`] appends exactly one
31//!   [`NodeRow`], exactly one subtype-table row matching `kind_row`,
32//!   optionally one [`ActorFailureRow`], and all of its
33//!   [`ChildRow`]s.
34//! - **CS-6 (resolution-error-boundary):** Resolver transport/query
35//!   failure aborts capture with `Err`. Only successfully resolved
36//!   payloads with `NodeProperties::Error` populate
37//!   `resolution_errors`.
38//! - **CS-7 (typed-visited-key):** Traversal dedup uses typed
39//!   `NodeRef` identity, not stringified IDs.
40//! - **CS-8 (snapshot-ts-capture-start):** `snapshot.snapshot_ts` is
41//!   captured once at traversal start and is independent of per-node
42//!   `as_of`.
43

67/// All row vectors produced by a single snapshot capture.
68#[derive(Debug)]
69pub struct SnapshotData {
70    /// Capture metadata (CS-1: exactly one per successful capture).
71    pub snapshot: SnapshotRow,
72    /// One row per resolved node (CS-2: each `NodeRef` resolved
73    /// once).
74    pub nodes: Vec<NodeRow>,
75    /// One row per parent→child edge (CS-3, CS-4: edges emitted
76    /// per-parent, so a multiply-reachable node has multiple edges).
77    pub children: Vec<ChildRow>,
78    /// Singleton on success — CS-2 applied to the single root entry
79    /// point means root is resolved exactly once.
80    pub root_nodes: Vec<RootNodeRow>,
81    /// One row per resolved host node.
82    pub host_nodes: Vec<HostNodeRow>,
83    /// One row per resolved proc node.
84    pub proc_nodes: Vec<ProcNodeRow>,
85    /// One row per resolved actor node.
86    pub actor_nodes: Vec<ActorNodeRow>,
87    /// One row per actor with `failure_info: Some(…)` (CV-3).
88    pub actor_failures: Vec<ActorFailureRow>,
89    /// One row per successfully resolved `NodeProperties::Error`
90    /// (CS-6: distinct from resolver transport failures).
91    pub resolution_errors: Vec<ResolutionErrorRow>,
92}
93
94impl SnapshotData {
95    /// Fold a single converted node into the accumulator (CS-5).
96    ///
97    /// This is the only place that branches on [`NodeKindRow`]. BFS
98    /// calls this and nothing else.
99    pub fn push_converted(&mut self, converted: ConvertedNode) {
100        self.nodes.push(converted.node);
101        self.children.extend(converted.children);
102        if let Some(f) = converted.actor_failure {
103            self.actor_failures.push(f);
104        }
105        match converted.kind_row {
106            NodeKindRow::Root(r) => self.root_nodes.push(r),
107            NodeKindRow::Host(h) => self.host_nodes.push(h),
108            NodeKindRow::Proc(p) => self.proc_nodes.push(p),
109            NodeKindRow::Actor(a) => self.actor_nodes.push(a),
110            NodeKindRow::ResolutionError(e) => self.resolution_errors.push(e),
111        }
112    }
113}

115/// Capture a full mesh snapshot by BFS from root.
116///
117/// The `resolve` closure is called once per distinct `NodeRef`
118/// (CS-2). Resolver transport failures abort capture immediately
119/// (CS-6). Successfully resolved `NodeProperties::Error` payloads are
120/// valid rows, not capture failures.
121pub async fn capture_snapshot<F, Fut>(snapshot_id: &str, resolve: F) -> anyhow::Result<SnapshotData>
122where
123    F: Fn(&NodeRef) -> Fut,
124    Fut: Future<Output = anyhow::Result<NodePayload>>,
125{
126    // CS-1, CS-8: one snapshot row, timestamp at capture start.
127    let snapshot = SnapshotRow {
128        snapshot_id: snapshot_id.to_owned(),
129        snapshot_ts: to_micros(SystemTime::now())
130            .context("failed to compute snapshot timestamp")?,
131    };
132
133    let mut data = SnapshotData {
134        snapshot,
135        nodes: Vec::new(),
136        children: Vec::new(),
137        root_nodes: Vec::new(),
138        host_nodes: Vec::new(),
139        proc_nodes: Vec::new(),
140        actor_nodes: Vec::new(),
141        actor_failures: Vec::new(),
142        resolution_errors: Vec::new(),
143    };
144
145    // CS-7: typed visited key.
146    let mut visited: HashSet<NodeRef> = HashSet::new();
147    let mut queue: VecDeque<NodeRef> = VecDeque::new();
148    queue.push_back(NodeRef::Root);
149
150    while let Some(node_ref) = queue.pop_front() {
151        // Dedup on dequeue — queue may contain already-visited refs.
152        if !visited.insert(node_ref.clone()) {
153            continue;
154        }
155
156        // CS-6: resolver failure is an immediate capture error.
157        let payload = resolve(&node_ref)
158            .await
159            .with_context(|| format!("failed to resolve {}", node_ref))?;
160
161        // Enqueue all children unconditionally; dedup happens on
162        // dequeue.
163        for child_ref in &payload.children {
164            queue.push_back(child_ref.clone());
165        }
166
167        // Project and fold.
168        let converted = convert_node(snapshot_id, &payload)
169            .with_context(|| format!("failed to convert {}", node_ref))?;
170        data.push_converted(converted);
171    }
172
173    Ok(data)
174}

222    /// Stub resolver backed by a map. Returns the payload for known
223    /// refs, error for unknown.
224    fn stub_resolver(
225        map: HashMap<NodeRef, NodePayload>,
226    ) -> impl Fn(&NodeRef) -> std::future::Ready<anyhow::Result<NodePayload>> {
227        move |node_ref: &NodeRef| {
228            let result = map
229                .get(node_ref)
230                .cloned()
231                .ok_or_else(|| anyhow::anyhow!("unknown ref: {}", node_ref));
232            std::future::ready(result)
233        }
234    }

246                .ok_or_else(|| anyhow::anyhow!("unknown ref: {}", node_ref));
247            std::future::ready(result)
248        }
