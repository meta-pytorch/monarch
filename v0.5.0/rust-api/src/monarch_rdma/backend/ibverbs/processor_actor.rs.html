9//! `IbvProcessorActor` — data-path child of `IbvManagerActor`.
10//!
11//! Owns the peer queue pairs handed off from the manager via
12//! [`RequestQueuePair`], caches local MR registrations via
13//! [`RegisterMr`], and runs batches of [`IbvOp`]s submitted through
14//! [`SubmitOps`]. Generic over a [`Manager`] actor type and a
15//! [`QueuePair`] type so unit tests can swap in mocks.

42/// A queue-pair value the processor owns end-to-end. The trait is a
43/// marker today; data-path methods (post send / poll completion)
44/// land alongside the real per-batch WR accounting in a follow-up.
45pub(super) trait QueuePair: std::fmt::Debug + Send + Sync + 'static {}

47/// Local-only message: ask the manager to register an MR for
48/// `[addr, addr + size)` and return the resulting view.
49#[cfg_attr(not(test), expect(dead_code, reason = "wired up in D105213930"))]
50#[derive(Debug)]
51pub(super) struct RegisterMr {
52    pub addr: usize,
53    pub size: usize,
54    pub reply: OncePortHandle<Result<IbvMemoryRegionView, String>>,
55}

57/// Local-only message: ask the manager to set up (and transfer
58/// ownership of) a peer queue pair. The manager hands the QP value
59/// across the reply port and forgets it on its own side — the
60/// processor becomes the sole owner.
61#[cfg_attr(not(test), expect(dead_code, reason = "wired up in D105213930"))]
62#[derive(Debug)]
63pub(super) struct RequestQueuePair<M: Referable, Qp: QueuePair> {
64    pub qp_key: QpKey,
65    #[expect(dead_code, reason = "read by IbvManagerActor handler in D105213930")]
66    pub remote_manager: ActorRef<M>,
67    pub reply: OncePortHandle<Result<Qp, String>>,
68}

70/// Bundle of trait bounds for an actor type that can serve the

73pub(super) trait Manager<Qp: QueuePair>:
74    Actor + Referable + Handler<RegisterMr> + Handler<RequestQueuePair<Self, Qp>>

92/// Single-purpose child actor that runs the ibverbs data path.
93/// Generic over a [`Manager`] (slow path) and [`QueuePair`] (data
94/// path) so it can be unit-tested with mocks.
95#[derive(Debug)]
96pub(super) struct IbvProcessorActor<M: Actor, Qp> {
97    manager: ActorHandle<M>,
98    config: IbvConfig,
99    mr_lru: LruCache<(usize, usize), IbvMemoryRegionView>,
100    /// Peer QPs the processor took ownership of via
101    /// [`RequestQueuePair`]. Each entry is its own `Arc<Mutex<Qp>>`
102    /// so the data path can hand out an
103    /// [`OwnedMutexGuard`](tokio::sync::OwnedMutexGuard) per QP and
104    /// lock distinct QPs concurrently from the parallel per-QP
105    /// futures in `process_batch`.
106    peer_qps: HashMap<QpKey, Arc<Mutex<Qp>>>,
107}
108
109#[cfg_attr(not(test), expect(dead_code, reason = "wired up in D105213930"))]
110impl<M, Qp> IbvProcessorActor<M, Qp>
111where
112    M: Manager<Qp>,
113    Qp: QueuePair,
114{
115    /// `mr_lru_capacity` should be non-zero. A zero value is
116    /// clamped to one with a `tracing::warn`; we treat the
117    /// degenerate single-entry LRU as the floor rather than
118    /// crashing the manager.
119    pub(super) fn new(manager: ActorHandle<M>, config: IbvConfig, mr_lru_capacity: usize) -> Self {
120        let mr_lru_capacity = NonZeroUsize::new(mr_lru_capacity).unwrap_or_else(|| {
121            tracing::warn!(
122                "RDMA_MR_LRU_CACHE_SIZE was 0; clamping to 1 (LRU disabled in practice)"
123            );
124            NonZeroUsize::MIN
125        });
126        Self {
127            manager,
128            config,
129            mr_lru: LruCache::new(mr_lru_capacity),
130            peer_qps: HashMap::new(),
131        }
132    }
133
134    /// Resolve `(addr, size)` via the LRU cache, asking the manager
135    /// on miss.
136    async fn resolve_mr(
137        &mut self,
138        cx: &Context<'_, Self>,
139        addr: usize,
140        size: usize,
141    ) -> Result<IbvMemoryRegionView, String> {
142        if let Some(mrv) = self.mr_lru.get(&(addr, size)).cloned() {
143            return Ok(mrv);
144        }
145        let (reply, rx) = cx
146            .mailbox()
147            .open_once_port::<Result<IbvMemoryRegionView, String>>();
148        self.manager.post(cx, RegisterMr { addr, size, reply });
149        let mrv = rx
150            .recv()
151            .await
152            .map_err(|e| {
153                format!(
154                    "MR registration port closed [virtual_addr=0x{:x}, size={}]: {}",
155                    addr, size, e
156                )
157            })?
158            .map_err(|e| {
159                format!(
160                    "MR registration failed [virtual_addr=0x{:x}, size={}]: {}",
161                    addr, size, e
162                )
163            })?;
164        self.mr_lru.put((addr, size), mrv.clone());
165        Ok(mrv)
166    }
167
168    /// Return an [`OwnedMutexGuard`] over the QP for `qp_key`,
169    /// requesting a fresh QP from the manager and inserting it into
170    /// `peer_qps` on miss. The QP stays in the map for the lifetime
171    /// of the actor; the caller drives the QP through the guard and
172    /// the lock releases when the guard drops. Locking is via
173    /// `try_lock_owned` because the actor serializes its handlers
174    /// and only one `submit_ops` is in flight at a time, so the
175    /// guard is always uncontended at acquisition.
176    async fn get_or_request_qp(
177        &mut self,
178        cx: &Context<'_, Self>,
179        qp_key: &QpKey,
180        remote_manager: ActorRef<M>,
181    ) -> Result<OwnedMutexGuard<Qp>, String> {
182        if !self.peer_qps.contains_key(qp_key) {
183            let (reply, rx) = cx.mailbox().open_once_port::<Result<Qp, String>>();
184            self.manager.post(
185                cx,
186                RequestQueuePair {
187                    qp_key: qp_key.clone(),
188                    remote_manager,
189                    reply,
190                },
191            );
192            let qp = rx
193                .recv()
194                .await
195                .map_err(|e| {
196                    format!(
197                        "QP setup port closed for {} -> {} on {}: {}",
198                        qp_key.self_device, qp_key.other_id, qp_key.other_device, e
199                    )
200                })?
201                .map_err(|e| {
202                    format!(
203                        "QP setup failed for {} -> {} on {}: {}",
204                        qp_key.self_device, qp_key.other_id, qp_key.other_device, e
205                    )
206                })?;
207            self.peer_qps
208                .insert(qp_key.clone(), Arc::new(Mutex::new(qp)));
209        }
210        Ok(Arc::clone(self.peer_qps.get(qp_key).expect("just ensured"))
211            .try_lock_owned()
212            .expect("no other holders"))
213    }
214
215    async fn process_batch(
216        &mut self,
217        cx: &Context<'_, Self>,
218        ops: Vec<IbvOp<M>>,
219        timeout: Duration,
220    ) -> Vec<Result<(), String>> {
221        let n = ops.len();
222        let mut results: Vec<Result<(), String>> = (0..n)
223            .map(|i| Err(format!("op {} not processed (internal bug)", i)))
224            .collect();
225
226        // Resolve MRs and group ops by QP key.
227        let mut by_qp: HashMap<QpKey, Vec<(usize, IbvOp<M>, IbvMemoryRegionView)>> = HashMap::new();
228        for (i, op) in ops.into_iter().enumerate() {
229            match self
230                .resolve_mr(cx, op.local_memory.addr(), op.local_memory.size())
231                .await
232            {
233                Ok(mrv) => {
234                    let qp_key = QpKey {
235                        self_device: mrv.device_name.clone(),
236                        other_id: op.remote_manager.actor_addr().id().clone(),
237                        other_device: op.remote_buffer.device_name.clone(),
238                    };
239                    by_qp.entry(qp_key).or_default().push((i, op, mrv));
240                }
241                Err(e) => {
242                    results[i] = Err(e);
243                }
244            }
245        }
246
247        // Acquire an `OwnedMutexGuard` per QP and build a per-QP
248        // future. The guards don't borrow `self`, so the resulting
249        // futures can run concurrently in `join_all`.
250        let max_send_wr = self.config.max_send_wr;
251        let max_rd_atomic = self.config.max_rd_atomic as u32;
252        let mut group_futures = Vec::with_capacity(by_qp.len());
253        for (qp_key, group) in by_qp {
254            let remote_manager = group[0].1.remote_manager.clone();
255            let guard = match self.get_or_request_qp(cx, &qp_key, remote_manager).await {
256                Ok(g) => g,
257                Err(msg) => {
258                    for (orig_idx, _, _) in group {
259                        results[orig_idx] = Err(msg.clone());
260                    }
261                    continue;
262                }
263            };
264            group_futures.push(async move {
265                let mut qp = guard;
266                process_qp_group(&mut *qp, group, timeout, max_send_wr, max_rd_atomic).await
267            });
268        }
269
270        for group_results in futures::future::join_all(group_futures).await {
271            for (orig_idx, res) in group_results {
272                results[orig_idx] = res;
273            }
274        }
275
276        results
277    }
278}

280/// Stub. The real per-QP processing — `PostedOps`-driven posting,
281/// CQ polling, and WR accounting — lands in a follow-up. Marks
282/// every op in `group` as successful so the actor can be exercised
283/// end-to-end without a working data path.
284async fn process_qp_group<M, Qp>(
285    _qp: &mut Qp,
286    group: Vec<(usize, IbvOp<M>, IbvMemoryRegionView)>,
287    _timeout: Duration,
288    _max_send_wr: u32,
289    _max_rd_atomic: u32,
290) -> Vec<(usize, Result<(), String>)>
291where
292    M: Referable,
293    Qp: QueuePair,
294{
295    group
296        .into_iter()
297        .map(|(orig_idx, _, _)| (orig_idx, Ok(())))
298        .collect()
299}

312#[async_trait]
313impl<M, Qp> Handler<SubmitOps<M>> for IbvProcessorActor<M, Qp>
314where
315    M: Manager<Qp>,
316    Qp: QueuePair,
317{
318    async fn handle(&mut self, cx: &Context<Self>, msg: SubmitOps<M>) -> anyhow::Result<()> {
319        let SubmitOps {
320            ops,
321            timeout,
322            reply,
323        } = msg;
324        let results = self.process_batch(cx, ops, timeout).await;
325        reply.post(cx, results);
326        Ok(())
327    }

612    async fn submit_ops_mr_lru_eviction_re_requests() {

614        // evicts the least-recently-used entry. A subsequent op for

625            ],
626        )
627        .await;
628        // (0x1000, 4096) is no longer in the cache. Requesting it
629        // again triggers a fresh register_mr call.
630        let _ = submit(&h, vec![fake_op(0x1000, 4096, "dev0")]).await;
631        assert_eq!(
632            h.mgr_inner.lock().unwrap().register_mr_calls,
633            vec![
634                (0x1000, 4096),
635                (0x2000, 4096),
